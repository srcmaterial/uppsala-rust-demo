132    add_bos: bool,
133    ignore_merges: bool,
134    ty: PreTokenizerType,
135}

137impl GGUFPreTokenizerConfig {
138    pub(crate) fn build(
139        &self,
140        vocab: HashMap<String, u32>,
141        types: Vec<u8>,
142        merges: Vec<(String, String)>,
143        bos: &str,
144        eos: &str,
145    ) -> Result<Tokenizer, tokenizers::Error> {
146        let mut special_tokens: Vec<_> = vocab
147            .iter()
148            .filter_map(|(k, v)| {
149                // 1=normal, 2=unknown, 3=control, 4=user defined, 5=unused, 6=byte
150                let special_token = types[*v as usize];
151                if special_token == 1 {
152                    return None;
153                }
154                Some(AddedToken::from(k.to_string(), true))
155            })
156            .collect();
157        let bos_token = vocab[bos];
158        let bpe_tokenizer = BpeBuilder::new()
159            .vocab_and_merges(vocab, merges)
160            .ignore_merges(self.ignore_merges)
161            .build()?;
162
163        let byte_level_pre = ByteLevel::new(false, true, false);
164        let byte_level_post = ByteLevel::new(true, false, true);
165        let byte_level_decoder = ByteLevel::new(true, true, true);
166
167        let mut tokenizer = Tokenizer::new(bpe_tokenizer);
168        let mut pre_tokenizers = Vec::new();
169        for regex in self.ty.regexes() {
170            let split = tokenizers::pre_tokenizers::split::Split::new(
171                SplitPattern::Regex(regex.to_string()),
172                tokenizers::SplitDelimiterBehavior::Isolated,
173                false,
174            )?;
175            pre_tokenizers.push(split.into());
176        }
177        pre_tokenizers.push(byte_level_pre.into());
178        let pre_tokenizer = tokenizers::pre_tokenizers::sequence::Sequence::new(pre_tokenizers);
179        tokenizer.with_pre_tokenizer(Some(pre_tokenizer));
180        tokenizer.with_decoder(Some(byte_level_decoder));
181        let mut post_processors = Vec::new();
182        post_processors.push(byte_level_post.into());
183        if self.add_bos {
184            let special_toks = vec![SpecialToken::from((bos_token, bos.to_string()))];
185            post_processors.push(
186                TemplateProcessing::builder()
187                    .single(
188                        tokenizers::processors::template::Template::try_from(vec![
189                            format!("{bos}:0"),
190                            "$A:0".to_string(),
191                        ])
192                        .unwrap(),
193                    )
194                    .pair(
195                        tokenizers::processors::template::Template::try_from(vec![
196                            format!("{bos}:0"),
197                            "$A:0".to_string(),
198                            format!("{bos}:1"),
199                            "$B:1".to_string(),
200                        ])
201                        .unwrap(),
202                    )
203                    .special_tokens(special_toks)
204                    .build()
205                    .unwrap()
206                    .into(),
207            );
208        }
209        tokenizer.with_post_processor(Some(tokenizers::processors::sequence::Sequence::new(
210            post_processors,
211        )));
212        special_tokens.push(AddedToken::from(bos.to_string(), true));
213        special_tokens.push(AddedToken::from(eos.to_string(), true));
214        tokenizer.add_special_tokens(&special_tokens);
215
216        Ok(tokenizer)
217    }
218}
