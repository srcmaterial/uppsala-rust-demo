4pub(crate) fn fast_cpu_silu(tensor: &Tensor) -> candle_core::Result<Tensor> {
5    use rayon::iter::ParallelIterator;
6    use rayon::slice::ParallelSliceMut;
7
8    static SILU_CACHE: std::sync::OnceLock<Vec<f32>> = std::sync::OnceLock::new();
9
10    fn silu_cache() -> &'static Vec<f32> {
11        SILU_CACHE.get_or_init(|| {
12            let f16_count = 2 << 16;
13            let mut cache = Vec::with_capacity(f16_count);
14            for i in 0..f16_count {
15                let x = half::f16::from_bits(i as u16).to_f32();
16                cache.push(x / (1. + (-x).exp()));
17            }
18            cache
19        })
20    }
21
22    #[inline(always)]
23    fn silu(x: &mut f32) {
24        let cache: &[f32] = silu_cache();
25        let as_f16 = half::f16::from_f32(*x);
26        let as_f16 = as_f16.to_bits();
27        let as_f16 = as_f16 as usize;
28        *x = cache[as_f16];
29    }
30
31    #[inline(always)]
32    fn silu_chunk(chunk: &mut [f32; 16]) {
33        for entry in chunk {
34            silu(entry);
35        }
36    }
37
38    let device = tensor.device();
39
40    if matches!(device, candle_core::Device::Cpu) {
41        let shape = tensor.shape();
42
43        let mut as_vec = tensor.flatten_all()?.to_vec1::<f32>()?;
44        let mut iter = as_vec.par_chunks_exact_mut(16);
45        for item in iter.remainder() {
46            silu(item)
47        }
48        iter.for_each(|chunk| {
49            let chunk: &mut [f32; 16] = unsafe { chunk.try_into().unwrap_unchecked() };
50            silu_chunk(chunk)
51        });
52
53        Tensor::from_vec(as_vec, shape, device)
54    } else {
55        candle_nn::ops::silu(tensor)
56    }
57}
