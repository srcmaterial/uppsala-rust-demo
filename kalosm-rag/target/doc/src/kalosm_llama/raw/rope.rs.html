1use std::f32::consts::PI;
2
3use super::LlamaConfig;
4use candle_core::{DType, Device, Tensor};
5
6#[derive(Debug, Clone)]
7pub struct RopeCache {
8    sin: Tensor,
9    cos: Tensor,
10}
11
12impl RopeCache {
13    pub fn new(config: &LlamaConfig, dtype: DType, device: &Device) -> candle_core::Result<Self> {
14        let mut inverse_frequency = (0..config.head_dimension)
15            .step_by(2)
16            .map(|i| {
17                1. / (config
18                    .rope_theta
19                    .powf(i as f32 / config.head_dimension as f32))
20            })
21            .collect::<Vec<_>>();
22        if let Some(scaling_config) = &config.rope_scaling {
23            let original_max_position_embeddings = scaling_config.original_max_position_embeddings;
24            let factor = scaling_config.factor;
25            let high_freq_factor = scaling_config.high_freq_factor;
26            let low_freq_factor = scaling_config.low_freq_factor;
27            let low_freq_wavelen = original_max_position_embeddings as f32 / low_freq_factor;
28            let high_freq_wavelen = original_max_position_embeddings as f32 / high_freq_factor;
29            for freq in inverse_frequency.iter_mut() {
30                let wavelen = 2. * PI / *freq;
31                if wavelen > low_freq_wavelen {
32                    *freq /= factor
33                } else if wavelen == high_freq_wavelen {
34                    let smooth = (original_max_position_embeddings as f32 / wavelen
35                        - low_freq_factor)
36                        / (high_freq_factor - low_freq_factor);
37                    *freq = (1. - smooth) * *freq / factor + smooth * *freq
38                }
39            }
40        }
41        let inverse_frequency_len = inverse_frequency.len();
42        let mut inverse_frequency =
43            Tensor::from_vec(inverse_frequency, (1, inverse_frequency_len), device)?
44                .to_dtype(dtype)?;
45        if let Some(weight) = &config.rope_freq_weight {
46            inverse_frequency = inverse_frequency.mul(&weight.reshape((1, ()))?)?;
47        }
48
49        let llama_context_length_indices =
50            Tensor::arange(0f32, config.context_length as f32, device)?
51                .reshape((config.context_length, 1))?
52                .to_dtype(dtype)?;
53
54        let outer_product = llama_context_length_indices.matmul(&inverse_frequency)?;
55
56        let sin = outer_product.sin()?;
57        let cos = outer_product.cos()?;
58
59        Ok(Self { sin, cos })
60    }
61
62    fn forward_with_embed(
63        &self,
64        q: &Tensor,
65        k: &Tensor,
66        start_pos: usize,
67        apply_rotary_emb: fn(&Tensor, &Tensor, &Tensor) -> candle_core::Result<Tensor>,
68    ) -> candle_core::Result<(Tensor, Tensor)> {
69        let apply_rotary_emb = |sin: &Tensor, cos: &Tensor, x: &Tensor, index_pos| {
70            let (_b_sz, _n_head, seq_len, _n_embd) = x.dims4()?;
71            let cos = cos.narrow(0, index_pos, seq_len)?;
72            let sin = sin.narrow(0, index_pos, seq_len)?;
73            apply_rotary_emb(&x.contiguous()?, &cos, &sin)
74        };
75        let device = q.device();
76        let (q, k) = if matches!(device, Device::Cpu) {
77            std::thread::scope(|s| {
78                let q = s.spawn(|| apply_rotary_emb(&self.sin, &self.cos, q, start_pos));
79                let k = apply_rotary_emb(&self.sin, &self.cos, k, start_pos)?;
80                candle_core::Result::Ok((
81                    q.join()
82                        .map_err(|e| candle_core::Error::Msg(format!("Error in q: {:?}", e)))??,
83                    k,
84                ))
85            })?
86        } else {
87            let q = apply_rotary_emb(&self.sin, &self.cos, q, start_pos)?;
88            let k = apply_rotary_emb(&self.sin, &self.cos, k, start_pos)?;
89            (q, k)
90        };
91
92        Ok((q, k))
93    }
94
95    pub fn forward(
96        &self,
97        q: &Tensor,
98        k: &Tensor,
99        start_pos: usize,
100    ) -> candle_core::Result<(Tensor, Tensor)> {
101        self.forward_with_embed(q, k, start_pos, candle_nn::rotary_emb::rope)
102    }
103
104    pub fn forward_i(
105        &self,
106        q: &Tensor,
107        k: &Tensor,
108        start_pos: usize,
109    ) -> candle_core::Result<(Tensor, Tensor)> {
110        self.forward_with_embed(q, k, start_pos, candle_nn::rotary_emb::rope_i)
111    }
112}

114#[test]
115fn test_rope_cache() {
116    let config = LlamaConfig::mock_test();
117    let device = Device::cuda_if_available(0).unwrap();
118    let cache = RopeCache::new(&config, DType::F32, &device).unwrap();
119
120    let expected_cos = Tensor::new(
121        vec![
122            vec![1.0000f32],
123            vec![0.5403f32],
124            vec![-0.4161f32],
125            vec![-0.9900f32],
126            vec![-0.6536f32],
127            vec![0.2837f32],
128        ],
129        &device,
130    )
131    .unwrap();
132    let expected_sin = Tensor::new(
133        vec![
134            vec![0.0000f32],
135            vec![0.8415f32],
136            vec![0.9093f32],
137            vec![0.1411f32],
138            vec![-0.7568f32],
139            vec![-0.9589f32],
140        ],
141        &device,
142    )
143    .unwrap();
144
145    let cos_error: f32 = (cache.cos - expected_cos)
146        .unwrap()
147        .abs()
148        .unwrap()
149        .sum_all()
150        .unwrap()
151        .to_scalar()
152        .unwrap();
153    assert!(cos_error < 1e-2);
154    let sin_error: f32 = (cache.sin - expected_sin)
155        .unwrap()
156        .abs()
157        .unwrap()
158        .sum_all()
159        .unwrap()
160        .to_scalar()
161        .unwrap();
162    assert!(sin_error < 1e-2);
163}
