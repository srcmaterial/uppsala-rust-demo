18pub(crate) fn generate_structured<P: Parser>(
19    prompt: impl Display,
20    llm: &LlamaModel,
21    session: &mut LlamaSession,
22    parser: P,
23    parser_state: P::PartialState,
24    mut sampler: Arc<Mutex<dyn Sampler>>,
25    mut on_token: impl FnMut(String) -> Result<(), LlamaModelError>,
26    top_k: Option<usize>,
27    seed: Option<u64>,
28) -> Result<P::Output, LlamaModelError> {
29    let eos_token = llm.model.config.stop_token_string.clone();
30    let mut on_token = move |tok: String| {
31        if tok == eos_token {
32            return Ok(());
33        }
34        on_token(tok)
35    };
36    let mut session = session
37        .cache
38        .write()
39        .map_err(|err| LlamaModelError::Session(err.to_string()))?;
40    let tokenizer = &llm.tokenizer;
41
42    let prompt_text = prompt.to_string();
43    let prompt_tokens = tokenizer
44        .encode_fast(prompt_text, false)
45        .map_err(LlamaModelError::Tokenizer)?;
46    let mut prompt_tokens = prompt_tokens.get_ids();
47
48    // Prompt healing
49    // Trim the last token and add what it would decode to into the constraints
50    let last_token = if let Some((last, tokens)) = prompt_tokens.split_last() {
51        if tokenizer.get_added_tokens_decoder().contains_key(last) {
52            None
53        } else {
54            prompt_tokens = tokens;
55            Some(*last)
56        }
57    } else {
58        None
59    };
60
61    let mut unprocessed_token_count = prompt_tokens.len();
62    let mut token_stream = TokenOutputStream::new(tokenizer.clone());
63    for token in prompt_tokens {
64        token_stream
65            .next_token(*token)
66            .map_err(LlamaModelError::TokenOutputStreamError)?;
67    }
68
69    let remaining_prompt_text = last_token
70        .map(|token| {
71            token_stream
72                .peek_token(token)
73                .map_err(LlamaModelError::TokenOutputStreamError)
74        })
75        .transpose()?
76        .flatten()
77        .unwrap_or_default();
78
79    let parser = LiteralParser::new(remaining_prompt_text.clone())
80        .ignore_output_then(parser.with_initial_state(move || parser_state.clone()));
81    {
82        let mut parser_state = parser.create_parser_state();
83        for c in remaining_prompt_text.chars() {
84            let str = c.to_string();
85            let bytes = str.as_bytes();
86            let (parser_state_new, _) = parser
87                .parse(&parser_state, bytes)
88                .unwrap()
89                .unwrap_incomplete();
90            parser_state = parser_state_new;
91        }
92    }
93    let mut parser_state = parser.create_parser_state();
94    let mut strip_required_next = true;
95
96    let mut rng = if let Some(seed) = seed {
97        rand::rngs::StdRng::seed_from_u64(seed)
98    } else {
99        rand::rngs::StdRng::from_entropy()
100    };
101    let mut state_map = vec![];
102    let mut logits_indexed = Vec::new();
103    let mut token_cache = DetokenizationCache::new();
104    let mut logits = Logits::default();
105    let mut logit_probs = Vec::new();
106
107    loop {
108        let tokens = token_stream.tokens();
109        LlamaModel::forward(
110            &llm.model,
111            &llm.device,
112            &tokens[tokens.len() - unprocessed_token_count..],
113            Some(&mut *session),
114            &mut logit_probs,
115            &llm.tokenizer,
116        )?;
117        let resources = &mut SamplerResources {
118            previous_tokens: tokens,
119            rng: &mut rng,
120        };
121
122        // fill the state map with None for each token
123        token_cache.clear(logit_probs.len());
124        state_map.clear();
125        logits_indexed.clear();
126        logits.clear();
127        for (id, prob) in logit_probs.iter().enumerate() {
128            logits_indexed.push(Logit {
129                token_id: id as u32,
130                logit: *prob,
131                prob: 0f32,
132            });
133            state_map.push(None);
134        }
135
136        let mut valid_tokens = false;
137
138        // If we don't have a top k, then we can just cache the entire detokenization
139        if top_k.is_none() {
140            token_cache.expand(
141                &(0..logit_probs.len() as u32).collect::<Vec<_>>(),
142                &token_stream,
143            );
144        }
145
146        const DETOKENIZATION_INITIAL_BATCH_SIZE: usize = 64;
147
148        // Constraints tend to be either very difficult to satisfy or very easy to satisfy
149        // We exponentially increase the batch size as a balance between the two
150        // If the first half of the tokens are invalid, it is unlikely that the first 64 tokens of the second half will be valid
151        let mut detokenization_batch_size = DETOKENIZATION_INITIAL_BATCH_SIZE;
152
153        let mut partitioned_logits_index = top_k.map(|_| 0);
154
155        for i in 0..logits_indexed.len() {
156            // If we have top k enabled, and there are less than top k - committed logits sorted, we need to expand the partitioned logits
157            if let (Some(top_k), Some(partitioned_index)) = (top_k, partitioned_logits_index) {
158                // If the remaining logits are less than the top k, no need to partition
159                let remaining_needed = top_k - logits.len();
160                let remaining_possible = partitioned_index - i;
161                if remaining_possible <= remaining_needed {
162                    // We batch together updates to the cache by detokenization_batch_size
163                    let logits_to_update = (remaining_needed.max(detokenization_batch_size))
164                        .min(logits_indexed.len() - 1 - i);
165                    let new_partitioned_index = i + logits_to_update;
166
167                    // If we eliminated a logit, our partitioning of the logits is no longer valid
168                    logits_indexed[i..].select_nth_unstable_by(logits_to_update, cmp_logits);
169                    logits_indexed[i..=new_partitioned_index].sort_unstable_by(cmp_logits);
170                    // Expand the cache to include the new logits
171                    partitioned_logits_index = Some(new_partitioned_index);
172                    token_cache.expand_with_logits(
173                        &logits_indexed[i..=new_partitioned_index],
174                        &token_stream,
175                    );
176
177                    // Double the batch size for next time
178                    detokenization_batch_size = detokenization_batch_size.saturating_mul(4);
179                }
180            }
181
182            let Logit {
183                token_id, logit, ..
184            } = logits_indexed[i];
185            let Some(text) = token_cache.get(token_id as usize) else {
186                continue;
187            };
188            if let Ok(result) = parser.parse(&parser_state, text.as_bytes()) {
189                let parsed_bytes = match result {
190                    ParseStatus::Finished { remaining, .. } => text.len() - remaining.len(),
191                    ParseStatus::Incomplete { .. } => text.len(),
192                };
193                let result = result.without_remaining();
194                state_map[token_id as usize] = Some((result, parsed_bytes));
195                valid_tokens = true;
196                logits.push(Logit {
197                    token_id,
198                    logit,
199                    prob: 0f32,
200                });
201                // If we only need to keep the top k logits, then we can quit early once we have enough
202                if let Some(top_k) = top_k {
203                    if logits.len() >= top_k {
204                        break;
205                    }
206                }
207            }
208        }
209
210        // If there are no valid tokens, return an error
211        if !valid_tokens {
212            return Err(LlamaModelError::NoValidTokens);
213        }
214        let token_id = sampler
215            .sample_token(resources, &mut logits)
216            .map_err(|err| LlamaModelError::SamplerError(err.into()))?
217            .ok_or(LlamaModelError::NoValidTokens)?;
218
219        unprocessed_token_count = 1;
220        let (result, parsed_bytes) = state_map
221            .get_mut(token_id as usize)
222            .unwrap()
223            .take()
224            .unwrap_or_else(|| panic!("Token {} not found in state map", token_id));
225        let mut token = token_stream
226            .next_token(token_id)
227            .map_err(LlamaModelError::TokenOutputStreamError)?
228            .unwrap();
229        token.truncate(parsed_bytes);
230        tracing::trace!("Adding token {} to parser", token);
231        // If we are still loading the initial prompt, don't send that part of the text
232        if strip_required_next {
233            if let Some(stripped) = token.strip_prefix(&remaining_prompt_text) {
234                token = stripped.to_string();
235            }
236            strip_required_next = false;
237        }
238        on_token(token)?;
239
240        if let Some(result) = update_state(
241            &parser,
242            &mut parser_state,
243            result,
244            tokenizer,
245            &mut token_stream,
246            &mut on_token,
247            &mut unprocessed_token_count,
248        )? {
249            return Ok(result);
250        }
251    }
252}

254fn cmp_logits(a: &Logit, b: &Logit) -> std::cmp::Ordering {
255    // SAFETY: Logits should never be NaN or Inf
256    let compare = b.logit.partial_cmp(&a.logit);
257    debug_assert!(compare.is_some());
258    unsafe { compare.unwrap_unchecked() }
259}

261#[allow(unused, clippy::all)]
262fn update_state<P: Parser>(
263    parser: &P,
264    parser_state: &mut P::PartialState,
265    result: ParseStatus<P::PartialState, P::Output>,
266    tokenizer: &Tokenizer,
267    token_stream: &mut TokenOutputStream,
268    on_token: &mut impl FnMut(String) -> Result<(), LlamaModelError>,
269    unprocessed_token_count: &mut usize,
270) -> Result<Option<P::Output>, LlamaModelError> {
271    match result {
272        kalosm_sample::ParseStatus::Incomplete {
273            new_state,
274            required_next,
275        } => {
276            *parser_state = new_state;
277            if required_next.is_empty() {
278                Ok(None)
279            } else {
280                // The token may decode to a string that is a valid prefix of the required next token, but in a way that doesn't let us decode the required next tokens
281                let Some(mut extra_tokens) = token_stream
282                    .encode_after(&required_next)
283                    .map_err(|err| LlamaModelError::TokenOutputStreamError(err))?
284                else {
285                    return Ok(None);
286                };
287                // Remove the last token to avoid influencing the next token
288                extra_tokens.pop();
289                // If there are no new tokens, continue generating tokens normally
290                if extra_tokens.is_empty() {
291                    return Ok(None);
292                }
293
294                let mut all_required_next = String::new();
295                if let Some(next) = token_stream
296                    .peek_next_tokens(extra_tokens.iter().copied())
297                    .map_err(|err| LlamaModelError::TokenOutputStreamError(err))?
298                {
299                    all_required_next = next;
300                }
301                // The token may decode to a string that is a valid prefix of the required next token, but in a way that doesn't encode the same way.
302                // Make sure the final text we are adding is actually valid
303                if !required_next.starts_with(&all_required_next) {
304                    return Ok(None);
305                }
306                token_stream
307                    .next_tokens(&extra_tokens)
308                    .map_err(|err| LlamaModelError::TokenOutputStreamError(err))?;
309                *unprocessed_token_count += extra_tokens.len();
310                on_token(all_required_next.clone())?;
311                let mut result = parser
312                    .parse(parser_state, all_required_next.as_bytes())
313                    .unwrap_or_else(|_| {
314                        unreachable!("Required next should always be valid attempted to add {:?} but got error", all_required_next)
315                });
316                update_state(
317                    parser,
318                    parser_state,
319                    result,
320                    tokenizer,
321                    token_stream,
322                    on_token,
323                    unprocessed_token_count,
324                )
325            }
326        }
327        kalosm_sample::ParseStatus::Finished { result, .. } => Ok(Some(result)),
328    }
329}

331struct SamplerResources<'a, 'b, R: rand::Rng> {
332    rng: &'a mut R,
333    previous_tokens: &'b [u32],
334}
335
336impl<R> Debug for SamplerResources<'_, '_, R>
337where
338    R: rand::Rng,
339{
340    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
341        f.debug_struct("SamplerResources")
342            .field("previous_tokens", &self.previous_tokens)
343            .finish()
344    }
345}

347impl<R> HasSamplerResources for SamplerResources<'_, '_, R>
348where
349    R: rand::Rng,
350{
351    fn with_rng_mut(

355        fun(self.rng);
356        Ok(())
357    }

359    fn with_last_tokens(&self, fun: &mut dyn FnMut(&[u32])) -> Result<(), SamplerError> {
360        fun(self.previous_tokens);
361        Ok(())
362    }

365#[derive(Clone)]
366enum TokenCacheStatus {
367    Empty,
368    Invalid,
369    Valid(String),
370}
371
372struct DetokenizationCache {
373    cache: Box<[TokenCacheStatus]>,
374    vec: Vec<Option<String>>,
375}
376
377impl DetokenizationCache {
378    fn new() -> Self {
379        Self {
380            cache: Box::new([]),
381            vec: Vec::new(),
382        }
383    }
384
385    fn get(&self, index: usize) -> Option<&str> {
386        match &self.cache[index] {
387            TokenCacheStatus::Empty => panic!("cache for token {} is empty", index),
388            TokenCacheStatus::Invalid => None,
389            TokenCacheStatus::Valid(token) => Some(token),
390        }
391    }
392
393    fn expand_with_logits(&mut self, tokens: &[Logit], stream: &TokenOutputStream) {
394        stream.peek_tokens(
395            tokens.into_par_iter().map(|logit| logit.token_id),
396            &mut self.vec,
397        );
398
399        for (logit, token) in tokens.iter().zip(self.vec.drain(..)) {
400            self.cache[logit.token_id as usize] = match token {
401                Some(token) => TokenCacheStatus::Valid(token),
402                None => TokenCacheStatus::Invalid,
403            };
404        }
405    }
406
407    fn expand(&mut self, tokens: &[u32], stream: &TokenOutputStream) {
408        stream.peek_tokens(tokens.into_par_iter().copied(), &mut self.vec);
409
410        for (&i, token) in tokens.iter().zip(self.vec.drain(..)) {
411            self.cache[i as usize] = match token {
412                Some(token) => TokenCacheStatus::Valid(token),
413                None => TokenCacheStatus::Invalid,
414            };
415        }
416    }
417
418    fn clear(&mut self, size: usize) {
419        if self.cache.len() == size {
420            for token in self.cache.iter_mut() {
421                *token = TokenCacheStatus::Empty;
422            }
423        } else {
424            self.cache = vec![TokenCacheStatus::Empty; size].into_boxed_slice();
425        }
426        self.vec.clear();
427    }
