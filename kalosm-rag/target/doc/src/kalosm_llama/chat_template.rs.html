12pub(crate) struct HuggingFaceChatTemplate {
13    environment: Environment<'static>,
14}
15
16impl HuggingFaceChatTemplate {
17    pub(crate) fn create(chat_template: impl Display) -> Result<Self, minijinja::Error> {
18        let chat_template = chat_template.to_string();
19        let mut environment = Environment::new();
20
21        // enable python compatibility methods because most models are tested with python
22        environment.set_unknown_method_callback(pycompat::unknown_method_callback);
23
24        // add the raise_exception function from huggingface templates to the environment
25        let raise_exception = |err_text: String| -> Result<String, minijinja::Error> {
26            Err(minijinja::Error::new(
27                ErrorKind::InvalidOperation,
28                format!("The template raised an exception: {}", err_text),
29            ))
30        };
31        environment.add_function("raise_exception", raise_exception);
32
33        // compile the template expression in the environment
34        environment.add_template_owned("main", chat_template)?;
35
36        Ok(Self { environment })
37    }
38
39    pub(crate) fn format(
40        &self,
41        bos_token: &str,
42        eos_token: &str,
43        messages: &[ChatMessage],
44        add_generation_prompt: bool,
45    ) -> Result<String, minijinja::Error> {
46        let tools: Option<()> = None;
47        let ctx = context! { bos_token, eos_token, messages, add_generation_prompt, tools };
48        let template = self.environment.get_template("main")?;
49        let result = template.render(&ctx)?;
50        Ok(result)
51    }
52}

55fn test_qwen_chat_template() {
56    let template = r#"{%- if tools %}
57    {{- '<|im_start|>system\n' }}
58    {%- if messages[0]['role'] == 'system' %}
59        {{- messages[0]['content'] }}
60    {%- else %}
61        {{- 'You are Qwen, created by Alibaba Cloud. You are a helpful assistant.' }}
62    {%- endif %}
63    {{- "\n\n# Tools\n\nYou may call one or more functions to assist with the user query.\n\nYou are provided with function signatures within <tools></tools> XML tags:\n<tools>" }}
64    {%- for tool in tools %}
65        {{- "\n" }}
66        {{- tool | tojson }}
67    {%- endfor %}
68    {{- "\n</tools>\n\nFor each function call, return a json object with function name and arguments within <tool_call></tool_call> XML tags:\n<tool_call>\n{{\"name\": <function-name>, \"arguments\": <args-json-object>}}\n</tool_call><|im_end|>\n" }}
69{%- else %}
70    {%- if messages[0]['role'] == 'system' %}
71        {{- '<|im_start|>system\n' + messages[0]['content'] + '<|im_end|>\n' }}
72    {%- else %}
73        {{- '<|im_start|>system\nYou are Qwen, created by Alibaba Cloud. You are a helpful assistant.<|im_end|>\n' }}
74    {%- endif %}
75{%- endif %}
76{%- for message in messages %}
77    {%- if (message.role == "user") or (message.role == "system" and not loop.first) or (message.role == "assistant" and not message.tool_calls) %}
78        {{- '<|im_start|>' + message.role + '\n' + message.content + '<|im_end|>' + '\n' }}
79    {%- elif message.role == "assistant" %}
80        {{- '<|im_start|>' + message.role }}
81        {%- if message.content %}
82            {{- '\n' + message.content }}
83        {%- endif %}
84        {%- for tool_call in message.tool_calls %}
85            {%- if tool_call.function is defined %}
86                {%- set tool_call = tool_call.function %}
87            {%- endif %}
88            {{- '\n<tool_call>\n{"name": "' }}
89            {{- tool_call.name }}
90            {{- '", "arguments": ' }}
91            {{- tool_call.arguments | tojson }}
92            {{- '}\n</tool_call>' }}
93        {%- endfor %}
94        {{- '<|im_end|>\n' }}
95    {%- elif message.role == "tool" %}
96        {%- if (loop.index0 == 0) or (messages[loop.index0 - 1].role != "tool") %}
97            {{- '<|im_start|>user' }}
98        {%- endif %}
99        {{- '\n<tool_response>\n' }}
100        {{- message.content }}
101        {{- '\n</tool_response>' }}
102        {%- if loop.last or (messages[loop.index0 + 1].role != "tool") %}
103            {{- '<|im_end|>\n' }}
104        {%- endif %}
105    {%- endif %}
106{%- endfor %}
107{%- if add_generation_prompt %}
108    {{- '<|im_start|>assistant\n' }}
109{%- endif %}"#;
110
111    let template = HuggingFaceChatTemplate::create(template).unwrap();
112
113    let inputs = [
114        ChatMessage::new(MessageType::UserMessage, "Hello, how are you?".to_string()),
115        ChatMessage::new(
116            MessageType::ModelAnswer,
117            "I'm doing great. How can I help you today?".to_string(),
118        ),
119        ChatMessage::new(
120            MessageType::UserMessage,
121            "I'd like to show off how chat templating works!".to_string(),
122        ),
123    ];
124
125    let result = template
126        .format("<|endoftext|>", "<|im_end|>", &inputs, false)
127        .unwrap();
128    assert_eq!(
129        result,
130        r#"<|im_start|>system
131You are Qwen, created by Alibaba Cloud. You are a helpful assistant.<|im_end|>
132<|im_start|>user
133Hello, how are you?<|im_end|>
134<|im_start|>assistant
135I'm doing great. How can I help you today?<|im_end|>
136<|im_start|>user
137I'd like to show off how chat templating works!<|im_end|>
138"#
139    );

176#[test]
177fn test_mistral_chat_template() {
178    let template = "{%- if messages[0]['role'] == 'system' %}\n    {%- set system_message = messages[0]['content'] %}\n    {%- set loop_messages = messages[1:] %}\n{%- else %}\n    {%- set loop_messages = messages %}\n{%- endif %}\n\n{{- bos_token }}\n{%- for message in loop_messages %}\n    {%- if (message['role'] == 'user') != (loop.index0 % 2 == 0) %}\n        {{- raise_exception('After the optional system message, conversation roles must alternate user/assistant/user/assistant/...') }}\n    {%- endif %}\n    {%- if message['role'] == 'user' %}\n        {%- if loop.first and system_message is defined %}\n            {{- ' [INST] ' + system_message + '\\n\\n' + message['content'] + ' [/INST]' }}\n        {%- else %}\n            {{- ' [INST] ' + message['content'] + ' [/INST]' }}\n        {%- endif %}\n    {%- elif message['role'] == 'assistant' %}\n        {{- ' ' + message['content'] + eos_token}}\n    {%- else %}\n        {{- raise_exception('Only user and assistant roles are supported, with the exception of an initial optional system message!') }}\n    {%- endif %}\n{%- endfor %}\n";
179
180    let template = HuggingFaceChatTemplate::create(template).unwrap();
181
182    let inputs = [
183        ChatMessage::new(MessageType::UserMessage, "Hello, how are you?".to_string()),
184        ChatMessage::new(
185            MessageType::ModelAnswer,
186            "I'm doing great. How can I help you today?".to_string(),
187        ),
188        ChatMessage::new(
189            MessageType::UserMessage,
190            "I'd like to show off how chat templating works!".to_string(),
191        ),
192    ];
193
194    let result = template.format("<s>", "</s>", &inputs, false).unwrap();
195    assert_eq!(
196        result,
197        r#"<s> [INST] Hello, how are you? [/INST] I'm doing great. How can I help you today?</s> [INST] I'd like to show off how chat templating works! [/INST]"#
198    )
