22/// An error that can occur when running a [`LlamaModel`].
23#[derive(Debug, thiserror::Error)]
24pub enum LlamaModelError {
25    /// An error from candle while running the model.
26    #[error("Candle error: {0}")]
27    Candle(#[from] candle_core::Error),
28
29    /// An error from tokenizers while running the model.
30    #[error("Tokenizer error: {0}")]
31    Tokenizer(tokenizers::Error),
32
33    /// An error while sampling tokens.
34    #[error("Sampler error: {0}")]
35    SamplerError(Box<dyn std::error::Error + Send + Sync>),
36
37    /// A streaming detokenization error.
38    #[error("Token output stream error: {0}")]
39    TokenOutputStreamError(TokenOutputStreamError),
40
41    /// An error while writing to the session cache.
42    #[error("Session cache error: {0}")]
43    Session(String),
44
45    /// No valid tokens were sampled during structured generation
46    #[error("No valid tokens were sampled")]
47    NoValidTokens,
48
49    /// The model has already stopped.
50    #[error("Model stopped")]
51    ModelStopped,
52
53    /// No chat template was provided
54    #[error("No chat template was provided")]
55    NoChatTemplate,
56
57    /// Error running the chat template
58    #[error("Error running the chat template: {0}")]
59    ChatTemplateError(#[from] minijinja::Error),
60}

66    pub(crate) tokenizer: Arc<Tokenizer>,

70    pub(crate) fn forward(
71        model: &Model,
72        device: &Device,
73        tokens: &[u32],
74        cache: Option<&mut LlamaCache>,
75        logits_vec: &mut Vec<f32>,
76        #[allow(unused)] tokenizer: &Tokenizer,
77    ) -> candle_core::Result<()> {
78        if tokens.is_empty() {
79            candle_core::bail!("Cannot run model on empty input");
80        }
81
82        #[cfg(debug_assertions)]
83        {
84            tracing::trace!(
85                "Running model with tokens: {:?}",
86                tokenizer.decode(tokens, false)
87            );
88        }
89
90        let logits = model.forward(tokens, device, cache)?;
91
92        let logits = logits.squeeze(0)?.to_dtype(DType::F32)?;
93        copy_tensor_into_vec(&logits, logits_vec)?;
94
95        Ok(())
96    }

98    /// Create a new sync Llama model from a builder.
99    pub(crate) async fn from_builder(
100        builder: crate::LlamaBuilder,
101        mut handler: impl FnMut(ModelLoadingProgress) + Send + Sync + 'static,
102    ) -> Result<Self, LlamaSourceError> {
103        let device = builder.get_device()?;
104
105        // Download the model and tokenizer. These are relatively cheep operations that can be run in the async runtime
106        let tokenizer_path = match &builder.source.tokenizer {
107            Some(tokenizer) => {
108                let tokenizer_source = format!("Tokenizer ({})", tokenizer);
109                let mut create_progress =
110                    ModelLoadingProgress::downloading_progress(tokenizer_source);
111                let tokenizer_path = builder
112                    .source
113                    .cache
114                    .get(tokenizer, |progress| handler(create_progress(progress)))
115                    .await?;
116                Some(tokenizer_path)
117            }
118            None => None,
119        };
120
121        // Download the config file if it exists
122        let config_path = match &builder.source.config {
123            Some(config) => {
124                let config_source = format!("Config ({})", config);
125                let mut create_progress = ModelLoadingProgress::downloading_progress(config_source);
126                let config_path = builder
127                    .source
128                    .cache
129                    .get(config, |progress| handler(create_progress(progress)))
130                    .await?;
131                Some(config_path)
132            }
133            None => None,
134        };
135
136        let source = format!("Model ({})", builder.source.model);
137        let mut create_progress = ModelLoadingProgress::downloading_progress(source);
138        let filename = builder
139            .source
140            .model(|progress| handler(create_progress(progress)))
141            .await?;
142
143        // Then actually load the model and tokenizer. This is expensive, so we do it in a blocking task
144        let (model, tokenizer) = tokio::task::spawn_blocking({
145            let device = device.clone();
146            move || {
147                let tokenizer = match tokenizer_path {
148                    Some(tokenizer_path) => {
149                        let tokenizer = Tokenizer::from_file(tokenizer_path)
150                            .map_err(LlamaSourceError::Tokenizer)?;
151                        Some(tokenizer)
152                    }
153                    None => None,
154                };
155
156                let config = match config_path {
157                    Some(config_path) => {
158                        let config = std::fs::read_to_string(config_path).map_err(|err| {
159                            LlamaSourceError::Config(serde_json::Error::custom(err))
160                        })?;
161                        let config: LlamaConfigJson =
162                            serde_json::from_str(&config).map_err(LlamaSourceError::Config)?;
163                        config.rope_scaling
164                    }
165                    None => None,
166                };
167
168                let mut file = std::fs::File::open(&filename)
169                    .expect("The path returned by LlamaSource::model should be valid");
170                let override_stop_token_string = builder.source.override_stop_token_string;
171                match filename.extension().and_then(|v| v.to_str()) {
172                    Some("gguf") => {
173                        let model = gguf_file::Content::read(&mut file)?;
174                        let tokenizer = match tokenizer {
175                            Some(tokenizer) => tokenizer,
176                            None => {
177                                let tokenizer_model = model
178                                    .metadata
179                                    .get("tokenizer.ggml.model")
180                                    .ok_or(LlamaSourceError::NoTokenizer)?
181                                    .to_string()
182                                    .map_err(|_| LlamaSourceError::NoTokenizer)?;
183                                if tokenizer_model != "gpt2" {
184                                    return Err(LlamaSourceError::NoTokenizer);
185                                }
186                                let pre = model
187                                    .metadata
188                                    .get("tokenizer.ggml.pre")
189                                    .ok_or(LlamaSourceError::NoTokenizer)?
190                                    .to_string()
191                                    .map_err(|_| LlamaSourceError::NoTokenizer)?;
192                                let add_bos_token = model
193                                    .metadata
194                                    .get("tokenizer.ggml.add_bos_token")
195                                    .and_then(|v| v.to_bool().ok());
196                                let config = get_pre_tokenizer(pre, add_bos_token);
197
198                                let tokens: Result<Vec<_>, _> = model
199                                    .metadata
200                                    .get("tokenizer.ggml.tokens")
201                                    .ok_or(LlamaSourceError::NoTokenizer)?
202                                    .to_vec()
203                                    .map_err(|_| LlamaSourceError::NoTokenizer)?
204                                    .iter()
205                                    .map(|v| v.to_string().map(|s| s.to_string()))
206                                    .collect();
207                                let tokens = tokens.map_err(|_| LlamaSourceError::NoTokenizer)?;
208                                let types: Result<Vec<_>, _> = model
209                                    .metadata
210                                    .get("tokenizer.ggml.token_type")
211                                    .ok_or(LlamaSourceError::NoTokenizer)?
212                                    .to_vec()
213                                    .map_err(|_| LlamaSourceError::NoTokenizer)?
214                                    .iter()
215                                    .map(|v| {
216                                        v.to_i32()
217                                            .map(|v| v as u8)
218                                            .or_else(|_| v.to_i64().map(|v| v as u8))
219                                            .or_else(|_| v.to_i16().map(|v| v as u8))
220                                            .or_else(|_| v.to_i8().map(|v| v as u8))
221                                            .or_else(|_| v.to_u64().map(|v| v as u8))
222                                            .or_else(|_| v.to_u32().map(|v| v as u8))
223                                            .or_else(|_| v.to_u16().map(|v| v as u8))
224                                            .or_else(|_| v.to_u8())
225                                    })
226                                    .collect();
227                                let types = types.map_err(|_| LlamaSourceError::NoTokenizer)?;
228                                let vocab: HashMap<_, _> = tokens
229                                    .iter()
230                                    .enumerate()
231                                    .map(|(id, v)| (v.clone(), id as u32))
232                                    .collect();
233                                let merges = model
234                                    .metadata
235                                    .get("tokenizer.ggml.merges")
236                                    .ok_or(LlamaSourceError::NoTokenizer)?;
237                                let merges: Result<Vec<_>, _> = merges
238                                    .to_vec()
239                                    .map_err(|_| LlamaSourceError::NoTokenizer)?
240                                    .iter()
241                                    .map(|v| {
242                                        v.to_string()
243                                            .map_err(|_| LlamaSourceError::NoTokenizer)
244                                            .and_then(|v| {
245                                                v.split_once(' ')
246                                                    .ok_or(LlamaSourceError::NoTokenizer)
247                                            })
248                                            .map(|(a, b)| (a.to_string(), b.to_string()))
249                                    })
250                                    .collect();
251                                let merges = merges.map_err(|_| LlamaSourceError::NoTokenizer)?;
252
253                                let eos = model
254                                    .metadata
255                                    .get("tokenizer.ggml.eos_token_id")
256                                    .ok_or(LlamaSourceError::NoTokenizer)?;
257                                let eos =
258                                    eos.to_u32().map_err(|_| LlamaSourceError::NoTokenizer)?;
259                                let eos = &tokens[eos as usize];
260
261                                let bos = model
262                                    .metadata
263                                    .get("tokenizer.ggml.bos_token_id")
264                                    .ok_or(LlamaSourceError::NoTokenizer)?;
265                                let bos =
266                                    bos.to_u32().map_err(|_| LlamaSourceError::NoTokenizer)?;
267                                let bos = &tokens[bos as usize];
268
269                                config
270                                    .build(vocab, types, merges, bos, eos)
271                                    .map_err(LlamaSourceError::Tokenizer)?
272                            }
273                        };
274                        let model = Model::from_gguf(
275                            model,
276                            &mut file,
277                            &device,
278                            override_stop_token_string,
279                            config,
280                        )?;
281                        Ok((model, tokenizer))
282                    }
283                    Some("ggml" | "bin") | Some(_) | None => {
284                        let model = ggml_file::Content::read(&mut file, &device)?;
285                        let tokenizer = tokenizer.ok_or(LlamaSourceError::NoTokenizer)?;
286
287                        let gqa = builder.source.group_query_attention;
288                        let vocab = tokenizer.get_vocab(true);
289                        let start_token_string = match vocab
290                            .get("<s>")
291                            .map(|v| (*v, "<s>".to_string()))
292                            .or_else(|| {
293                                vocab
294                                    .get("<|start_of_text|>")
295                                    .map(|v| (*v, "<|start_of_text|>".to_string()))
296                            })
297                            .or_else(|| {
298                                vocab
299                                    .get("<|startoftext|>")
300                                    .map(|v| (*v, "<|startoftext|>".to_string()))
301                            }) {
302                            Some((_, string)) => string,
303                            None => String::new(),
304                        };
305                        let (stop_token, stop_token_string) = match vocab
306                            .get("</s>")
307                            .map(|v| (*v, "</s>".to_string()))
308                            .or_else(|| {
309                                vocab
310                                    .get("<|end_of_text|>")
311                                    .map(|v| (*v, "<|end_of_text|>".to_string()))
312                            })
313                            .or_else(|| {
314                                vocab
315                                    .get("<|endoftext|>")
316                                    .map(|v| (*v, "<|endoftext|>".to_string()))
317                            }) {
318                            Some((token, string)) => (token, string),
319                            None => return Err(LlamaSourceError::NoStopToken),
320                        };
321                        let model = Model::from_ggml(
322                            model,
323                            gqa as usize,
324                            &device,
325                            start_token_string,
326                            stop_token,
327                            stop_token_string,
328                            config,
329                        )?;
330                        Ok((model, tokenizer))
331                    }
332                }
333            }
334        })
335        .await
336        .map_err(|_| LlamaSourceError::ModelLoadingPanic)??;
337
338        Ok(Self {
339            model,
340            tokenizer: Arc::new(tokenizer),
341            device,
342        })
343    }

345    pub(crate) fn _infer(
346        &mut self,
347        settings: InferenceSettings,
348        mut on_token: Box<dyn FnMut(String) -> Result<(), LlamaModelError> + Send + Sync>,
349        finished: &tokio::sync::oneshot::Sender<Result<(), LlamaModelError>>,
350    ) -> Result<(), LlamaModelError> {
351        let InferenceSettings {
352            prompt,
353            stop_on,
354            mut sampler,
355            session,
356            max_tokens,
357            seed,
358        } = settings;
359
360        let mut session = session
361            .cache
362            .write()
363            .map_err(|err| LlamaModelError::Session(err.to_string()))?;
364
365        let tokens = self
366            .tokenizer
367            .encode_fast(prompt, false)
368            .map_err(LlamaModelError::Tokenizer)?;
369        let tokens = tokens.get_ids();
370        let mut text_stream = TokenOutputStream::new(self.tokenizer.clone());
371        for &token in tokens {
372            text_stream
373                .next_token(token)
374                .map_err(LlamaModelError::TokenOutputStreamError)?;
375        }
376
377        let mut logit_probs = Vec::new();
378        Self::forward(
379            &self.model,
380            &self.device,
381            tokens,
382            Some(&mut session),
383            &mut logit_probs,
384            &self.tokenizer,
385        )?;
386        let mut logits = Logits::try_from_iter_top_k(logit_probs, 512)
387            .expect("model output should be valid logits");
388        // This stores a buffer of text that has been generated to check against the stop_on string. It should never be longer than the stop_on string.
389        let mut queued_text_matching_stop_on = String::new();
390        let stop_on_lowercase = stop_on.as_ref().map(|s| s.to_lowercase());
391        let stop_on_lowercase = stop_on_lowercase.as_deref();
392        let stop_token = self.model.config.stop_token;
393        let mut tokens_generated = 0;
394        let mut logit_probs = Vec::new();
395
396        'generate: while !finished.is_closed() && tokens_generated < max_tokens {
397            let new_token = text_stream
398                .sample_token(&mut sampler, logits, stop_on.as_deref(), seed)
399                .map_err(LlamaModelError::TokenOutputStreamError)?;
400            Self::forward(
401                &self.model,
402                &self.device,
403                &[new_token],
404                Some(&mut session),
405                &mut logit_probs,
406                &self.tokenizer,
407            )?;
408            if new_token == stop_token {
409                tracing::trace!("Stopping on stop token");
410                break;
411            }
412            if let Some(mut new_text) = text_stream
413                .next_token(new_token)
414                .map_err(LlamaModelError::TokenOutputStreamError)?
415            {
416                tokens_generated += 1;
417                if let Some(stop_on) = stop_on_lowercase {
418                    let lowercase = new_text.to_lowercase();
419
420                    // Check if the string ends with the start of the stop_on string
421                    let mut before_stop_on = None;
422                    let remaining_stop_on = stop_on
423                        .strip_prefix(&queued_text_matching_stop_on)
424                        .unwrap_or(stop_on);
425
426                    // If the remaining stop_on string is empty, we have found a match
427                    if remaining_stop_on.is_empty() {
428                        break;
429                    }
430
431                    for (i, _) in lowercase.char_indices() {
432                        let end_of_new_text = &lowercase[i..];
433                        if end_of_new_text.is_empty() {
434                            break;
435                        }
436
437                        // Check if we have matched all of the stop_on string
438                        if end_of_new_text.starts_with(remaining_stop_on) {
439                            queued_text_matching_stop_on += end_of_new_text;
440                            break 'generate;
441                        }
442
443                        // Check if the string ends with the start of the stop_on string
444                        if remaining_stop_on.starts_with(end_of_new_text) {
445                            before_stop_on = Some(lowercase[..i].to_string());
446                            queued_text_matching_stop_on += end_of_new_text;
447                            break;
448                        }
449                    }
450
451                    match before_stop_on {
452                        Some(before_stop_on) => {
453                            on_token(before_stop_on)?;
454                        }
455                        None => {
456                            new_text =
457                                std::mem::take(&mut queued_text_matching_stop_on) + &new_text;
458                            on_token(new_text)?;
459                        }
460                    }
461                } else {
462                    on_token(new_text)?;
463                }
464            }
465            logits = Logits::try_from_iter_top_k(logit_probs.iter().copied(), 512)
466                .expect("model output should be valid logits");
467        }
468
469        // Flush the queued text
470        if let Some(stop_string) = stop_on_lowercase {
471            if !queued_text_matching_stop_on.starts_with(stop_string) {
472                on_token(queued_text_matching_stop_on)?;
473            }
474        }
475
476        Ok(())
477    }
