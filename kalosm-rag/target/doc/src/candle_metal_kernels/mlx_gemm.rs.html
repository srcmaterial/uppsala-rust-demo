1use crate::utils::EncoderProvider;
2use crate::{ConstantValues, Kernels, MetalKernelError, Source, Value};
3use metal::{Buffer, ComputeCommandEncoderRef, Device, MTLSize, NSUInteger};
4use std::ffi::c_void;
5
6#[derive(Copy, Clone, PartialEq, Eq, Hash, Debug)]
7pub enum GemmDType {
8    BF16,
9    F16,
10    F32,
11}
12
13#[allow(clippy::too_many_arguments)]
14pub fn call_mlx_gemm(
15    device: &Device,
16    ep: impl EncoderProvider,
17    kernels: &Kernels,
18    dtype: GemmDType,
19    (b, m, n, k): (usize, usize, usize, usize),
20    lhs_stride: &[usize],
21    lhs_offset: usize,
22    lhs_buffer: &Buffer,
23    rhs_stride: &[usize],
24    rhs_offset: usize,
25    rhs_buffer: &Buffer,
26    output: &Buffer,
27) -> Result<(), MetalKernelError> {
28    #[derive(Debug)]
29    #[repr(C)]
30    struct GemmParams {
31        m: i32,
32        n: i32,
33        k: i32,
34        lda: i32,
35        ldb: i32,
36        ldd: i32,
37        tiles_n: i32,
38        tiles_m: i32,
39        batch_stride_a: isize,
40        batch_stride_b: isize,
41        batch_stride_d: isize,
42        swizzle_log: i32,
43        gemm_k_iterations_aligned: i32,
44        batch_ndim: i32,
45    }
46    assert!(rhs_stride.len() >= 2);
47    assert!(lhs_stride.len() >= 2);
48    let rhs_m1 = rhs_stride[rhs_stride.len() - 1];
49    let rhs_m2 = rhs_stride[rhs_stride.len() - 2];
50    let lhs_m1 = lhs_stride[lhs_stride.len() - 1];
51    let lhs_m2 = lhs_stride[lhs_stride.len() - 2];
52    // lhs has shape b, m, k
53    // We also allow for the case where the stride on the minor dimension is not as expected but
54    // there is a single element.
55    let (lda, a_trans) = if (lhs_m1 == 1 || k == 1) && (lhs_m2 == k || m == 1) {
56        (k as i32, false)
57    } else if (lhs_m1 == m || k == 1) && (lhs_m2 == 1 || m == 1) {
58        (m as i32, true)
59    } else {
60        return Err(MetalKernelError::MatMulNonContiguous {
61            lhs_stride: lhs_stride.to_vec(),
62            rhs_stride: rhs_stride.to_vec(),
63            mnk: (m, n, k),
64        })?;
65    };
66    // rhs has shape b, k, n
67    let (ldb, b_trans) = if (rhs_m1 == 1 || n == 1) && (rhs_m2 == n || k == 1) {
68        (n as i32, false)
69    } else if (rhs_m1 == k || n == 1) && (rhs_m2 == 1 || k == 1) {
70        (k as i32, true)
71    } else {
72        return Err(MetalKernelError::MatMulNonContiguous {
73            lhs_stride: lhs_stride.to_vec(),
74            rhs_stride: rhs_stride.to_vec(),
75            mnk: (m, n, k),
76        })?;
77    };
78    let (bm, bn, bk, wn, wm) = (32, 32, 16, 2, 2);
79    // https://github.com/ml-explore/mlx/blob/02efb310cac667bc547d1b96f21596c221f84fe7/mlx/backend/metal/matmul.cpp#L422
80    let constants = Some(ConstantValues::new(vec![
81        (10, Value::Bool(/* has_batch */ b > 1)),
82        (100, Value::Bool(/* use_out_source */ false)),
83        (110, Value::Bool(/* do_axpby */ false)),
84        (200, Value::Bool(/* align_m */ m % bm == 0)),
85        (201, Value::Bool(/* align_n */ n % bn == 0)),
86        (202, Value::Bool(/* align_k */ k % bk == 0)),
87        (300, Value::Bool(/* do_gather */ false)),
88    ]));
89
90    let swizzle_log = 0;
91    let tile = 1 << swizzle_log;
92    let tn = n.div_ceil(bn);
93    let tm = m.div_ceil(bm);
94    let tn = tn * tile;
95    let tm = tm.div_ceil(tile);
96
97    let batch_stride_a = if lhs_stride.len() > 2 {
98        lhs_stride[lhs_stride.len() - 3]
99    } else {
100        m * k
101    };
102    let batch_stride_b = if rhs_stride.len() > 2 {
103        rhs_stride[rhs_stride.len() - 3]
104    } else {
105        n * k
106    };
107
108    let gemm_params = GemmParams {
109        m: m as i32,
110        n: n as i32,
111        k: k as i32,
112        lda,
113        ldb,
114        ldd: n as i32,
115        tiles_n: tn as i32,
116        tiles_m: tm as i32,
117        swizzle_log,
118        batch_stride_a: batch_stride_a as isize,
119        batch_stride_b: batch_stride_b as isize,
120        batch_stride_d: (m * n) as isize,
121        batch_ndim: 1i32,
122        gemm_k_iterations_aligned: (k / bk) as i32,
123    };
124    let batch_strides = [gemm_params.batch_stride_a, gemm_params.batch_stride_b];
125
126    // TODO(laurent): generate the name
127    // template [[host_name("gemm_" #tname "_"  #iname "_" #oname "_bm" #bm "_bn" #bn "_bk" #bk "_wm" #wm "_wn" #wn)]]
128    let name = match (dtype, a_trans, b_trans) {
129        (GemmDType::F32, false, false) => "gemm_nn_f32_f32_32_32_16_2_2",
130        (GemmDType::F32, true, false) => "gemm_tn_f32_f32_32_32_16_2_2",
131        (GemmDType::F32, false, true) => "gemm_nt_f32_f32_32_32_16_2_2",
132        (GemmDType::F32, true, true) => "gemm_tt_f32_f32_32_32_16_2_2",
133        (GemmDType::BF16, false, false) => "gemm_nn_bf16_bf16_32_32_16_2_2",
134        (GemmDType::BF16, true, false) => "gemm_tn_bf16_bf16_32_32_16_2_2",
135        (GemmDType::BF16, false, true) => "gemm_nt_bf16_bf16_32_32_16_2_2",
136        (GemmDType::BF16, true, true) => "gemm_tt_bf16_bf16_32_32_16_2_2",
137        (GemmDType::F16, false, false) => "gemm_nn_f16_f16_32_32_16_2_2",
138        (GemmDType::F16, true, false) => "gemm_tn_f16_f16_32_32_16_2_2",
139        (GemmDType::F16, false, true) => "gemm_nt_f16_f16_32_32_16_2_2",
140        (GemmDType::F16, true, true) => "gemm_tt_f16_f16_32_32_16_2_2",
141    };
142    let pipeline = kernels.load_pipeline_with_constants(device, Source::Gemm, name, constants)?;
143    let encoder = ep.encoder();
144    let encoder: &ComputeCommandEncoderRef = encoder.as_ref();
145    encoder.set_compute_pipeline_state(&pipeline);
146    encoder.set_buffer(0, Some(lhs_buffer), lhs_offset as NSUInteger);
147    encoder.set_buffer(1, Some(rhs_buffer), rhs_offset as NSUInteger);
148    encoder.set_buffer(3, Some(output), 0);
149    encoder.set_bytes(
150        4,
151        std::mem::size_of::<GemmParams>() as u64,
152        &gemm_params as *const GemmParams as *const c_void,
153    );
154    encoder.set_bytes(
155        6, // batch_shape
156        std::mem::size_of::<i32>() as u64,
157        &(b as i32) as *const i32 as *const c_void,
158    );
159    encoder.set_bytes(
160        7,
161        (std::mem::size_of::<isize>() * batch_strides.len()) as u64,
162        batch_strides.as_ptr() as *const c_void,
163    );
164
165    let grid_size = MTLSize {
166        width: tn as u64,
167        height: tm as u64,
168        depth: /* batch_size_out */ b as u64,
169    };
170    let group_size = MTLSize {
171        width: 32,
172        height: wn,
173        depth: wm,
174    };
175    encoder.use_resource(lhs_buffer, metal::MTLResourceUsage::Read);
176    encoder.use_resource(rhs_buffer, metal::MTLResourceUsage::Read);
177    encoder.use_resource(output, metal::MTLResourceUsage::Write);
178    encoder.dispatch_thread_groups(grid_size, group_size);
179    Ok(())
180}
