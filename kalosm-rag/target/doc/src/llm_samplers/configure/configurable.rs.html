8/// Configurable samplers implement this trait. "Configurable" means
9/// they allow access to their their options by key/type and allow configuration
10/// based on descriptions.
11///
12/// There are default implementations for for all the methods, so in the general
13/// case you will only need to implement the `OPTIONS` constant with the option
14/// definitions.
15pub trait ConfigurableSampler<UI = u32, F = f32>: HasSamplerMetadata<UI, F>
16where
17    UI: ConfigurableNumValue,
18    F: ConfigurableNumValue,
19{
20    /// Given an option key and [SamplerOptionValue] attempts to set the option
21    /// to the specified value.
22    ///
23    /// The default implementation will call [Self::pre_set_option] before setting
24    /// the option and [Self::post_set_option] afterward. For an example of
25    /// the latter you can look at the Mirostat samplers: when `tau` is set, they'll
26    /// automatically set `mu` to `tau * 2`.
27    fn set_option(&mut self, key: &str, val: SamplerOptionValue) -> Result<()> {
28        configurable_sampler::set_option(self, key, val)?;
29        Ok(())
30    }
31
32    /// Called before an option is set and is passed a mutable reference
33    /// to the [SamplerOptionValue]. It is also passed an index into
34    /// the options definition list.
35    ///
36    /// The default implementation is a no-op.
37    #[allow(unused_variables)]
38    fn pre_set_option(
39        &mut self,
40        md: &SamplerOptionMetadata,
41        val: &mut SamplerOptionValue,
42    ) -> Result<()> {
43        Ok(())
44    }
45
46    /// Called before an option is set is passed an index into
47    /// the options definition list.
48    ///
49    /// The default implementation is a no-op.
50    #[allow(unused_variables)]
51    fn post_set_option(&mut self, md: &SamplerOptionMetadata) -> Result<()> {
52        Ok(())
53    }
54
55    /// Gets an option by name.
56    fn get_option(&self, key: &str) -> Result<SamplerOptionValue> {
57        configurable_sampler::get_option(self, key)
58    }
59
60    /// Updates a sampler's configurable options based on a string in the
61    /// format:
62    ///
63    /// `key1=value1:key2=value2:keyN=valueN`
64    ///
65    /// The key be a prefix of the option name as long as it's not
66    /// ambiguous. It's also possible to just specify the value,
67    /// which is equivalent to `=value` (i.e. a blank key name).
68    ///
69    /// Values in this default implementation cannot contain `=` or `:`
70    /// and whitespace at the beginning and end of parts are stripped.
71    fn configure(&mut self, s: &str) -> Result<()> {
72        configurable_sampler::configure(self, s)?;
73        Ok(())
74    }
75}

77/// Since Rust traits don't allow calling base default methods from
78/// a more specific implementation, the [ConfigurableSampler] trait
79/// default methods are implemented in terms of the functions
80/// in this submodule.
81///
82/// See the trait for descriptions of the methods.
83pub mod configurable_sampler {
84    use super::*;
85
86    pub fn set_option<'a, CS, UI, F>(
87        slf: &'a mut CS,
88        key: &str,
89        mut val: SamplerOptionValue,
90    ) -> Result<&'a mut CS>
91    where
92        CS: ConfigurableSampler<UI, F> + HasSamplerMetadata<UI, F> + ?Sized,
93        UI: ConfigurableNumValue,
94        F: ConfigurableNumValue,
95    {
96        let key = key.trim();
97        let (omd, optidx) = {
98            let opts = slf.sampler_options_mut();
99            if let (omd, Some(optidx)) = opts.find_option_definition(key)? {
100                (omd, optidx)
101            } else {
102                Err(ConfigureSamplerError::CannotAccessOptionValue(
103                    key.to_string(),
104                ))?
105            }
106        };
107
108        slf.pre_set_option(&omd, &mut val)?;
109
110        let mut opts = slf.sampler_options_mut();
111        let acc = opts[optidx]
112            .1
113            .take()
114            .ok_or_else(|| ConfigureSamplerError::CannotAccessOptionValue(key.to_string()))?;
115
116        match (acc, val) {
117            (SamplerOptionValueMut::Float(rv), SamplerOptionValue::Float(v)) => {
118                *rv = F::from_f64(v)
119                    .ok_or_else(|| ConfigureSamplerError::ConversionFailure(key.to_string()))?
120            }
121
122            (SamplerOptionValueMut::UInt(rv), SamplerOptionValue::UInt(v)) => {
123                *rv = UI::from_u64(v)
124                    .ok_or_else(|| ConfigureSamplerError::ConversionFailure(key.to_string()))?
125            }
126            (SamplerOptionValueMut::Bool(rv), SamplerOptionValue::Bool(v)) => *rv = v,
127            (SamplerOptionValueMut::String(rv), SamplerOptionValue::String(v)) => {
128                *rv = Cow::from(v.to_string())
129            }
130            _ => Err(ConfigureSamplerError::UnknownOrBadType(key.to_string()))?,
131        }
132        slf.post_set_option(&omd)?;
133        Ok(slf)
134    }
135
136    pub fn get_option<'a, CS, UI, F>(slf: &'a CS, key: &str) -> Result<SamplerOptionValue<'a>>
137    where
138        CS: ConfigurableSampler<UI, F> + HasSamplerMetadata<UI, F> + ?Sized,
139        UI: ConfigurableNumValue,
140        F: ConfigurableNumValue,
141    {
142        let key = key.trim();
143
144        let mut opts = slf.sampler_options();
145
146        let (_omd, Some(optidx)) = opts.find_option_definition(key)? else {
147            Err(ConfigureSamplerError::CannotAccessOptionValue(key.to_string()))?
148        };
149
150        Ok(match opts[optidx].1.take().expect("Impossible") {
151            SamplerOptionValue::UInt(v) => SamplerOptionValue::UInt(
152                <u64 as NumCast>::from(v)
153                    .ok_or_else(|| ConfigureSamplerError::ConversionFailure(key.to_string()))?,
154            ),
155            SamplerOptionValue::Float(v) => SamplerOptionValue::Float(
156                <f64 as NumCast>::from(v)
157                    .ok_or_else(|| ConfigureSamplerError::ConversionFailure(key.to_string()))?,
158            ),
159            SamplerOptionValue::Bool(v) => SamplerOptionValue::Bool(v),
160            SamplerOptionValue::String(v) => SamplerOptionValue::String(Cow::from(v.to_string())),
161        })
162    }
163
164    pub fn configure<CS, UI, F>(slf: &mut CS, s: &str) -> Result<()>
165    where
166        CS: ConfigurableSampler<UI, F> + HasSamplerMetadata<UI, F> + ?Sized,
167        UI: ConfigurableNumValue,
168        F: ConfigurableNumValue,
169    {
170        let opts = SamplerOptions::from(
171            slf.sampler_options_mut()
172                .iter()
173                .map(|(md, acc)| (md.clone(), acc.is_some().then_some(()))),
174        );
175        s.trim()
176            .split(':')
177            .map(str::trim)
178            .filter(|s| !s.is_empty())
179            .try_for_each(|kv| {
180                let (k, v) = kv.split_once('=').unwrap_or(("", kv));
181                let (omd, Some(_)) = opts.find_option_definition(k)? else {
182                    Err(ConfigureSamplerError::UnknownOrBadType(k.to_string()))?
183                };
184
185                slf.set_option(
186                    omd.key,
187                    SamplerOptionValue::parse_value(omd.option_type, v.trim())?,
188                )?;
189                anyhow::Ok(())
190            })?;
191        Ok(())
192    }
