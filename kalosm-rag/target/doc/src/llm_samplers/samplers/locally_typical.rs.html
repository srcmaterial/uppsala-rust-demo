6/// # Locally typical sampling
7///
8/// An approach to sampling that attempts to maximize natural
9/// and human-like output.
10///
11/// See: <https://arxiv.org/abs/2202.00666>
12///
13/// **Properties**:
14/// - Modifies logits
15/// - Filters logits
16///
17/// **Parameters**:
18/// - `min_keep`: Minimum number of entries to keep. Setting this to `0` is not recommended. (default: `1`)
19/// - `p`: Referred to as τ in the paper. It suggests using 0.2
20///   as a value for story generation and `0.95` for "abstractive summarization"
21///   (presumably this means more factual output). (default: `1.0`)
22#[derive(Debug, Clone, Copy, PartialEq)]
23pub struct SampleLocallyTypical {
24    pub(crate) p: L,
25    pub(crate) min_keep: usize,
26}

53impl Sampler for SampleLocallyTypical {
54    fn sample<'a>(
55        &mut self,
56        _res: &mut dyn HasSamplerResources,
57        logits: &'a mut Logits,
58    ) -> anyhow::Result<&'a mut Logits> {
59        use std::ops::ControlFlow::*;
60
61        let Self { p, min_keep } = *self;
62        let min_keep = if min_keep == 0 { 0 } else { min_keep - 1 };
63        logits.ensure_softmax()?;
64
65        let ent = logits
66            .iter()
67            .fold(0f32, |ent, l| ent + -l.prob * l.prob.ln());
68
69        let mut shifted = logits
70            .iter()
71            .map(|l| (l.clone(), (-l.prob.ln() - ent).abs()))
72            .collect::<Vec<_>>();
73        {
74            let mut sort_err = Ok(());
75            shifted.sort_by(|a, b| {
76                a.1.partial_cmp(&b.1).unwrap_or_else(|| {
77                    sort_err = Err(SamplerError::InternalError(String::from(
78                        "Impossible: logit comparison failed?",
79                    )));
80                    Ordering::Less
81                })
82            });
83            sort_err?;
84        }
85
86        let mut cum_sum = 0f32;
87        let last_idx = match shifted.iter().enumerate().try_fold(
88            shifted.len(),
89            |last_idx, (idx, (logit, _score))| {
90                cum_sum += logit.prob;
91                if cum_sum > p && idx >= min_keep {
92                    return Break(idx + 1);
93                }
94                Continue(last_idx)
95            },
96        ) {
97            Continue(i) => i,
98            Break(i) => i,
99        };
100        logits.clear();
101        logits.set_sorted(false);
102        logits.set_softmax(false);
103        shifted
104            .into_iter()
105            .take(last_idx)
106            .for_each(|(logit, _score)| logits.push(logit));
107        Ok(logits)
108    }
109}
