3/// # Min-P sampling
4/// This sampler prunes tokens that don't meet a certain percentage
5/// of the most probable token. For example if `p` is `0.05` then
6/// after `min_keep` is satisfied, other tokens must be at least 5%
7/// of the most probable token.
8///
9/// Credit to @kalomaze on GitHub for design. See this link for a more in-depth
10/// explanation: https://github.com/ggerganov/llama.cpp/issues/3483#issuecomment-1783920998
11
12///
13/// **Properties**:
14/// - Filters logits
15///
16/// **Parameters**:
17/// - `min_keep`: Minimum number of entries to keep. (default: `1`)
18/// - `p`: Threshold value. Use `0.0` to disable. (default: `0.9`)
19#[derive(Debug, Clone, Copy, PartialEq)]
20pub struct SampleMinP {
21    pub(crate) p: L,
22    pub(crate) min_keep: usize,
23}
24
25impl Default for SampleMinP {
26    fn default() -> Self {
27        Self {
28            p: 0.05f32,
29            min_keep: 1,
30        }
31    }
32}
33
34impl SampleMinP {
35    pub fn new(p: L, min_keep: usize) -> Self {
36        Self { p, min_keep }
37    }
38
39    pub fn min_keep(mut self, val: usize) -> Self {
40        self.min_keep = val;
41        self
42    }
43
44    pub fn p(mut self, val: L) -> Self {
45        self.p = val;
46        self
47    }
48}
49
50impl Sampler for SampleMinP {
51    fn sample<'a>(
52        &mut self,
53        _res: &mut dyn HasSamplerResources,
54        logits: &'a mut Logits,
55    ) -> anyhow::Result<&'a mut Logits> {
56        let Self { p, min_keep } = *self;
57        if p == 0f32 || logits.is_empty() {
58            return Ok(logits);
59        }
60
61        logits.ensure_softmax()?;
62
63        if logits.len() <= min_keep {
64            return Ok(logits);
65        }
66
67        let threshold = logits[0].prob * p;
68        let last_idx = logits
69            .iter()
70            .enumerate()
71            .skip(1.max(min_keep))
72            .find(|(_, l)| l.prob < threshold)
73            .map(|(idx, _)| idx)
74            .unwrap_or_else(|| logits.len());
75        if last_idx != logits.len() {
76            logits.truncate(last_idx);
77            logits.set_softmax(false);
78        }
79        Ok(logits)
80    }
81}
