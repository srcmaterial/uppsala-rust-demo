3pub(crate) fn render_token(token: &Token<'_>) -> String {

10        Token::Hash(name) | Token::IDHash(name) => format!("#{}", name.clone()),
11        Token::QuotedString(value) => format!("\"{}\"", value.clone()),

23            has_sign: signed,

25            int_value: _,
26            unit,
27        } => format!("{}{}", render_int(*signed, *num), unit),
28        Token::WhiteSpace(_) => String::from(" "),
29        Token::Comment(comment) => format!("/* {} */", comment),
30        Token::Function(name) => format!("{}()", name.clone()),
31        Token::BadString(string) => format!("<Bad String {:?}>", string.clone()),
32        Token::BadUrl(url) => format!("<Bad URL {:?}>", url.clone()),
33        // Single-character token
34        sc_token => render_single_char_token(sc_token),
35    }

38fn render_single_char_token(token: &Token) -> String {
39    String::from(match token {
40        Token::Colon => ":",
41        Token::Semicolon => ";",
42        Token::Comma => ",",
43        Token::IncludeMatch => "~=",
44        Token::DashMatch => "|=",
45        Token::PrefixMatch => "^=",
46        Token::SuffixMatch => "$=",
47        Token::SubstringMatch => "*=",
48        Token::CDO => "<!--",
49        Token::CDC => "-->",
50        Token::ParenthesisBlock => "<(",
51        Token::SquareBracketBlock => "<[",
52        Token::CurlyBracketBlock => "<{",
53        Token::CloseParenthesis => "<)",
54        Token::CloseSquareBracket => "<]",
55        Token::CloseCurlyBracket => "<}",
56        other => panic!(
57            "Token {:?} is not supposed to match as a single-character token!",
58            other
59        ),
60    })
61}

63fn render_number(signed: bool, num: f32, token: &Token) -> String {
64    let num = render_int(signed, num);
65
66    match token {
67        Token::Number { .. } => num,
68        Token::Percentage { .. } => format!("{}%", num),
69        _ => panic!("render_number is not supposed to be called on a non-numerical token"),
70    }
71}

73fn render_int(signed: bool, num: f32) -> String {
74    if signed {
75        render_int_signed(num)
76    } else {
77        render_int_unsigned(num)
78    }
79}

81fn render_int_signed(num: f32) -> String {
