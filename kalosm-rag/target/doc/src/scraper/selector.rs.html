17/// Wrapper around CSS selectors.
18///
19/// Represents a "selector group", i.e. a comma-separated list of selectors.
20#[derive(Debug, Clone, PartialEq, Eq)]
21pub struct Selector {
22    /// The CSS selectors.
23    selectors: SelectorList<Simple>,
24}
25
26impl Selector {
27    /// Parses a CSS selector group.
28    pub fn parse(selectors: &'_ str) -> Result<Self, SelectorErrorKind> {
29        let mut parser_input = cssparser::ParserInput::new(selectors);
30        let mut parser = cssparser::Parser::new(&mut parser_input);
31
32        SelectorList::parse(&Parser, &mut parser, ParseRelative::No)
33            .map(|selectors| Self { selectors })
34            .map_err(SelectorErrorKind::from)
35    }
36
37    /// Returns true if the element matches this selector.
38    pub fn matches(&self, element: &ElementRef) -> bool {
39        self.matches_with_scope(element, None)
40    }
41
42    /// Returns true if the element matches this selector.
43    /// The optional `scope` argument is used to specify which element has `:scope` pseudo-class.
44    /// When it is `None`, `:scope` will match the root element.
45    pub fn matches_with_scope(&self, element: &ElementRef, scope: Option<ElementRef>) -> bool {
46        self.matches_with_scope_and_cache(element, scope, &mut NthIndexCache::default())
47    }
48
49    // The `nth_index_cache` must not be used after `self` is dropped
50    // to avoid incorrect results (even though no undefined behaviour is possible)
51    // due to the usage of selector memory addresses as cache keys.
52    pub(crate) fn matches_with_scope_and_cache(
53        &self,
54        element: &ElementRef,
55        scope: Option<ElementRef>,
56        nth_index_cache: &mut NthIndexCache,
57    ) -> bool {
58        let mut context = matching::MatchingContext::new(
59            matching::MatchingMode::Normal,
60            None,
61            nth_index_cache,
62            matching::QuirksMode::NoQuirks,
63            matching::NeedsSelectorFlags::No,
64            matching::IgnoreNthChildForInvalidation::No,
65        );
66        context.scope_element = scope.map(|x| selectors::Element::opaque(&x));
67        self.selectors
68            .0
69            .iter()
70            .any(|s| matching::matches_selector(s, 0, None, element, &mut context))
71    }
72}

111/// Wraps [`String`] so that it can be used with [`selectors`]
112#[derive(Debug, Clone, PartialEq, Eq)]
113pub struct CssString(pub String);
114
115impl<'a> From<&'a str> for CssString {
116    fn from(val: &'a str) -> Self {
117        Self(val.to_owned())
118    }
119}
120
121impl AsRef<str> for CssString {
122    fn as_ref(&self) -> &str {
123        &self.0
124    }
125}
126
127impl ToCss for CssString {
128    fn to_css<W>(&self, dest: &mut W) -> fmt::Result
129    where
130        W: fmt::Write,
131    {
132        cssparser::serialize_string(&self.0, dest)
133    }
134}
