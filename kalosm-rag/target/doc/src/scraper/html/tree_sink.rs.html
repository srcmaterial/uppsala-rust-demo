11/// Note: does not support the `<template>` element.
12impl TreeSink for Html {
13    type Output = Self;
14    type Handle = NodeId;
15
16    fn finish(self) -> Self {
17        self

20    // Signal a parse error.
21    fn parse_error(&mut self, msg: Cow<'static, str>) {
22        #[cfg(feature = "errors")]
23        self.errors.push(msg);
24        #[cfg(not(feature = "errors"))]
25        let _ = msg;
26    }

28    // Set the document's quirks mode.
29    fn set_quirks_mode(&mut self, mode: QuirksMode) {
30        self.quirks_mode = mode;

33    // Get a handle to the Document node.
34    fn get_document(&mut self) -> Self::Handle {
35        self.tree.root().id()
36    }

38    // Do two handles refer to the same node?
39    fn same_node(&self, x: &Self::Handle, y: &Self::Handle) -> bool {
40        x == y
41    }

43    // What is the name of this element?
44    //
45    // Should never be called on a non-element node; feel free to panic!.
46    fn elem_name(&self, target: &Self::Handle) -> ExpandedName {
47        self.tree
48            .get(*target)
49            .unwrap()
50            .value()
51            .as_element()
52            .unwrap()
53            .name
54            .expanded()
55    }
56
57    // Create an element.
58    //
59    // When creating a template element (name.expanded() == expanded_name!(html "template")), an
60    // associated document fragment called the "template contents" should also be created. Later
61    // calls to self.get_template_contents() with that given element return it.
62    fn create_element(
63        &mut self,
64        name: QualName,
65        attrs: Vec<Attribute>,
66        _flags: ElementFlags,
67    ) -> Self::Handle {
68        let fragment = name.expanded() == expanded_name!(html "template");
69
70        let mut node = self.tree.orphan(Node::Element(Element::new(name, attrs)));
71
72        if fragment {
73            node.append(Node::Fragment);
74        }
75
76        node.id()
77    }
78
79    // Create a comment node.
80    fn create_comment(&mut self, text: StrTendril) -> Self::Handle {
81        self.tree
82            .orphan(Node::Comment(Comment {
83                comment: make_tendril(text),
84            }))
85            .id()
86    }

88    // Append a DOCTYPE element to the Document node.
89    fn append_doctype_to_document(
90        &mut self,
91        name: StrTendril,
92        public_id: StrTendril,
93        system_id: StrTendril,
94    ) {
95        let name = make_tendril(name);
96        let public_id = make_tendril(public_id);
97        let system_id = make_tendril(system_id);
98        let doctype = Doctype {
99            name,
100            public_id,
101            system_id,
102        };
103        self.tree.root_mut().append(Node::Doctype(doctype));
104    }

106    // Append a node as the last child of the given node. If this would produce adjacent sibling
107    // text nodes, it should concatenate the text instead.
108    //
109    // The child node will not already have a parent.
110    fn append(&mut self, parent: &Self::Handle, child: NodeOrText<Self::Handle>) {
111        let mut parent = self.tree.get_mut(*parent).unwrap();
112
113        match child {
114            NodeOrText::AppendNode(id) => {
115                parent.append_id(id);
116            }
117
118            NodeOrText::AppendText(text) => {
119                let text = make_tendril(text);
120
121                let did_concat = parent.last_child().map_or(false, |mut n| match n.value() {
122                    Node::Text(t) => {
123                        t.text.push_tendril(&text);
124                        true
125                    }
126                    _ => false,
127                });
128
129                if !did_concat {
130                    parent.append(Node::Text(Text { text }));
131                }
132            }
133        }
134    }
135
136    // Append a node as the sibling immediately before the given node. If that node has no parent,
137    // do nothing and return Err(new_node).
138    //
139    // The tree builder promises that sibling is not a text node. However its old previous sibling,
140    // which would become the new node's previous sibling, could be a text node. If the new node is
141    // also a text node, the two should be merged, as in the behavior of append.
142    //
143    // NB: new_node may have an old parent, from which it should be removed.
144    fn append_before_sibling(
145        &mut self,
146        sibling: &Self::Handle,
147        new_node: NodeOrText<Self::Handle>,
148    ) {
149        if let NodeOrText::AppendNode(id) = new_node {
150            self.tree.get_mut(id).unwrap().detach();
151        }
152
153        let mut sibling = self.tree.get_mut(*sibling).unwrap();
154        if sibling.parent().is_some() {
155            match new_node {
156                NodeOrText::AppendNode(id) => {
157                    sibling.insert_id_before(id);
158                }
159
160                NodeOrText::AppendText(text) => {
161                    let text = make_tendril(text);
162
163                    let did_concat =
164                        sibling
165                            .prev_sibling()
166                            .map_or(false, |mut n| match n.value() {
167                                Node::Text(t) => {
168                                    t.text.push_tendril(&text);
169                                    true
170                                }
171                                _ => false,
172                            });
173
174                    if !did_concat {
175                        sibling.insert_before(Node::Text(Text { text }));
176                    }
177                }
178            }
179        }
180    }
181
182    // Detach the given node from its parent.
183    fn remove_from_parent(&mut self, target: &Self::Handle) {
184        self.tree.get_mut(*target).unwrap().detach();
185    }
186
187    // Remove all the children from node and append them to new_parent.
188    fn reparent_children(&mut self, node: &Self::Handle, new_parent: &Self::Handle) {
189        self.tree
190            .get_mut(*new_parent)
191            .unwrap()
192            .reparent_from_id_append(*node);
193    }
194
195    // Add each attribute to the given element, if no attribute with that name already exists. The
196    // tree builder promises this will never be called with something else than an element.
197    fn add_attrs_if_missing(&mut self, target: &Self::Handle, attrs: Vec<Attribute>) {
198        let mut node = self.tree.get_mut(*target).unwrap();
199        let element = match *node.value() {
200            Node::Element(ref mut e) => e,
201            _ => unreachable!(),
202        };
203
204        for attr in attrs {
205            element
206                .attrs
207                .entry(attr.name)
208                .or_insert_with(|| make_tendril(attr.value));
209        }
210    }
211
212    // Get a handle to a template's template contents.
213    //
214    // The tree builder promises this will never be called with something else than a template
215    // element.
216    fn get_template_contents(&mut self, target: &Self::Handle) -> Self::Handle {
217        self.tree.get(*target).unwrap().first_child().unwrap().id()
218    }
219
220    // Mark a HTML <script> element as "already started".
221    fn mark_script_already_started(&mut self, _node: &Self::Handle) {}
222
223    // Create Processing Instruction.
224    fn create_pi(&mut self, target: StrTendril, data: StrTendril) -> Self::Handle {
225        let target = make_tendril(target);
226        let data = make_tendril(data);
227        self.tree
228            .orphan(Node::ProcessingInstruction(ProcessingInstruction {
229                target,
230                data,
231            }))
232            .id()
233    }
234
235    fn append_based_on_parent_node(
236        &mut self,
237        element: &Self::Handle,
238        prev_element: &Self::Handle,
239        child: NodeOrText<Self::Handle>,
240    ) {
241        if self.tree.get(*element).unwrap().parent().is_some() {
242            self.append_before_sibling(element, child)
243        } else {
244            self.append(prev_element, child)
245        }
246    }
