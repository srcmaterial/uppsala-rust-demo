36/// The options available when building the arroy database.
37struct BuildOption<'a> {
38    n_trees: Option<usize>,
39    split_after: Option<usize>,
40    cancel: Box<dyn Fn() -> bool + 'a + Sync + Send>,
41}
42
43impl Default for BuildOption<'_> {
44    fn default() -> Self {
45        Self { n_trees: None, split_after: None, cancel: Box::new(|| false) }
46    }
47}

88    /// Provide a closure that can cancel the indexing process early if needed.
89    /// There is no guarantee on when the process is going to cancel itself, but
90    /// arroy will try to stop as soon as possible once the closure returns `true`.
91    ///
92    /// Since the closure is not mutable and will be called from multiple threads
93    /// at the same time it's encouraged to make it quick to execute. A common
94    /// way to use it is to fetch an `AtomicBool` inside it that can be set
95    /// from another thread without lock.
96    ///
97    /// # Example
98    ///
99    /// ```no_run
100    /// # use arroy::{Writer, distances::Euclidean};
101    /// # let (writer, wtxn): (Writer<Euclidean>, heed::RwTxn) = todo!();
102    /// use rand::rngs::StdRng;
103    /// use rand::SeedableRng;
104    /// use std::sync::atomic::{AtomicBool, Ordering};
105    ///
106    /// let stops_after = AtomicBool::new(false);
107    ///
108    /// // Cancel the task after one minute
109    /// std::thread::spawn(|| {
110    ///     let one_minute = std::time::Duration::from_secs(60);
111    ///     std::thread::sleep(one_minute);
112    ///     stops_after.store(true, Ordering::Relaxed);
113    /// });
114    ///
115    /// let mut rng = StdRng::seed_from_u64(92);
116    /// writer.builder(&mut rng).split_after(1000).build(&mut wtxn);
117    /// ```
118    pub fn cancel(&mut self, cancel: impl Fn() -> bool + 'a + Sync + Send) -> &mut Self {
119        self.inner.cancel = Box::new(cancel);
120        self
121    }

123    /// Generates a forest of `n_trees` trees.
124    ///
125    /// More trees give higher precision when querying at the cost of more disk usage.
126    ///
127    /// This function is using rayon to spawn threads. It can be configured
128    /// by using the [`rayon::ThreadPoolBuilder`] and the
129    /// [`rayon::ThreadPool::install`].
130    ///
131    /// # Example
132    ///
133    /// ```no_run
134    /// # use arroy::{Writer, distances::Euclidean};
135    /// # let (writer, wtxn): (Writer<Euclidean>, heed::RwTxn) = todo!();
136    /// use rand::rngs::StdRng;
137    /// use rand::SeedableRng;
138    /// let mut rng = StdRng::seed_from_u64(92);
139    /// writer.builder(&mut rng).build(&mut wtxn);
140    /// ```
141    pub fn build(&mut self, wtxn: &mut RwTxn) -> Result<()> {
142        self.writer.build(wtxn, self.rng, &self.inner)
143    }

165    /// Returns a writer after having deleted the tree nodes and rewrote all the items
166    /// for the new [`Distance`] format to be able to modify items safely.
167    pub fn prepare_changing_distance<ND: Distance>(self, wtxn: &mut RwTxn) -> Result<Writer<ND>> {
168        if TypeId::of::<ND>() != TypeId::of::<D>() {
169            clear_tree_nodes(wtxn, self.database, self.index)?;
170
171            let mut cursor = self
172                .database
173                .remap_key_type::<PrefixCodec>()
174                .prefix_iter_mut(wtxn, &Prefix::item(self.index))?
175                .remap_key_type::<KeyCodec>();
176            while let Some((item_id, node)) = cursor.next().transpose()? {
177                match node {
178                    Node::Leaf(Leaf { header: _, vector }) => {
179                        let vector = vector.to_vec();
180                        let vector = UnalignedVector::from_vec(vector);
181                        let new_leaf = Node::Leaf(Leaf { header: ND::new_header(&vector), vector });
182                        unsafe {
183                            // safety: We do not keep a reference to the current value, we own it.
184                            cursor.put_current_with_options::<NodeCodec<ND>>(
185                                PutFlags::empty(),
186                                &item_id,
187                                &new_leaf,
188                            )?
189                        };
190                    }
191                    Node::Descendants(_) | Node::SplitPlaneNormal(_) => panic!(),
192                }
193            }
194        }
195
196        let Writer { database, index, dimensions, tmpdir } = self;
197        Ok(Writer { database: database.remap_data_type(), index, dimensions, tmpdir })
198    }

258    /// Add an item associated to a vector in the database.
259    pub fn add_item(&self, wtxn: &mut RwTxn, item: ItemId, vector: &[f32]) -> Result<()> {
260        if vector.len() != self.dimensions {
261            return Err(Error::InvalidVecDimension {
262                expected: self.dimensions,
263                received: vector.len(),
264            });
265        }
266
267        let vector = UnalignedVector::from_slice(vector);
268        let leaf = Leaf { header: D::new_header(&vector), vector };
269        self.database.put(wtxn, &Key::item(self.index, item), &Node::Leaf(leaf))?;
270        self.database.remap_data_type::<Unit>().put(wtxn, &Key::updated(self.index, item), &())?;
271
272        Ok(())
273    }

275    /// Attempt to append an item into the database. It is generaly faster to append an item than insert it.
276    ///
277    /// There are two conditions for an item to be successfully appended:
278    ///  - The last item ID in the database is smaller than the one appended.
279    ///  - The index of the database is the highest one.
280    pub fn append_item(&self, wtxn: &mut RwTxn, item: ItemId, vector: &[f32]) -> Result<()> {
281        if vector.len() != self.dimensions {
282            return Err(Error::InvalidVecDimension {
283                expected: self.dimensions,
284                received: vector.len(),
285            });
286        }
287
288        let vector = UnalignedVector::from_slice(vector);
289        let leaf = Leaf { header: D::new_header(&vector), vector };
290        let key = Key::item(self.index, item);
291        match self.database.put_with_flags(wtxn, PutFlags::APPEND, &key, &Node::Leaf(leaf)) {
292            Ok(()) => (),
293            Err(heed::Error::Mdb(MdbError::KeyExist)) => return Err(Error::InvalidItemAppend),
294            Err(e) => return Err(e.into()),
295        }
296        // We cannot append here because the items appear after the updated keys
297        self.database.remap_data_type::<Unit>().put(wtxn, &Key::updated(self.index, item), &())?;
298
299        Ok(())
300    }

302    /// Deletes an item stored in this database and returns `true` if it existed.
303    pub fn del_item(&self, wtxn: &mut RwTxn, item: ItemId) -> Result<bool> {
304        if self.database.delete(wtxn, &Key::item(self.index, item))? {
305            self.database.remap_data_type::<Unit>().put(
306                wtxn,
307                &Key::updated(self.index, item),
308                &(),
309            )?;
310
311            Ok(true)
312        } else {
313            Ok(false)
314        }
315    }

317    /// Removes everything in the database, user items and internal tree nodes.
318    pub fn clear(&self, wtxn: &mut RwTxn) -> Result<()> {
319        let mut cursor = self
320            .database
321            .remap_key_type::<PrefixCodec>()
322            .prefix_iter_mut(wtxn, &Prefix::all(self.index))?
323            .remap_types::<DecodeIgnore, DecodeIgnore>();
324
325        while let Some((_id, _node)) = cursor.next().transpose()? {
326            // safety: we don't have any reference to the database
327            unsafe { cursor.del_current() }?;
328        }
329
330        Ok(())
331    }

358    fn build<R: Rng + SeedableRng>(
359        &self,
360        wtxn: &mut RwTxn,
361        rng: &mut R,
362        options: &BuildOption,
363    ) -> Result<()> {
364        log::debug!("started preprocessing the items...");
365
366        if (options.cancel)() {
367            return Err(Error::BuildCancelled);
368        }
369
370        D::preprocess(wtxn, |wtxn| {
371            Ok(self
372                .database
373                .remap_key_type::<PrefixCodec>()
374                .prefix_iter_mut(wtxn, &Prefix::item(self.index))?
375                .remap_key_type::<KeyCodec>())
376        })?;
377
378        if (options.cancel)() {
379            return Err(Error::BuildCancelled);
380        }
381
382        let item_indices = self.item_indices(wtxn)?;
383        let n_items = item_indices.len();
384
385        if self.fit_in_descendant(options, item_indices.len()) {
386            log::debug!("We can fit every elements in a single descendant node, we can skip all the build process");
387            // No item left in the index, we can clear every tree
388
389            self.database.remap_data_type::<Bytes>().delete_range(
390                wtxn,
391                &(Key::tree(self.index, 0)..=Key::tree(self.index, ItemId::MAX)),
392            )?;
393
394            let mut roots = Vec::new();
395
396            if !item_indices.is_empty() {
397                // if we have more than 0 elements we need to create a descendant node
398
399                self.database.put(
400                    wtxn,
401                    &Key::tree(self.index, 0),
402                    &Node::Descendants(Descendants { descendants: Cow::Borrowed(&item_indices) }),
403                )?;
404                roots.push(0);
405            }
406
407            log::debug!("reset the updated items...");
408            let mut updated_iter = self
409                .database
410                .remap_types::<PrefixCodec, DecodeIgnore>()
411                .prefix_iter_mut(wtxn, &Prefix::updated(self.index))?
412                .remap_key_type::<KeyCodec>();
413            while updated_iter.next().transpose()?.is_some() {
414                // Safe because we don't hold any reference to the database currently
415                unsafe {
416                    updated_iter.del_current()?;
417                }
418            }
419            drop(updated_iter);
420
421            log::debug!("write the metadata...");
422            let metadata = Metadata {
423                dimensions: self.dimensions.try_into().unwrap(),
424                items: item_indices,
425                roots: ItemIds::from_slice(&roots),
426                distance: D::name(),
427            };
428            self.database.remap_data_type::<MetadataCodec>().put(
429                wtxn,
430                &Key::metadata(self.index),
431                &metadata,
432            )?;
433
434            return Ok(());
435        }
436
437        log::debug!("reset and retrieve the updated items...");
438        let mut updated_items = RoaringBitmap::new();
439        let mut updated_iter = self
440            .database
441            .remap_types::<PrefixCodec, DecodeIgnore>()
442            .prefix_iter_mut(wtxn, &Prefix::updated(self.index))?
443            .remap_key_type::<KeyCodec>();
444        while let Some((key, _)) = updated_iter.next().transpose()? {
445            let inserted = updated_items.push(key.node.item);
446            debug_assert!(inserted, "The keys should be sorted by LMDB");
447            // Safe because we don't hold any reference to the database currently
448            unsafe {
449                updated_iter.del_current()?;
450            }
451        }
452        drop(updated_iter);
453
454        // while iterating on the nodes we want to delete all the modified element even if they are being inserted right after.
455        let to_delete = &updated_items;
456        let to_insert = &item_indices & &updated_items;
457
458        let metadata = self
459            .database
460            .remap_data_type::<MetadataCodec>()
461            .get(wtxn, &Key::metadata(self.index))?;
462        let mut roots =
463            metadata.as_ref().map_or_else(Vec::new, |metadata| metadata.roots.iter().collect());
464
465        log::debug!("Getting a reference to your {} items...", n_items);
466
467        if (options.cancel)() {
468            return Err(Error::BuildCancelled);
469        }
470
471        let used_node_ids = self.used_tree_node(wtxn)?;
472        let nb_tree_nodes = used_node_ids.len();
473
474        let concurrent_node_ids = ConcurrentNodeIds::new(used_node_ids);
475        let frozzen_reader = FrozzenReader {
476            leafs: &ImmutableLeafs::new(wtxn, self.database, self.index, item_indices.len())?,
477            trees: &ImmutableTrees::new(wtxn, self.database, self.index, nb_tree_nodes)?,
478            // The globally incrementing node ids that are shared between threads.
479            concurrent_node_ids: &concurrent_node_ids,
480        };
481
482        let mut nodes_to_write = Vec::new();
483
484        // If there is metadata it means that we already have trees and we must update them
485        if let Some(ref metadata) = metadata {
486            log::debug!(
487                "started inserting new items {} in {} trees...",
488                n_items,
489                metadata.roots.len()
490            );
491            let (new_roots, mut tmp_nodes_reader) =
492                self.update_trees(options, rng, metadata, &to_insert, to_delete, &frozzen_reader)?;
493            nodes_to_write.append(&mut tmp_nodes_reader);
494            roots = new_roots;
495        }
496
497        log::debug!("started building trees for {} items...", n_items);
498        log::debug!(
499            "running {} parallel tree building...",
500            options.n_trees.map_or_else(|| "an unknown number of".to_string(), |n| n.to_string())
501        );
502
503        // Once we updated the current trees we also need to create the new missing trees
504        // So we can run the normal path of building trees from scratch.
505        let n_trees_to_build = options
506            .n_trees
507            .zip(metadata)
508            .map(|(n_trees, metadata)| n_trees.saturating_sub(metadata.roots.len()))
509            .or(options.n_trees);
510        let (mut thread_roots, mut tmp_nodes) =
511            self.build_trees(options, rng, n_trees_to_build, &item_indices, &frozzen_reader)?;
512        nodes_to_write.append(&mut tmp_nodes);
513
514        log::debug!("started updating the tree nodes of {} trees...", tmp_nodes.len());
515        for (i, tmp_node) in nodes_to_write.iter().enumerate() {
516            log::debug!(
517                "started deleting the {} tree nodes of the {i}nth trees...",
518                tmp_node.len()
519            );
520            for item_id in tmp_node.to_delete() {
521                let key = Key::tree(self.index, item_id);
522                self.database.remap_data_type::<Bytes>().delete(wtxn, &key)?;
523            }
524            log::debug!(
525                "started inserting the {} tree nodes of the {i}nth trees...",
526                tmp_node.len()
527            );
528            for (item_id, item_bytes) in tmp_node.to_insert() {
529                let key = Key::tree(self.index, item_id);
530                self.database.remap_data_type::<Bytes>().put(wtxn, &key, item_bytes)?;
531            }
532        }
533
534        if thread_roots.is_empty() {
535            // we may have too many nodes
536            log::debug!("Deleting the extraneous trees if there is some...");
537            self.delete_extra_trees(
538                wtxn,
539                options,
540                &mut roots,
541                options.n_trees,
542                concurrent_node_ids.used(),
543                n_items,
544            )?;
545        } else {
546            roots.append(&mut thread_roots);
547        }
548
549        log::debug!("write the metadata...");
550        let metadata = Metadata {
551            dimensions: self.dimensions.try_into().unwrap(),
552            items: item_indices,
553            roots: ItemIds::from_slice(&roots),
554            distance: D::name(),
555        };
556        match self.database.remap_data_type::<MetadataCodec>().put(
557            wtxn,
558            &Key::metadata(self.index),
559            &metadata,
560        ) {
561            Ok(_) => Ok(()),
562            Err(e) => Err(e.into()),
563        }
564    }

566    fn update_trees<R: Rng + SeedableRng>(
567        &self,
568        opt: &BuildOption,
569        rng: &mut R,
570        metadata: &Metadata,
571        to_insert: &RoaringBitmap,
572        to_delete: &RoaringBitmap,
573        frozen_reader: &FrozzenReader<D>,
574    ) -> Result<(Vec<ItemId>, Vec<TmpNodesReader>)> {
575        let roots: Vec<_> = metadata.roots.iter().collect();
576
577        repeatn(rng.next_u64(), metadata.roots.len())
578            .zip(roots)
579            .map(|(seed, root)| {
580                log::debug!("started updating tree {root:X}...");
581                let mut rng = R::seed_from_u64(seed.wrapping_add(root as u64));
582                let mut tmp_nodes: TmpNodes<NodeCodec<D>> = match self.tmpdir.as_ref() {
583                    Some(path) => TmpNodes::new_in(path)?,
584                    None => TmpNodes::new()?,
585                };
586                let root_node = NodeId::tree(root);
587                let (node_id, _items) = self.update_nodes_in_file(
588                    opt,
589                    frozen_reader,
590                    &mut rng,
591                    root_node,
592                    to_insert,
593                    to_delete,
594                    &mut tmp_nodes,
595                )?;
596                assert!(node_id.mode != NodeMode::Item, "update_nodes_in_file returned an item even though there was more than a single element");
597
598                log::debug!("finished updating tree {root:X}");
599                Ok((node_id.unwrap_tree(), tmp_nodes.into_bytes_reader()?))
600            })
601            .collect()
602    }

604    /// Update the nodes that changed and delete the deleted nodes all at once.
605    /// Run in O(n) on the total number of nodes. Return a tuple containing the
606    /// node ID you should use instead of the current_node and the number of
607    /// items in the subtree.
608    #[allow(clippy::too_many_arguments)]
609    fn update_nodes_in_file<R: Rng>(
610        &self,
611        opt: &BuildOption,
612        frozen_reader: &FrozzenReader<D>,
613        rng: &mut R,
614        current_node: NodeId,
615        to_insert: &RoaringBitmap,
616        to_delete: &RoaringBitmap,
617        tmp_nodes: &mut TmpNodes<NodeCodec<D>>,
618    ) -> Result<(NodeId, RoaringBitmap)> {
619        if (opt.cancel)() {
620            return Err(Error::BuildCancelled);
621        }
622        match current_node.mode {
623            NodeMode::Item => {
624                // We were called on a specific item, we should create a descendants node
625                let mut new_items = RoaringBitmap::from_iter([current_node.item]);
626                new_items -= to_delete;
627                new_items |= to_insert;
628
629                if new_items.len() == 1 {
630                    let item_id = new_items.iter().next().unwrap();
631                    if item_id == current_node.item {
632                        Ok((current_node, new_items))
633                    } else {
634                        Ok((NodeId::item(item_id), new_items))
635                    }
636                } else if self.fit_in_descendant(opt, new_items.len()) {
637                    let node_id = frozen_reader.concurrent_node_ids.next()?;
638                    let node_id = NodeId::tree(node_id);
639                    tmp_nodes.put(
640                        node_id.item,
641                        &Node::Descendants(Descendants {
642                            descendants: Cow::Owned(new_items.clone()),
643                        }),
644                    )?;
645                    Ok((node_id, new_items))
646                } else {
647                    let new_id =
648                        self.make_tree_in_file(opt, frozen_reader, rng, &new_items, tmp_nodes)?;
649
650                    return Ok((new_id, new_items));
651                }
652            }
653            NodeMode::Tree => {
654                match frozen_reader.trees.get(current_node.item)?.unwrap() {
655                    Node::Leaf(_) => unreachable!(),
656                    Node::Descendants(Descendants { descendants }) => {
657                        let mut new_descendants = descendants.clone().into_owned();
658                        // remove all the deleted IDs before inserting the new elements.
659                        new_descendants -= to_delete;
660
661                        // insert all of our IDs in the descendants
662                        new_descendants |= to_insert;
663
664                        if descendants.as_ref() == &new_descendants {
665                            // if nothing changed, do nothing
666                            Ok((current_node, descendants.into_owned()))
667                        } else if !self.fit_in_descendant(opt, new_descendants.len()) {
668                            // if it doesn't fit in one descendent we need to craft a new whole subtree
669                            tmp_nodes.remove(current_node.item);
670                            let new_id = self.make_tree_in_file(
671                                opt,
672                                frozen_reader,
673                                rng,
674                                &new_descendants,
675                                tmp_nodes,
676                            )?;
677
678                            Ok((new_id, new_descendants))
679                        } else if new_descendants.len() == 1 {
680                            tmp_nodes.remove(current_node.item);
681                            let item = new_descendants.iter().next().unwrap();
682                            Ok((NodeId::item(item), new_descendants))
683                        } else {
684                            // otherwise we can just update our descendants
685                            tmp_nodes.put(
686                                current_node.item,
687                                &Node::Descendants(Descendants {
688                                    descendants: Cow::Owned(new_descendants.clone()),
689                                }),
690                            )?;
691                            Ok((current_node, new_descendants))
692                        }
693                    }
694                    Node::SplitPlaneNormal(SplitPlaneNormal { normal, left, right }) => {
695                        // Split the to_insert into two bitmaps on the left and right of this normal
696                        let mut left_ids = RoaringBitmap::new();
697                        let mut right_ids = RoaringBitmap::new();
698
699                        if normal.is_zero() {
700                            randomly_split_children(rng, to_insert, &mut left_ids, &mut right_ids);
701                        } else {
702                            for leaf in to_insert {
703                                let node = frozen_reader.leafs.get(leaf)?.unwrap();
704                                match D::side(&normal, &node, rng) {
705                                    Side::Left => left_ids.insert(leaf),
706                                    Side::Right => right_ids.insert(leaf),
707                                };
708                            }
709                        }
710
711                        let (new_left, left_items) = self.update_nodes_in_file(
712                            opt,
713                            frozen_reader,
714                            rng,
715                            left,
716                            &left_ids,
717                            to_delete,
718                            tmp_nodes,
719                        )?;
720                        let (new_right, right_items) = self.update_nodes_in_file(
721                            opt,
722                            frozen_reader,
723                            rng,
724                            right,
725                            &right_ids,
726                            to_delete,
727                            tmp_nodes,
728                        )?;
729
730                        let total_items = left_items | right_items;
731
732                        if self.fit_in_descendant(opt, total_items.len()) {
733                            // Since we're shrinking we KNOW that new_left and new_right are descendants
734                            // thus we can delete them directly knowing there is no sub-tree to look at.
735                            if new_left.mode == NodeMode::Tree {
736                                tmp_nodes.remove(new_left.item);
737                            }
738                            if new_right.mode == NodeMode::Tree {
739                                tmp_nodes.remove(new_right.item);
740                            }
741
742                            tmp_nodes.put(
743                                current_node.item,
744                                &Node::Descendants(Descendants {
745                                    descendants: Cow::Owned(total_items.clone()),
746                                }),
747                            )?;
748
749                            // we should merge both branch and update ourselves to be a single descendant node
750                            Ok((current_node, total_items))
751                        } else {
752                            // if either the left or the right changed we must update ourselves inplace
753                            if new_left != left || new_right != right {
754                                tmp_nodes.put(
755                                    current_node.item,
756                                    &Node::SplitPlaneNormal(SplitPlaneNormal {
757                                        normal,
758                                        left: new_left,
759                                        right: new_right,
760                                    }),
761                                )?;
762                            }
763
764                            // TODO: Should we update the normals if something changed?
765
766                            Ok((current_node, total_items))
767                        }
768                    }
769                }
770            }
771            NodeMode::Metadata => unreachable!(),
772            NodeMode::Updated => todo!(),
773        }
774    }

776    fn build_trees<R: Rng + SeedableRng>(
777        &self,
778        opt: &BuildOption,
779        rng: &mut R,
780        n_trees: Option<usize>,
781        item_indices: &RoaringBitmap,
782        frozen_reader: &FrozzenReader<D>,
783    ) -> Result<(Vec<ItemId>, Vec<TmpNodesReader>)> {
784        let n_items = item_indices.len();
785        let concurrent_node_ids = frozen_reader.concurrent_node_ids;
786
787        repeatn(rng.next_u64(), n_trees.unwrap_or(usize::MAX))
788            .enumerate()
789            // Stop generating trees once the specified number of tree nodes are generated
790            // but continue to generate trees if the number of trees is unspecified
791            .take_any_while(|_| match n_trees {
792                Some(_) => true,
793                None => concurrent_node_ids.used() < n_items,
794            })
795            .map(|(i, seed)| {
796                log::debug!("started generating tree {i:X}...");
797                let mut rng = R::seed_from_u64(seed.wrapping_add(i as u64));
798                let mut tmp_nodes = match self.tmpdir.as_ref() {
799                    Some(path) => TmpNodes::new_in(path)?,
800                    None => TmpNodes::new()?,
801                };
802                let root_id =
803                    self.make_tree_in_file(opt, frozen_reader, &mut rng, item_indices, &mut tmp_nodes)?;
804                assert!(
805                    root_id.mode != NodeMode::Item,
806                    "make_tree_in_file returned an item even though there was more than a single element"
807                );
808                log::debug!("finished generating tree {i:X}");
809                // make_tree will NEVER return a leaf when called as root
810                Ok((root_id.unwrap_tree(), tmp_nodes.into_bytes_reader()?))
811            })
812            .collect()
813    }

815    /// Creates a tree of nodes from the frozzen items that lives
816    /// in the database and generates descendants, split normal
817    /// and root nodes in files that will be stored in the database later.
818    fn make_tree_in_file<R: Rng>(
819        &self,
820        opt: &BuildOption,
821        reader: &FrozzenReader<D>,
822        rng: &mut R,
823        item_indices: &RoaringBitmap,
824        tmp_nodes: &mut TmpNodes<NodeCodec<D>>,
825    ) -> Result<NodeId> {
826        if (opt.cancel)() {
827            return Err(Error::BuildCancelled);
828        }
829        if item_indices.len() == 1 {
830            return Ok(NodeId::item(item_indices.min().unwrap()));
831        }
832
833        if self.fit_in_descendant(opt, item_indices.len()) {
834            let item_id = reader.concurrent_node_ids.next()?;
835            let item = Node::Descendants(Descendants { descendants: Cow::Borrowed(item_indices) });
836            tmp_nodes.put(item_id, &item)?;
837            return Ok(NodeId::tree(item_id));
838        }
839
840        let children = ImmutableSubsetLeafs::from_item_ids(reader.leafs, item_indices);
841        let mut children_left = Vec::with_capacity(children.len() as usize);
842        let mut children_right = Vec::with_capacity(children.len() as usize);
843        let mut remaining_attempts = 3;
844
845        let mut normal = loop {
846            children_left.clear();
847            children_right.clear();
848
849            let normal = D::create_split(&children, rng)?;
850            for item_id in item_indices.iter() {
851                let node = children.get(item_id)?.unwrap();
852                match D::side(&normal, &node, rng) {
853                    Side::Left => children_left.push(item_id),
854                    Side::Right => children_right.push(item_id),
855                };
856            }
857
858            if split_imbalance(children_left.len() as u64, children_right.len() as u64) < 0.95
859                || remaining_attempts == 0
860            {
861                break normal;
862            }
863
864            remaining_attempts -= 1;
865        };
866
867        // If we didn't find a hyperplane, just randomize sides as a last option
868        // and set the split plane to zero as a dummy plane.
869        let (children_left, children_right) =
870            if split_imbalance(children_left.len() as u64, children_right.len() as u64) > 0.99 {
871                let mut children_left = RoaringBitmap::new();
872                let mut children_right = RoaringBitmap::new();
873                randomly_split_children(rng, item_indices, &mut children_left, &mut children_right);
874                UnalignedVector::reset(&mut normal);
875
876                (children_left, children_right)
877            } else {
878                (
879                    RoaringBitmap::from_sorted_iter(children_left).unwrap(),
880                    RoaringBitmap::from_sorted_iter(children_right).unwrap(),
881                )
882            };
883
884        let normal = SplitPlaneNormal {
885            normal,
886            left: self.make_tree_in_file(opt, reader, rng, &children_left, tmp_nodes)?,
887            right: self.make_tree_in_file(opt, reader, rng, &children_right, tmp_nodes)?,
888        };
889
890        let new_node_id = reader.concurrent_node_ids.next()?;
891        tmp_nodes.put(new_node_id, &Node::SplitPlaneNormal(normal))?;
892
893        Ok(NodeId::tree(new_node_id))
894    }

896    /// Delete any extraneous trees.
897    fn delete_extra_trees(
898        &self,
899        wtxn: &mut RwTxn,
900        opt: &BuildOption,
901        roots: &mut Vec<ItemId>,
902        nb_trees: Option<usize>,
903        nb_tree_nodes: u64,
904        nb_items: u64,
905    ) -> Result<()> {
906        if roots.is_empty() {
907            return Ok(());
908        }
909        let nb_trees = match nb_trees {
910            Some(nb_trees) => nb_trees,
911            None => {
912                // 1. Estimate the number of nodes per tree; the division is safe because we ensured there was at least one root node above.
913                let nodes_per_tree = nb_tree_nodes / roots.len() as u64;
914                // 2. Estimate the number of tree we need to have AT LEAST as much tree-nodes than items
915                (nb_items / nodes_per_tree) as usize
916            }
917        };
918
919        if roots.len() > nb_trees {
920            // we have too many trees and must delete some of them
921            let to_delete = roots.len() - nb_trees;
922
923            // we want to delete the oldest tree first since they're probably
924            // the less precise one
925            let new_roots = roots.split_off(to_delete);
926            let to_delete = mem::replace(roots, new_roots);
927            log::debug!("Deleting {} trees", to_delete.len());
928
929            for tree in to_delete {
930                if (opt.cancel)() {
931                    return Err(Error::BuildCancelled);
932                }
933                self.delete_tree(wtxn, NodeId::tree(tree))?;
934            }
935        }
936
937        Ok(())
938    }

940    fn delete_tree(&self, wtxn: &mut RwTxn, node: NodeId) -> Result<()> {
941        let key = Key::new(self.index, node);
942        match self.database.get(wtxn, &key)?.ok_or(Error::missing_key(key))? {
943            // the leafs are shared between the trees, we MUST NOT delete them.
944            Node::Leaf(_) => Ok(()),
945            Node::Descendants(_) => {
946                self.database.delete(wtxn, &key).map(|_| ()).map_err(Error::from)
947            }
948            Node::SplitPlaneNormal(SplitPlaneNormal { normal: _, left, right }) => {
949                self.delete_tree(wtxn, left)?;
950                self.delete_tree(wtxn, right)?;
951                self.database.delete(wtxn, &key).map(|_| ()).map_err(Error::from)
952            }
953        }
954    }

956    // Fetches the item's ids, not the tree nodes ones.
957    fn item_indices(&self, wtxn: &mut RwTxn<'_>) -> heed::Result<RoaringBitmap> {
958        let mut indices = RoaringBitmap::new();
959        for result in self
960            .database
961            .remap_types::<PrefixCodec, DecodeIgnore>()
962            .prefix_iter(wtxn, &Prefix::item(self.index))?
963            .remap_key_type::<KeyCodec>()
964        {
965            let (i, _) = result?;
966            indices.push(i.node.unwrap_item());
967        }
968
969        Ok(indices)
970    }

973/// Represents the final version of the leafs and contains
974/// useful informations to synchronize the building threads.
975#[derive(Clone)]
976struct FrozzenReader<'a, D: Distance> {
977    leafs: &'a ImmutableLeafs<'a, D>,
978    trees: &'a ImmutableTrees<'a, D>,
979    concurrent_node_ids: &'a ConcurrentNodeIds,
980}

982/// Randomly and efficiently splits the items into the left and right children vectors.
983fn randomly_split_children<R: Rng>(
984    rng: &mut R,
985    item_indices: &RoaringBitmap,
986    children_left: &mut RoaringBitmap,
987    children_right: &mut RoaringBitmap,
988) {
989    children_left.clear();
990    children_right.clear();
991
992    // Split it in half and put the right half into the right children's vector
993    for item_id in item_indices {
994        match Side::random(rng) {
995            Side::Left => children_left.push(item_id),
996            Side::Right => children_right.push(item_id),
997        };
998    }
999}

1001/// Clears everything but the leafs nodes (items).
1002/// Starts from the last node and stops at the first leaf.
1003fn clear_tree_nodes<D: Distance>(
1004    wtxn: &mut RwTxn,
1005    database: Database<D>,
1006    index: u16,
1007) -> Result<()> {
1008    database.delete(wtxn, &Key::metadata(index))?;
1009    let mut cursor = database
1010        .remap_types::<PrefixCodec, DecodeIgnore>()
1011        .prefix_iter_mut(wtxn, &Prefix::tree(index))?
1012        .remap_key_type::<DecodeIgnore>();
1013    while let Some((_id, _node)) = cursor.next().transpose()? {
1014        // safety: we keep no reference into the database between operations
1015        unsafe { cursor.del_current()? };
1016    }
1017
1018    Ok(())
1019}

1021fn split_imbalance(left_indices_len: u64, right_indices_len: u64) -> f64 {
1022    let ls = left_indices_len as f64;
1023    let rs = right_indices_len as f64;
1024    let f = ls / (ls + rs + f64::EPSILON); // Avoid 0/0
1025    f.max(1.0 - f)
1026}
