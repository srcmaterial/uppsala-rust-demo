19pub fn euclidean_distance(u: &UnalignedVector<f32>, v: &UnalignedVector<f32>) -> f32 {
20    #[cfg(target_arch = "x86_64")]
21    {
22        if is_x86_feature_detected!("avx")
23            && is_x86_feature_detected!("fma")
24            && u.len() >= MIN_DIM_SIZE_AVX
25        {
26            return unsafe { euclid_similarity_avx(u, v) };
27        }
28    }
29
30    #[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
31    {
32        if is_x86_feature_detected!("sse") && u.len() >= MIN_DIM_SIZE_SIMD {
33            return unsafe { euclid_similarity_sse(u, v) };
34        }
35    }
36
37    #[cfg(all(target_arch = "aarch64", target_feature = "neon"))]
38    {
39        if std::arch::is_aarch64_feature_detected!("neon") && u.len() >= MIN_DIM_SIZE_SIMD {
40            return unsafe { euclid_similarity_neon(u, v) };
41        }
42    }
43
44    euclidean_distance_non_optimized(u, v)
45}

85/// For the binary quantized dot product:
86/// 1. We need to multiply two scalars, in our case the only allowed values are -1 and 1:
87/// ```text
88/// -1 * -1 =  1
89/// -1 *  1 = -1
90///  1 * -1 = -1
91///  1 *  1 =  1
92/// ```
93///
94/// This looks like a negative xor already, if we replace the -1 by the binary quantized 0, and the 1 stays 1s:
95/// ```text
96/// ! 0 ^ 0 = 1
97/// ! 0 ^ 1 = 0
98/// ! 1 ^ 0 = 0
99/// ! 1 ^ 1 = 1
100/// ```
101/// Is equivalent to `!(a ^ b)`.
102///
103/// 2. Then we need to do the sum of the results:
104///    2.1 First we must do the sum of the operation on the `Word`s
105///        /!\ We must be careful here because `1 - 0` actually translates to `1 - 1 = 0`.
106///        `word.count_ones() - word.count_zeroes()` should do it:
107/// ```text
108///  00 => -2
109///  01 => 0
110///  10 => 0
111///  11 => 2
112/// ```
113///  /!\ We must also take care to use signed integer to be able to go into negatives
114///
115/// 2.2 Finally we must sum the result of all the words
116///     - By taking care of not overflowing: The biggest vectors contains like 5000 dimensions, a i16 could be enough. A i32 should be perfect.
117///     - We can do the sum straight away without any more tricks
118///     - We can cast the result to an f32 as expected
119pub fn dot_product_binary_quantized(
120    u: &UnalignedVector<BinaryQuantized>,
121    v: &UnalignedVector<BinaryQuantized>,
122) -> f32 {
123    u.as_bytes()
124        .iter()
125        .zip(v.as_bytes())
126        .map(|(u, v)| {
127            let ret = !(u ^ v);
128            ret.count_ones() as i32 - ret.count_zeros() as i32
129        })
130        .sum::<i32>() as f32
131}
