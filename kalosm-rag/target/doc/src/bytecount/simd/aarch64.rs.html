11#[target_feature(enable = "neon")]
12unsafe fn u8x16_from_offset(slice: &[u8], offset: usize) -> uint8x16_t {
13    debug_assert!(
14        offset + 16 <= slice.len(),
15        "{} + 16 ≥ {}",
16        offset,
17        slice.len()
18    );
19    vld1q_u8(slice.as_ptr().add(offset) as *const _) // TODO: does this need to be aligned?
20}

38unsafe fn sum4(u1: uint8x16_t, u2: uint8x16_t, u3: uint8x16_t, u4: uint8x16_t) -> usize {
39    ((vaddlvq_u8(u1) + vaddlvq_u8(u2)) + (vaddlvq_u8(u3) + vaddlvq_u8(u4))) as usize

42#[target_feature(enable = "neon")]
43pub unsafe fn chunk_count(haystack: &[u8], needle: u8) -> usize {
44    assert!(haystack.len() >= 16);
45
46    let mut offset = 0;
47    let mut count = 0;
48
49    let needles = vdupq_n_u8(needle);
50
51    // 16320
52    while haystack.len() >= offset + 64 * 255 {
53        let (mut count1, mut count2, mut count3, mut count4) =
54            (vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0));
55        for _ in 0..255 {
56            let uint8x16x4_t(h1, h2, h3, h4) = u8x16_x4_from_offset(haystack, offset);
57            count1 = vsubq_u8(count1, vceqq_u8(h1, needles));
58            count2 = vsubq_u8(count2, vceqq_u8(h2, needles));
59            count3 = vsubq_u8(count3, vceqq_u8(h3, needles));
60            count4 = vsubq_u8(count4, vceqq_u8(h4, needles));
61            offset += 64;
62        }
63        count += sum4(count1, count2, count3, count4);
64    }
65
66    // 64
67    let (mut count1, mut count2, mut count3, mut count4) =
68        (vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0));
69    for _ in 0..(haystack.len() - offset) / 64 {
70        let uint8x16x4_t(h1, h2, h3, h4) = u8x16_x4_from_offset(haystack, offset);
71        count1 = vsubq_u8(count1, vceqq_u8(h1, needles));
72        count2 = vsubq_u8(count2, vceqq_u8(h2, needles));
73        count3 = vsubq_u8(count3, vceqq_u8(h3, needles));
74        count4 = vsubq_u8(count4, vceqq_u8(h4, needles));
75        offset += 64;
76    }
77    count += sum4(count1, count2, count3, count4);
78
79    let mut counts = vdupq_n_u8(0);
80    // 16
81    for i in 0..(haystack.len() - offset) / 16 {
82        counts = vsubq_u8(
83            counts,
84            vceqq_u8(u8x16_from_offset(haystack, offset + i * 16), needles),
85        );
86    }
87    if haystack.len() % 16 != 0 {
88        counts = vsubq_u8(
89            counts,
90            vandq_u8(
91                vceqq_u8(u8x16_from_offset(haystack, haystack.len() - 16), needles),
92                u8x16_from_offset(&MASK, haystack.len() % 16),
93            ),
94        );
95    }
96    count + sum(counts)
97}

99#[target_feature(enable = "neon")]
100unsafe fn is_following_utf8_byte(u8s: uint8x16_t) -> uint8x16_t {
101    vceqq_u8(
102        vandq_u8(u8s, vdupq_n_u8(0b1100_0000)),
103        vdupq_n_u8(0b1000_0000),
104    )
105}

107#[target_feature(enable = "neon")]
108pub unsafe fn chunk_num_chars(utf8_chars: &[u8]) -> usize {
109    assert!(utf8_chars.len() >= 16);
110
111    let mut offset = 0;
112    let mut count = 0;
113
114    // 4080
115    while utf8_chars.len() >= offset + 64 * 255 {
116        let (mut count1, mut count2, mut count3, mut count4) =
117            (vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0));
118
119        for _ in 0..255 {
120            let uint8x16x4_t(h1, h2, h3, h4) = u8x16_x4_from_offset(utf8_chars, offset);
121            count1 = vsubq_u8(count1,is_following_utf8_byte(h1));
122            count2 = vsubq_u8(count2,is_following_utf8_byte(h2));
123            count3 = vsubq_u8(count3,is_following_utf8_byte(h3));
124            count4 = vsubq_u8(count4,is_following_utf8_byte(h4));
125            offset += 64;
126        }
127        count += sum4(count1, count2, count3, count4);
128    }
129
130    // 4080
131    let (mut count1, mut count2, mut count3, mut count4) =
132        (vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0), vdupq_n_u8(0));
133        for _ in 0..(utf8_chars.len() - offset) / 64 {
134            let uint8x16x4_t(h1, h2, h3, h4) = u8x16_x4_from_offset(utf8_chars, offset);
135            count1 = vsubq_u8(count1, is_following_utf8_byte(h1));
136            count2 = vsubq_u8(count2, is_following_utf8_byte(h2));
137            count3 = vsubq_u8(count3, is_following_utf8_byte(h3));
138            count4 = vsubq_u8(count4, is_following_utf8_byte(h4));
139            offset += 64;
140        }
141        count += sum4(count1, count2, count3, count4);
142    // 16
143    let mut counts = vdupq_n_u8(0);
144    for i in 0..(utf8_chars.len() - offset) / 16 {
145        counts = vsubq_u8(
146            counts,
147            is_following_utf8_byte(u8x16_from_offset(utf8_chars, offset + i * 16)),
148        );
149    }
150    if utf8_chars.len() % 16 != 0 {
151        counts = vsubq_u8(
152            counts,
153            vandq_u8(
154                is_following_utf8_byte(u8x16_from_offset(utf8_chars, utf8_chars.len() - 16)),
155                u8x16_from_offset(&MASK, utf8_chars.len() % 16),
156            ),
157        );
158    }
159    count += sum(counts);
160
161    utf8_chars.len() - count
162}
