9/// Returns the current time in UTC as unix timestamp.
10///
11/// To format this timestamp, use the [`datetimeformat`](crate::filters::datetimeformat) filter.
12#[cfg(feature = "datetime")]
13#[cfg_attr(docsrs, doc(cfg(feature = "datetime")))]
14pub fn now() -> Value {
15    let now = time::OffsetDateTime::now_utc();
16    Value::from(((now.unix_timestamp_nanos() / 1000) as f64) / 1_000_000.0)
17}

19/// Returns a cycler.
20///
21/// Similar to `loop.cycle`, but can be used outside loops or across
22/// multiple loops. For example, render a list of folders and files in a
23/// list, alternating giving them "odd" and "even" classes.
24///
25/// ```jinja
26/// {% set row_class = cycler("odd", "even") %}
27/// <ul class="browser">
28/// {% for folder in folders %}
29///   <li class="folder {{ row_class.next() }}">{{ folder }}
30/// {% endfor %}
31/// {% for file in files %}
32///   <li class="file {{ row_class.next() }}">{{ file }}
33/// {% endfor %}
34/// </ul>
35/// ```
36pub fn cycler(items: Vec<Value>) -> Result<Value, Error> {
37    #[derive(Debug)]
38    pub struct Cycler {
39        items: Vec<Value>,
40        pos: AtomicUsize,
41    }
42
43    impl Object for Cycler {
44        fn repr(self: &Arc<Self>) -> ObjectRepr {
45            ObjectRepr::Plain
46        }
47
48        fn call_method(
49            self: &Arc<Self>,
50            _state: &State<'_, '_>,
51            method: &str,
52            args: &[Value],
53        ) -> Result<Value, Error> {
54            match method {
55                "next" => {
56                    let () = from_args(args)?;
57                    let idx = self.pos.load(Ordering::Relaxed);
58                    self.pos
59                        .store((idx + 1) % self.items.len(), Ordering::Relaxed);
60                    Ok(self.items[idx].clone())
61                }
62                _ => Err(Error::from(ErrorKind::UnknownMethod)),
63            }
64        }
65    }
66
67    if items.is_empty() {
68        Err(Error::new(
69            ErrorKind::InvalidOperation,
70            "at least one value required",
71        ))
72    } else {
73        Ok(Value::from_object(Cycler {
74            items,
75            pos: AtomicUsize::new(0),
76        }))
77    }
78}

80/// A tiny helper that can be used to “join” multiple sections.  A
81/// joiner is passed a string and will return that string every time
82/// it’s called, except the first time (in which case it returns an
83/// empty string). You can use this to join things:
84///
85/// ```jinja
86/// {% set pipe = joiner("|") %}
87/// {% if categories %} {{ pipe() }}
88/// Categories: {{ categories|join(", ") }}
89/// {% endif %}
90/// {% if author %} {{ pipe() }}
91/// Author: {{ author() }}
92/// {% endif %}
93/// {% if can_edit %} {{ pipe() }}
94/// <a href="?action=edit">Edit</a>
95/// {% endif %}
96/// ```
97pub fn joiner(sep: Option<Value>) -> Value {
98    #[derive(Debug)]
99    struct Joiner {
100        sep: Value,
101        used: AtomicBool,
102    }
103
104    impl Object for Joiner {
105        fn repr(self: &Arc<Self>) -> ObjectRepr {
106            ObjectRepr::Plain
107        }
108
109        fn call(self: &Arc<Self>, _state: &State<'_, '_>, args: &[Value]) -> Result<Value, Error> {
110            let () = from_args(args)?;
111            let used = self.used.swap(true, Ordering::Relaxed);
112            if used {
113                Ok(self.sep.clone())
114            } else {
115                Ok(Value::from(""))
116            }
117        }
118    }
119
120    Value::from_object(Joiner {
121        sep: sep.unwrap_or_else(|| Value::from(", ")),
122        used: AtomicBool::new(false),
123    })
124}

126/// Returns a random number in a given range.
127///
128/// If only one parameter is provided it's taken as exclusive upper
129/// bound with 0 as lower bound, otherwise two parameters need to be
130/// passed for the lower and upper bound.  Only integers are permitted.
131///
132/// The random number generated can be seeded with the `RAND_SEED`
133/// global context variable.
134#[cfg(feature = "rand")]
135#[cfg_attr(docsrs, doc(cfg(feature = "rand")))]
136pub fn randrange(state: &State, n: i64, m: Option<i64>) -> i64 {
137    let (lower, upper) = match m {
138        None => (0, n),
139        Some(m) => (n, m),
140    };
141
142    crate::rand::XorShiftRng::for_state(state).random_range(lower, upper)
143}
