9/// Detect only a language by a given text.
10///
11/// # Example
12/// ```
13/// use whatlang::{detect_lang, Lang};
14/// let lang = detect_lang("There is no reason not to learn Esperanto.").unwrap();
15/// assert_eq!(lang, Lang::Eng);
16/// ```
17pub fn detect_lang(text: &str) -> Option<Lang> {
18    detect(text).map(|output| output.lang())
19}
20
21/// Detect a language and a script by a given text.
22///
23/// # Example
24/// ```
25/// use whatlang::{detect_lang, Lang};
26/// let lang = detect_lang("There is no reason not to learn Esperanto.").unwrap();
27/// assert_eq!(lang, Lang::Eng);
28/// ```
29pub fn detect(text: &str) -> Option<Info> {
30    let opts = Options::default();
31    detect_with_options(text, &opts)
32}
33
34pub fn detect_with_options(text: &str, options: &Options) -> Option<Info> {
35    let query = Query {
36        text,
37        filter_list: &options.filter_list,
38        method: options.method,
39    };
40    detect_by_query(&query)
41}
42
43pub fn detect_by_query(query: &Query) -> Option<Info> {
44    let raw_script_info = raw_detect_script(query.text);
45    let script = raw_script_info.main_script()?;
46
47    match script.to_lang_group() {
48        ScriptLangGroup::One(lang) => Some(Info::new(script, lang, 1.0)),
49        ScriptLangGroup::Multi(multi_lang_script) => {
50            detect_by_query_based_on_script(query, multi_lang_script)
51        }
52        ScriptLangGroup::Mandarin => {
53            Some(detect_lang_base_on_mandarin_script(query, &raw_script_info))
54        }
55    }
56}
57
58fn detect_by_query_based_on_script(
59    query: &Query,
60    multi_lang_script: MultiLangScript,
61) -> Option<Info> {
62    let iquery = query.to_internal(multi_lang_script);
63    match query.method {
64        Method::Alphabet => alphabets::detect(&iquery),
65        Method::Trigram => trigrams::detect(&iquery),
66        Method::Combined => combined::detect(&iquery),
67    }
68}

70// Sometimes Mandarin can be Japanese.
71// See https://github.com/greyblake/whatlang-rs/pull/45
72pub(crate) fn detect_lang_base_on_mandarin_script(
73    query: &Query,
74    raw_script_info: &RawScriptInfo,
75) -> Info {
76    let (lang, confidence) = if query.filter_list.is_allowed(Lang::Cmn) {
77        let mandarin_count = raw_script_info.count(Script::Mandarin);
78        let katakana_count = raw_script_info.count(Script::Katakana);
79        let hiragana_count = raw_script_info.count(Script::Hiragana);
80        let japanese_count = katakana_count + hiragana_count;
81        let total = mandarin_count + japanese_count;
82
83        let jpn_pct = japanese_count as f64 / total as f64;
84
85        // If at least 5% of characters are Japanese(Katakana or Hiragana) then it's likely to be Japanese language
86        // See https://github.com/greyblake/whatlang-rs/issues/88
87        if jpn_pct > 0.2 {
88            (Lang::Jpn, 1.0)
89        } else if jpn_pct > 0.05 {
90            (Lang::Jpn, 0.5)
91        } else if jpn_pct > 0.02 {
92            (Lang::Cmn, 0.5)
93        } else {
94            (Lang::Cmn, 1.0)
95        }
96    } else {
97        (Lang::Jpn, 1.0)
98    };
99    Info::new(Script::Mandarin, lang, confidence)
100}
