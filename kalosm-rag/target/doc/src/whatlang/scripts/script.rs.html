126impl FromStr for Script {

129    fn from_str(s: &str) -> Result<Self, Self::Err> {
130        match s.to_lowercase().trim() {
131            "latin" => Ok(Script::Latin),
132            "cyrillic" => Ok(Script::Cyrillic),
133            "arabic" => Ok(Script::Arabic),
134            "devanagari" => Ok(Script::Devanagari),
135            "hiragana" => Ok(Script::Hiragana),
136            "katakana" => Ok(Script::Katakana),
137            "ethiopic" => Ok(Script::Ethiopic),
138            "hebrew" => Ok(Script::Hebrew),
139            "bengali" => Ok(Script::Bengali),
140            "georgian" => Ok(Script::Georgian),
141            "mandarin" => Ok(Script::Mandarin),
142            "hangul" => Ok(Script::Hangul),
143            "greek" => Ok(Script::Greek),
144            "kannada" => Ok(Script::Kannada),
145            "tamil" => Ok(Script::Tamil),
146            "thai" => Ok(Script::Thai),
147            "gujarati" => Ok(Script::Gujarati),
148            "gurmukhi" => Ok(Script::Gurmukhi),
149            "telugu" => Ok(Script::Telugu),
150            "malayalam" => Ok(Script::Malayalam),
151            "oriya" => Ok(Script::Oriya),
152            "myanmar" => Ok(Script::Myanmar),
153            "sinhala" => Ok(Script::Sinhala),
154            "khmer" => Ok(Script::Khmer),
155            "armenian" => Ok(Script::Armenian),
156            _ => Err(ParseError::Script(s.to_string())),
157        }
158    }

161#[cfg(test)]
162mod tests {
163    use super::*;
164
165    #[test]
166    fn test_all() {
167        assert_eq!(Script::all().len(), 25);
168        let all = Script::all();
169        assert!(all.contains(&Script::Cyrillic));
170        assert!(all.contains(&Script::Arabic));
171        assert!(all.contains(&Script::Latin));
172    }
173
174    #[test]
175    fn test_from_str() {
176        for &script in Script::all() {
177            let s = script.name();
178            assert_eq!(s.parse::<Script>().unwrap(), script);
179            assert_eq!(s.to_lowercase().parse::<Script>().unwrap(), script);
180            assert_eq!(s.to_uppercase().parse::<Script>().unwrap(), script);
181        }
182
183        let result = "foobar".parse::<Script>();
184        assert!(matches!(result, Err(ParseError::Script(_))));
185    }
186
187    #[test]
188    fn test_langs() {
189        // Vec of all langs obtained with script.langs()
190        let script_langs: Vec<Lang> = Script::all()
191            .iter()
192            .map(|script| script.langs())
193            .flatten()
194            .copied()
195            .collect();
196
197        // Ensure all langs belong at least to one script
198        for lang in Lang::all() {
199            assert!(script_langs.contains(&lang));
200        }
201    }
202
203    #[test]
204    fn test_script_display() {
205        assert_eq!(Script::Georgian.to_string(), "Georgian");
206        assert_eq!(Script::Cyrillic.to_string(), "Cyrillic");
207        assert_eq!(Script::Arabic.to_string(), "Arabic");
208    }
209
210    #[cfg(feature = "serde")]
211    #[test]
212    fn test_serialize_and_deserialize() {
213        let scripts = vec![Script::Georgian, Script::Cyrillic];
214        let json_scripts = serde_json::to_string(&scripts).unwrap();
215        assert_eq!(json_scripts, r#"["Georgian","Cyrillic"]"#);
216        let parsed_scripts: Vec<Script> = serde_json::from_str(&json_scripts).unwrap();
217        assert_eq!(parsed_scripts, scripts);
218    }
