13#[inline]
14pub fn get_trigrams_with_positions(text: &LowercaseText) -> TrigramsWithPositions {
15    let CountResult {
16        total_trigrams,
17        trigram_occurances,
18    } = count(text);
19    let trigram_positions = trigram_occurances_to_positions(trigram_occurances);
20    TrigramsWithPositions {
21        _total_trigrams: total_trigrams,
22        trigram_positions,
23    }
24}

26#[inline]
27#[allow(clippy::unnecessary_sort_by)]
28fn trigram_occurances_to_positions(
29    trigram_occurances: HashMap<Trigram, u32>,
30) -> HashMap<Trigram, u32> {
31    // Sort in descending order by number of occurrences and trigrams
32    let mut count_vec: Vec<_> = trigram_occurances
33        .into_iter()
34        .map(|(trigram, count)| (count, trigram))
35        .collect();
36    count_vec.sort_unstable_by(|a, b| b.cmp(a));
37
38    count_vec
39        .into_iter()
40        .take(TEXT_TRIGRAMS_SIZE) // we're interested only in the first 600 (2 * MAX_TRIGRAM_DISTANCE)
41        .enumerate()
42        .map(|(i, (_, trigram))| (trigram, i as u32))
43        .collect()
44}

46struct CountResult {
47    total_trigrams: u32,
48    trigram_occurances: HashMap<Trigram, u32>,
49}
50
51#[inline]
52fn count(text: &LowercaseText) -> CountResult {
53    let hash_capacity = calculate_initial_hash_capacity(text);
54    let mut trigram_occurances: HashMap<Trigram, u32> = HashMap::with_capacity(hash_capacity);
55    let mut total_trigrams = 0;
56
57    // iterate through the string and count trigrams
58    let mut chars_iter = text
59        .chars()
60        .map(to_trigram_char)
61        //.flat_map(char::to_lowercase)
62        .chain(Some(' '));
63    let mut c1 = ' ';
64    // unwrap is safe, because we always chain a space character on the end of the iterator
65    let mut c2 = chars_iter.next().unwrap();
66    for cur_char in chars_iter {
67        let c3 = cur_char;
68        if !(c2 == ' ' && (c1 == ' ' || c3 == ' ')) {
69            let trigram = Trigram(c1, c2, c3);
70            let count = trigram_occurances.entry(trigram).or_insert(0);
71            *count += 1;
72            total_trigrams += 1;
73        }
74        c1 = c2;
75        c2 = c3;
76    }
77
78    CountResult {
79        total_trigrams,
80        trigram_occurances,
81    }
82}

84// Convert punctuations and digits to a space.
85#[inline]
86fn to_trigram_char(ch: char) -> char {
87    if is_stop_char(ch) {
88        ' '
89    } else {
90        ch
91    }
92}

94// In order to improve performance, define the initial capacity for trigrams hash map,
95// based on the size of the input text.
96#[inline]
97fn calculate_initial_hash_capacity(text: &str) -> usize {
98    const MAX_INITIAL_HASH_CAPACITY: usize = 2048;
99
100    let len = text.len();
101    if len > MAX_INITIAL_HASH_CAPACITY {
102        MAX_INITIAL_HASH_CAPACITY
103    } else {
104        len
105    }
106}

126        // valuable chars, that matters

129        // punctuations, digits, etc..
130        //
131        // 0x00 - 0x40

133        // 0x5B - 0x60
134        assert_not_valuable_trigram_chars(&['[', ']', '^', '\\', '`']);
135        // 0x7B - 0x7E
136        assert_not_valuable_trigram_chars(&['[', '|', '{', '}', '~']);

139    fn assert_count(text: &str, pairs: &[(&str, u32)]) {
140        let lowercase_text = LowercaseText::new(text);
141        let CountResult {
142            total_trigrams: _,
143            trigram_occurances,
144        } = count(&lowercase_text);
145        for &(trigram_str, expected_n) in pairs.iter() {
146            let chars: Vec<char> = trigram_str.chars().collect();
147            let trigram = Trigram(chars[0], chars[1], chars[2]);
148            let actual_n = trigram_occurances[&trigram];
149            assert_eq!(
150                actual_n, expected_n,
151                "trigram '{:?}' expected to occur {} times, got {}",
152                trigram, expected_n, actual_n
153            );
154        }
155        assert_eq!(trigram_occurances.len(), pairs.len());
156    }
