99    // Old naive implementation, that is not very effective but easy to understand
100    fn naive_alphabet_calculate_scores(
101        text: &LowercaseText,
102        filter_list: &FilterList,
103    ) -> RawOutcome {
104        let mut raw_scores: Vec<(Lang, i32)> = Script::Latin
105            .langs()
106            .iter()
107            .filter(|&&l| filter_list.is_allowed(l))
108            .map(|&l| (l, 0i32))
109            .collect();
110
111        let max_raw_score = text.chars().filter(|&ch| !is_stop_char(ch)).count();
112
113        for (lang, score) in &mut raw_scores {
114            let alphabet: Vec<char> = LATIN_ALPHABETS
115                .iter()
116                .find(|(l, _)| l == lang)
117                .unwrap()
118                .1
119                .chars()
120                .collect();
121
122            for ch in text.chars() {
123                if is_stop_char(ch) {
124                    continue;
125                };
126                if alphabet.contains(&ch) {
127                    *score += 1;
128                } else {
129                    *score -= 1;
130                }
131            }
132        }
133
134        raw_scores.sort_by(|a, b| b.1.cmp(&a.1));
135
136        let raw_scores: Vec<(Lang, usize)> = raw_scores
137            .into_iter()
138            .map(|(l, s)| {
139                let score = if s < 0 { 0usize } else { s as usize };
140                (l, score)
141            })
142            .collect();
143
144        let mut normalized_scores = vec![];
145
146        for &(lang, raw_score) in &raw_scores {
147            let normalized_score = raw_score as f64 / max_raw_score as f64;
148            normalized_scores.push((lang, normalized_score));
149        }
150
151        RawOutcome {
152            count: max_raw_score,
153            raw_scores,
154            scores: normalized_scores,
155        }
156    }

158    #[test]
159    fn test_alphabet_calculate_scores_against_harmaja_hauras() {
160        let text =
161            LowercaseText::new("Ja kulkee kylmä hetki pariimme, Olet hauras kuin jää, Ja kulke");
162        let filter = FilterList::All;
163
164        let outcome = alphabet_calculate_scores(&text, &filter);
165        assert_eq!(outcome.count, 50);
166        assert_eq!(outcome.raw_scores.len(), 36);
167        assert_eq!(outcome.scores.len(), 36);
168
169        let raw_scores_for = |lang: Lang| {
170            outcome
171                .raw_scores
172                .iter()
173                .find(|(l, _)| *l == lang)
174                .unwrap()
175                .1
176        };
177        let scores_for = |lang: Lang| outcome.scores.iter().find(|(l, _)| *l == lang).unwrap().1;
178
179        assert_eq!(raw_scores_for(Lang::Fin), 50);
180        assert_eq!(raw_scores_for(Lang::Deu), 50);
181        assert_eq!(raw_scores_for(Lang::Epo), 42);
182
183        assert_eq!(scores_for(Lang::Fin), 1.0);
184        assert_eq!(scores_for(Lang::Deu), 1.0);
185        assert_eq!(scores_for(Lang::Epo), 0.84);
186    }

188    #[test]
189    fn test_works_as_the_old_naive_implementation() {
190        let filter = FilterList::All;
191        let texts = [
192            "Tyrkiske språk eller turkotatariske språk er en språkgruppe bestående av minst 35 språk",
193            "René Descartes est un mathématicien, physicien et philosophe français, né le 31 mars 1596 à La Haye-en-Touraine",
194            "Die Sonne scheint in das Büro der Grabdenkmalsfirma Heinrich Kroll & Söhne. Es ist April 923, und das Geschäf geht gut.",
195        ];
196
197        for text in texts {
198            let lowercase_text = LowercaseText::new(text);
199            let outcome = alphabet_calculate_scores(&lowercase_text, &filter);
200            let naive_outcome = naive_alphabet_calculate_scores(&lowercase_text, &filter);
201
202            // We can just compare outcome against naive_outcome, because ordering maybe different,
203            // what is acceptable.
204            assert_eq!(
205                outcome.count, naive_outcome.count,
206                "count failed. Text: {}",
207                text
208            );
209            for (lang, raw_naive_score) in naive_outcome.raw_scores.into_iter() {
210                let lookup_raw_score = |lang| {
211                    outcome
212                        .raw_scores
213                        .iter()
214                        .find(|(l, _)| *l == lang)
215                        .unwrap()
216                        .1
217                };
218                let raw_score = lookup_raw_score(lang);
219                assert_eq!(
220                    raw_score, raw_naive_score,
221                    "raw_score VS raw_naive_score failed. Lang={}, Text: {}",
222                    lang, text
223                );
224            }
225            for (lang, naive_score) in naive_outcome.scores.into_iter() {
226                let lookup_score =
227                    |lang| outcome.scores.iter().find(|(l, _)| *l == lang).unwrap().1;
228                let score = lookup_score(lang);
229                assert_eq!(
230                    score, naive_score,
231                    "score VS naive_score failed. Lang={}, Text: {}",
232                    lang, text
233                );
234            }
235        }
236    }
