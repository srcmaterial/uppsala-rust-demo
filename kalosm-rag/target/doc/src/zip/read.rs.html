101impl<'a> Read for CryptoReader<'a> {
102    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
103        match self {
104            CryptoReader::Plaintext(r) => r.read(buf),
105            CryptoReader::ZipCrypto(r) => r.read(buf),
106            #[cfg(feature = "aes-crypto")]
107            CryptoReader::Aes { reader: r, .. } => r.read(buf),
108        }
109    }

112impl<'a> CryptoReader<'a> {
113    /// Consumes this decoder, returning the underlying reader.
114    pub fn into_inner(self) -> io::Take<&'a mut dyn Read> {
115        match self {
116            CryptoReader::Plaintext(r) => r,
117            CryptoReader::ZipCrypto(r) => r.into_inner(),
118            #[cfg(feature = "aes-crypto")]
119            CryptoReader::Aes { reader: r, .. } => r.into_inner(),
120        }
121    }
122
123    /// Returns `true` if the data is encrypted using AE2.
124    pub const fn is_ae2_encrypted(&self) -> bool {
125        #[cfg(feature = "aes-crypto")]
126        return matches!(
127            self,
128            CryptoReader::Aes {
129                vendor_version: AesVendorVersion::Ae2,
130                ..
131            }
132        );
133        #[cfg(not(feature = "aes-crypto"))]
134        false
135    }

174impl<'a> ZipFileReader<'a> {
175    /// Consumes this decoder, returning the underlying reader.
176    pub fn into_inner(self) -> io::Take<&'a mut dyn Read> {
177        match self {
178            ZipFileReader::NoReader => panic!("ZipFileReader was in an invalid state"),
179            ZipFileReader::Raw(r) => r,
180            ZipFileReader::Stored(r) => r.into_inner().into_inner(),
181            #[cfg(feature = "_deflate-any")]
182            ZipFileReader::Deflated(r) => r.into_inner().into_inner().into_inner(),
183            #[cfg(feature = "deflate64")]
184            ZipFileReader::Deflate64(r) => r.into_inner().into_inner().into_inner().into_inner(),
185            #[cfg(feature = "bzip2")]
186            ZipFileReader::Bzip2(r) => r.into_inner().into_inner().into_inner(),
187            #[cfg(feature = "zstd")]
188            ZipFileReader::Zstd(r) => r.into_inner().finish().into_inner().into_inner(),
189            #[cfg(feature = "lzma")]
190            ZipFileReader::Lzma(r) => {
191                let inner: Box<_> = r.into_inner().finish().unwrap().into();
192                Read::take(Box::leak(inner), u64::MAX)
193            }
194        }
195    }

205pub(crate) fn find_content<'a>(
206    data: &ZipFileData,
207    reader: &'a mut (impl Read + Seek),
208) -> ZipResult<io::Take<&'a mut dyn Read>> {
209    // Parse local header
210    reader.seek(io::SeekFrom::Start(data.header_start))?;
211    let signature = reader.read_u32_le()?;
212    if signature != spec::LOCAL_FILE_HEADER_SIGNATURE {
213        return Err(ZipError::InvalidArchive("Invalid local file header"));
214    }
215    let data_start = match data.data_start.get() {
216        None => {
217            reader.seek(io::SeekFrom::Current(22))?;
218            let file_name_length = reader.read_u16_le()? as u64;
219            let extra_field_length = reader.read_u16_le()? as u64;
220            let magic_and_header = 4 + 22 + 2 + 2;
221            let data_start =
222                data.header_start + magic_and_header + file_name_length + extra_field_length;
223            data.data_start.get_or_init(|| data_start);
224            data_start
225        }
226        Some(start) => *start,
227    };
228
229    reader.seek(io::SeekFrom::Start(data_start))?;
230    Ok((reader as &mut dyn Read).take(data.compressed_size))
231}

233#[allow(clippy::too_many_arguments)]
234pub(crate) fn make_crypto_reader<'a>(
235    compression_method: CompressionMethod,
236    crc32: u32,
237    last_modified_time: DateTime,
238    using_data_descriptor: bool,
239    reader: io::Take<&'a mut dyn Read>,
240    password: Option<&[u8]>,
241    aes_info: Option<(AesMode, AesVendorVersion)>,
242    #[cfg(feature = "aes-crypto")] compressed_size: u64,
243) -> ZipResult<CryptoReader<'a>> {
244    #[allow(deprecated)]
245    {
246        if let CompressionMethod::Unsupported(_) = compression_method {
247            return unsupported_zip_error("Compression method not supported");
248        }
249    }
250
251    let reader = match (password, aes_info) {
252        #[cfg(not(feature = "aes-crypto"))]
253        (Some(_), Some(_)) => {
254            return Err(ZipError::UnsupportedArchive(
255                "AES encrypted files cannot be decrypted without the aes-crypto feature.",
256            ))
257        }
258        #[cfg(feature = "aes-crypto")]
259        (Some(password), Some((aes_mode, vendor_version))) => {
260            match AesReader::new(reader, aes_mode, compressed_size).validate(password)? {
261                None => return Err(InvalidPassword),
262                Some(r) => CryptoReader::Aes {
263                    reader: r,
264                    vendor_version,
265                },
266            }
267        }
268        (Some(password), None) => {
269            let validator = if using_data_descriptor {
270                ZipCryptoValidator::InfoZipMsdosTime(last_modified_time.timepart())
271            } else {
272                ZipCryptoValidator::PkzipCrc32(crc32)
273            };
274            match ZipCryptoReader::new(reader, password).validate(validator)? {
275                None => return Err(InvalidPassword),
276                Some(r) => CryptoReader::ZipCrypto(r),
277            }
278        }
279        (None, Some(_)) => return Err(InvalidPassword),
280        (None, None) => CryptoReader::Plaintext(reader),
281    };
282    Ok(reader)
283}

285pub(crate) fn make_reader(
286    compression_method: CompressionMethod,
287    crc32: u32,
288    reader: CryptoReader,
289) -> ZipFileReader {
290    let ae2_encrypted = reader.is_ae2_encrypted();
291
292    match compression_method {
293        CompressionMethod::Stored => {
294            ZipFileReader::Stored(Crc32Reader::new(reader, crc32, ae2_encrypted))
295        }
296        #[cfg(feature = "_deflate-any")]
297        CompressionMethod::Deflated => {
298            let deflate_reader = DeflateDecoder::new(reader);
299            ZipFileReader::Deflated(Crc32Reader::new(deflate_reader, crc32, ae2_encrypted))
300        }
301        #[cfg(feature = "deflate64")]
302        CompressionMethod::Deflate64 => {
303            let deflate64_reader = Deflate64Decoder::new(reader);
304            ZipFileReader::Deflate64(Crc32Reader::new(deflate64_reader, crc32, ae2_encrypted))
305        }
306        #[cfg(feature = "bzip2")]
307        CompressionMethod::Bzip2 => {
308            let bzip2_reader = BzDecoder::new(reader);
309            ZipFileReader::Bzip2(Crc32Reader::new(bzip2_reader, crc32, ae2_encrypted))
310        }
311        #[cfg(feature = "zstd")]
312        CompressionMethod::Zstd => {
313            let zstd_reader = ZstdDecoder::new(reader).unwrap();
314            ZipFileReader::Zstd(Crc32Reader::new(zstd_reader, crc32, ae2_encrypted))
315        }
316        #[cfg(feature = "lzma")]
317        CompressionMethod::Lzma => {
318            let reader = LzmaDecoder::new(reader);
319            ZipFileReader::Lzma(Crc32Reader::new(Box::new(reader), crc32, ae2_encrypted))
320        }
321        _ => panic!("Compression method not supported"),
322    }
323}

333impl<R> ZipArchive<R> {
334    pub(crate) fn from_finalized_writer(
335        files: IndexMap<Box<str>, ZipFileData>,
336        comment: Vec<u8>,
337        reader: R,
338        central_start: u64,
339    ) -> ZipResult<Self> {
340        if files.is_empty() {
341            return Err(ZipError::InvalidArchive(
342                "attempt to finalize empty zip writer into readable",
343            ));
344        }
345        /* This is where the whole file starts. */
346        let (_, first_header) = files.first().unwrap();
347        let initial_offset = first_header.header_start;
348        let shared = Arc::new(zip_archive::Shared {
349            files,
350            offset: initial_offset,
351            dir_start: central_start,
352        });
353        Ok(Self {
354            reader,
355            shared,
356            comment: comment.into_boxed_slice().into(),
357        })
358    }
359}

361impl<R: Read + Seek> ZipArchive<R> {
362    pub(crate) fn merge_contents<W: Write + io::Seek>(
363        &mut self,
364        mut w: W,
365    ) -> ZipResult<IndexMap<Box<str>, ZipFileData>> {
366        let mut new_files = self.shared.files.clone();
367        if new_files.is_empty() {
368            return Ok(IndexMap::new());
369        }
370        /* The first file header will probably start at the beginning of the file, but zip doesn't
371         * enforce that, and executable zips like PEX files will have a shebang line so will
372         * definitely be greater than 0.
373         *
374         * assert_eq!(0, new_files[0].header_start); // Avoid this.
375         */
376
377        let new_initial_header_start = w.stream_position()?;
378        /* Push back file header starts for all entries in the covered files. */
379        new_files.values_mut().try_for_each(|f| {
380            /* This is probably the only really important thing to change. */
381            f.header_start = f.header_start.checked_add(new_initial_header_start).ok_or(
382                ZipError::InvalidArchive("new header start from merge would have been too large"),
383            )?;
384            /* This is only ever used internally to cache metadata lookups (it's not part of the
385             * zip spec), and 0 is the sentinel value. */
386            f.central_header_start = 0;
387            /* This is an atomic variable so it can be updated from another thread in the
388             * implementation (which is good!). */
389            if let Some(old_data_start) = f.data_start.take() {
390                let new_data_start = old_data_start.checked_add(new_initial_header_start).ok_or(
391                    ZipError::InvalidArchive("new data start from merge would have been too large"),
392                )?;
393                f.data_start.get_or_init(|| new_data_start);
394            }
395            Ok::<_, ZipError>(())
396        })?;
397
398        /* Rewind to the beginning of the file.
399         *
400         * NB: we *could* decide to start copying from new_files[0].header_start instead, which
401         * would avoid copying over e.g. any pex shebangs or other file contents that start before
402         * the first zip file entry. However, zip files actually shouldn't care about garbage data
403         * in *between* real entries, since the central directory header records the correct start
404         * location of each, and keeping track of that math is more complicated logic that will only
405         * rarely be used, since most zips that get merged together are likely to be produced
406         * specifically for that purpose (and therefore are unlikely to have a shebang or other
407         * preface). Finally, this preserves any data that might actually be useful.
408         */
409        self.reader.rewind()?;
410        /* Find the end of the file data. */
411        let length_to_read = self.shared.dir_start;
412        /* Produce a Read that reads bytes up until the start of the central directory header.
413         * This "as &mut dyn Read" trick is used elsewhere to avoid having to clone the underlying
414         * handle, which it really shouldn't need to anyway. */
415        let mut limited_raw = (&mut self.reader as &mut dyn Read).take(length_to_read);
416        /* Copy over file data from source archive directly. */
417        io::copy(&mut limited_raw, &mut w)?;
418
419        /* Return the files we've just written to the data stream. */
420        Ok(new_files)
421    }

423    fn get_directory_info_zip32(
424        footer: &spec::CentralDirectoryEnd,
425        cde_start_pos: u64,
426    ) -> ZipResult<CentralDirectoryInfo> {
427        // Some zip files have data prepended to them, resulting in the
428        // offsets all being too small. Get the amount of error by comparing
429        // the actual file position we found the CDE at with the offset
430        // recorded in the CDE.
431        let archive_offset = cde_start_pos
432            .checked_sub(footer.central_directory_size as u64)
433            .and_then(|x| x.checked_sub(footer.central_directory_offset as u64))
434            .ok_or(ZipError::InvalidArchive(
435                "Invalid central directory size or offset",
436            ))?;
437
438        let directory_start = footer.central_directory_offset as u64 + archive_offset;
439        let number_of_files = footer.number_of_files_on_this_disk as usize;
440        Ok(CentralDirectoryInfo {
441            archive_offset,
442            directory_start,
443            number_of_files,
444            disk_number: footer.disk_number as u32,
445            disk_with_central_directory: footer.disk_with_central_directory as u32,
446        })
447    }

449    fn get_directory_info_zip64(
450        reader: &mut R,
451        footer: &spec::CentralDirectoryEnd,
452        cde_start_pos: u64,
453    ) -> ZipResult<Vec<ZipResult<CentralDirectoryInfo>>> {
454        // See if there's a ZIP64 footer. The ZIP64 locator if present will
455        // have its signature 20 bytes in front of the standard footer. The
456        // standard footer, in turn, is 22+N bytes large, where N is the
457        // comment length. Therefore:
458        reader.seek(io::SeekFrom::End(
459            -(20 + 22 + footer.zip_file_comment.len() as i64),
460        ))?;
461        let locator64 = spec::Zip64CentralDirectoryEndLocator::parse(reader)?;
462
463        // We need to reassess `archive_offset`. We know where the ZIP64
464        // central-directory-end structure *should* be, but unfortunately we
465        // don't know how to precisely relate that location to our current
466        // actual offset in the file, since there may be junk at its
467        // beginning. Therefore we need to perform another search, as in
468        // read::CentralDirectoryEnd::find_and_parse, except now we search
469        // forward. There may be multiple results because of Zip64 central-directory signatures in
470        // ZIP comment data.
471
472        let mut results = Vec::new();
473
474        let search_upper_bound = cde_start_pos
475            .checked_sub(60) // minimum size of Zip64CentralDirectoryEnd + Zip64CentralDirectoryEndLocator
476            .ok_or(ZipError::InvalidArchive(
477                "File cannot contain ZIP64 central directory end",
478            ))?;
479        let search_results = spec::Zip64CentralDirectoryEnd::find_and_parse(
480            reader,
481            locator64.end_of_central_directory_offset,
482            search_upper_bound,
483        )?;
484        search_results.into_iter().for_each(|(footer64, archive_offset)| {
485            results.push({
486                let directory_start_result = footer64
487                    .central_directory_offset
488                    .checked_add(archive_offset)
489                    .ok_or(ZipError::InvalidArchive(
490                        "Invalid central directory size or offset",
491                    ));
492                directory_start_result.and_then(|directory_start| {
493                    if directory_start > search_upper_bound {
494                        Err(ZipError::InvalidArchive(
495                            "Invalid central directory size or offset",
496                        ))
497                    } else if footer64.number_of_files_on_this_disk > footer64.number_of_files {
498                        Err(ZipError::InvalidArchive(
499                            "ZIP64 footer indicates more files on this disk than in the whole archive",
500                        ))
501                    } else if footer64.version_needed_to_extract > footer64.version_made_by {
502                        Err(ZipError::InvalidArchive(
503                            "ZIP64 footer indicates a new version is needed to extract this archive than the \
504    version that wrote it",
505                        ))
506                    } else {
507                        Ok(CentralDirectoryInfo {
508                            archive_offset,
509                            directory_start,
510                            number_of_files: footer64.number_of_files as usize,
511                            disk_number: footer64.disk_number,
512                            disk_with_central_directory: footer64.disk_with_central_directory,
513                        })
514                    }
515                })
516            });
517        });
518        Ok(results)
519    }

521    /// Get the directory start offset and number of files. This is done in a
522    /// separate function to ease the control flow design.
523    pub(crate) fn get_metadata(
524        reader: &mut R,
525        footer: &spec::CentralDirectoryEnd,
526        cde_start_pos: u64,
527    ) -> ZipResult<Shared> {
528        // Check if file has a zip64 footer
529        let mut results = Self::get_directory_info_zip64(reader, footer, cde_start_pos)
530            .unwrap_or_else(|e| vec![Err(e)]);
531        let zip32_result = Self::get_directory_info_zip32(footer, cde_start_pos);
532        let mut invalid_errors = Vec::new();
533        let mut unsupported_errors = Vec::new();
534        let mut ok_results = Vec::new();
535        results.iter_mut().for_each(|result| {
536            if let Ok(central_dir) = result {
537                if let Ok(zip32_central_dir) = &zip32_result {
538                    // Both zip32 and zip64 footers exist, so check if the zip64 footer is valid; if not, try zip32
539                    if central_dir.number_of_files != zip32_central_dir.number_of_files
540                        && zip32_central_dir.number_of_files != u16::MAX as usize
541                    {
542                        *result = Err(ZipError::InvalidArchive(
543                            "ZIP32 and ZIP64 file counts don't match",
544                        ));
545                        return;
546                    }
547                    if central_dir.disk_number != zip32_central_dir.disk_number
548                        && zip32_central_dir.disk_number != u16::MAX as u32
549                    {
550                        *result = Err(ZipError::InvalidArchive(
551                            "ZIP32 and ZIP64 disk numbers don't match",
552                        ));
553                        return;
554                    }
555                    if central_dir.disk_with_central_directory
556                        != zip32_central_dir.disk_with_central_directory
557                        && zip32_central_dir.disk_with_central_directory != u16::MAX as u32
558                    {
559                        *result = Err(ZipError::InvalidArchive(
560                            "ZIP32 and ZIP64 last-disk numbers don't match",
561                        ));
562                    }
563                }
564            }
565        });
566        results.push(zip32_result);
567        results
568            .into_iter()
569            .map(|result| {
570                result.and_then(|dir_info| {
571                    // If the parsed number of files is greater than the offset then
572                    // something fishy is going on and we shouldn't trust number_of_files.
573                    let file_capacity =
574                        if dir_info.number_of_files > dir_info.directory_start as usize {
575                            0
576                        } else {
577                            dir_info.number_of_files
578                        };
579                    let mut files = IndexMap::with_capacity(file_capacity);
580                    reader.seek(io::SeekFrom::Start(dir_info.directory_start))?;
581                    for _ in 0..dir_info.number_of_files {
582                        let file = central_header_to_zip_file(reader, dir_info.archive_offset)?;
583                        files.insert(file.file_name.clone(), file);
584                    }
585                    if dir_info.disk_number != dir_info.disk_with_central_directory {
586                        unsupported_zip_error("Support for multi-disk files is not implemented")
587                    } else {
588                        Ok(Shared {
589                            files,
590                            offset: dir_info.archive_offset,
591                            dir_start: dir_info.directory_start,
592                        })
593                    }
594                })
595            })
596            .for_each(|result| match result {
597                Err(ZipError::UnsupportedArchive(e)) => {
598                    unsupported_errors.push(ZipError::UnsupportedArchive(e))
599                }
600                Err(e) => invalid_errors.push(e),
601                Ok(o) => ok_results.push(o),
602            });
603        if ok_results.is_empty() {
604            return Err(unsupported_errors
605                .into_iter()
606                .next()
607                .unwrap_or_else(|| invalid_errors.into_iter().next().unwrap()));
608        }
609        let shared = ok_results
610            .into_iter()
611            .max_by_key(|shared| shared.dir_start)
612            .unwrap();
613        reader.seek(io::SeekFrom::Start(shared.dir_start))?;
614        Ok(shared)
615    }

617    /// Read a ZIP archive, collecting the files it contains
618    ///
619    /// This uses the central directory record of the ZIP file, and ignores local file headers
620    pub fn new(mut reader: R) -> ZipResult<ZipArchive<R>> {
621        let (footer, cde_start_pos) = spec::CentralDirectoryEnd::find_and_parse(&mut reader)?;
622        let shared = Self::get_metadata(&mut reader, &footer, cde_start_pos)?;
623        Ok(ZipArchive {
624            reader,
625            shared: shared.into(),
626            comment: footer.zip_file_comment.into(),
627        })
628    }
629    /// Extract a Zip archive into a directory, overwriting files if they
630    /// already exist. Paths are sanitized with [`ZipFile::enclosed_name`].
631    ///
632    /// Extraction is not atomic; If an error is encountered, some of the files
633    /// may be left on disk.
634    pub fn extract<P: AsRef<Path>>(&mut self, directory: P) -> ZipResult<()> {
635        use std::fs;
636
637        for i in 0..self.len() {
638            let mut file = self.by_index(i)?;
639            let filepath = file
640                .enclosed_name()
641                .ok_or(ZipError::InvalidArchive("Invalid file path"))?;
642
643            let outpath = directory.as_ref().join(filepath);
644
645            if file.is_dir() {
646                fs::create_dir_all(&outpath)?;
647            } else {
648                if let Some(p) = outpath.parent() {
649                    if !p.exists() {
650                        fs::create_dir_all(p)?;
651                    }
652                }
653                let mut outfile = fs::File::create(&outpath)?;
654                io::copy(&mut file, &mut outfile)?;
655            }
656            // Get and Set permissions
657            #[cfg(unix)]
658            {
659                use std::os::unix::fs::PermissionsExt;
660                if let Some(mode) = file.unix_mode() {
661                    fs::set_permissions(&outpath, fs::Permissions::from_mode(mode))?;
662                }
663            }
664        }
665        Ok(())
666    }

839/// Parse a central directory entry to collect the information for the file.
840pub(crate) fn central_header_to_zip_file<R: Read + Seek>(
841    reader: &mut R,
842    archive_offset: u64,
843) -> ZipResult<ZipFileData> {
844    let central_header_start = reader.stream_position()?;
845
846    // Parse central header
847    let signature = reader.read_u32_le()?;
848    if signature != spec::CENTRAL_DIRECTORY_HEADER_SIGNATURE {
849        Err(ZipError::InvalidArchive("Invalid Central Directory header"))
850    } else {
851        central_header_to_zip_file_inner(reader, archive_offset, central_header_start)
852    }
853}
854
855/// Parse a central directory entry to collect the information for the file.
856fn central_header_to_zip_file_inner<R: Read>(
857    reader: &mut R,
858    archive_offset: u64,
859    central_header_start: u64,
860) -> ZipResult<ZipFileData> {
861    let version_made_by = reader.read_u16_le()?;
862    let _version_to_extract = reader.read_u16_le()?;
863    let flags = reader.read_u16_le()?;
864    let encrypted = flags & 1 == 1;
865    let is_utf8 = flags & (1 << 11) != 0;
866    let using_data_descriptor = flags & (1 << 3) != 0;
867    let compression_method = reader.read_u16_le()?;
868    let last_mod_time = reader.read_u16_le()?;
869    let last_mod_date = reader.read_u16_le()?;
870    let crc32 = reader.read_u32_le()?;
871    let compressed_size = reader.read_u32_le()?;
872    let uncompressed_size = reader.read_u32_le()?;
873    let file_name_length = reader.read_u16_le()? as usize;
874    let extra_field_length = reader.read_u16_le()? as usize;
875    let file_comment_length = reader.read_u16_le()? as usize;
876    let _disk_number = reader.read_u16_le()?;
877    let _internal_file_attributes = reader.read_u16_le()?;
878    let external_file_attributes = reader.read_u32_le()?;
879    let offset = reader.read_u32_le()? as u64;
880    let mut file_name_raw = vec![0; file_name_length];
881    reader.read_exact(&mut file_name_raw)?;
882    let mut extra_field = vec![0; extra_field_length];
883    reader.read_exact(&mut extra_field)?;
884    let mut file_comment_raw = vec![0; file_comment_length];
885    reader.read_exact(&mut file_comment_raw)?;
886
887    let file_name: Box<str> = match is_utf8 {
888        true => String::from_utf8_lossy(&file_name_raw).into(),
889        false => file_name_raw.from_cp437().into(),
890    };
891    let file_comment: Box<str> = match is_utf8 {
892        true => String::from_utf8_lossy(&file_comment_raw).into(),
893        false => file_comment_raw.from_cp437().into(),
894    };
895
896    // Construct the result
897    let mut result = ZipFileData {
898        system: System::from((version_made_by >> 8) as u8),
899        version_made_by: version_made_by as u8,
900        encrypted,
901        using_data_descriptor,
902        compression_method: {
903            #[allow(deprecated)]
904            CompressionMethod::from_u16(compression_method)
905        },
906        compression_level: None,
907        last_modified_time: DateTime::from_msdos(last_mod_date, last_mod_time),
908        crc32,
909        compressed_size: compressed_size as u64,
910        uncompressed_size: uncompressed_size as u64,
911        file_name,
912        file_name_raw: file_name_raw.into(),
913        extra_field: Some(Arc::new(extra_field)),
914        central_extra_field: None,
915        file_comment,
916        header_start: offset,
917        central_header_start,
918        data_start: OnceLock::new(),
919        external_attributes: external_file_attributes,
920        large_file: false,
921        aes_mode: None,
922        extra_fields: Vec::new(),
923    };
924
925    match parse_extra_field(&mut result) {
926        Ok(..) | Err(ZipError::Io(..)) => {}
927        Err(e) => return Err(e),
928    }
929
930    let aes_enabled = result.compression_method == CompressionMethod::AES;
931    if aes_enabled && result.aes_mode.is_none() {
932        return Err(ZipError::InvalidArchive(
933            "AES encryption without AES extra data field",
934        ));
935    }
936
937    // Account for shifted zip offsets.
938    result.header_start = result
939        .header_start
940        .checked_add(archive_offset)
941        .ok_or(ZipError::InvalidArchive("Archive header is too large"))?;
942
943    Ok(result)
944}
945
946fn parse_extra_field(file: &mut ZipFileData) -> ZipResult<()> {
947    let Some(extra_field) = &file.extra_field else {
948        return Ok(());
949    };
950    let mut reader = io::Cursor::new(extra_field.as_ref());
951
952    while (reader.position() as usize) < extra_field.len() {
953        let kind = reader.read_u16_le()?;
954        let len = reader.read_u16_le()?;
955        let mut len_left = len as i64;
956        match kind {
957            // Zip64 extended information extra field
958            0x0001 => {
959                if file.uncompressed_size == spec::ZIP64_BYTES_THR {
960                    file.large_file = true;
961                    file.uncompressed_size = reader.read_u64_le()?;
962                    len_left -= 8;
963                }
964                if file.compressed_size == spec::ZIP64_BYTES_THR {
965                    file.large_file = true;
966                    file.compressed_size = reader.read_u64_le()?;
967                    len_left -= 8;
968                }
969                if file.header_start == spec::ZIP64_BYTES_THR {
970                    file.header_start = reader.read_u64_le()?;
971                    len_left -= 8;
972                }
973            }
974            0x9901 => {
975                // AES
976                if len != 7 {
977                    return Err(ZipError::UnsupportedArchive(
978                        "AES extra data field has an unsupported length",
979                    ));
980                }
981                let vendor_version = reader.read_u16_le()?;
982                let vendor_id = reader.read_u16_le()?;
983                let mut out = [0u8];
984                reader.read_exact(&mut out)?;
985                let aes_mode = out[0];
986                let compression_method = reader.read_u16_le()?;
987
988                if vendor_id != 0x4541 {
989                    return Err(ZipError::InvalidArchive("Invalid AES vendor"));
990                }
991                let vendor_version = match vendor_version {
992                    0x0001 => AesVendorVersion::Ae1,
993                    0x0002 => AesVendorVersion::Ae2,
994                    _ => return Err(ZipError::InvalidArchive("Invalid AES vendor version")),
995                };
996                match aes_mode {
997                    0x01 => file.aes_mode = Some((AesMode::Aes128, vendor_version)),
998                    0x02 => file.aes_mode = Some((AesMode::Aes192, vendor_version)),
999                    0x03 => file.aes_mode = Some((AesMode::Aes256, vendor_version)),
1000                    _ => return Err(ZipError::InvalidArchive("Invalid AES encryption strength")),
1001                };
1002                file.compression_method = {
1003                    #[allow(deprecated)]
1004                    CompressionMethod::from_u16(compression_method)
1005                };
1006            }
1007            0x5455 => {
1008                // extended timestamp
1009                // https://libzip.org/specifications/extrafld.txt
1010
1011                file.extra_fields.push(ExtraField::ExtendedTimestamp(
1012                    ExtendedTimestamp::try_from_reader(&mut reader, len)?,
1013                ));
1014
1015                // the reader for ExtendedTimestamp consumes `len` bytes
1016                len_left = 0;
1017            }
1018            _ => {
1019                // Other fields are ignored
1020            }
1021        }
1022
1023        // We could also check for < 0 to check for errors
1024        if len_left > 0 {
1025            reader.seek(io::SeekFrom::Current(len_left))?;
1026        }
1027    }
1028    Ok(())
1029}

1200impl<'a> Drop for ZipFile<'a> {
1201    fn drop(&mut self) {
1202        // self.data is Owned, this reader is constructed by a streaming reader.
1203        // In this case, we want to exhaust the reader so that the next file is accessible.
1204        if let Cow::Owned(_) = self.data {
1205            let mut buffer = [0; 1 << 16];
1206
1207            // Get the inner `Take` reader so all decryption, decompression and CRC calculation is skipped.
1208            let mut reader: io::Take<&mut dyn Read> = match &mut self.reader {
1209                ZipFileReader::NoReader => {
1210                    let innerreader = self.crypto_reader.take();
1211                    innerreader.expect("Invalid reader state").into_inner()
1212                }
1213                reader => {
1214                    let innerreader = std::mem::replace(reader, ZipFileReader::NoReader);
1215                    innerreader.into_inner()
1216                }
1217            };
1218
1219            #[allow(clippy::unused_io_amount)]
1220            loop {
1221                match reader.read(&mut buffer) {
1222                    Ok(0) => break,
1223                    Ok(_read) => (),
1224                    Err(e) => {
1225                        panic!("Could not consume all of the output of the current ZipFile: {e:?}")
1226                    }
1227                }
1228            }
1229        }
1230    }
1231}

1233/// Read ZipFile structures from a non-seekable reader.
1234///
1235/// This is an alternative method to read a zip file. If possible, use the ZipArchive functions
1236/// as some information will be missing when reading this manner.
1237///
1238/// Reads a file header from the start of the stream. Will return `Ok(Some(..))` if a file is
1239/// present at the start of the stream. Returns `Ok(None)` if the start of the central directory
1240/// is encountered. No more files should be read after this.
1241///
1242/// The Drop implementation of ZipFile ensures that the reader will be correctly positioned after
1243/// the structure is done.
1244///
1245/// Missing fields are:
1246/// * `comment`: set to an empty string
1247/// * `data_start`: set to 0
1248/// * `external_attributes`: `unix_mode()`: will return None
1249pub fn read_zipfile_from_stream<'a, R: Read>(reader: &'a mut R) -> ZipResult<Option<ZipFile<'_>>> {
1250    let signature = reader.read_u32_le()?;
1251
1252    match signature {
1253        spec::LOCAL_FILE_HEADER_SIGNATURE => (),
1254        spec::CENTRAL_DIRECTORY_HEADER_SIGNATURE => return Ok(None),
1255        _ => return Err(ZipError::InvalidArchive("Invalid local file header")),
1256    }
1257
1258    let version_made_by = reader.read_u16_le()?;
1259    let flags = reader.read_u16_le()?;
1260    let encrypted = flags & 1 == 1;
1261    let is_utf8 = flags & (1 << 11) != 0;
1262    let using_data_descriptor = flags & (1 << 3) != 0;
1263    #[allow(deprecated)]
1264    let compression_method = CompressionMethod::from_u16(reader.read_u16_le()?);
1265    let last_mod_time = reader.read_u16_le()?;
1266    let last_mod_date = reader.read_u16_le()?;
1267    let crc32 = reader.read_u32_le()?;
1268    let compressed_size = reader.read_u32_le()?;
1269    let uncompressed_size = reader.read_u32_le()?;
1270    let file_name_length = reader.read_u16_le()? as usize;
1271    let extra_field_length = reader.read_u16_le()? as usize;
1272
1273    let mut file_name_raw = vec![0; file_name_length];
1274    reader.read_exact(&mut file_name_raw)?;
1275    let mut extra_field = vec![0; extra_field_length];
1276    reader.read_exact(&mut extra_field)?;
1277
1278    let file_name: Box<str> = match is_utf8 {
1279        true => String::from_utf8_lossy(&file_name_raw).into(),
1280        false => file_name_raw.clone().from_cp437().into(),
1281    };
1282
1283    let mut result = ZipFileData {
1284        system: System::from((version_made_by >> 8) as u8),
1285        version_made_by: version_made_by as u8,
1286        encrypted,
1287        using_data_descriptor,
1288        compression_method,
1289        compression_level: None,
1290        last_modified_time: DateTime::from_msdos(last_mod_date, last_mod_time),
1291        crc32,
1292        compressed_size: compressed_size as u64,
1293        uncompressed_size: uncompressed_size as u64,
1294        file_name,
1295        file_name_raw: file_name_raw.into(),
1296        extra_field: Some(Arc::new(extra_field)),
1297        central_extra_field: None,
1298        file_comment: String::with_capacity(0).into_boxed_str(), // file comment is only available in the central directory
1299        // header_start and data start are not available, but also don't matter, since seeking is
1300        // not available.
1301        header_start: 0,
1302        data_start: OnceLock::new(),
1303        central_header_start: 0,
1304        // The external_attributes field is only available in the central directory.
1305        // We set this to zero, which should be valid as the docs state 'If input came
1306        // from standard input, this field is set to zero.'
1307        external_attributes: 0,
1308        large_file: false,
1309        aes_mode: None,
1310        extra_fields: Vec::new(),
1311    };
1312
1313    match parse_extra_field(&mut result) {
1314        Ok(..) | Err(ZipError::Io(..)) => {}
1315        Err(e) => return Err(e),
1316    }
1317
1318    if encrypted {
1319        return unsupported_zip_error("Encrypted files are not supported");
1320    }
1321    if using_data_descriptor {
1322        return unsupported_zip_error("The file length is not available in the local header");
1323    }
1324
1325    let limit_reader = (reader as &'a mut dyn Read).take(result.compressed_size);
1326
1327    let result_crc32 = result.crc32;
1328    let result_compression_method = result.compression_method;
1329    let crypto_reader = make_crypto_reader(
1330        result_compression_method,
1331        result_crc32,
1332        result.last_modified_time,
1333        result.using_data_descriptor,
1334        limit_reader,
1335        None,
1336        None,
1337        #[cfg(feature = "aes-crypto")]
1338        result.compressed_size,
1339    )?;
1340
1341    Ok(Some(ZipFile {
1342        data: Cow::Owned(result),
1343        crypto_reader: None,
1344        reader: make_reader(result_compression_method, result_crc32, crypto_reader),
1345    }))
1346}
