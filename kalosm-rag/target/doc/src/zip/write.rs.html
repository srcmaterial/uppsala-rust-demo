206impl arbitrary::Arbitrary<'_> for FileOptions<ExtendedFileOptions> {
207    fn arbitrary(u: &mut arbitrary::Unstructured) -> arbitrary::Result<Self> {
208        let mut options = FullFileOptions {
209            compression_method: CompressionMethod::arbitrary(u)?,
210            compression_level: None,
211            last_modified_time: DateTime::arbitrary(u)?,
212            permissions: Option::<u32>::arbitrary(u)?,
213            large_file: bool::arbitrary(u)?,
214            encrypt_with: Option::<ZipCryptoKeys>::arbitrary(u)?,
215            alignment: u16::arbitrary(u)?,
216            #[cfg(feature = "deflate-zopfli")]
217            zopfli_buffer_size: None,
218            ..Default::default()
219        };
220        match options.compression_method {
221            #[cfg(feature = "deflate-zopfli")]
222            CompressionMethod::Deflated => {
223                if bool::arbitrary(u)? {
224                    let level = u.int_in_range(0..=24)?;
225                    options.compression_level = Some(level);
226                    if level > Compression::best().level().try_into().unwrap() {
227                        options.zopfli_buffer_size = Some(1 << u.int_in_range(9..=30)?);
228                    }
229                }
230            }
231            Stored => {
232                if bool::arbitrary(u)? {
233                    options.compression_level = Some(1);
234                }
235            }
236            _ => {
237                if bool::arbitrary(u)? {
238                    options.compression_level = Some(u.int_in_range(0..=10)?);
239                }
240            }
241        }
242        u.arbitrary_loop(Some(0), Some((u16::MAX / 4) as u32), |u| {
243            options
244                .add_extra_data(
245                    u16::arbitrary(u)?,
246                    &Vec::<u8>::arbitrary(u)?,
247                    bool::arbitrary(u)?,
248                )
249                .map_err(|_| arbitrary::Error::IncorrectFormat)?;
250            Ok(core::ops::ControlFlow::Continue(()))
251        })?;
252        Ok(options)
253    }

347impl FileOptions<ExtendedFileOptions> {
348    /// Adds an extra data field.
349    pub fn add_extra_data(
350        &mut self,
351        header_id: u16,
352        data: &[u8],
353        central_only: bool,
354    ) -> ZipResult<()> {
355        validate_extra_data(header_id, data)?;
356        let len = data.len() + 4;
357        if self.extended_options.extra_data.len()
358            + self.extended_options.central_extra_data.len()
359            + len
360            > u16::MAX as usize
361        {
362            Err(InvalidArchive(
363                "Extra data field would be longer than allowed",
364            ))
365        } else {
366            let field = if central_only {
367                &mut self.extended_options.central_extra_data
368            } else {
369                &mut self.extended_options.extra_data
370            };
371            let vec = Arc::get_mut(field);
372            let vec = match vec {
373                Some(exclusive) => exclusive,
374                None => {
375                    *field = Arc::new(field.to_vec());
376                    Arc::get_mut(field).unwrap()
377                }
378            };
379            vec.reserve_exact(data.len() + 4);
380            vec.write_u16_le(header_id)?;
381            vec.write_u16_le(data.len() as u16)?;
382            vec.write_all(data)?;
383            Ok(())
384        }
385    }
386
387    /// Removes the extra data fields.
388    #[must_use]
389    pub fn clear_extra_data(mut self) -> Self {
390        if self.extended_options.extra_data.len() > 0 {
391            self.extended_options.extra_data = Arc::new(vec![]);
392        }
393        if self.extended_options.central_extra_data.len() > 0 {
394            self.extended_options.central_extra_data = Arc::new(vec![]);
395        }
396        self
397    }

420impl<W: Write + Seek> Write for ZipWriter<W> {
421    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
422        if !self.writing_to_file {
423            return Err(io::Error::new(
424                io::ErrorKind::Other,
425                "No file has been started",
426            ));
427        }
428        if buf.is_empty() {
429            return Ok(0);
430        }
431        match self.inner.ref_mut() {
432            Some(ref mut w) => {
433                let write_result = w.write(buf);
434                if let Ok(count) = write_result {
435                    self.stats.update(&buf[0..count]);
436                    if self.stats.bytes_written > spec::ZIP64_BYTES_THR
437                        && !self.files.last_mut().unwrap().1.large_file
438                    {
439                        self.abort_file().unwrap();
440                        return Err(io::Error::new(
441                            io::ErrorKind::Other,
442                            "Large file option has not been set",
443                        ));
444                    }
445                }
446                write_result
447            }
448            None => Err(io::Error::new(
449                io::ErrorKind::BrokenPipe,
450                "write(): ZipWriter was already closed",
451            )),
452        }
453    }
454
455    fn flush(&mut self) -> io::Result<()> {
456        match self.inner.ref_mut() {
457            Some(ref mut w) => w.flush(),
458            None => Err(io::Error::new(
459                io::ErrorKind::BrokenPipe,
460                "flush(): ZipWriter was already closed",
461            )),
462        }
463    }
464}

475    pub fn new_append(mut readwriter: A) -> ZipResult<ZipWriter<A>> {
476        let (footer, cde_start_pos) = spec::CentralDirectoryEnd::find_and_parse(&mut readwriter)?;
477        let metadata = ZipArchive::get_metadata(&mut readwriter, &footer, cde_start_pos)?;
478
479        Ok(ZipWriter {
480            inner: Storer(MaybeEncrypted::Unencrypted(readwriter)),
481            files: metadata.files,
482            stats: Default::default(),
483            writing_to_file: false,
484            comment: footer.zip_file_comment,
485            writing_raw: true, // avoid recomputing the last file's header
486            flush_on_finish_file: false,
487        })

490    /// `flush_on_finish_file` is designed to support a streaming `inner` that may unload flushed
491    /// bytes. It flushes a file's header and body once it starts writing another file. A ZipWriter
492    /// will not try to seek back into where a previous file was written unless
493    /// either [`ZipWriter::abort_file`] is called while [`ZipWriter::is_writing_file`] returns
494    /// false, or [`ZipWriter::deep_copy_file`] is called. In the latter case, it will only need to
495    /// read previously-written files and not overwrite them.
496    ///
497    /// Note: when using an `inner` that cannot overwrite flushed bytes, do not wrap it in a
498    /// [std::io::BufWriter], because that has a [Seek::seek] method that implicitly calls
499    /// [BufWriter::flush], and ZipWriter needs to seek backward to update each file's header with
500    /// the size and checksum after writing the body.
501    ///
502    /// This setting is false by default.
503    pub fn set_flush_on_finish_file(&mut self, flush_on_finish_file: bool) {
504        self.flush_on_finish_file = flush_on_finish_file;
505    }

508impl<A: Read + Write + Seek> ZipWriter<A> {
509    /// Adds another copy of a file already in this archive. This will produce a larger but more
510    /// widely-compatible archive compared to [Self::shallow_copy_file]. Does not copy alignment.
511    pub fn deep_copy_file(&mut self, src_name: &str, dest_name: &str) -> ZipResult<()> {
512        self.finish_file()?;
513        let write_position = self.inner.get_plain().stream_position()?;
514        let src_index = self.index_by_name(src_name)?;
515        let src_data = &self.files[src_index];
516        let data_start = *src_data.data_start.get().unwrap_or(&0);
517        let compressed_size = src_data.compressed_size;
518        debug_assert!(compressed_size <= write_position - data_start);
519        let uncompressed_size = src_data.uncompressed_size;
520
521        let raw_values = ZipRawValues {
522            crc32: src_data.crc32,
523            compressed_size,
524            uncompressed_size,
525        };
526        let mut reader = BufReader::new(ZipFileReader::Raw(find_content(
527            src_data,
528            self.inner.get_plain(),
529        )?));
530        let mut copy = Vec::with_capacity(compressed_size as usize);
531        reader.read_to_end(&mut copy)?;
532        drop(reader);
533        self.inner
534            .get_plain()
535            .seek(SeekFrom::Start(write_position))?;
536        if src_data.extra_field.is_some() || src_data.central_extra_field.is_some() {
537            let mut options = FileOptions::<ExtendedFileOptions> {
538                compression_method: src_data.compression_method,
539                compression_level: src_data.compression_level,
540                last_modified_time: src_data.last_modified_time,
541                permissions: src_data.unix_mode(),
542                large_file: src_data.large_file,
543                encrypt_with: None,
544                extended_options: ExtendedFileOptions {
545                    extra_data: src_data.extra_field.clone().unwrap_or_default(),
546                    central_extra_data: src_data.central_extra_field.clone().unwrap_or_default(),
547                },
548                alignment: 1,
549                #[cfg(feature = "deflate-zopfli")]
550                zopfli_buffer_size: None,
551            };
552            if let Some(perms) = src_data.unix_mode() {
553                options = options.unix_permissions(perms);
554            }
555            Self::normalize_options(&mut options);
556            self.start_entry(dest_name, options, Some(raw_values))?;
557        } else {
558            let mut options = FileOptions::<()> {
559                compression_method: src_data.compression_method,
560                compression_level: src_data.compression_level,
561                last_modified_time: src_data.last_modified_time,
562                permissions: src_data.unix_mode(),
563                large_file: src_data.large_file,
564                encrypt_with: None,
565                extended_options: (),
566                alignment: 1,
567                #[cfg(feature = "deflate-zopfli")]
568                zopfli_buffer_size: None,
569            };
570            if let Some(perms) = src_data.unix_mode() {
571                options = options.unix_permissions(perms);
572            }
573            Self::normalize_options(&mut options);
574            self.start_entry(dest_name, options, Some(raw_values))?;
575        }
576
577        self.writing_to_file = true;
578        self.writing_raw = true;
579        if let Err(e) = self.write_all(&copy) {
580            self.abort_file().unwrap();
581            return Err(e.into());
582        }
583        self.finish_file()
584    }

587    ///
588    /// This function ensures that the '/' path separator is used and normalizes `.` and `..`. It
589    /// ignores any `..` or Windows drive letter that would produce a path outside the ZIP file's
590    /// root.
591    pub fn deep_copy_file_from_path<T: AsRef<Path>, U: AsRef<Path>>(

599    /// Write the zip file into the backing stream, then produce a readable archive of that data.
600    ///
601    /// This method avoids parsing the central directory records at the end of the stream for
602    /// a slight performance improvement over running [`ZipArchive::new()`] on the output of
603    /// [`Self::finish()`].
604    ///
605    ///```
606    /// # fn main() -> Result<(), zip::result::ZipError> {
607    /// use std::io::{Cursor, prelude::*};
608    /// use zip::{ZipArchive, ZipWriter, write::SimpleFileOptions};
609    ///
610    /// let buf = Cursor::new(Vec::new());
611    /// let mut zip = ZipWriter::new(buf);
612    /// let options = SimpleFileOptions::default();
613    /// zip.start_file("a.txt", options)?;
614    /// zip.write_all(b"hello\n")?;
615    ///
616    /// let mut zip = zip.finish_into_readable()?;
617    /// let mut s: String = String::new();
618    /// zip.by_name("a.txt")?.read_to_string(&mut s)?;
619    /// assert_eq!(s, "hello\n");
620    /// # Ok(())
621    /// # }
622    ///```
623    pub fn finish_into_readable(mut self) -> ZipResult<ZipArchive<A>> {
624        let central_start = self.finalize()?;
625        let inner = mem::replace(&mut self.inner, Closed).unwrap();
626        let comment = mem::take(&mut self.comment);
627        let files = mem::take(&mut self.files);
628        let archive = ZipArchive::from_finalized_writer(files, comment, inner, central_start)?;
629        Ok(archive)
630    }
631}

685    /// Start a new file for with the requested options.
686    fn start_entry<S, T: FileOptionExtension>(
687        &mut self,
688        name: S,
689        options: FileOptions<T>,
690        raw_values: Option<ZipRawValues>,
691    ) -> ZipResult<()>
692    where
693        S: Into<Box<str>>,
694    {
695        self.finish_file()?;
696
697        let raw_values = raw_values.unwrap_or(ZipRawValues {
698            crc32: 0,
699            compressed_size: 0,
700            uncompressed_size: 0,
701        });
702
703        {
704            let header_start = self.inner.get_plain().stream_position()?;
705
706            let permissions = options.permissions.unwrap_or(0o100644);
707            let file = ZipFileData {
708                system: System::Unix,
709                version_made_by: DEFAULT_VERSION,
710                encrypted: options.encrypt_with.is_some(),
711                using_data_descriptor: false,
712                compression_method: options.compression_method,
713                compression_level: options.compression_level,
714                last_modified_time: options.last_modified_time,
715                crc32: raw_values.crc32,
716                compressed_size: raw_values.compressed_size,
717                uncompressed_size: raw_values.uncompressed_size,
718                file_name: name.into(),
719                file_name_raw: vec![].into_boxed_slice(), // Never used for saving
720                extra_field: options.extended_options.extra_data().cloned(),
721                central_extra_field: options.extended_options.central_extra_data().cloned(),
722                file_comment: String::with_capacity(0).into_boxed_str(),
723                header_start,
724                data_start: OnceLock::new(),
725                central_header_start: 0,
726                external_attributes: permissions << 16,
727                large_file: options.large_file,
728                aes_mode: None,
729                extra_fields: Vec::new(),
730            };
731            let index = self.insert_file_data(file)?;
732            let file = &mut self.files[index];
733            let writer = self.inner.get_plain();
734            writer.write_u32_le(spec::LOCAL_FILE_HEADER_SIGNATURE)?;
735            // version needed to extract
736            writer.write_u16_le(file.version_needed())?;
737            // general purpose bit flag
738            let flag = if !file.file_name.is_ascii() {
739                1u16 << 11
740            } else {
741                0
742            } | if file.encrypted { 1u16 << 0 } else { 0 };
743            writer.write_u16_le(flag)?;
744            // Compression method
745            #[allow(deprecated)]
746            writer.write_u16_le(file.compression_method.to_u16())?;
747            // last mod file time and last mod file date
748            writer.write_u16_le(file.last_modified_time.timepart())?;
749            writer.write_u16_le(file.last_modified_time.datepart())?;
750            // crc-32
751            writer.write_u32_le(file.crc32)?;
752            // compressed size and uncompressed size
753            if file.large_file {
754                writer.write_u32_le(spec::ZIP64_BYTES_THR as u32)?;
755                writer.write_u32_le(spec::ZIP64_BYTES_THR as u32)?;
756            } else {
757                writer.write_u32_le(file.compressed_size as u32)?;
758                writer.write_u32_le(file.uncompressed_size as u32)?;
759            }
760            // file name length
761            writer.write_u16_le(file.file_name.as_bytes().len() as u16)?;
762            // extra field length
763            let mut extra_field_length = file.extra_field_len();
764            if file.large_file {
765                extra_field_length += 20;
766            }
767            if extra_field_length + file.central_extra_field_len() > u16::MAX as usize {
768                let _ = self.abort_file();
769                return Err(InvalidArchive("Extra data field is too large"));
770            }
771            let extra_field_length = extra_field_length as u16;
772            writer.write_u16_le(extra_field_length)?;
773            // file name
774            writer.write_all(file.file_name.as_bytes())?;
775            // zip64 extra field
776            if file.large_file {
777                write_local_zip64_extra_field(writer, file)?;
778            }
779            if let Some(extra_field) = &file.extra_field {
780                writer.write_all(extra_field)?;
781            }
782            let mut header_end = writer.stream_position()?;
783            if options.alignment > 1 {
784                let align = options.alignment as u64;
785                let unaligned_header_bytes = header_end % align;
786                if unaligned_header_bytes != 0 {
787                    let pad_length = (align - unaligned_header_bytes) as usize;
788                    let Some(new_extra_field_length) =
789                        (pad_length as u16).checked_add(extra_field_length)
790                    else {
791                        let _ = self.abort_file();
792                        return Err(InvalidArchive(
793                            "Extra data field would be larger than allowed after aligning",
794                        ));
795                    };
796                    if pad_length >= 4 {
797                        // Add an extra field to the extra_data
798                        let pad_body = vec![0; pad_length - 4];
799                        writer.write_all(b"za").map_err(ZipError::from)?; // 0x617a
800                        writer
801                            .write_u16_le(pad_body.len() as u16)
802                            .map_err(ZipError::from)?;
803                        writer.write_all(&pad_body).map_err(ZipError::from)?;
804                    } else {
805                        // extra_data padding is too small for an extra field header, so pad with
806                        // zeroes
807                        let pad = vec![0; pad_length];
808                        writer.write_all(&pad).map_err(ZipError::from)?;
809                    }
810                    header_end = writer.stream_position()?;
811
812                    // Update extra field length in local file header.
813                    writer.seek(SeekFrom::Start(file.header_start + 28))?;
814                    writer.write_u16_le(new_extra_field_length)?;
815                    writer.seek(SeekFrom::Start(header_end))?;
816                    debug_assert_eq!(header_end % align, 0);
817                }
818            }
819            if let Some(keys) = options.encrypt_with {
820                let mut zipwriter = crate::zipcrypto::ZipCryptoWriter {
821                    writer: mem::replace(&mut self.inner, Closed).unwrap(),
822                    buffer: vec![],
823                    keys,
824                };
825                let crypto_header = [0u8; 12];
826
827                zipwriter.write_all(&crypto_header)?;
828                header_end = zipwriter.writer.stream_position()?;
829                self.inner = Storer(MaybeEncrypted::Encrypted(zipwriter));
830            }
831            self.stats.start = header_end;
832            debug_assert!(file.data_start.get().is_none());
833            file.data_start.get_or_init(|| header_end);
834            self.writing_to_file = true;
835            self.stats.bytes_written = 0;
836            self.stats.hasher = Hasher::new();
837        }
838        Ok(())
839    }

841    fn insert_file_data(&mut self, file: ZipFileData) -> ZipResult<usize> {
842        if self.files.contains_key(&file.file_name) {
843            return Err(InvalidArchive("Duplicate filename"));
844        }
845        let name = file.file_name.to_owned();
846        self.files.insert(name.clone(), file);
847        Ok(self.files.get_index_of(&name).unwrap())
848    }

850    fn finish_file(&mut self) -> ZipResult<()> {
851        if !self.writing_to_file {
852            return Ok(());
853        }
854
855        let make_plain_writer = self.inner.prepare_next_writer(
856            Stored,
857            None,
858            #[cfg(feature = "deflate-zopfli")]
859            None,
860        )?;
861        self.inner.switch_to(make_plain_writer)?;
862        self.switch_to_non_encrypting_writer()?;
863        let writer = self.inner.get_plain();
864
865        if !self.writing_raw {
866            let file = match self.files.last_mut() {
867                None => return Ok(()),
868                Some((_, f)) => f,
869            };
870            file.crc32 = self.stats.hasher.clone().finalize();
871            file.uncompressed_size = self.stats.bytes_written;
872
873            let file_end = writer.stream_position()?;
874            debug_assert!(file_end >= self.stats.start);
875            file.compressed_size = file_end - self.stats.start;
876
877            update_local_file_header(writer, file)?;
878            writer.seek(SeekFrom::Start(file_end))?;
879        }
880        if self.flush_on_finish_file {
881            if let Err(e) = writer.flush() {
882                self.abort_file()?;
883                return Err(e.into());
884            }
885        }
886
887        self.writing_to_file = false;
888        Ok(())
889    }
890
891    fn switch_to_non_encrypting_writer(&mut self) -> Result<(), ZipError> {
892        match mem::replace(&mut self.inner, Closed) {
893            Storer(MaybeEncrypted::Encrypted(writer)) => {
894                let crc32 = self.stats.hasher.clone().finalize();
895                self.inner = Storer(MaybeEncrypted::Unencrypted(writer.finish(crc32)?))
896            }
897            Storer(MaybeEncrypted::Unencrypted(w)) => {
898                self.inner = Storer(MaybeEncrypted::Unencrypted(w))
899            }
900            _ => unreachable!(),
901        }
902        Ok(())
903    }

905    /// Removes the file currently being written from the archive if there is one, or else removes
906    /// the file most recently written.
907    pub fn abort_file(&mut self) -> ZipResult<()> {
908        let (_, last_file) = self.files.pop().ok_or(ZipError::FileNotFound)?;
909        let make_plain_writer = self.inner.prepare_next_writer(
910            Stored,
911            None,
912            #[cfg(feature = "deflate-zopfli")]
913            None,
914        )?;
915        self.inner.switch_to(make_plain_writer)?;
916        self.switch_to_non_encrypting_writer()?;
917        // Make sure this is the last file, and that no shallow copies of it remain; otherwise we'd
918        // overwrite a valid file and corrupt the archive
919        let rewind_safe: bool = match last_file.data_start.get() {
920            None => self.files.is_empty(),
921            Some(last_file_start) => self.files.values().all(|file| {
922                file.data_start
923                    .get()
924                    .is_some_and(|start| start < last_file_start)
925            }),
926        };
927        if rewind_safe {
928            self.inner
929                .get_plain()
930                .seek(SeekFrom::Start(last_file.header_start))?;
931        }
932        self.writing_to_file = false;
933        Ok(())
934    }

966    /// Copy over the entire contents of another archive verbatim.
967    ///
968    /// This method extracts file metadata from the `source` archive, then simply performs a single
969    /// big [`io::copy()`](io::copy) to transfer all the actual file contents without any
970    /// decompression or decryption. This is more performant than the equivalent operation of
971    /// calling [`Self::raw_copy_file()`] for each entry from the `source` archive in sequence.
972    ///
973    ///```
974    /// # fn main() -> Result<(), zip::result::ZipError> {
975    /// use std::io::{Cursor, prelude::*};
976    /// use zip::{ZipArchive, ZipWriter, write::SimpleFileOptions};
977    ///
978    /// let buf = Cursor::new(Vec::new());
979    /// let mut zip = ZipWriter::new(buf);
980    /// zip.start_file("a.txt", SimpleFileOptions::default())?;
981    /// zip.write_all(b"hello\n")?;
982    /// let src = ZipArchive::new(zip.finish()?)?;
983    ///
984    /// let buf = Cursor::new(Vec::new());
985    /// let mut zip = ZipWriter::new(buf);
986    /// zip.start_file("b.txt", SimpleFileOptions::default())?;
987    /// zip.write_all(b"hey\n")?;
988    /// let src2 = ZipArchive::new(zip.finish()?)?;
989    ///
990    /// let buf = Cursor::new(Vec::new());
991    /// let mut zip = ZipWriter::new(buf);
992    /// zip.merge_archive(src)?;
993    /// zip.merge_archive(src2)?;
994    /// let mut result = ZipArchive::new(zip.finish()?)?;
995    ///
996    /// let mut s: String = String::new();
997    /// result.by_name("a.txt")?.read_to_string(&mut s)?;
998    /// assert_eq!(s, "hello\n");
999    /// s.clear();
1000    /// result.by_name("b.txt")?.read_to_string(&mut s)?;
1001    /// assert_eq!(s, "hey\n");
1002    /// # Ok(())
1003    /// # }
1004    ///```
1005    pub fn merge_archive<R>(&mut self, mut source: ZipArchive<R>) -> ZipResult<()>
1006    where
1007        R: Read + io::Seek,
1008    {
1009        self.finish_file()?;
1010
1011        /* Ensure we accept the file contents on faith (and avoid overwriting the data).
1012         * See raw_copy_file_rename(). */
1013        self.writing_to_file = true;
1014        self.writing_raw = true;
1015
1016        let writer = self.inner.get_plain();
1017        /* Get the file entries from the source archive. */
1018        let new_files = source.merge_contents(writer)?;
1019
1020        /* These file entries are now ours! */
1021        self.files.extend(new_files);
1022
1023        Ok(())
1024    }

1049    /// Add a new file using the already compressed data from a ZIP file being read and renames it, this
1050    /// allows faster copies of the `ZipFile` since there is no need to decompress and compress it again.
1051    /// Any `ZipFile` metadata is copied and not checked, for example the file CRC.
1052
1053    /// ```no_run
1054    /// use std::fs::File;
1055    /// use std::io::{Read, Seek, Write};
1056    /// use zip::{ZipArchive, ZipWriter};
1057    ///
1058    /// fn copy_rename<R, W>(
1059    ///     src: &mut ZipArchive<R>,
1060    ///     dst: &mut ZipWriter<W>,
1061    /// ) -> zip::result::ZipResult<()>
1062    /// where
1063    ///     R: Read + Seek,
1064    ///     W: Write + Seek,
1065    /// {
1066    ///     // Retrieve file entry by name
1067    ///     let file = src.by_name("src_file.txt")?;
1068    ///
1069    ///     // Copy and rename the previously obtained file entry to the destination zip archive
1070    ///     dst.raw_copy_file_rename(file, "new_name.txt")?;
1071    ///
1072    ///     Ok(())
1073    /// }
1074    /// ```
1075    pub fn raw_copy_file_rename<S>(&mut self, mut file: ZipFile, name: S) -> ZipResult<()>
1076    where
1077        S: Into<Box<str>>,
1078    {
1079        let mut options = SimpleFileOptions::default()
1080            .large_file(file.compressed_size().max(file.size()) > spec::ZIP64_BYTES_THR)
1081            .last_modified_time(file.last_modified())
1082            .compression_method(file.compression());
1083        if let Some(perms) = file.unix_mode() {
1084            options = options.unix_permissions(perms);
1085        }
1086        Self::normalize_options(&mut options);
1087
1088        let raw_values = ZipRawValues {
1089            crc32: file.crc32(),
1090            compressed_size: file.compressed_size(),
1091            uncompressed_size: file.size(),
1092        };
1093
1094        self.start_entry(name, options, Some(raw_values))?;
1095        self.writing_to_file = true;
1096        self.writing_raw = true;
1097
1098        io::copy(file.get_raw_reader(), self)?;
1099
1100        Ok(())
1101    }

1211        name: N,
1212        target: T,
1213        mut options: FileOptions<E>,
1214    ) -> ZipResult<()>
1215    where
1216        N: Into<Box<str>>,
1217        T: Into<Box<str>>,
1218    {
1219        if options.permissions.is_none() {
1220            options.permissions = Some(0o777);
1221        }
1222        *options.permissions.as_mut().unwrap() |= 0o120000;
1223        // The symlink target is stored as file content. And compressing the target path
1224        // likely wastes space. So always store.
1225        options.compression_method = Stored;
1226
1227        self.start_entry(name, options, None)?;
1228        self.writing_to_file = true;
1229        if let Err(e) = self.write_all(target.into().as_bytes()) {
1230            self.abort_file().unwrap();
1231            return Err(e.into());
1232        }
1233        self.finish_file()?;
1234
1235        Ok(())
1236    }

1238    /// Add a symlink entry, taking Paths to the location and target as arguments.
1239    ///
1240    /// This function ensures that the '/' path separator is used and normalizes `.` and `..`. It
1241    /// ignores any `..` or Windows drive letter that would produce a path outside the ZIP file's
1242    /// root.
1243    pub fn add_symlink_from_path<P: AsRef<Path>, T: AsRef<Path>, E: FileOptionExtension>(
1244        &mut self,
1245        path: P,
1246        target: T,
1247        options: FileOptions<E>,
1248    ) -> ZipResult<()> {
1249        self.add_symlink(path_to_string(path), path_to_string(target), options)
1250    }

1252    fn finalize(&mut self) -> ZipResult<u64> {
1253        self.finish_file()?;
1254
1255        let central_start = {
1256            let central_start = self.write_central_and_footer()?;
1257            let writer = self.inner.get_plain();
1258            let footer_end = writer.stream_position()?;
1259            let file_end = writer.seek(SeekFrom::End(0))?;
1260            if footer_end < file_end {
1261                // Data from an aborted file is past the end of the footer, so rewrite the footer at
1262                // the actual end.
1263                let central_and_footer_size = footer_end - central_start;
1264                writer.seek(SeekFrom::End(-(central_and_footer_size as i64)))?;
1265                self.write_central_and_footer()?;
1266            }
1267            central_start
1268        };
1269
1270        Ok(central_start)
1271    }

1273    fn write_central_and_footer(&mut self) -> Result<u64, ZipError> {
1274        let writer = self.inner.get_plain();
1275
1276        let central_start = writer.stream_position()?;
1277        for file in self.files.values() {
1278            write_central_directory_header(writer, file)?;
1279        }
1280        let central_size = writer.stream_position()? - central_start;
1281
1282        if self.files.len() > spec::ZIP64_ENTRY_THR
1283            || central_size.max(central_start) > spec::ZIP64_BYTES_THR
1284        {
1285            let zip64_footer = spec::Zip64CentralDirectoryEnd {
1286                version_made_by: DEFAULT_VERSION as u16,
1287                version_needed_to_extract: DEFAULT_VERSION as u16,
1288                disk_number: 0,
1289                disk_with_central_directory: 0,
1290                number_of_files_on_this_disk: self.files.len() as u64,
1291                number_of_files: self.files.len() as u64,
1292                central_directory_size: central_size,
1293                central_directory_offset: central_start,
1294            };
1295
1296            zip64_footer.write(writer)?;
1297
1298            let zip64_footer = spec::Zip64CentralDirectoryEndLocator {
1299                disk_with_central_directory: 0,
1300                end_of_central_directory_offset: central_start + central_size,
1301                number_of_disks: 1,
1302            };
1303
1304            zip64_footer.write(writer)?;
1305        }
1306
1307        let number_of_files = self.files.len().min(spec::ZIP64_ENTRY_THR) as u16;
1308        let footer = spec::CentralDirectoryEnd {
1309            disk_number: 0,
1310            disk_with_central_directory: 0,
1311            zip_file_comment: self.comment.clone(),
1312            number_of_files_on_this_disk: number_of_files,
1313            number_of_files,
1314            central_directory_size: central_size.min(spec::ZIP64_BYTES_THR) as u32,
1315            central_directory_offset: central_start.min(spec::ZIP64_BYTES_THR) as u32,
1316        };
1317
1318        footer.write(writer)?;
1319        Ok(central_start)
1320    }

1370        compression_level: Option<i64>,
1371        #[cfg(feature = "deflate-zopfli")] zopfli_buffer_size: Option<usize>,
1372    ) -> ZipResult<SwitchWriterFunction<W>> {
1373        if let Closed = self {
1374            return Err(
1375                io::Error::new(io::ErrorKind::BrokenPipe, "ZipWriter was already closed").into(),
1376            );
1377        }
1378
1379        {
1380            #[allow(deprecated)]
1381            #[allow(unreachable_code)]
1382            match compression {
1383                Stored => {
1384                    if compression_level.is_some() {
1385                        Err(ZipError::UnsupportedArchive(
1386                            "Unsupported compression level",
1387                        ))
1388                    } else {
1389                        Ok(Box::new(|bare| Storer(bare)))
1390                    }
1391                }
1392                #[cfg(feature = "_deflate-any")]
1393                CompressionMethod::Deflated => {
1394                    let default = if cfg!(feature = "deflate-zopfli") {
1395                        24
1396                    } else {
1397                        Compression::default().level() as i64
1398                    };
1399
1400                    let level = clamp_opt(
1401                        compression_level.unwrap_or(default),
1402                        deflate_compression_level_range(),
1403                    )
1404                    .ok_or(ZipError::UnsupportedArchive(
1405                        "Unsupported compression level",
1406                    ))? as u32;
1407
1408                    #[cfg(feature = "deflate-zopfli")]
1409                    {
1410                        let best_non_zopfli = Compression::best().level();
1411                        if level > best_non_zopfli {
1412                            let options = Options {
1413                                iteration_count: NonZeroU64::try_from(
1414                                    (level - best_non_zopfli) as u64,
1415                                )
1416                                .unwrap(),
1417                                ..Default::default()
1418                            };
1419                            return Ok(Box::new(move |bare| match zopfli_buffer_size {
1420                                Some(size) => GenericZipWriter::BufferedZopfliDeflater(
1421                                    BufWriter::with_capacity(
1422                                        size,
1423                                        zopfli::DeflateEncoder::new(
1424                                            options,
1425                                            Default::default(),
1426                                            bare,
1427                                        ),
1428                                    ),
1429                                ),
1430                                None => GenericZipWriter::ZopfliDeflater(
1431                                    zopfli::DeflateEncoder::new(options, Default::default(), bare),
1432                                ),
1433                            }));
1434                        }
1435                    }
1436
1437                    #[cfg(any(
1438                        feature = "deflate",
1439                        feature = "deflate-zlib",
1440                        feature = "deflate-zlib-ng",
1441                    ))]
1442                    {
1443                        return Ok(Box::new(move |bare| {
1444                            GenericZipWriter::Deflater(DeflateEncoder::new(
1445                                bare,
1446                                Compression::new(level),
1447                            ))
1448                        }));
1449                    }
1450                    unreachable!()
1451                }
1452                #[cfg(feature = "deflate64")]
1453                CompressionMethod::Deflate64 => Err(ZipError::UnsupportedArchive(
1454                    "Compressing Deflate64 is not supported",
1455                )),
1456                #[cfg(feature = "bzip2")]
1457                CompressionMethod::Bzip2 => {
1458                    let level = clamp_opt(
1459                        compression_level.unwrap_or(bzip2::Compression::default().level() as i64),

1461                    )
1462                    .ok_or(ZipError::UnsupportedArchive(
1463                        "Unsupported compression level",
1464                    ))? as u32;
1465                    Ok(Box::new(move |bare| {
1466                        GenericZipWriter::Bzip2(BzEncoder::new(
1467                            bare,
1468                            bzip2::Compression::new(level),
1469                        ))
1470                    }))
1471                }
1472                CompressionMethod::AES => Err(ZipError::UnsupportedArchive(
1473                    "AES compression is not supported for writing",
1474                )),
1475                #[cfg(feature = "zstd")]
1476                CompressionMethod::Zstd => {
1477                    let level = clamp_opt(
1478                        compression_level.unwrap_or(zstd::DEFAULT_COMPRESSION_LEVEL as i64),
1479                        zstd::compression_level_range(),
1480                    )
1481                    .ok_or(ZipError::UnsupportedArchive(
1482                        "Unsupported compression level",
1483                    ))?;
1484                    Ok(Box::new(move |bare| {

1487                }
1488                #[cfg(feature = "lzma")]
1489                CompressionMethod::Lzma => {
1490                    Err(UnsupportedArchive("LZMA isn't supported for compression"))
1491                }
1492                CompressionMethod::Unsupported(..) => {

1494                }
1495            }
1496        }

1611fn update_local_file_header<T: Write + Seek>(writer: &mut T, file: &ZipFileData) -> ZipResult<()> {
1612    const CRC32_OFFSET: u64 = 14;
1613    writer.seek(SeekFrom::Start(file.header_start + CRC32_OFFSET))?;
1614    writer.write_u32_le(file.crc32)?;
1615    if file.large_file {
1616        update_local_zip64_extra_field(writer, file)?;
1617    } else {
1618        // check compressed size as well as it can also be slightly larger than uncompressed size
1619        if file.compressed_size > spec::ZIP64_BYTES_THR {
1620            return Err(ZipError::Io(io::Error::new(
1621                io::ErrorKind::Other,
1622                "Large file option has not been set",
1623            )));
1624        }
1625        writer.write_u32_le(file.compressed_size as u32)?;
1626        // uncompressed size is already checked on write to catch it as soon as possible
1627        writer.write_u32_le(file.uncompressed_size as u32)?;
1628    }
1629    Ok(())
1630}

1632fn write_central_directory_header<T: Write>(writer: &mut T, file: &ZipFileData) -> ZipResult<()> {
1633    // buffer zip64 extra field to determine its variable length
1634    let mut zip64_extra_field = [0; 28];
1635    let zip64_extra_field_length =
1636        write_central_zip64_extra_field(&mut zip64_extra_field.as_mut(), file)?;
1637
1638    // central file header signature
1639    writer.write_u32_le(spec::CENTRAL_DIRECTORY_HEADER_SIGNATURE)?;
1640    // version made by
1641    let version_made_by = (file.system as u16) << 8 | (file.version_made_by as u16);
1642    writer.write_u16_le(version_made_by)?;
1643    // version needed to extract
1644    writer.write_u16_le(file.version_needed())?;
1645    // general purpose bit flag
1646    let flag = if !file.file_name.is_ascii() {
1647        1u16 << 11
1648    } else {
1649        0
1650    } | if file.encrypted { 1u16 << 0 } else { 0 };
1651    writer.write_u16_le(flag)?;
1652    // compression method
1653    #[allow(deprecated)]
1654    writer.write_u16_le(file.compression_method.to_u16())?;
1655    // last mod file time + date
1656    writer.write_u16_le(file.last_modified_time.timepart())?;
1657    writer.write_u16_le(file.last_modified_time.datepart())?;
1658    // crc-32
1659    writer.write_u32_le(file.crc32)?;
1660    // compressed size
1661    writer.write_u32_le(file.compressed_size.min(spec::ZIP64_BYTES_THR) as u32)?;
1662    // uncompressed size
1663    writer.write_u32_le(file.uncompressed_size.min(spec::ZIP64_BYTES_THR) as u32)?;
1664    // file name length
1665    writer.write_u16_le(file.file_name.as_bytes().len() as u16)?;
1666    // extra field length
1667    writer.write_u16_le(
1668        zip64_extra_field_length
1669            + file.extra_field_len() as u16
1670            + file.central_extra_field_len() as u16,
1671    )?;
1672    // file comment length
1673    writer.write_u16_le(0)?;
1674    // disk number start
1675    writer.write_u16_le(0)?;
1676    // internal file attributes
1677    writer.write_u16_le(0)?;
1678    // external file attributes
1679    writer.write_u32_le(file.external_attributes)?;
1680    // relative offset of local header
1681    writer.write_u32_le(file.header_start.min(spec::ZIP64_BYTES_THR) as u32)?;
1682    // file name
1683    writer.write_all(file.file_name.as_bytes())?;
1684    // zip64 extra field
1685    writer.write_all(&zip64_extra_field[..zip64_extra_field_length as usize])?;
1686    // extra field
1687    if let Some(extra_field) = &file.extra_field {
1688        writer.write_all(extra_field)?;
1689    }
1690    if let Some(central_extra_field) = &file.central_extra_field {
1691        writer.write_all(central_extra_field)?;
1692    }
1693    // file comment
1694    // <none>
1695
1696    Ok(())
1697}

1699fn validate_extra_data(header_id: u16, data: &[u8]) -> ZipResult<()> {
1700    if data.len() > u16::MAX as usize {
1701        return Err(ZipError::Io(io::Error::new(
1702            io::ErrorKind::Other,
1703            "Extra-data field can't exceed u16::MAX bytes",
1704        )));
1705    }
1706    if header_id == 0x0001 {
1707        return Err(ZipError::Io(io::Error::new(
1708            io::ErrorKind::Other,
1709            "No custom ZIP64 extra data allowed",
1710        )));
1711    }
1712
1713    #[cfg(not(feature = "unreserved"))]
1714    {
1715        if header_id <= 31
1716            || EXTRA_FIELD_MAPPING
1717                .iter()
1718                .any(|&mapped| mapped == header_id)
1719        {
1720            return Err(ZipError::Io(io::Error::new(
1721                io::ErrorKind::Other,
1722                format!(
1723                    "Extra data header ID {header_id:#06} requires crate feature \"unreserved\"",
1724                ),
1725            )));
1726        }
1727    }
1728
1729    Ok(())
1730}

1738    writer.write_u64_le(file.compressed_size)?;

1757fn write_central_zip64_extra_field<T: Write>(writer: &mut T, file: &ZipFileData) -> ZipResult<u16> {
1758    // The order of the fields in the zip64 extended
1759    // information record is fixed, but the fields MUST
1760    // only appear if the corresponding Local or Central
1761    // directory record field is set to 0xFFFF or 0xFFFFFFFF.
1762    let mut size = 0;
1763    let uncompressed_size = file.uncompressed_size > spec::ZIP64_BYTES_THR;
1764    let compressed_size = file.compressed_size > spec::ZIP64_BYTES_THR;
1765    let header_start = file.header_start > spec::ZIP64_BYTES_THR;
1766    if uncompressed_size {
1767        size += 8;
1768    }
1769    if compressed_size {
1770        size += 8;
1771    }
1772    if header_start {
1773        size += 8;
1774    }
1775    if size > 0 {
1776        writer.write_u16_le(0x0001)?;
1777        writer.write_u16_le(size)?;
1778        size += 4;
1779
1780        if uncompressed_size {
1781            writer.write_u64_le(file.uncompressed_size)?;
1782        }
1783        if compressed_size {
1784            writer.write_u64_le(file.compressed_size)?;
1785        }
1786        if header_start {
1787            writer.write_u64_le(file.header_start)?;
1788        }
1789        // Excluded fields:
1790        // u32: disk start number
1791    }
1792    Ok(size)
1793}
