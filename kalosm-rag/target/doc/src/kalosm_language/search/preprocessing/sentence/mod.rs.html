8/// The default sentence chunker. Unlike [`SentenceChunker`], this is Send + Sync
9#[derive(Debug, Clone, Copy, Default)]
10pub struct DefaultSentenceChunker;
11
12impl Chunker for DefaultSentenceChunker {
13    type Error<E: Send + Sync + 'static> = E;
14
15    /// Chunk a document into embedded snippets.
16    fn chunk<E: Embedder + Send>(
17        &self,
18        document: &Document,
19        embedder: &E,
20    ) -> impl Future<Output = Result<Vec<Chunk>, Self::Error<E::Error>>> + Send {
21        let default = SentenceChunker::default();
22        // Split the document into sentences. We first just collect the sentences as strings and byte ranges
23        let mut initial_chunks = Vec::new();
24        let body = document.body();
25        let ranges = default.split_sentences(document.body());
26        for chunk in &ranges {
27            initial_chunks.push(body[chunk.clone()].to_string());
28        }
29
30        embed_chunk(embedder, initial_chunks, ranges)
31    }
32}

34/// A [`Chunker`] that splits a string into sentences with a given [SRX](https://www.unicode.org/uli/pas/srx/srx20.html) rules.
35///
36/// This uses the [srx](https://crates.io/crates/srx) crate to parse and apply the rules.
37#[derive(Debug, Clone)]
38pub struct SentenceChunker {
39    srx: Rc<SRX>,
40}
41
42impl SentenceChunker {
43    /// Create a new sentence chunker from a xml rules string
44    pub fn new(rules: &str) -> Self {
45        Self {
46            srx: SRX::from_str(rules)
47                .expect("the rules file is valid")
48                .into(),
49        }
50    }
51
52    /// Create a new sentence chunker from anything that implements [`std::io::Read`](std::io::Read) in the srx rules format
53    pub fn load(reader: impl std::io::Read) -> Result<Self, srx::Error> {
54        Ok(Self {
55            srx: SRX::from_reader(reader)?.into(),
56        })
57    }
58
59    /// Split the body of a document into a list of ranges with sentences
60    pub fn split_sentences(&self, string: &str) -> Vec<std::ops::Range<usize>> {
61        // Try to autodetect the language of the document
62        let language = whatlang::detect_lang(string)
63            .map(|lang_code| lang_code.code())
64            .unwrap_or("en");
65
66        // Then get the language specific rules to split the document into sentences
67        let rules = self.srx.language_rules(language);
68
69        rules.split_ranges(string)
70    }
71}

73impl Default for SentenceChunker {
74    fn default() -> Self {
75        // The rules are expensive to parse (~1 second), so we cache them in a static once cell
76        thread_local! {
77            static DEFAULT_RULES: OnceCell<Rc<SRX>> = const { OnceCell::new() };
78        }
79
80        let rules = DEFAULT_RULES.with(|default| {
81            default
82                .get_or_init(|| {
83                    // Defaults to the language tool ruleset: https://github.com/languagetool-org/languagetool/blob/master/languagetool-core/src/main/resources/org/languagetool/resource/segment.srx
84                    let rules = SRX::from_str(include_str!("./assets/segment.srx"))
85                        .expect("the rules file is valid");
86                    Rc::new(rules)
87                })
88                .clone()
89        });
90
91        Self { srx: rules }
92    }
93}

95/// A strategy for chunking a document into smaller pieces.
96impl Chunker for SentenceChunker {
97    type Error<E: Send + Sync + 'static> = E;
98
99    /// Chunk a document into embedded snippets.
100    fn chunk<E: Embedder + Send>(
101        &self,
102        document: &Document,
103        embedder: &E,
104    ) -> impl Future<Output = Result<Vec<Chunk>, Self::Error<E::Error>>> + Send {
105        // Split the document into sentences. We first just collect the sentences as strings and byte ranges
106        let mut initial_chunks = Vec::new();
107        let body = document.body();
108        let ranges = self.split_sentences(document.body());
109        for chunk in &ranges {
110            initial_chunks.push(body[chunk.clone()].to_string());
111        }
112
113        embed_chunk(embedder, initial_chunks, ranges)
114    }
115}
116
117async fn embed_chunk<E: Embedder + Send>(
118    embedder: &E,
119    initial_chunks: Vec<String>,
120    ranges: Vec<std::ops::Range<usize>>,
121) -> Result<Vec<Chunk>, E::Error> {
122    // Next embed them all in one big batch
123    let embeddings = embedder.embed_vec(initial_chunks).await?;
124
125    // Now merge the embeddings and ranges into chunks
126    let mut chunks = Vec::new();
127    for (embedding, chunk) in embeddings.into_iter().zip(ranges) {
128        let chunk = Chunk {
129            byte_range: chunk,
130            embeddings: vec![embedding],
131        };
132        chunks.push(chunk);
133    }
134
135    Ok(chunks)
136}
