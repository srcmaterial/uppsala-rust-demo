12/// A pdf document that can be read from the file system.
13#[derive(Debug, Clone)]
14pub struct PdfDocument {
15    path: PathBuf,
16}
17
18impl TryFrom<PathBuf> for PdfDocument {
19    type Error = FsDocumentError;
20
21    fn try_from(path: PathBuf) -> Result<Self, Self::Error> {
22        if !path.is_file() {
23            return Err(std::io::Error::from(std::io::ErrorKind::NotFound).into());
24        }
25        if path.extension().unwrap() != "pdf" {
26            return Err(FsDocumentError::WrongFileType);
27        }
28        Ok(Self { path })
29    }
30}
31
32impl IntoDocument for PdfDocument {
33    type Error = FsDocumentError<lopdf::Error>;
34
35    async fn into_document(self) -> Result<Document, Self::Error> {
36        let path = &self.path;
37
38        let doc = load_pdf(&self.path).await?;
39        let title = doc
40            .get_toc()
41            .map_err(FsDocumentError::Decode)?
42            .toc
43            .into_iter()
44            .min_by_key(|toc| toc.level)
45            .map(|toc| toc.title.to_string())
46            .unwrap_or_default();
47        let text = get_pdf_text(&doc)?;
48        if !text.errors.is_empty() {
49            tracing::error!(
50                "Encountered errors while extracting text from PDF at {path:?}: {:?}",
51                text.errors
52            );
53        }
54
55        let mut all_text = String::new();
56        for paragraph in text.text.values().flatten() {
57            all_text.push_str(paragraph);
58            all_text.push('\n');
59        }
60
61        Ok(Document::from_parts(title, all_text))
62    }
63}

90#[derive(Debug, Deserialize, Serialize)]
91struct PdfText {
92    text: BTreeMap<u32, Vec<String>>, // Key is page number
93    errors: Vec<String>,
94}
95
96fn filter_func(object_id: (u32, u16), object: &mut Object) -> Option<((u32, u16), Object)> {
97    if IGNORE.contains(&object.type_name().unwrap_or_default()) {
98        return None;
99    }
100    if let Ok(d) = object.as_dict_mut() {
101        d.remove(b"Producer");
102        d.remove(b"ModDate");
103        d.remove(b"Creator");
104        d.remove(b"ProcSet");
105        d.remove(b"Procset");
106        d.remove(b"XObject");
107        d.remove(b"MediaBox");
108        d.remove(b"Annots");
109        if d.is_empty() {
110            return None;
111        }
112    }
113    Some((object_id, object.to_owned()))
114}

116async fn load_pdf<P: AsRef<Path>>(path: P) -> Result<PdfDoc, Error> {
117    PdfDoc::load_filtered(path, filter_func)
118        .await
119        .map_err(|e| Error::new(ErrorKind::Other, e.to_string()))
120}

122fn get_pdf_text(doc: &PdfDoc) -> Result<PdfText, Error> {
123    let mut pdf_text: PdfText = PdfText {
124        text: BTreeMap::new(),
125        errors: Vec::new(),
126    };
127    let pages: Vec<Result<(u32, Vec<String>), Error>> = doc
128        .get_pages()
129        .into_iter()
130        .map(
131            |(page_num, page_id): (u32, (u32, u16))| -> Result<(u32, Vec<String>), Error> {
132                let text = doc.extract_text(&[page_num]).map_err(|e| {
133                    Error::new(
134                        ErrorKind::Other,
135                        format!("Failed to extract text from page {page_num} id={page_id:?}: {e:}"),
136                    )
137                })?;
138                Ok((
139                    page_num,
140                    text.split('\n')
141                        .map(|s| s.trim_end().to_string())
142                        .collect::<Vec<String>>(),
143                ))
144            },
145        )
146        .collect();
147    for page in pages {
148        match page {
149            Ok((page_num, lines)) => {
150                pdf_text.text.insert(page_num, lines);
151            }
152            Err(e) => {
153                pdf_text.errors.push(e.to_string());
154            }
155        }
156    }
157    Ok(pdf_text)
158}
