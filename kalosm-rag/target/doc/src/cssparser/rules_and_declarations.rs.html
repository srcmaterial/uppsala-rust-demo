13/// Parse `!important`.
14///
15/// Typical usage is `input.try_parse(parse_important).is_ok()`
16/// at the end of a `DeclarationParser::parse_value` implementation.
17pub fn parse_important<'i, 't>(input: &mut Parser<'i, 't>) -> Result<(), BasicParseError<'i>> {
18    input.expect_delim('!')?;
19    input.expect_ident_matching("important")
20}

22/// A trait to provide various parsing of declaration values.
23///
24/// For example, there could be different implementations for property declarations in style rules
25/// and for descriptors in `@font-face` rules.
26pub trait DeclarationParser<'i> {
27    /// The finished representation of a declaration.
28    type Declaration;
29
30    /// The error type that is included in the ParseError value that can be returned.
31    type Error: 'i;
32
33    /// Parse the value of a declaration with the given `name`.
34    ///
35    /// Return the finished representation for the declaration
36    /// as returned by `DeclarationListParser::next`,
37    /// or `Err(())` to ignore the entire declaration as invalid.
38    ///
39    /// Declaration name matching should be case-insensitive in the ASCII range.
40    /// This can be done with `std::ascii::Ascii::eq_ignore_ascii_case`,
41    /// or with the `match_ignore_ascii_case!` macro.
42    ///
43    /// The given `input` is a "delimited" parser
44    /// that ends wherever the declaration value should end.
45    /// (In declaration lists, before the next semicolon or end of the current block.)
46    ///
47    /// If `!important` can be used in a given context,
48    /// `input.try_parse(parse_important).is_ok()` should be used at the end
49    /// of the implementation of this method and the result should be part of the return value.
50    fn parse_value<'t>(
51        &mut self,
52        name: CowRcStr<'i>,
53        input: &mut Parser<'i, 't>,
54    ) -> Result<Self::Declaration, ParseError<'i, Self::Error>> {
55        Err(input.new_error(BasicParseErrorKind::UnexpectedToken(Token::Ident(name))))
56    }
57}

59/// A trait to provide various parsing of at-rules.
60///
61/// For example, there could be different implementations for top-level at-rules
62/// (`@media`, `@font-face`, â€¦)
63/// and for page-margin rules inside `@page`.
64///
65/// Default implementations that reject all at-rules are provided,
66/// so that `impl AtRuleParser<(), ()> for ... {}` can be used
67/// for using `DeclarationListParser` to parse a declarations list with only qualified rules.
68pub trait AtRuleParser<'i> {
69    /// The intermediate representation of prelude of an at-rule.
70    type Prelude;
71
72    /// The finished representation of an at-rule.
73    type AtRule;
74
75    /// The error type that is included in the ParseError value that can be returned.
76    type Error: 'i;
77
78    /// Parse the prelude of an at-rule with the given `name`.
79    ///
80    /// Return the representation of the prelude and the type of at-rule,
81    /// or `Err(())` to ignore the entire at-rule as invalid.
82    ///
83    /// The prelude is the part after the at-keyword
84    /// and before the `;` semicolon or `{ /* ... */ }` block.
85    ///
86    /// At-rule name matching should be case-insensitive in the ASCII range.
87    /// This can be done with `std::ascii::Ascii::eq_ignore_ascii_case`,
88    /// or with the `match_ignore_ascii_case!` macro.
89    ///
90    /// The given `input` is a "delimited" parser
91    /// that ends wherever the prelude should end.
92    /// (Before the next semicolon, the next `{`, or the end of the current block.)
93    fn parse_prelude<'t>(
94        &mut self,
95        name: CowRcStr<'i>,
96        input: &mut Parser<'i, 't>,
97    ) -> Result<Self::Prelude, ParseError<'i, Self::Error>> {
98        Err(input.new_error(BasicParseErrorKind::AtRuleInvalid(name)))
99    }
100
101    /// End an at-rule which doesn't have block. Return the finished
102    /// representation of the at-rule.
103    ///
104    /// The location passed in is source location of the start of the prelude.
105    ///
106    /// This is only called when `parse_prelude` returned `WithoutBlock`, and
107    /// either the `;` semicolon indeed follows the prelude, or parser is at
108    /// the end of the input.
109    fn rule_without_block(
110        &mut self,
111        prelude: Self::Prelude,
112        start: &ParserState,
113    ) -> Result<Self::AtRule, ()> {
114        let _ = prelude;
115        let _ = start;
116        Err(())
117    }
118
119    /// Parse the content of a `{ /* ... */ }` block for the body of the at-rule.
120    ///
121    /// The location passed in is source location of the start of the prelude.
122    ///
123    /// Return the finished representation of the at-rule
124    /// as returned by `RuleListParser::next` or `DeclarationListParser::next`,
125    /// or `Err(())` to ignore the entire at-rule as invalid.
126    ///
127    /// This is only called when `parse_prelude` returned `WithBlock`, and a block
128    /// was indeed found following the prelude.
129    fn parse_block<'t>(
130        &mut self,
131        prelude: Self::Prelude,
132        start: &ParserState,
133        input: &mut Parser<'i, 't>,
134    ) -> Result<Self::AtRule, ParseError<'i, Self::Error>> {
135        let _ = prelude;
136        let _ = start;
137        Err(input.new_error(BasicParseErrorKind::AtRuleBodyInvalid))
138    }
139}

141/// A trait to provide various parsing of qualified rules.
142///
143/// For example, there could be different implementations for top-level qualified rules (i.e. style
144/// rules with Selectors as prelude) and for qualified rules inside `@keyframes` (keyframe rules
145/// with keyframe selectors as prelude).
146///
147/// Default implementations that reject all qualified rules are provided, so that
148/// `impl QualifiedRuleParser<(), ()> for ... {}` can be used for example for using
149/// `RuleListParser` to parse a rule list with only at-rules (such as inside
150/// `@font-feature-values`).
151pub trait QualifiedRuleParser<'i> {
152    /// The intermediate representation of a qualified rule prelude.
153    type Prelude;
154
155    /// The finished representation of a qualified rule.
156    type QualifiedRule;
157
158    /// The error type that is included in the ParseError value that can be returned.
159    type Error: 'i;
160
161    /// Parse the prelude of a qualified rule. For style rules, this is as Selector list.
162    ///
163    /// Return the representation of the prelude,
164    /// or `Err(())` to ignore the entire at-rule as invalid.
165    ///
166    /// The prelude is the part before the `{ /* ... */ }` block.
167    ///
168    /// The given `input` is a "delimited" parser
169    /// that ends where the prelude should end (before the next `{`).
170    fn parse_prelude<'t>(
171        &mut self,
172        input: &mut Parser<'i, 't>,
173    ) -> Result<Self::Prelude, ParseError<'i, Self::Error>> {
174        Err(input.new_error(BasicParseErrorKind::QualifiedRuleInvalid))
175    }
176
177    /// Parse the content of a `{ /* ... */ }` block for the body of the qualified rule.
178    ///
179    /// The location passed in is source location of the start of the prelude.
180    ///
181    /// Return the finished representation of the qualified rule
182    /// as returned by `RuleListParser::next`,
183    /// or `Err(())` to ignore the entire at-rule as invalid.
184    fn parse_block<'t>(
185        &mut self,
186        prelude: Self::Prelude,
187        start: &ParserState,
188        input: &mut Parser<'i, 't>,
189    ) -> Result<Self::QualifiedRule, ParseError<'i, Self::Error>> {
190        let _ = prelude;
191        let _ = start;
192        Err(input.new_error(BasicParseErrorKind::QualifiedRuleInvalid))
193    }
194}

196/// Provides an iterator for rule bodies and declaration lists.
197pub struct RuleBodyParser<'i, 't, 'a, P, I, E> {
198    /// The input given to the parser.
199    pub input: &'a mut Parser<'i, 't>,
200    /// The parser given to `DeclarationListParser::new`
201    pub parser: &'a mut P,
202
203    _phantom: std::marker::PhantomData<(I, E)>,
204}

206/// A parser for a rule body item.
207pub trait RuleBodyItemParser<'i, DeclOrRule, Error: 'i>:
208    DeclarationParser<'i, Declaration = DeclOrRule, Error = Error>
209    + QualifiedRuleParser<'i, QualifiedRule = DeclOrRule, Error = Error>
210    + AtRuleParser<'i, AtRule = DeclOrRule, Error = Error>
211{
212    /// Whether we should attempt to parse declarations. If you know you won't, returning false
213    /// here is slightly faster.
214    fn parse_declarations(&self) -> bool;
215    /// Whether we should attempt to parse qualified rules. If you know you won't, returning false
216    /// would be slightly faster.
217    fn parse_qualified(&self) -> bool;
218}

220impl<'i, 't, 'a, P, I, E> RuleBodyParser<'i, 't, 'a, P, I, E> {
221    /// Create a new `DeclarationListParser` for the given `input` and `parser`.
222    ///
223    /// Note that all CSS declaration lists can on principle contain at-rules.
224    /// Even if no such valid at-rule exists (yet),
225    /// this affects error handling: at-rules end at `{}` blocks, not just semicolons.
226    ///
227    /// The given `parser` therefore needs to implement
228    /// both `DeclarationParser` and `AtRuleParser` traits.
229    /// However, the latter can be an empty `impl`
230    /// since `AtRuleParser` provides default implementations of its methods.
231    ///
232    /// The return type for finished declarations and at-rules also needs to be the same,
233    /// since `<DeclarationListParser as Iterator>::next` can return either.
234    /// It could be a custom enum.
235    pub fn new(input: &'a mut Parser<'i, 't>, parser: &'a mut P) -> Self {
236        Self {
237            input,
238            parser,
239            _phantom: std::marker::PhantomData,
240        }
241    }
242}

244/// https://drafts.csswg.org/css-syntax/#consume-a-blocks-contents
245impl<'i, 't, 'a, I, P, E: 'i> Iterator for RuleBodyParser<'i, 't, 'a, P, I, E>
246where
247    P: RuleBodyItemParser<'i, I, E>,
248{
249    type Item = Result<I, (ParseError<'i, E>, &'i str)>;
250
251    fn next(&mut self) -> Option<Self::Item> {
252        loop {
253            self.input.skip_whitespace();
254            let start = self.input.state();
255            match self.input.next_including_whitespace_and_comments().ok()? {
256                Token::CloseCurlyBracket |
257                Token::WhiteSpace(..) |
258                Token::Semicolon |
259                Token::Comment(..) => continue,
260                Token::AtKeyword(ref name) => {
261                    let name = name.clone();
262                    return Some(parse_at_rule(&start, name, self.input, &mut *self.parser));
263                }
264                // https://drafts.csswg.org/css-syntax/#consume-a-declaration bails out just to
265                // keep parsing as a qualified rule if the token is not an ident, so we implement
266                // that in a slightly more straight-forward way
267                Token::Ident(ref name) if self.parser.parse_declarations() => {
268                    let name = name.clone();
269                    let result = {
270                        let parser = &mut self.parser;
271                        parse_until_after(self.input, Delimiter::Semicolon, |input| {
272                            input.expect_colon()?;
273                            parser.parse_value(name, input)
274                        })
275                    };
276                    if result.is_err() && self.parser.parse_qualified() {
277                        self.input.reset(&start);
278                        // We ignore the resulting error here. The property declaration parse error
279                        // is likely to be more relevant.
280                        if let Ok(qual) = parse_qualified_rule(
281                            &start,
282                            self.input,
283                            &mut *self.parser,
284                            Delimiter::Semicolon | Delimiter::CurlyBracketBlock,
285                        ) {
286                            return Some(Ok(qual))
287                        }
288                    }
289
290                    return Some(result.map_err(|e| (e, self.input.slice_from(start.position()))));
291                }
292                token => {
293                    let result = if self.parser.parse_qualified() {
294                        self.input.reset(&start);
295                        let delimiters = if self.parser.parse_declarations() {
296                            Delimiter::Semicolon | Delimiter::CurlyBracketBlock
297                        } else {
298                            Delimiter::CurlyBracketBlock
299                        };
300                        parse_qualified_rule(&start, self.input, &mut *self.parser, delimiters)
301                    } else {
302                        let token = token.clone();
303                        self.input.parse_until_after(Delimiter::Semicolon, |_| {
304                            Err(start.source_location().new_unexpected_token_error(token))
305                        })
306                    };
307                    return Some(result.map_err(|e| (e, self.input.slice_from(start.position()))));
308                }
309            }
310        }
311    }
312}

314/// Provides an iterator for rule list parsing at the top-level of a stylesheet.
315pub struct StyleSheetParser<'i, 't, 'a, P> {
316    /// The input given.
317    pub input: &'a mut Parser<'i, 't>,
318
319    /// The parser given.
320    pub parser: &'a mut P,
321
322    any_rule_so_far: bool,
323}
324
325impl<'i, 't, 'a, R, P, E: 'i> StyleSheetParser<'i, 't, 'a, P>
326where
327    P: QualifiedRuleParser<'i, QualifiedRule = R, Error = E>
328        + AtRuleParser<'i, AtRule = R, Error = E>,
329{
330    /// The given `parser` needs to implement both `QualifiedRuleParser` and `AtRuleParser` traits.
331    /// However, either of them can be an empty `impl` since the traits provide default
332    /// implementations of their methods.
333    ///
334    /// The return type for finished qualified rules and at-rules also needs to be the same,
335    /// since `<RuleListParser as Iterator>::next` can return either. It could be a custom enum.
336    pub fn new(input: &'a mut Parser<'i, 't>, parser: &'a mut P) -> Self {
337        Self {
338            input,
339            parser,
340            any_rule_so_far: false,
341        }
342    }
343}

345/// `RuleListParser` is an iterator that yields `Ok(_)` for a rule or `Err(())` for an invalid one.
346impl<'i, 't, 'a, R, P, E: 'i> Iterator for StyleSheetParser<'i, 't, 'a, P>
347where
348    P: QualifiedRuleParser<'i, QualifiedRule = R, Error = E>
349        + AtRuleParser<'i, AtRule = R, Error = E>,
350{
351    type Item = Result<R, (ParseError<'i, E>, &'i str)>;
352
353    fn next(&mut self) -> Option<Self::Item> {
354        loop {
355            self.input.skip_cdc_and_cdo();
356            let start = self.input.state();
357            let at_keyword = match self.input.next_byte()? {
358                b'@' => match self.input.next_including_whitespace_and_comments() {
359                    Ok(&Token::AtKeyword(ref name)) => Some(name.clone()),
360                    _ => {
361                        self.input.reset(&start);
362                        None
363                    }
364                },
365                _ => None,
366            };
367
368            if let Some(name) = at_keyword {
369                let first_stylesheet_rule = !self.any_rule_so_far;
370                self.any_rule_so_far = true;
371                if first_stylesheet_rule && name.eq_ignore_ascii_case("charset") {
372                    let delimiters = Delimiter::Semicolon | Delimiter::CurlyBracketBlock;
373                    let _: Result<(), ParseError<()>> =
374                        self.input.parse_until_after(delimiters, |_| Ok(()));
375                } else {
376                    return Some(parse_at_rule(
377                        &start,
378                        name.clone(),
379                        self.input,
380                        &mut *self.parser,
381                    ));
382                }
383            } else {
384                self.any_rule_so_far = true;
385                let result = parse_qualified_rule(
386                    &start,
387                    self.input,
388                    &mut *self.parser,
389                    Delimiter::CurlyBracketBlock,
390                );
391                return Some(result.map_err(|e| (e, self.input.slice_from(start.position()))));
392            }
393        }
394    }
395}

397/// Parse a single declaration, such as an `( /* ... */ )` parenthesis in an `@supports` prelude.
398pub fn parse_one_declaration<'i, 't, P, E>(
399    input: &mut Parser<'i, 't>,
400    parser: &mut P,
401) -> Result<<P as DeclarationParser<'i>>::Declaration, (ParseError<'i, E>, &'i str)>
402where
403    P: DeclarationParser<'i, Error = E>,
404{
405    let start_position = input.position();
406    input
407        .parse_entirely(|input| {
408            let name = input.expect_ident()?.clone();
409            input.expect_colon()?;
410            parser.parse_value(name, input)
411        })
412        .map_err(|e| (e, input.slice_from(start_position)))
413}

415/// Parse a single rule, such as for CSSOMâ€™s `CSSStyleSheet.insertRule`.
416pub fn parse_one_rule<'i, 't, R, P, E>(
417    input: &mut Parser<'i, 't>,
418    parser: &mut P,
419) -> Result<R, ParseError<'i, E>>
420where
421    P: QualifiedRuleParser<'i, QualifiedRule = R, Error = E>
422        + AtRuleParser<'i, AtRule = R, Error = E>,
423{
424    input.parse_entirely(|input| {
425        input.skip_whitespace();
426        let start = input.state();
427        let at_keyword = if input.next_byte() == Some(b'@') {
428            match *input.next_including_whitespace_and_comments()? {
429                Token::AtKeyword(ref name) => Some(name.clone()),
430                _ => {
431                    input.reset(&start);
432                    None
433                }
434            }
435        } else {
436            None
437        };
438
439        if let Some(name) = at_keyword {
440            parse_at_rule(&start, name, input, parser).map_err(|e| e.0)
441        } else {
442            parse_qualified_rule(&start, input, parser, Delimiter::CurlyBracketBlock)
443        }
444    })
445}

447fn parse_at_rule<'i, 't, P, E>(
448    start: &ParserState,
449    name: CowRcStr<'i>,
450    input: &mut Parser<'i, 't>,
451    parser: &mut P,
452) -> Result<<P as AtRuleParser<'i>>::AtRule, (ParseError<'i, E>, &'i str)>
453where
454    P: AtRuleParser<'i, Error = E>,
455{
456    let delimiters = Delimiter::Semicolon | Delimiter::CurlyBracketBlock;
457    let result = parse_until_before(input, delimiters, |input| parser.parse_prelude(name, input));
458    match result {
459        Ok(prelude) => {
460            let result = match input.next() {
461                Ok(&Token::Semicolon) | Err(_) => parser
462                    .rule_without_block(prelude, start)
463                    .map_err(|()| input.new_unexpected_token_error(Token::Semicolon)),
464                Ok(&Token::CurlyBracketBlock) => {
465                    parse_nested_block(input, |input| parser.parse_block(prelude, start, input))
466                }
467                Ok(_) => unreachable!(),
468            };
469            result.map_err(|e| (e, input.slice_from(start.position())))
470        }
471        Err(error) => {
472            let end_position = input.position();
473            match input.next() {
474                Ok(&Token::CurlyBracketBlock) | Ok(&Token::Semicolon) | Err(_) => {}
475                _ => unreachable!(),
476            };
477            Err((error, input.slice(start.position()..end_position)))
478        }
479    }
480}

482fn parse_qualified_rule<'i, 't, P, E>(
483    start: &ParserState,
484    input: &mut Parser<'i, 't>,
485    parser: &mut P,
486    delimiters: Delimiters,
487) -> Result<<P as QualifiedRuleParser<'i>>::QualifiedRule, ParseError<'i, E>>
488where
489    P: QualifiedRuleParser<'i, Error = E>,
490{
491    let prelude = parse_until_before(input, delimiters, |input| parser.parse_prelude(input));
492    input.expect_curly_bracket_block()?;
493    // Do this here so that we consume the `{` even if the prelude is `Err`.
494    let prelude = prelude?;
495    parse_nested_block(input, |input| parser.parse_block(prelude, &start, input))
496}
