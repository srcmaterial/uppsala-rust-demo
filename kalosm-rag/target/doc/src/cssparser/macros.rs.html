7/// Expands to a `match` expression with string patterns,
8/// matching case-insensitively in the ASCII range.
9///
10/// The patterns must not contain ASCII upper case letters. (They must be already be lower-cased.)
11///
12/// # Example
13///
14/// ```rust
15/// # fn main() {}  // Make doctest not wrap everything in its own main
16/// # fn dummy(function_name: &String) { let _ =
17/// cssparser::match_ignore_ascii_case! { &function_name,
18///     "rgb" => parse_rgb(..),
19/// #   #[cfg(not(something))]
20///     "rgba" => parse_rgba(..),
21///     "hsl" => parse_hsl(..),
22///     "hsla" => parse_hsla(..),
23///     _ => Err(format!("unknown function: {}", function_name))
24/// }
25/// # ;}
26/// # use std::ops::RangeFull;
27/// # fn parse_rgb(_: RangeFull) -> Result<(), String> { Ok(()) }
28/// # fn parse_rgba(_: RangeFull) -> Result<(), String> { Ok(()) }
29/// # fn parse_hsl(_: RangeFull) -> Result<(), String> { Ok(()) }
30/// # fn parse_hsla(_: RangeFull) -> Result<(), String> { Ok(()) }
31/// ```
32#[macro_export]
33macro_rules! match_ignore_ascii_case {
34    ( $input:expr,
35        $(
36            $( #[$meta: meta] )*
37            $( $pattern: pat )|+ $( if $guard: expr )? => $then: expr
38        ),+
39        $(,)?
40    ) => {
41        {
42            // This dummy module works around the feature gate
43            // `error[E0658]: procedural macros cannot be expanded to statements`
44            // by forcing the macro to be in an item context
45            // rather than expression/statement context,
46            // even though the macro only expands to items.
47            mod cssparser_internal {
48                $crate::_cssparser_internal_max_len! {
49                    $( $( $pattern )+ )+
50                }
51            }
52            $crate::_cssparser_internal_to_lowercase!($input, cssparser_internal::MAX_LENGTH => lowercase);
53            // "A" is a short string that we know is different for every string pattern,
54            // since weâ€™ve verified that none of them include ASCII upper case letters.
55            match lowercase.unwrap_or("A") {
56                $(
57                    $( #[$meta] )*
58                    $( $pattern )|+ $( if $guard )? => $then,
59                )+
60            }
61        }
62    };
63}

65/// Define a function `$name(&str) -> Option<&'static $ValueType>`
66///
67/// The function finds a match for the input string
68/// in a [`phf` map](https://github.com/sfackler/rust-phf)
69/// and returns a reference to the corresponding value.
70/// Matching is case-insensitive in the ASCII range.
71///
72/// ## Example:
73///
74/// ```rust
75/// # fn main() {}  // Make doctest not wrap everything in its own main
76///
77/// fn color_rgb(input: &str) -> Option<(u8, u8, u8)> {
78///     cssparser::ascii_case_insensitive_phf_map! {
79///         keyword -> (u8, u8, u8) = {
80///             "red" => (255, 0, 0),
81///             "green" => (0, 255, 0),
82///             "blue" => (0, 0, 255),
83///         }
84///     }
85///     keyword(input).cloned()
86/// }
87#[macro_export]
88macro_rules! ascii_case_insensitive_phf_map {
89    ($name: ident -> $ValueType: ty = { $( $key: tt => $value: expr ),+ }) => {
90        ascii_case_insensitive_phf_map!($name -> $ValueType = { $( $key => $value, )+ })
91    };
92    ($name: ident -> $ValueType: ty = { $( $key: tt => $value: expr, )+ }) => {
93        pub fn $name(input: &str) -> Option<&'static $ValueType> {
94            // This dummy module works around a feature gate,
95            // see comment on the similar module in `match_ignore_ascii_case!` above.
96            mod _cssparser_internal {
97                $crate::_cssparser_internal_max_len! {
98                    $( $key )+
99                }
100            }
101            use $crate::_cssparser_internal_phf as phf;
102            static MAP: phf::Map<&'static str, $ValueType> = phf::phf_map! {
103                $(
104                    $key => $value,
105                )*
106            };
107            $crate::_cssparser_internal_to_lowercase!(input, _cssparser_internal::MAX_LENGTH => lowercase);
108            lowercase.and_then(|s| MAP.get(s))
109        }
110    }
111}

113/// Create a new array of MaybeUninit<T> items, in an uninitialized state.
114#[inline(always)]
115pub fn _cssparser_internal_create_uninit_array<const N: usize>() -> [MaybeUninit<u8>; N] {
116    unsafe {
117        // SAFETY: An uninitialized `[MaybeUninit<_>; LEN]` is valid.
118        // See: https://doc.rust-lang.org/stable/core/mem/union.MaybeUninit.html#method.uninit_array
119        MaybeUninit::<[MaybeUninit<u8>; N]>::uninit().assume_init()
120    }
121}

123/// Implementation detail of match_ignore_ascii_case! and ascii_case_insensitive_phf_map! macros.
124///
125/// **This macro is not part of the public API. It can change or be removed between any versions.**
126///
127/// Define a local variable named `$output`
128/// and assign it the result of calling `_cssparser_internal_to_lowercase`
129/// with a stack-allocated buffer of length `$BUFFER_SIZE`.
130#[macro_export]
131#[doc(hidden)]
132macro_rules! _cssparser_internal_to_lowercase {
133    ($input: expr, $BUFFER_SIZE: expr => $output: ident) => {
134        let mut buffer = $crate::_cssparser_internal_create_uninit_array::<{ $BUFFER_SIZE }>();
135        let input: &str = $input;
136        let $output = $crate::_cssparser_internal_to_lowercase(&mut buffer, input);
137    };
138}

140/// Implementation detail of match_ignore_ascii_case! and ascii_case_insensitive_phf_map! macros.
141///
142/// **This function is not part of the public API. It can change or be removed between any versions.**
143///
144/// If `input` is larger than buffer, return `None`.
145/// Otherwise, return `input` ASCII-lowercased, using `buffer` as temporary space if necessary.
146#[doc(hidden)]
147#[allow(non_snake_case)]
148#[inline]
149pub fn _cssparser_internal_to_lowercase<'a>(
150    buffer: &'a mut [MaybeUninit<u8>],
151    input: &'a str,
152) -> Option<&'a str> {
153    let buffer = buffer.get_mut(..input.len())?;
154
155    #[cold]
156    fn make_ascii_lowercase<'a>(
157        buffer: &'a mut [MaybeUninit<u8>],
158        input: &'a str,
159        first_uppercase: usize,
160    ) -> &'a str {
161        // This cast doesn't change the pointer's validity
162        // since `u8` has the same layout as `MaybeUninit<u8>`:
163        let input_bytes =
164            unsafe { &*(input.as_bytes() as *const [u8] as *const [MaybeUninit<u8>]) };
165
166        buffer.copy_from_slice(&*input_bytes);
167
168        // Same as above re layout, plus these bytes have been initialized:
169        let buffer = unsafe { &mut *(buffer as *mut [MaybeUninit<u8>] as *mut [u8]) };
170
171        buffer[first_uppercase..].make_ascii_lowercase();
172        // `buffer` was initialized to a copy of `input`
173        // (which is `&str` so well-formed UTF-8)
174        // then ASCII-lowercased (which preserves UTF-8 well-formedness):
175        unsafe { ::std::str::from_utf8_unchecked(buffer) }
176    }
177
178    Some(
179        match input.bytes().position(|byte| matches!(byte, b'A'..=b'Z')) {
180            Some(first_uppercase) => make_ascii_lowercase(buffer, input, first_uppercase),
181            // common case: input is already lower-case
182            None => input,
183        },
184    )
185}
