12/// Trait for things the can serialize themselves in CSS syntax.
13pub trait ToCss {
14    /// Serialize `self` in CSS syntax, writing to `dest`.
15    fn to_css<W>(&self, dest: &mut W) -> fmt::Result
16    where
17        W: fmt::Write;
18
19    /// Serialize `self` in CSS syntax and return a string.
20    ///
21    /// (This is a convenience wrapper for `to_css` and probably should not be overridden.)
22    #[inline]
23    fn to_css_string(&self) -> String {
24        let mut s = String::new();
25        self.to_css(&mut s).unwrap();
26        s
27    }
28}

30#[inline]
31fn write_numeric<W>(value: f32, int_value: Option<i32>, has_sign: bool, dest: &mut W) -> fmt::Result
32where
33    W: fmt::Write,
34{
35    // `value.value >= 0` is true for negative 0.
36    if has_sign && value.is_sign_positive() {
37        dest.write_str("+")?;
38    }
39
40    let notation = if value == 0.0 && value.is_sign_negative() {
41        // Negative zero. Work around #20596.
42        dest.write_str("-0")?;
43        Notation {
44            decimal_point: false,
45            scientific: false,
46        }
47    } else {
48        dtoa_short::write(dest, value)?
49    };
50
51    if int_value.is_none() && value.fract() == 0. {
52        if !notation.decimal_point && !notation.scientific {
53            dest.write_str(".0")?;
54        }
55    }
56    Ok(())
57}

59impl<'a> ToCss for Token<'a> {
60    fn to_css<W>(&self, dest: &mut W) -> fmt::Result
61    where
62        W: fmt::Write,
63    {
64        match *self {
65            Token::Ident(ref value) => serialize_identifier(&**value, dest)?,
66            Token::AtKeyword(ref value) => {
67                dest.write_str("@")?;
68                serialize_identifier(&**value, dest)?;
69            }
70            Token::Hash(ref value) => {
71                dest.write_str("#")?;
72                serialize_name(value, dest)?;
73            }
74            Token::IDHash(ref value) => {
75                dest.write_str("#")?;
76                serialize_identifier(&**value, dest)?;
77            }
78            Token::QuotedString(ref value) => serialize_string(&**value, dest)?,
79            Token::UnquotedUrl(ref value) => {
80                dest.write_str("url(")?;
81                serialize_unquoted_url(&**value, dest)?;
82                dest.write_str(")")?;
83            }
84            Token::Delim(value) => dest.write_char(value)?,
85
86            Token::Number {
87                value,
88                int_value,
89                has_sign,
90            } => write_numeric(value, int_value, has_sign, dest)?,
91            Token::Percentage {
92                unit_value,
93                int_value,
94                has_sign,
95            } => {
96                write_numeric(unit_value * 100., int_value, has_sign, dest)?;
97                dest.write_str("%")?;
98            }
99            Token::Dimension {
100                value,
101                int_value,
102                has_sign,
103                ref unit,
104            } => {
105                write_numeric(value, int_value, has_sign, dest)?;
106                // Disambiguate with scientific notation.
107                let unit = &**unit;
108                // TODO(emilio): This doesn't handle e.g. 100E1m, which gets us
109                // an unit of "E1m"...
110                if unit == "e" || unit == "E" || unit.starts_with("e-") || unit.starts_with("E-") {
111                    dest.write_str("\\65 ")?;
112                    serialize_name(&unit[1..], dest)?;
113                } else {
114                    serialize_identifier(unit, dest)?;
115                }
116            }
117
118            Token::WhiteSpace(content) => dest.write_str(content)?,
119            Token::Comment(content) => {
120                dest.write_str("/*")?;
121                dest.write_str(content)?;
122                dest.write_str("*/")?
123            }
124            Token::Colon => dest.write_str(":")?,
125            Token::Semicolon => dest.write_str(";")?,
126            Token::Comma => dest.write_str(",")?,
127            Token::IncludeMatch => dest.write_str("~=")?,
128            Token::DashMatch => dest.write_str("|=")?,
129            Token::PrefixMatch => dest.write_str("^=")?,
130            Token::SuffixMatch => dest.write_str("$=")?,
131            Token::SubstringMatch => dest.write_str("*=")?,
132            Token::CDO => dest.write_str("<!--")?,
133            Token::CDC => dest.write_str("-->")?,
134
135            Token::Function(ref name) => {
136                serialize_identifier(&**name, dest)?;
137                dest.write_str("(")?;
138            }
139            Token::ParenthesisBlock => dest.write_str("(")?,
140            Token::SquareBracketBlock => dest.write_str("[")?,
141            Token::CurlyBracketBlock => dest.write_str("{")?,
142
143            Token::BadUrl(ref contents) => {
144                dest.write_str("url(")?;
145                dest.write_str(contents)?;
146                dest.write_char(')')?;
147            }
148            Token::BadString(ref value) => {
149                // During tokenization, an unescaped newline after a quote causes
150                // the token to be a BadString instead of a QuotedString.
151                // The BadString token ends just before the newline
152                // (which is in a separate WhiteSpace token),
153                // and therefore does not have a closing quote.
154                dest.write_char('"')?;
155                CssStringWriter::new(dest).write_str(value)?;
156            }
157            Token::CloseParenthesis => dest.write_str(")")?,
158            Token::CloseSquareBracket => dest.write_str("]")?,
159            Token::CloseCurlyBracket => dest.write_str("}")?,
160        }
161        Ok(())
162    }
163}

165fn hex_escape<W>(ascii_byte: u8, dest: &mut W) -> fmt::Result
166where
167    W: fmt::Write,
168{
169    static HEX_DIGITS: &'static [u8; 16] = b"0123456789abcdef";
170    let b3;
171    let b4;
172    let bytes = if ascii_byte > 0x0F {
173        let high = (ascii_byte >> 4) as usize;
174        let low = (ascii_byte & 0x0F) as usize;
175        b4 = [b'\\', HEX_DIGITS[high], HEX_DIGITS[low], b' '];
176        &b4[..]
177    } else {
178        b3 = [b'\\', HEX_DIGITS[ascii_byte as usize], b' '];
179        &b3[..]
180    };
181    dest.write_str(unsafe { str::from_utf8_unchecked(&bytes) })
182}
183
184fn char_escape<W>(ascii_byte: u8, dest: &mut W) -> fmt::Result
185where
186    W: fmt::Write,
187{
188    let bytes = [b'\\', ascii_byte];
189    dest.write_str(unsafe { str::from_utf8_unchecked(&bytes) })
190}

192/// Write a CSS identifier, escaping characters as necessary.
193pub fn serialize_identifier<W>(mut value: &str, dest: &mut W) -> fmt::Result
194where
195    W: fmt::Write,
196{
197    if value.is_empty() {
198        return Ok(());
199    }
200
201    if value.starts_with("--") {
202        dest.write_str("--")?;
203        serialize_name(&value[2..], dest)
204    } else if value == "-" {
205        dest.write_str("\\-")
206    } else {
207        if value.as_bytes()[0] == b'-' {
208            dest.write_str("-")?;
209            value = &value[1..];
210        }
211        if let digit @ b'0'..=b'9' = value.as_bytes()[0] {
212            hex_escape(digit, dest)?;
213            value = &value[1..];
214        }
215        serialize_name(value, dest)
216    }
217}

219/// Write a CSS name, like a custom property name.
220///
221/// You should only use this when you know what you're doing, when in doubt,
222/// consider using `serialize_identifier`.
223pub fn serialize_name<W>(value: &str, dest: &mut W) -> fmt::Result
224where
225    W: fmt::Write,
226{
227    let mut chunk_start = 0;
228    for (i, b) in value.bytes().enumerate() {
229        let escaped = match b {
230            b'0'..=b'9' | b'A'..=b'Z' | b'a'..=b'z' | b'_' | b'-' => continue,
231            _ if !b.is_ascii() => continue,
232            b'\0' => Some("\u{FFFD}"),
233            _ => None,
234        };
235        dest.write_str(&value[chunk_start..i])?;
236        if let Some(escaped) = escaped {
237            dest.write_str(escaped)?;
238        } else if (b >= b'\x01' && b <= b'\x1F') || b == b'\x7F' {
239            hex_escape(b, dest)?;
240        } else {
241            char_escape(b, dest)?;
242        }
243        chunk_start = i + 1;
244    }
245    dest.write_str(&value[chunk_start..])
246}

248fn serialize_unquoted_url<W>(value: &str, dest: &mut W) -> fmt::Result
249where
250    W: fmt::Write,
251{
252    let mut chunk_start = 0;
253    for (i, b) in value.bytes().enumerate() {
254        let hex = match b {
255            b'\0'..=b' ' | b'\x7F' => true,
256            b'(' | b')' | b'"' | b'\'' | b'\\' => false,
257            _ => continue,
258        };
259        dest.write_str(&value[chunk_start..i])?;
260        if hex {
261            hex_escape(b, dest)?;
262        } else {
263            char_escape(b, dest)?;
264        }
265        chunk_start = i + 1;
266    }
267    dest.write_str(&value[chunk_start..])
268}

270/// Write a double-quoted CSS string token, escaping content as necessary.
271pub fn serialize_string<W>(value: &str, dest: &mut W) -> fmt::Result
272where
273    W: fmt::Write,
274{
275    dest.write_str("\"")?;
276    CssStringWriter::new(dest).write_str(value)?;
277    dest.write_str("\"")?;
278    Ok(())
279}

281/// A `fmt::Write` adapter that escapes text for writing as a double-quoted CSS string.
282/// Quotes are not included.
283///
284/// Typical usage:
285///
286/// ```{rust,ignore}
287/// fn write_foo<W>(foo: &Foo, dest: &mut W) -> fmt::Result where W: fmt::Write {
288///     dest.write_str("\"")?;
289///     {
290///         let mut string_dest = CssStringWriter::new(dest);
291///         // Write into string_dest...
292///     }
293///     dest.write_str("\"")?;
294///     Ok(())
295/// }
296/// ```
297pub struct CssStringWriter<'a, W> {
298    inner: &'a mut W,
299}
300
301impl<'a, W> CssStringWriter<'a, W>
302where
303    W: fmt::Write,
304{
305    /// Wrap a text writer to create a `CssStringWriter`.
306    pub fn new(inner: &'a mut W) -> CssStringWriter<'a, W> {
307        CssStringWriter { inner: inner }
308    }
309}
310
311impl<'a, W> fmt::Write for CssStringWriter<'a, W>
312where
313    W: fmt::Write,
314{
315    fn write_str(&mut self, s: &str) -> fmt::Result {
316        let mut chunk_start = 0;
317        for (i, b) in s.bytes().enumerate() {
318            let escaped = match b {
319                b'"' => Some("\\\""),
320                b'\\' => Some("\\\\"),
321                b'\0' => Some("\u{FFFD}"),
322                b'\x01'..=b'\x1F' | b'\x7F' => None,
323                _ => continue,
324            };
325            self.inner.write_str(&s[chunk_start..i])?;
326            match escaped {
327                Some(x) => self.inner.write_str(x)?,
328                None => hex_escape(b, self.inner)?,
329            };
330            chunk_start = i + 1;
331        }
332        self.inner.write_str(&s[chunk_start..])
333    }
334}

375/// A category of token. See the `needs_separator_when_before` method.
376#[derive(Copy, Clone, Eq, PartialEq, Debug)]
377pub struct TokenSerializationType(TokenSerializationTypeVariants);
378
379impl TokenSerializationType {
380    /// Return a value that represents the absence of a token, e.g. before the start of the input.
381    pub fn nothing() -> TokenSerializationType {
382        TokenSerializationType(TokenSerializationTypeVariants::Nothing)
383    }
384
385    /// If this value is `TokenSerializationType::nothing()`, set it to the given value instead.
386    pub fn set_if_nothing(&mut self, new_value: TokenSerializationType) {
387        if self.0 == TokenSerializationTypeVariants::Nothing {
388            self.0 = new_value.0
389        }
390    }
391
392    /// Return true if, when a token of category `self` is serialized just before
393    /// a token of category `other` with no whitespace in between,
394    /// an empty comment `/**/` needs to be inserted between them
395    /// so that they are not re-parsed as a single token.
396    ///
397    /// See https://drafts.csswg.org/css-syntax/#serialization
398    ///
399    /// See https://github.com/w3c/csswg-drafts/issues/4088 for the
400    /// `DelimPercent` bits.
401    pub fn needs_separator_when_before(self, other: TokenSerializationType) -> bool {
402        use self::TokenSerializationTypeVariants::*;
403        match self.0 {
404            Ident => matches!(
405                other.0,
406                Ident
407                    | Function
408                    | UrlOrBadUrl
409                    | DelimMinus
410                    | Number
411                    | Percentage
412                    | Dimension
413                    | CDC
414                    | OpenParen
415            ),
416            AtKeywordOrHash | Dimension => matches!(
417                other.0,
418                Ident | Function | UrlOrBadUrl | DelimMinus | Number | Percentage | Dimension | CDC
419            ),
420            DelimHash | DelimMinus => matches!(
421                other.0,
422                Ident | Function | UrlOrBadUrl | DelimMinus | Number | Percentage | Dimension
423            ),
424            Number => matches!(
425                other.0,
426                Ident
427                    | Function
428                    | UrlOrBadUrl
429                    | DelimMinus
430                    | Number
431                    | Percentage
432                    | DelimPercent
433                    | Dimension
434            ),
435            DelimAt => matches!(other.0, Ident | Function | UrlOrBadUrl | DelimMinus),
436            DelimDotOrPlus => matches!(other.0, Number | Percentage | Dimension),
437            DelimAssorted | DelimAsterisk => matches!(other.0, DelimEquals),
438            DelimBar => matches!(other.0, DelimEquals | DelimBar | DashMatch),
439            DelimSlash => matches!(other.0, DelimAsterisk | SubstringMatch),
440            Nothing | WhiteSpace | Percentage | UrlOrBadUrl | Function | CDC | OpenParen
441            | DashMatch | SubstringMatch | DelimQuestion | DelimEquals | DelimPercent | Other => {
442                false
443            }
444        }
445    }
446}

448#[derive(Copy, Clone, Eq, PartialEq, Debug)]
449enum TokenSerializationTypeVariants {
450    Nothing,
451    WhiteSpace,
452    AtKeywordOrHash,
453    Number,
454    Dimension,
455    Percentage,
456    UrlOrBadUrl,
457    Function,
458    Ident,
459    CDC,
460    DashMatch,
461    SubstringMatch,
462    OpenParen,      // '('
463    DelimHash,      // '#'
464    DelimAt,        // '@'
465    DelimDotOrPlus, // '.', '+'
466    DelimMinus,     // '-'
467    DelimQuestion,  // '?'
468    DelimAssorted,  // '$', '^', '~'
469    DelimEquals,    // '='
470    DelimBar,       // '|'
471    DelimSlash,     // '/'
472    DelimAsterisk,  // '*'
473    DelimPercent,   // '%'
474    Other,          // anything else
475}

477impl<'a> Token<'a> {
478    /// Categorize a token into a type that determines when `/**/` needs to be inserted
479    /// between two tokens when serialized next to each other without whitespace in between.
480    ///
481    /// See the `TokenSerializationType::needs_separator_when_before` method.
482    pub fn serialization_type(&self) -> TokenSerializationType {
483        use self::TokenSerializationTypeVariants::*;
484        TokenSerializationType(match *self {
485            Token::Ident(_) => Ident,
486            Token::AtKeyword(_) | Token::Hash(_) | Token::IDHash(_) => AtKeywordOrHash,
487            Token::UnquotedUrl(_) | Token::BadUrl(_) => UrlOrBadUrl,
488            Token::Delim('#') => DelimHash,
489            Token::Delim('@') => DelimAt,
490            Token::Delim('.') | Token::Delim('+') => DelimDotOrPlus,
491            Token::Delim('-') => DelimMinus,
492            Token::Delim('?') => DelimQuestion,
493            Token::Delim('$') | Token::Delim('^') | Token::Delim('~') => DelimAssorted,
494            Token::Delim('%') => DelimPercent,
495            Token::Delim('=') => DelimEquals,
496            Token::Delim('|') => DelimBar,
497            Token::Delim('/') => DelimSlash,
498            Token::Delim('*') => DelimAsterisk,
499            Token::Number { .. } => Number,
500            Token::Percentage { .. } => Percentage,
501            Token::Dimension { .. } => Dimension,
502            Token::WhiteSpace(_) => WhiteSpace,
503            Token::Comment(_) => DelimSlash,
504            Token::DashMatch => DashMatch,
505            Token::SubstringMatch => SubstringMatch,
506            Token::CDC => CDC,
507            Token::Function(_) => Function,
508            Token::ParenthesisBlock => OpenParen,
509            Token::SquareBracketBlock
510            | Token::CurlyBracketBlock
511            | Token::CloseParenthesis
512            | Token::CloseSquareBracket
513            | Token::CloseCurlyBracket
514            | Token::QuotedString(_)
515            | Token::BadString(_)
516            | Token::Delim(_)
517            | Token::Colon
518            | Token::Semicolon
519            | Token::Comma
520            | Token::CDO
521            | Token::IncludeMatch
522            | Token::PrefixMatch
523            | Token::SuffixMatch => Other,
524        })
525    }
