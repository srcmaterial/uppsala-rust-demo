20impl Embedding {
21    /// Compute the cosine similarity between this embedding and another embedding.
22    pub fn cosine_similarity(&self, other: &Self) -> f32 {
23        let sum_ij = self
24            .embedding
25            .iter()
26            .zip(other.embedding.iter())
27            .map(|(a, b)| a * b)
28            .sum::<f32>();
29        let sum_i2 = other.embedding.iter().map(|a| a * a).sum::<f32>();
30        let sum_j2 = self.embedding.iter().map(|a| a * a).sum::<f32>();
31        sum_ij / (sum_i2 * sum_j2).sqrt()
32    }
33}

115fn embedding_serialization() {

117    let first_float: Vec<f32> = embedding.vector().to_vec();
118    let bytes = postcard::to_stdvec(&embedding).unwrap();
119    let embedding: Embedding = postcard::from_bytes(&bytes).unwrap();
120    let second_float: Vec<f32> = embedding.vector().to_vec();
