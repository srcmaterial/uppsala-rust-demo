299impl<S> DynChatModel for S
300where
301    S: ChatModel<
302        Error: Send + Sync + Error + 'static,
303        ChatSession: ChatSession<Error: Error + Send + Sync + 'static>
304                         + Clone
305                         + Send
306                         + Sync
307                         + 'static,
308    >,
309{
310    fn add_messages_with_callback_boxed<'a>(
311        &'a self,
312        session: &'a mut BoxedChatSession,
313        messages: &[super::ChatMessage],
314        sampler: crate::GenerationParameters,
315        mut on_token: BoxedTokenClosure,
316    ) -> BoxedMaybeFuture<'a> {
317        let session = session.session.as_any_mut();
318
319        let Some(session) = session.downcast_mut::<S::ChatSession>() else {
320            return Box::pin(async move {
321                Err(Box::new(MismatchedSessionType) as Box<dyn Error + Send + Sync>)
322            });
323        };
324        let on_token = move |token: String| {
325            if let Err(err) = on_token(token) {
326                tracing::error!("Error running on_token callback: {}", err);
327            }
328            Ok(())
329        };
330        let future = self.add_messages_with_callback(session, messages, sampler, on_token);
331        // Double box prevents a rust compiler error with lifetimes. See https://github.com/rust-lang/rust/issues/102211
332        let future: Pin<Box<dyn Future<Output = Result<(), _>> + Send>> = Box::pin(future);
333        Box::pin(async move {
334            future
335                .await
336                .map_err(|e| Box::new(e) as Box<dyn Error + Send + Sync + 'static>)
337        })
338    }
339}
