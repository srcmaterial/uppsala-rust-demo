11#[doc = include_str!("../../docs/completion_session.md")]
12pub trait TextCompletionSession {
13    /// The type of error the session may return during operations.
14    type Error: Send + Sync + 'static;
15
16    /// Serialize the session into bytes. This method is identical to [`TextCompletionSession::to_bytes`] except it can re-use an existing [`Vec`] buffer.
17    fn write_to(&self, into: &mut Vec<u8>) -> Result<(), Self::Error>;
18
19    /// # Loading sessions
20    ///
21    /// Sessions can be deserialized to and from bytes using the [`TextCompletionSession::from_bytes`] method.
22    /// Caching a session avoids re-processing the text again when the session is resumed.
23    ///
24    /// ```rust, no_run
25    /// use kalosm::language::*;
26    /// use std::io::Write;
27    ///
28    /// #[tokio::main]
29    /// async fn main() {
30    ///     let mut llm = Llama::new().await.unwrap();
31    ///     let mut session = llm.new_session().unwrap();
32    ///
33    ///     // Feed some text into the session
34    ///     llm.stream_text_with_callback(
35    ///         &mut session,
36    ///         "The capital of France is ",
37    ///         GenerationParameters::new().with_max_length(0),
38    ///         |_| Ok(()),
39    ///     )
40    ///     .await
41    ///     .unwrap();
42    ///
43    ///     // Save the session to bytes
44    ///     let session_as_bytes = session.to_bytes().unwrap();
45    ///
46    ///     // And write those bytes to a file
47    ///     std::fs::write("session.bin", session_as_bytes).unwrap();
48    /// }
49    /// ```
50    fn to_bytes(&self) -> Result<Vec<u8>, Self::Error> {
51        let mut bytes = Vec::new();
52        self.write_to(&mut bytes)?;
53        Ok(bytes)
54    }
55
56    /// # Loading sessions
57    ///
58    /// Sessions can be deserialized to and from bytes using the [`TextCompletionSession::from_bytes`] method.
59    /// Caching a session avoids re-processing the text again when the session is resumed.
60    ///
61    /// ```rust, no_run
62    /// use kalosm::language::*;
63    /// use std::io::Write;
64    ///
65    /// #[tokio::main]
66    /// async fn main() {
67    ///     let mut llm = Llama::new().await.unwrap();
68    ///     // Load a text completion session from a file
69    ///     let mut session =
70    ///         LlamaSession::from_bytes(std::fs::read("session.bin").unwrap().as_slice()).unwrap();
71    ///
72    ///     // Feed some more text into the session
73    ///     llm.stream_text_with_callback(
74    ///         &mut session,
75    ///         "The capital of France is ",
76    ///         GenerationParameters::new(),
77    ///         |token| {
78    ///             println!("{token}");
79    ///             Ok(())
80    ///         },
81    ///     )
82    ///     .await
83    ///     .unwrap();
84    /// }
85    /// ```
86    fn from_bytes(bytes: &[u8]) -> Result<Self, Self::Error>
87    where
88        Self: std::marker::Sized;
89
90    /// # Cloning Sessions
91    ///
92    /// Not all models support cloning sessions, but if a model does support cloning sessions, you can clone a session using the [`TextCompletionSession::try_clone`] method to clone a session state while retaining the original session.
93    ///
94    /// ```rust, no_run
95    /// use kalosm::language::*;
96    /// use std::io::Write;
97    ///
98    /// #[tokio::main]
99    /// async fn main() {
100    ///     let mut llm = Llama::new().await.unwrap();
101    ///     let mut session = llm.new_session().unwrap();
102    ///
103    ///     // Feed some text into the session
104    ///     llm.stream_text_with_callback(
105    ///         &mut session,
106    ///         "The capital of France is ",
107    ///         GenerationParameters::new().with_max_length(0),
108    ///         |_| Ok(()),
109    ///     )
110    ///     .await
111    ///     .unwrap();
112    ///
113    ///     // Clone the session
114    ///     let cloned_session = session.try_clone().unwrap();
115    ///
116    ///     // Feed some more text into the cloned session
117    ///     llm.stream_text_with_callback(
118    ///         &mut session,
119    ///         "The capital of France is ",
120    ///         GenerationParameters::new(),
121    ///         |token| {
122    ///             println!("{token}");
123    ///             Ok(())
124    ///         },
125    ///     )
126    ///     .await
127    ///     .unwrap();
128    /// }
129    /// ```
130    fn try_clone(&self) -> Result<Self, Self::Error>
131    where
132        Self: std::marker::Sized;
133}

165/// A trait for creating a text completion session for a model. While it the core trait
166/// every text completion model must implement, most methods to use models that implement
167/// this trait are implemented in the [`TextCompletionModelExt`] trait.
168///
169/// # Example
170///
171/// ```rust, no_run
172/// use kalosm::language::*;
173///
174/// #[tokio::main]
175/// async fn main() {
176///     // Create a new model which implements the CreateTextCompletionSession trait
177///     let mut llm = Llama::new().await.unwrap();
178///     // Create a new session for the model
179///     let mut session = llm.new_session().unwrap();
180/// }
181/// ```
182pub trait CreateTextCompletionSession {
183    /// The type of error this model may return during operations.
184    type Error: Send + Sync + 'static;
185
186    /// The type of the session that this model uses.
187    type Session: TextCompletionSession;
188
189    /// Create a new session for this model.
190    ///
191    /// # Example
192    /// ```rust, no_run
193    /// # use kalosm::language::*;
194    /// # #[tokio::main]
195    /// # async fn main() {
196    /// // Create a new model which implements the CreateTextCompletionSession trait
197    /// let llm = Llama::new().await.unwrap();
198    /// // Create a new session for the model
199    /// let session = llm.new_session().unwrap();
200    /// # }
201    /// ```
202    fn new_session(&self) -> Result<Self::Session, Self::Error>;
203}

216/// A trait for unstructured text completion models. This trait is required for any text completion models
217/// even if they do not support structured generation. While this trait is implemented for all text completion models,
218/// most methods to use models that implement this trait are implemented in the [`TextCompletionModelExt`] trait.
219///
220/// # Example
221///
222/// ```rust, no_run
223/// use kalosm::language::*;
224///
225/// #[tokio::main]
226/// async fn main() {
227///     // Create a new model which implements the CreateTextCompletionSession trait
228///     let mut llm = Llama::new().await.unwrap();
229///     // Create a new session for the model
230///     let mut session = llm.new_session().unwrap();
231///     // Feed some text into the session using the raw text completion api that accepts a session, prompt, sampler, and on token callback
232///     llm.stream_text_with_callback(&mut session, "The capital of France is ", GenerationParameters::new(), |token| {println!("{token}"); Ok(())}).await.unwrap();
233/// }
234/// ```
235pub trait TextCompletionModel<Sampler = GenerationParameters>: CreateTextCompletionSession {
236    /// Generate text with the given prompt.
237    ///
238    /// See [`TextCompletionModelExt::complete`] for nicer API with an example.
239    fn stream_text_with_callback<'a>(
240        &'a self,
241        session: &'a mut Self::Session,
242        text: &str,
243        sampler: Sampler,
244        on_token: impl FnMut(String) -> Result<(), Self::Error> + Send + Sync + 'static,
245    ) -> impl Future<Output = Result<(), Self::Error>> + Send + 'a;
246}

248/// A trait for text completion models that support structured generation. While this trait is implemented for
249/// all structured text completion models, most methods to use models that implement this trait are implemented
250/// in the [`TextCompletionModelExt`] trait.
251///
252/// # Example
253///
254/// ```rust, no_run
255/// use kalosm::language::*;
256///
257/// #[tokio::main]
258/// async fn main() {
259///     // Create a new model which implements the CreateTextCompletionSession trait
260///     let mut llm = Llama::new().await.unwrap();
261///     // Create a new session for the model
262///     let mut session = llm.new_session().unwrap();
263///     // Create a parser for your data. Different models accept different types of parsers. The Llama model accepts
264///     // any parsers that implements the `Parse` trait.
265///     let parser = i32::new_parser();
266///     // Feed some text into the session using the raw structured text completion api that accepts a session, prompt, sampler, and on token callback
267///     llm.stream_text_with_callback_and_parser(&mut session, "5 * 5 = ", GenerationParameters::new(), parser, |token| {println!("{token}"); Ok(())}).await.unwrap();
268/// }
269/// ```
270pub trait StructuredTextCompletionModel<
271    Constraints: ModelConstraints,
272    Sampler = GenerationParameters,
273>: TextCompletionModel<Sampler>
274{
275    /// Generate text with the given prompt and the given constraints.
276    ///
277    /// See [`TextCompletionModelExt::complete`] for nicer API with an example.
278    fn stream_text_with_callback_and_parser<'a>(
279        &'a self,
280        session: &'a mut Self::Session,
281        text: &str,
282        sampler: Sampler,
283        parser: Constraints,
284        on_token: impl FnMut(String) -> Result<(), Self::Error> + Send + Sync + 'static,
285    ) -> impl Future<Output = Result<Constraints::Output, Self::Error>> + Send + 'a;
