6#[derive(Debug, Eq)]
7struct Merge {
8    pos: usize,
9    rank: u32,
10    new_id: u32,
11}
12
13impl PartialEq for Merge {
14    fn eq(&self, other: &Self) -> bool {
15        self.rank == other.rank && self.pos == other.pos
16    }
17}
18
19impl PartialOrd for Merge {
20    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
21        // By manually implementing this, we make the containing BinaryHeap a
22        // min-heap ordered first on the rank, and the pos otherwise
23        Some(self.cmp(other))
24    }
25}
26
27impl Ord for Merge {
28    fn cmp(&self, other: &Self) -> Ordering {
29        if self.rank != other.rank {
30            other.rank.cmp(&self.rank)
31        } else {
32            other.pos.cmp(&self.pos)
33        }
34    }
35}

37#[derive(Debug, Clone, Copy)]
38struct Symbol {
39    c: u32,
40    prev: isize,
41    next: isize,
42    len: usize,
43}
44impl Symbol {
45    /// Merges the current Symbol with the other one.
46    /// In order to update prev/next, we consider Self to be the Symbol on the left,
47    /// and other to be the next one on the right.
48    pub fn merge_with(&mut self, other: &Self, new_c: u32) {
49        self.c = new_c;
50        self.len += other.len;
51        self.next = other.next;
52    }
53}

87    pub(super) fn add(&mut self, c: u32, byte_len: usize) {
88        let (prev, next) = {
89            let len = self.symbols.len() as isize;
90            if let Some(last) = self.symbols.last_mut() {
91                // Update `next` on the previous one
92                last.next = len;
93                (len - 1, -1)
94            } else {
95                (-1, -1)
96            }
97        };
98        self.symbols.push(Symbol {
99            c,
100            prev,
101            next,
102            len: byte_len,
103        });
104    }
105
106    pub(super) fn merge(
107        &mut self,
108        c1: u32,
109        c2: u32,
110        replacement: u32,
111        max_length: usize,
112    ) -> Vec<(Pair, i32)> {
113        let mut changes: Vec<(Pair, i32)> = vec![];
114        let mut i = 0;
115        loop {
116            if i >= self.symbols.len() {
117                break;
118            }
119
120            // Found a pair
121            if self.symbols[i].c == c1 && i + 1 < self.symbols.len() && self.symbols[i + 1].c == c2
122            {
123                let first = self.symbols[i];
124                let second = self.symbols[i + 1];
125
126                // Remove in place
127                let new_s = Symbol {
128                    c: replacement,
129                    prev: first.prev,
130                    next: second.next,
131                    len: first.len + second.len,
132                };
133
134                // If there are other characters before the pair
135                if i > 0 {
136                    changes.push(((self.symbols[i - 1].c, first.c), -1));
137                    if self.symbols[i - 1].len + new_s.len < max_length {
138                        changes.push(((self.symbols[i - 1].c, replacement), 1));
139                    }
140                }
141
142                self.symbols.insert(i, new_s); // Insert replacement before first char of pair
143                self.symbols.remove(i + 1); // Remove first char of pair
144                self.symbols.remove(i + 1); // And then the second
145
146                // If there are other characters after the pair
147                if i < self.symbols.len() - 1 {
148                    changes.push(((second.c, self.symbols[i + 1].c), -1));
149                    if self.symbols[i + 1].len + new_s.len < max_length {
150                        changes.push(((replacement, self.symbols[i + 1].c), 1));
151                    }
152                }
153            }
154
155            i += 1;
156        }
157
158        changes
159    }

161    pub(super) fn merge_all(&mut self, merges: &HashMap<Pair, (u32, u32)>, dropout: Option<f32>) {
162        let mut queue = BinaryHeap::with_capacity(self.symbols.len());
163        let mut skip = Vec::with_capacity(queue.len());
164
165        queue.extend(
166            self.symbols
167                .windows(2)
168                .enumerate()
169                .filter_map(|(index, window)| {
170                    let pair = (window[0].c, window[1].c);
171                    merges.get(&pair).map(|m| Merge {
172                        pos: index,
173                        rank: m.0,
174                        new_id: m.1,
175                    })
176                }),
177        );
178
179        while let Some(top) = queue.pop() {
180            if dropout
181                .map(|d| thread_rng().gen::<f32>() < d)
182                .unwrap_or(false)
183            {
184                skip.push(top);
185            } else {
186                // Re-insert the skipped elements
187                queue.extend(skip.drain(..));
188
189                if self.symbols[top.pos].len == 0 {
190                    continue;
191                }
192                // Do nothing if we are the last symbol
193                if self.symbols[top.pos].next == -1 {
194                    continue;
195                }
196
197                let next_pos = self.symbols[top.pos].next as usize;
198                let right = self.symbols[next_pos];
199
200                // Make sure we are not processing an expired queue entry
201                let target_new_pair = (self.symbols[top.pos].c, right.c);
202                if merges
203                    .get(&target_new_pair)
204                    .is_none_or(|(_, new_id)| *new_id != top.new_id)
205                {
206                    continue;
207                }
208
209                // Otherwise, let's merge
210                self.symbols[top.pos].merge_with(&right, top.new_id);
211                // Tag the right part as removed
212                self.symbols[next_pos].len = 0;
213
214                // Update `prev` on the new `next` to the current pos
215                if right.next > -1 && (right.next as usize) < self.symbols.len() {
216                    self.symbols[right.next as usize].prev = top.pos as isize;
217                }
218
219                // Insert the new pair formed with the previous symbol
220                let current = &self.symbols[top.pos];
221                if current.prev >= 0 {
222                    let prev = current.prev as usize;
223                    let prev_symbol = self.symbols[prev];
224                    let new_pair = (prev_symbol.c, current.c);
225                    if let Some((rank, new_id)) = merges.get(&new_pair) {
226                        queue.push(Merge {
227                            pos: current.prev as usize,
228                            rank: *rank,
229                            new_id: *new_id,
230                        });
231                    }
232                }
233
234                // Insert the new pair formed with the next symbol
235                let next = current.next as usize;
236                if next < self.symbols.len() {
237                    let next_symbol = self.symbols[next];
238                    let new_pair = (current.c, next_symbol.c);
239                    if let Some((rank, new_id)) = merges.get(&new_pair) {
240                        queue.push(Merge {
241                            pos: top.pos,
242                            rank: *rank,
243                            new_id: *new_id,
244                        });
245                    }
246                }
247            }
248        }
249
250        // Filter out the removed symbols
251        self.symbols.retain(|s| s.len != 0);
252    }

262    pub(super) fn get_offsets_iter(&self) -> impl Iterator<Item = (usize, usize)> + '_ {
263        let mut pos = 0;
264        self.symbols.iter().map(move |symbol| {
265            let new_pos = pos + symbol.len;
266            let offset = (pos, new_pos);
267            pos = new_pos;
268            offset
269        })
270    }

277    #[test]
278    fn test_merge() {
279        // Let's say we have the word 'hello' and a word-to-id vocab that looks
280        // like this: {'h': 0, 'e': 1, 'l': 2, 'o': 3}.
281        let mut word = Word::new();
282        word.add(0, 1); // 'h'
283        word.add(1, 1); // 'e'
284        word.add(2, 1); // 'l'
285        word.add(2, 1); // 'l'
286        word.add(3, 1); // 'o'
287
288        // We're going to perform a merge on the pair ('l', 'l') ~= (2, 2). Let's
289        // say that 'll' has the ID of 4 in the updated word-to-id vocab.
290        let changes = word.merge(2, 2, 4, usize::MAX);
291
292        // So the word should now look like this:
293        assert_eq!(
294            word.get_chars(),
295            &[
296                0u32, // 'h'
297                1u32, // 'e'
298                4u32, // 'll'
299                3u32, // 'o'
300            ]
301        );
302
303        // The return value `changes` will be used to update the pair counts during
304        // training. This merge affects the counts for the pairs
305        // ('e', 'l') ~= (1, 2),
306        // ('e', 'll') ~= (1, 4),
307        // ('l', 'o') ~= (2, 3), and
308        // ('ll', 'o') ~= (4, 3).
309        // So the changes should reflect that:
310        assert_eq!(
311            changes,
312            &[
313                ((1u32, 2u32), -1i32), // count for ('e', 'l') should be decreased by 1.
314                ((1u32, 4u32), 1i32),  // count for ('e', 'll') should be increased by 1.
315                ((2u32, 3u32), -1i32), // count for ('l', 'o') should be decreased by 1.
316                ((4u32, 3u32), 1i32),  // count for ('ll', 'o') should be increased by 1.
317            ]
318        );
319    }
