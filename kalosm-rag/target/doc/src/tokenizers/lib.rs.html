6//! The core of `tokenizers`, written in Rust.
7//! Provides an implementation of today's most used tokenizers, with a focus on performance and
8//! versatility.
9//!
10//! # What is a Tokenizer
11//!
12//! A Tokenizer works as a pipeline, it processes some raw text as input and outputs an `Encoding`.
13//! The various steps of the pipeline are:
14//!
15//! 1. The `Normalizer`: in charge of normalizing the text. Common examples of normalization are
16//!    the [unicode normalization standards](https://unicode.org/reports/tr15/#Norm_Forms), such as `NFD` or `NFKC`.
17//!    More details about how to use the `Normalizers` are available on the
18//!    [Hugging Face blog](https://huggingface.co/docs/tokenizers/components#normalizers)
19//! 2. The `PreTokenizer`: in charge of creating initial words splits in the text. The most common way of
20//!    splitting text is simply on whitespace.
21//! 3. The `Model`: in charge of doing the actual tokenization. An example of a `Model` would be
22//!    `BPE` or `WordPiece`.
23//! 4. The `PostProcessor`: in charge of post-processing the `Encoding` to add anything relevant
24//!    that, for example, a language model would need, such as special tokens.
25//!
26//! ## Loading a pretrained tokenizer from the Hub
27//! ```
28//! use tokenizers::tokenizer::{Result, Tokenizer};
29//!
30//! fn main() -> Result<()> {
31//!     # #[cfg(feature = "http")]
32//!     # {
33//!     // needs http feature enabled
34//!     let tokenizer = Tokenizer::from_pretrained("bert-base-cased", None)?;
35//!
36//!     let encoding = tokenizer.encode("Hey there!", false)?;
37//!     println!("{:?}", encoding.get_tokens());
38//!     # }
39//!     Ok(())
40//! }
41//! ```
42//!
43//! ## Deserialization and tokenization example
44//!
45//! ```no_run
46//! use tokenizers::tokenizer::{Result, Tokenizer, EncodeInput};
47//! use tokenizers::models::bpe::BPE;
48//!
49//! fn main() -> Result<()> {
50//!     let bpe_builder = BPE::from_file("./path/to/vocab.json", "./path/to/merges.txt");
51//!     let bpe = bpe_builder
52//!         .dropout(0.1)
53//!         .unk_token("[UNK]".into())
54//!         .build()?;
55//!
56//!     let mut tokenizer = Tokenizer::new(bpe);
57//!
58//!     let encoding = tokenizer.encode("Hey there!", false)?;
59//!     println!("{:?}", encoding.get_tokens());
60//!
61//!     Ok(())
62//! }
63//! ```
64//!
65//! ## Training and serialization example
66//!
67//! ```no_run
68//! use tokenizers::decoders::DecoderWrapper;
69//! use tokenizers::models::bpe::{BpeTrainerBuilder, BPE};
70//! use tokenizers::normalizers::{strip::Strip, unicode::NFC, utils::Sequence, NormalizerWrapper};
71//! use tokenizers::pre_tokenizers::byte_level::ByteLevel;
72//! use tokenizers::pre_tokenizers::PreTokenizerWrapper;
73//! use tokenizers::processors::PostProcessorWrapper;
74//! use tokenizers::{AddedToken, Model, Result, TokenizerBuilder};
75//!
76//! use std::path::Path;
77//!
78//! fn main() -> Result<()> {
79//!     let vocab_size: usize = 100;
80//!
81//!     let mut trainer = BpeTrainerBuilder::new()
82//!         .show_progress(true)
83//!         .vocab_size(vocab_size)
84//!         .min_frequency(0)
85//!         .special_tokens(vec![
86//!             AddedToken::from(String::from("<s>"), true),
87//!             AddedToken::from(String::from("<pad>"), true),
88//!             AddedToken::from(String::from("</s>"), true),
89//!             AddedToken::from(String::from("<unk>"), true),
90//!             AddedToken::from(String::from("<mask>"), true),
91//!         ])
92//!         .build();
93//!
94//!     let mut tokenizer = TokenizerBuilder::new()
95//!         .with_model(BPE::default())
96//!         .with_normalizer(Some(Sequence::new(vec![
97//!             Strip::new(true, true).into(),
98//!             NFC.into(),
99//!         ])))
100//!         .with_pre_tokenizer(Some(ByteLevel::default()))
101//!         .with_post_processor(Some(ByteLevel::default()))
102//!         .with_decoder(Some(ByteLevel::default()))
103//!         .build()?;
104//!
105//!     let pretty = false;
106//!     tokenizer
107//!         .train_from_files(
108//!             &mut trainer,
109//!             vec!["path/to/vocab.txt".to_string()],
110//!         )?
111//!         .save("tokenizer.json", pretty)?;
112//!
113//!     Ok(())
114//! }
115//! ```
116//!
117//! # Additional information
118//!
119//! - tokenizers is designed to leverage CPU parallelism when possible. The level of parallelism is determined
120//!   by the total number of core/threads your CPU provides but this can be tuned by setting the `RAYON_RS_NUM_THREADS`
121//!   environment variable. As an example setting `RAYON_RS_NUM_THREADS=4` will allocate a maximum of 4 threads.
122//!   **_Please note this behavior may evolve in the future_**
123//!
124//! # Features
125//! **progressbar**: The progress bar visualization is enabled by default. It might be disabled if
126//!   compilation for certain targets is not supported by the [termios](https://crates.io/crates/termios)
127//!   dependency of the [indicatif](https://crates.io/crates/indicatif) progress bar.
128//! **http**: This feature enables downloading the tokenizer via HTTP. It is disabled by default.
129//!   With this feature enabled, `Tokenizer::from_pretrained` becomes accessible.
