8// https://github.com/ggerganov/llama.cpp/blob/468ea24fb4633a0d681f7ac84089566c1c6190cb/llama.h#L37
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10enum Magic {
11    Ggjt,
12    Ggla,
13    Ggmf,
14    Ggml,
15    Ggsn,
16}
17
18impl TryFrom<u32> for Magic {
19    type Error = crate::Error;
20    fn try_from(value: u32) -> Result<Self> {
21        let magic = match value {
22            0x67676a74 => Self::Ggjt,
23            0x67676c61 => Self::Ggla,
24            0x67676d66 => Self::Ggmf,
25            0x67676d6c => Self::Ggml,
26            0x6767736e => Self::Ggsn,
27            _ => crate::bail!("unknown magic {value:08x}"),
28        };
29        Ok(magic)
30    }
31}

33#[derive(Debug, Clone, Copy, PartialEq, Eq)]
34pub enum VersionedMagic {
35    GgmlUnversioned,
36    GgmfV1,
37    GgjtV1,
38    GgjtV2,
39    GgjtV3,
40}
41
42impl VersionedMagic {
43    fn read<R: std::io::Read>(reader: &mut R) -> Result<Self> {
44        let magic = reader.read_u32::<LittleEndian>()?;
45        let magic = Magic::try_from(magic)?;
46        if magic == Magic::Ggml {
47            return Ok(Self::GgmlUnversioned);
48        }
49        let version = reader.read_u32::<LittleEndian>()?;
50        let versioned_magic = match (magic, version) {
51            (Magic::Ggmf, 1) => Self::GgmfV1,
52            (Magic::Ggjt, 1) => Self::GgjtV1,
53            (Magic::Ggjt, 2) => Self::GgjtV2,
54            (Magic::Ggjt, 3) => Self::GgjtV3,
55            _ => crate::bail!("ggml: unsupported magic/version {magic:?}/{version}"),
56        };
57        Ok(versioned_magic)
58    }
59
60    fn align32(&self) -> bool {
61        match self {
62            Self::GgmlUnversioned | Self::GgmfV1 => false,
63            Self::GgjtV1 | Self::GgjtV2 | Self::GgjtV3 => true,
64        }
65    }
66}

100#[derive(Debug, Clone, PartialEq)]
101pub struct Vocab {
102    pub token_score_pairs: Vec<(Vec<u8>, f32)>,
103}
104
105impl Vocab {
106    fn read<R: std::io::Read>(reader: &mut R, n_vocab: usize) -> Result<Self> {
107        // https://github.com/ggerganov/llama.cpp/blob/468ea24fb4633a0d681f7ac84089566c1c6190cb/llama.cpp#L556
108        let mut token_score_pairs = Vec::with_capacity(n_vocab);
109        for _index in 0..n_vocab {
110            let len = reader.read_u32::<LittleEndian>()? as usize;
111            let mut word = vec![0u8; len];
112            reader.read_exact(&mut word)?;
113            let score = reader.read_f32::<LittleEndian>()?;
114            token_score_pairs.push((word, score))
115        }
116        Ok(Self { token_score_pairs })
117    }
118}

120fn from_raw_data<T: super::GgmlType + Send + Sync + 'static>(
121    raw_data: &[u8],
122    size_in_bytes: usize,
123    dims: Vec<usize>,
124    device: &Device,
125) -> Result<super::QTensor> {
126    let raw_data_ptr = raw_data.as_ptr();
127    let n_blocks = size_in_bytes / std::mem::size_of::<T>();
128    let data = unsafe { std::slice::from_raw_parts(raw_data_ptr as *const T, n_blocks) };
129    let data: QStorage = match device {
130        Device::Cpu => QStorage::Cpu(Box::new(data.to_vec())),
131        Device::Metal(metal) => super::metal::load_quantized(metal, data)?,
132        Device::Cuda(cuda) => super::cuda::load_quantized(cuda, data)?,
133    };
134    super::QTensor::new(data, dims)
135}

137/// Creates a Tensor from a raw GGML tensor.
138pub fn qtensor_from_ggml(
139    ggml_dtype: GgmlDType,
140    raw_data: &[u8],
141    dims: Vec<usize>,
142    device: &Device,
143) -> Result<super::QTensor> {
144    let tensor_elems = dims.iter().product::<usize>();
145    let block_size = ggml_dtype.block_size();
146    if tensor_elems % block_size != 0 {
147        crate::bail!(
148            "the number of elements {tensor_elems} is not divisible by the block size {block_size}"
149        )
150    }
151    let size_in_bytes = tensor_elems / block_size * ggml_dtype.type_size();
152
153    match ggml_dtype {
154        GgmlDType::F32 => from_raw_data::<f32>(raw_data, size_in_bytes, dims, device),
155        GgmlDType::F16 => from_raw_data::<half::f16>(raw_data, size_in_bytes, dims, device),
156        GgmlDType::Q4_0 => {
157            from_raw_data::<k_quants::BlockQ4_0>(raw_data, size_in_bytes, dims, device)
158        }
159        GgmlDType::Q4_1 => {
160            from_raw_data::<k_quants::BlockQ4_1>(raw_data, size_in_bytes, dims, device)
161        }
162        GgmlDType::Q5_0 => {
163            from_raw_data::<k_quants::BlockQ5_0>(raw_data, size_in_bytes, dims, device)
164        }
165        GgmlDType::Q5_1 => {
166            from_raw_data::<k_quants::BlockQ5_1>(raw_data, size_in_bytes, dims, device)
167        }
168        GgmlDType::Q8_0 => {
169            from_raw_data::<k_quants::BlockQ8_0>(raw_data, size_in_bytes, dims, device)
170        }
171        GgmlDType::Q2K => {
172            from_raw_data::<k_quants::BlockQ2K>(raw_data, size_in_bytes, dims, device)
173        }
174        GgmlDType::Q3K => {
175            from_raw_data::<k_quants::BlockQ3K>(raw_data, size_in_bytes, dims, device)
176        }
177        GgmlDType::Q4K => {
178            from_raw_data::<k_quants::BlockQ4K>(raw_data, size_in_bytes, dims, device)
179        }
180        GgmlDType::Q5K => {
181            from_raw_data::<k_quants::BlockQ5K>(raw_data, size_in_bytes, dims, device)
182        }
183        GgmlDType::Q6K => {
184            from_raw_data::<k_quants::BlockQ6K>(raw_data, size_in_bytes, dims, device)
185        }
186        _ => crate::bail!("quantized type {ggml_dtype:?} is not supported yet"),
187    }
188}

190fn read_one_tensor<R: std::io::Seek + std::io::Read>(
191    reader: &mut R,
192    magic: VersionedMagic,
193    device: &Device,
194) -> Result<(String, super::QTensor)> {
195    let n_dims = reader.read_u32::<LittleEndian>()?;
196    let name_len = reader.read_u32::<LittleEndian>()?;
197    let ggml_dtype = reader.read_u32::<LittleEndian>()?;
198    let ggml_dtype = GgmlDType::from_u32(ggml_dtype)?;
199    let mut dims = vec![0u32; n_dims as usize];
200    reader.read_u32_into::<LittleEndian>(&mut dims)?;
201    // The dimensions are stored in reverse order, see for example:
202    // https://github.com/ggerganov/llama.cpp/blob/b5ffb2849d23afe73647f68eec7b68187af09be6/convert.py#L969
203    dims.reverse();
204    let mut name = vec![0u8; name_len as usize];
205    reader.read_exact(&mut name)?;
206    let name = String::from_utf8_lossy(&name).into_owned();
207
208    if magic.align32() {
209        let pos = reader.stream_position()?;
210        reader.seek(std::io::SeekFrom::Current(((32 - pos % 32) % 32) as i64))?;
211    }
212    let dims = dims.iter().map(|&u| u as usize).collect::<Vec<_>>();
213    let tensor_elems = dims.iter().product::<usize>();
214    let size_in_bytes = tensor_elems * ggml_dtype.type_size() / ggml_dtype.block_size();
215    // TODO: Mmap version to avoid copying the data around?
216    let mut raw_data = vec![0u8; size_in_bytes];
217    reader.read_exact(&mut raw_data)?;
218    match qtensor_from_ggml(ggml_dtype, &raw_data, dims, device) {
219        Ok(tensor) => Ok((name, tensor)),
220        Err(e) => crate::bail!("Error creating tensor {name}: {e}"),
221    }
222}

224pub struct Content {
225    pub magic: VersionedMagic,
226    pub hparams: HParams,
227    pub vocab: Vocab,
228    pub tensors: HashMap<String, super::QTensor>,
229    pub device: Device,
230}
231
232impl Content {
233    pub fn read<R: std::io::Seek + std::io::Read>(
234        reader: &mut R,
235        device: &Device,
236    ) -> Result<Content> {
237        // https://github.com/ggerganov/llama.cpp/blob/468ea24fb4633a0d681f7ac84089566c1c6190cb/llama.cpp#L505
238        let last_position = reader.seek(std::io::SeekFrom::End(0))?;
239        reader.seek(std::io::SeekFrom::Start(0))?;
240        let magic = VersionedMagic::read(reader)?;
241        let hparams = HParams::read(reader)?;
242        let vocab = Vocab::read(reader, hparams.n_vocab as usize)?;
243        let mut tensors = HashMap::new();
244
245        while reader.stream_position()? != last_position {
246            let (name, tensor) = read_one_tensor(reader, magic, device)?;
247            tensors.insert(name, tensor);
248        }
249        let device = device.clone();
250        Ok(Self {
251            magic,
252            hparams,
253            vocab,
254            tensors,
255            device,
256        })
257    }
258
259    pub fn remove(&mut self, name: &str) -> Result<super::QTensor> {
260        match self.tensors.remove(name) {
261            None => crate::bail!("cannot find tensor with name '{name}'"),
262            Some(tensor) => Ok(tensor),
263        }
264    }
