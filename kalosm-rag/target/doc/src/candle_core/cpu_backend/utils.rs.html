38pub trait Map2 {
39    const OP: &'static str;
40    fn f<T: WithDType>(&self, v1: &[T], l1: &Layout, v2: &[T], l2: &Layout) -> Result<Vec<T>>;
41
42    fn map(&self, v1: &C, l1: &Layout, v2: &C, l2: &Layout) -> Result<C> {
43        match (v1, v2) {
44            (C::U8(v1), C::U8(v2)) => Ok(C::U8(self.f(v1, l1, v2, l2)?)),
45            (C::U32(v1), C::U32(v2)) => Ok(C::U32(self.f(v1, l1, v2, l2)?)),
46            (C::I64(v1), C::I64(v2)) => Ok(C::I64(self.f(v1, l1, v2, l2)?)),
47            (C::BF16(v1), C::BF16(v2)) => Ok(C::BF16(self.f(v1, l1, v2, l2)?)),
48            (C::F16(v1), C::F16(v2)) => Ok(C::F16(self.f(v1, l1, v2, l2)?)),
49            (C::F32(v1), C::F32(v2)) => Ok(C::F32(self.f(v1, l1, v2, l2)?)),
50            (C::F64(v1), C::F64(v2)) => Ok(C::F64(self.f(v1, l1, v2, l2)?)),
51            _ => Err(Error::DTypeMismatchBinaryOp {
52                lhs: v1.dtype(),
53                rhs: v2.dtype(),
54                op: Self::OP,
55            }
56            .bt()),
57        }
58    }

84pub fn binary_map<T: Copy, U: Copy, F: FnMut(T, T) -> U>(
85    lhs_l: &Layout,
86    rhs_l: &Layout,
87    lhs: &[T],
88    rhs: &[T],
89    mut f: F,
90) -> Vec<U> {
91    match (lhs_l.contiguous_offsets(), rhs_l.contiguous_offsets()) {
92        (Some((o_l1, o_l2)), Some((o_r1, o_r2))) => lhs[o_l1..o_l2]
93            .iter()
94            .zip(rhs[o_r1..o_r2].iter())
95            .map(|(&l, &r)| f(l, r))
96            .collect(),
97        (Some((o_l1, o_l2)), None) => {
98            // TODO: Maybe we want to avoid going through the layout twice.
99            match rhs_l.offsets_b() {
100                Some(ob) => {
101                    let mut i_in_block = 0;
102                    let mut i_right_broadcast = 0;
103                    lhs[o_l1..o_l2]
104                        .iter()
105                        .map(|&l| {
106                            let r = unsafe { rhs.get_unchecked(i_in_block + ob.start) };
107                            i_right_broadcast += 1;
108                            if i_right_broadcast >= ob.right_broadcast {
109                                i_in_block += 1;
110                                i_right_broadcast = 0;
111                            }
112                            if i_in_block >= ob.len {
113                                i_in_block = 0
114                            }
115                            f(l, *r)
116                        })
117                        .collect()
118                }
119                None => lhs_l
120                    .strided_index()
121                    .zip(rhs_l.strided_index())
122                    .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
123                    .collect(),
124            }
125        }
126        (None, Some((o_r1, o_r2))) => {
127            // TODO: Maybe we want to avoid going through the layout twice.
128            match lhs_l.offsets_b() {
129                Some(ob) => {
130                    let mut i_in_block = 0;
131                    let mut i_right_broadcast = 0;
132                    rhs[o_r1..o_r2]
133                        .iter()
134                        .map(|&r| {
135                            let l = unsafe { lhs.get_unchecked(i_in_block + ob.start) };
136                            i_right_broadcast += 1;
137                            if i_right_broadcast >= ob.right_broadcast {
138                                i_in_block += 1;
139                                i_right_broadcast = 0;
140                            }
141                            if i_in_block >= ob.len {
142                                i_in_block = 0
143                            }
144                            f(*l, r)
145                        })
146                        .collect()
147                }
148                None => lhs_l
149                    .strided_index()
150                    .zip(rhs_l.strided_index())
151                    .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
152                    .collect(),
153            }
154        }
155        _ => lhs_l
156            .strided_index()
157            .zip(rhs_l.strided_index())
158            .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
159            .collect(),
160    }
161}

163// Similar to binary_map but with vectorized variants.
164pub fn binary_map_vec<T: Copy, F: FnMut(T, T) -> T, FV: FnMut(&[T], &[T], &mut [T])>(
165    lhs_l: &Layout,
166    rhs_l: &Layout,
167    lhs: &[T],
168    rhs: &[T],
169    mut f: F,
170    mut f_vec: FV,
171) -> Vec<T> {
172    let el_count = lhs_l.shape().elem_count();
173    match (lhs_l.contiguous_offsets(), rhs_l.contiguous_offsets()) {
174        (Some((o_l1, o_l2)), Some((o_r1, o_r2))) => {
175            let mut ys: Vec<T> = Vec::with_capacity(el_count);
176            let ys_to_set = ys.spare_capacity_mut();
177            let ys_to_set = unsafe {
178                std::mem::transmute::<&mut [std::mem::MaybeUninit<T>], &mut [T]>(ys_to_set)
179            };
180            f_vec(&lhs[o_l1..o_l2], &rhs[o_r1..o_r2], ys_to_set);
181            // SAFETY: values are all set by f_vec.
182            unsafe { ys.set_len(el_count) };
183            ys
184        }
185        (Some((o_l1, o_l2)), None) => match rhs_l.offsets_b() {
186            Some(ob) if ob.right_broadcast == 1 => {
187                let rhs = &rhs[ob.start..ob.start + ob.len];
188                let mut ys: Vec<T> = Vec::with_capacity(el_count);
189                let ys_to_set = ys.spare_capacity_mut();
190                let ys_to_set = unsafe {
191                    std::mem::transmute::<&mut [std::mem::MaybeUninit<T>], &mut [T]>(ys_to_set)
192                };
193                let mut dst_i = 0;
194                for src_i in (o_l1..o_l2).step_by(ob.len) {
195                    f_vec(
196                        &lhs[src_i..src_i + ob.len],
197                        rhs,
198                        &mut ys_to_set[dst_i..dst_i + ob.len],
199                    );
200                    dst_i += ob.len;
201                }
202                // SAFETY: values are all set by f_vec.
203                unsafe { ys.set_len(el_count) };
204                ys
205            }
206            Some(ob) => {
207                let rhs = &rhs[ob.start..ob.start + ob.len];
208                let mut ys = lhs[o_l1..o_l2].to_vec();
209                for idx_l in 0..ob.left_broadcast {
210                    let start = idx_l * ob.len * ob.right_broadcast;
211                    for (i, &r) in rhs.iter().enumerate() {
212                        let start = start + i * ob.right_broadcast;
213                        for v in ys[start..start + ob.right_broadcast].iter_mut() {
214                            *v = f(*v, r)
215                        }
216                    }
217                }
218                ys
219            }
220            None => lhs_l
221                .strided_index()
222                .zip(rhs_l.strided_index())
223                .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
224                .collect(),
225        },
226        (None, Some((o_r1, o_r2))) => match lhs_l.offsets_b() {
227            Some(ob) if ob.right_broadcast == 1 => {
228                let lhs = &lhs[ob.start..ob.start + ob.len];
229                let mut ys: Vec<T> = Vec::with_capacity(el_count);
230                let ys_to_set = ys.spare_capacity_mut();
231                let ys_to_set = unsafe {
232                    std::mem::transmute::<&mut [std::mem::MaybeUninit<T>], &mut [T]>(ys_to_set)
233                };
234                let mut dst_i = 0;
235                for src_i in (o_r1..o_r2).step_by(ob.len) {
236                    f_vec(
237                        lhs,
238                        &rhs[src_i..src_i + ob.len],
239                        &mut ys_to_set[dst_i..dst_i + ob.len],
240                    );
241                    dst_i += ob.len;
242                }
243                // SAFETY: values are all set by f_vec.
244                unsafe { ys.set_len(el_count) };
245                ys
246            }
247            Some(ob) => {
248                let lhs = &lhs[ob.start..ob.start + ob.len];
249                let mut ys = rhs[o_r1..o_r2].to_vec();
250                for idx_l in 0..ob.left_broadcast {
251                    let start = idx_l * ob.len * ob.right_broadcast;
252                    for (i, &l) in lhs.iter().enumerate() {
253                        let start = start + i * ob.right_broadcast;
254                        for v in ys[start..start + ob.right_broadcast].iter_mut() {
255                            *v = f(l, *v)
256                        }
257                    }
258                }
259                ys
260            }
261            None => lhs_l
262                .strided_index()
263                .zip(rhs_l.strided_index())
264                .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
265                .collect(),
266        },
267        _ => lhs_l
268            .strided_index()
269            .zip(rhs_l.strided_index())
270            .map(|(lhs_i, rhs_i)| f(lhs[lhs_i], rhs[rhs_i]))
271            .collect(),
272    }

275pub fn unary_map<T: Copy, U: Copy, F: FnMut(T) -> U>(
276    vs: &[T],
277    layout: &Layout,
278    mut f: F,
279) -> Vec<U> {
280    match layout.strided_blocks() {
281        crate::StridedBlocks::SingleBlock { start_offset, len } => vs
282            [start_offset..start_offset + len]
283            .iter()
284            .map(|&v| f(v))
285            .collect(),
286        crate::StridedBlocks::MultipleBlocks {
287            block_start_index,
288            block_len,
289        } => {
290            let mut result = Vec::with_capacity(layout.shape().elem_count());
291            // Specialize the case where block_len is one to avoid the second loop.
292            if block_len == 1 {
293                for index in block_start_index {
294                    let v = unsafe { vs.get_unchecked(index) };
295                    result.push(f(*v))
296                }
297            } else {
298                for index in block_start_index {
299                    for offset in 0..block_len {
300                        let v = unsafe { vs.get_unchecked(index + offset) };
301                        result.push(f(*v))
302                    }
303                }
304            }
305            result
306        }
307    }
308}

310pub fn unary_map_vec<T: Copy, U: Copy, F: FnMut(T) -> U, FV: FnMut(&[T], &mut [U])>(
311    vs: &[T],
312    layout: &Layout,
313    mut f: F,
314    mut f_vec: FV,
315) -> Vec<U> {
316    match layout.strided_blocks() {
317        crate::StridedBlocks::SingleBlock { start_offset, len } => {
318            let mut ys: Vec<U> = Vec::with_capacity(len);
319            let ys_to_set = ys.spare_capacity_mut();
320            let ys_to_set = unsafe {
321                std::mem::transmute::<&mut [std::mem::MaybeUninit<U>], &mut [U]>(ys_to_set)
322            };
323            f_vec(&vs[start_offset..start_offset + len], ys_to_set);
324            // SAFETY: values are all set by f_vec.
325            unsafe { ys.set_len(len) };
326            ys
327        }
328        crate::StridedBlocks::MultipleBlocks {
329            block_start_index,
330            block_len,
331        } => {
332            let el_count = layout.shape().elem_count();
333            // Specialize the case where block_len is one to avoid the second loop.
334            if block_len == 1 {
335                let mut result = Vec::with_capacity(el_count);
336                for index in block_start_index {
337                    let v = unsafe { vs.get_unchecked(index) };
338                    result.push(f(*v))
339                }
340                result
341            } else {
342                let mut ys: Vec<U> = Vec::with_capacity(el_count);
343                let ys_to_set = ys.spare_capacity_mut();
344                let ys_to_set = unsafe {
345                    std::mem::transmute::<&mut [std::mem::MaybeUninit<U>], &mut [U]>(ys_to_set)
346                };
347                let mut dst_index = 0;
348                for src_index in block_start_index {
349                    let vs = &vs[src_index..src_index + block_len];
350                    let ys = &mut ys_to_set[dst_index..dst_index + block_len];
351                    f_vec(vs, ys);
352                    dst_index += block_len;
353                }
354                // SAFETY: values are all set by f_vec.
355                unsafe { ys.set_len(el_count) };
356                ys
357            }
358        }
359    }
