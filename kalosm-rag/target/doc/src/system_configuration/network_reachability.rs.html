36/// Failure to determine reachability
37#[derive(Debug)]
38pub enum ReachabilityError {
39    /// `SCNetworkReachabilityGetFlags` call failed.
40    FailedToDetermineReachability,
41    ///  `SCNetworkReachabilityGetFlags` call returned unrecognized flags.
42    UnrecognizedFlags(u32),
43}
44
45/// Failure to schedule a reachability callback on a runloop.
46#[derive(Debug)]
47pub struct SchedulingError(());
48
49impl Display for SchedulingError {
50    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
51        write!(f, "Failed to schedule a reachability callback on a runloop")
52    }
53}
54
55impl Error for SchedulingError {}

85    /// Rustier interface for [`SCNetworkReachabilityFlags`].
86    ///
87    /// [`SCNetworkReachability`]: https://developer.apple.com/documentation/systemconfiguration/scnetworkreachabilityflags
88    pub struct ReachabilityFlags: u32 {
89        /// The specified node name or address can be reached via a transient connection, such as
90        /// PPP.
91        const TRANSIENT_CONNECTION = kSCNetworkReachabilityFlagsTransientConnection;
92        /// The specified node name or address can be reached using the current network
93        /// configuration.
94        const REACHABLE = kSCNetworkReachabilityFlagsReachable;
95        /// The specified node name or address can be reached using the current network
96        /// configuration, but a connection must first be established. If this flag is set, the
97        /// `CONNECTION_ON_TRAFFIC` flag, `CONNECTION_ON_DEMAND` flag, or `IS_WANN` flag is also
98        /// typically set to indicate the type of connection required. If the user must manually
99        /// make the connection, the `INTERVENTION_REQUIRED` flag is also set.
100        const CONNECTION_REQUIRED = kSCNetworkReachabilityFlagsConnectionRequired;
101        /// The specified node name or address can be reached using the current network
102        /// configuration, but a connection must first be established. Any traffic directed to the
103        /// specified name or address will initiate the connection.
104        const CONNECTION_ON_TRAFFIC = kSCNetworkReachabilityFlagsConnectionOnTraffic;
105        /// The specified node name or address can be reached using the current network
106        /// configuration, but a connection must first be established.
107        const INTERVENTION_REQUIRED = kSCNetworkReachabilityFlagsInterventionRequired;
108        /// The specified node name or address can be reached using the current network
109        /// configuration, but a connection must first be established.
110        const CONNECTION_ON_DEMAND = kSCNetworkReachabilityFlagsConnectionOnDemand;
111        /// The specified node name or address is one that is associated with a network interface on the current system.
112        const IS_LOCAL_ADDRESS = kSCNetworkReachabilityFlagsIsLocalAddress;
113        /// Network traffic to the specified node name or address will not go through a gateway, but
114        /// is routed directly to one of the interfaces in the system.
115        const IS_DIRECT = kSCNetworkReachabilityFlagsIsDirect;
116        /// The specified node name or address can be reached via a cellular connection, such as EDGE or GPRS.
117        const IS_WWAN = kSCNetworkReachabilityFlagsIsWWAN;
118    }

137impl SCNetworkReachability {
138    /// Construct a SCNetworkReachability struct with a local and a remote socket address.
139    ///
140    /// See [`SCNetworkReachabilityCreateWithAddressPair`] for details.
141    ///
142    /// [`SCNetworkReachabilityCreateWithAddressPair`]: https://developer.apple.com/documentation/systemconfiguration/1514908-scnetworkreachabilitycreatewitha?language=objc
143    pub fn from_addr_pair(local: SocketAddr, remote: SocketAddr) -> SCNetworkReachability {
144        let ptr = unsafe {
145            SCNetworkReachabilityCreateWithAddressPair(
146                std::ptr::null(),
147                &*to_c_sockaddr(local),
148                &*to_c_sockaddr(remote),
149            )
150        };
151
152        unsafe { Self::wrap_under_create_rule(ptr) }
153    }
154
155    /// Construct a Reachability from either a hostname or a network node
156    ///
157    /// See [`SCNetworkReachabilityCreateWithName`] for details.
158    ///
159    /// [`SCNetworkReachabilityCreateWithName`]: https://developer.apple.com/documentation/systemconfiguration/1514904-scnetworkreachabilitycreatewithn?language=objc
160    pub fn from_host(host: &CStr) -> Option<Self> {
161        let ptr = unsafe { SCNetworkReachabilityCreateWithName(ptr::null(), host.as_ptr()) };
162        if ptr.is_null() {
163            None
164        } else {
165            unsafe { Some(Self::wrap_under_create_rule(ptr)) }
166        }
167    }
168
169    /// Return a flag indicating whether the specified network address is reachable.
170    ///
171    /// See [`SCNetworkReachabilityGetFlags`] for details.
172    ///
173    /// [`SCNetworkReachabilityGetFlags`]: https://developer.apple.com/documentation/systemconfiguration/1514924-scnetworkreachabilitygetflags?language=objc
174    pub fn reachability(&self) -> Result<ReachabilityFlags, ReachabilityError> {
175        let mut raw_flags = 0u32;
176        if unsafe { SCNetworkReachabilityGetFlags(self.0, &mut raw_flags) } == 0u8 {
177            return Err(ReachabilityError::FailedToDetermineReachability);
178        }
179
180        ReachabilityFlags::from_bits(raw_flags)
181            .ok_or(ReachabilityError::UnrecognizedFlags(raw_flags))
182    }
183
184    /// Schedule callback with runloop.
185    ///
186    /// See [`SCNetworkReachabilityScheduleFromRunLoop`] for details.
187    ///
188    /// [`SCNetworkReachabilityScheduleFromRunLoop`]: https://developer.apple.com/documentation/systemconfiguration/1514894-scnetworkreachabilityschedulewit?language=objc
189    pub fn schedule_with_runloop(
190        &self,
191        run_loop: &CFRunLoop,
192        run_loop_mode: CFStringRef,
193    ) -> Result<(), SchedulingError> {
194        if unsafe {
195            SCNetworkReachabilityScheduleWithRunLoop(
196                self.0,
197                run_loop.to_void() as *mut _,
198                run_loop_mode,
199            )
200        } == 0u8
201        {
202            Err(SchedulingError(()))
203        } else {
204            Ok(())
205        }
206    }
207
208    /// Unschedule from run loop.
209    ///
210    /// See [`SCNetworkReachabilityUnscheduleFromRunLoop`] for details.
211    ///
212    /// [`SCNetworkReachabilityUnscheduleFromRunLoop`]: https://developer.apple.com/documentation/systemconfiguration/1514899-scnetworkreachabilityunschedulef?language=objc
213    pub fn unschedule_from_runloop(
214        &self,
215        run_loop: &CFRunLoop,
216        run_loop_mode: CFStringRef,
217    ) -> Result<(), UnschedulingError> {
218        if unsafe {
219            SCNetworkReachabilityUnscheduleFromRunLoop(
220                self.0,
221                run_loop.to_void() as *mut _,
222                run_loop_mode,
223            )
224        } == 0u8
225        {
226            Err(UnschedulingError(()))
227        } else {
228            Ok(())
229        }
230    }
231
232    /// Sets callback that is run whenever network connectivity changes. For the callback to be
233    /// invoked, the `SCNetworkReachability` has to be registered on a run loop. Calling this
234    /// function multiple times will clear the subsequently set callback.
235    ///
236    /// See [`SCNetworkReachabilityContext`] for details.
237    ///
238    /// [`SCNetworkReachabilityContext`]: https://developer.apple.com/documentation/systemconfiguration/1514903-scnetworkreachabilitysetcallback?language=objc
239    pub fn set_callback<F: Fn(ReachabilityFlags) + Sync + Send>(
240        &mut self,
241        callback: F,
242    ) -> Result<(), SetCallbackError> {
243        let callback = Arc::new(NetworkReachabilityCallbackContext::new(
244            self.clone(),
245            callback,
246        ));
247
248        let mut callback_context = SCNetworkReachabilityContext {
249            version: 0,
250            info: Arc::into_raw(callback) as *mut _,
251            retain: Some(NetworkReachabilityCallbackContext::<F>::retain_context),
252            release: Some(NetworkReachabilityCallbackContext::<F>::release_context),
253            copyDescription: Some(NetworkReachabilityCallbackContext::<F>::copy_ctx_description),
254        };
255
256        if unsafe {
257            SCNetworkReachabilitySetCallback(
258                self.0,
259                Some(NetworkReachabilityCallbackContext::<F>::callback),
260                &mut callback_context,
261            )
262        } == 0u8
263        {
264            Err(SetCallbackError {})
265        } else {
266            Ok(())
267        }
268    }

281struct NetworkReachabilityCallbackContext<T: Fn(ReachabilityFlags) + Sync + Send> {
282    _host: SCNetworkReachability,
283    callback: T,
284}
285
286impl<T: Fn(ReachabilityFlags) + Sync + Send> NetworkReachabilityCallbackContext<T> {
287    fn new(host: SCNetworkReachability, callback: T) -> Self {
288        Self {
289            _host: host,
290            callback,
291        }
292    }
293
294    extern "C" fn callback(
295        _target: SCNetworkReachabilityRef,
296        flags: SCNetworkReachabilityFlags,
297        context: *mut c_void,
298    ) {
299        let context: &mut Self = unsafe { &mut (*(context as *mut _)) };
300        (context.callback)(unsafe { ReachabilityFlags::from_bits_unchecked(flags) });
301    }
302
303    extern "C" fn copy_ctx_description(_ctx: *const c_void) -> CFStringRef {
304        let description = CFString::from_static_string("NetworkRechability's callback context");
305        let description_ref = description.as_concrete_TypeRef();
306        std::mem::forget(description);
307        description_ref
308    }
309
310    extern "C" fn release_context(ctx: *const c_void) {
311        unsafe {
312            let _ = Arc::from_raw(ctx as *mut Self);
313        }
314    }
315
316    extern "C" fn retain_context(ctx_ptr: *const c_void) -> *const c_void {
317        unsafe {
318            let ctx_ref: Arc<Self> = Arc::from_raw(ctx_ptr as *const Self);
319            let new_ctx = ctx_ref.clone();
320            std::mem::forget(ctx_ref);
321            Arc::into_raw(new_ctx) as *const c_void
322        }
323    }
324}

326/// Allocates a libc::sockaddr compatible struct and fills it with either a libc::sockaddr_in or a
327/// libc::sockaddr_in6, depending on the passed in standard library SocketAddr.
328fn to_c_sockaddr(addr: SocketAddr) -> Box<libc::sockaddr> {
329    let ptr = match addr {
330        SocketAddr::V4(addr) => Box::into_raw(Box::new(libc::sockaddr_in {
331            sin_len: std::mem::size_of::<libc::sockaddr_in>() as u8,
332            sin_family: libc::AF_INET as u8,
333            sin_port: addr.port(),
334            sin_addr: libc::in_addr {
335                s_addr: u32::from(*addr.ip()),
336            },
337            sin_zero: [0i8; 8],
338        })) as *mut c_void,
339        SocketAddr::V6(addr) => Box::into_raw(Box::new(libc::sockaddr_in6 {
340            sin6_len: std::mem::size_of::<libc::sockaddr_in6>() as u8,
341            sin6_family: libc::AF_INET6 as u8,
342            sin6_port: addr.port(),
343            sin6_flowinfo: 0,
344            sin6_addr: libc::in6_addr {
345                s6_addr: addr.ip().octets(),
346            },
347            sin6_scope_id: 0,
348        })) as *mut c_void,
349    };
350
351    unsafe { Box::from_raw(ptr as *mut _) }
352}
