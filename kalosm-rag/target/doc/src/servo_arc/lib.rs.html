11//! Fork of Arc for Servo. This has the following advantages over std::sync::Arc:
12//!
13//! * We don't waste storage on the weak reference count.
14//! * We don't do extra RMU operations to handle the possibility of weak references.
15//! * We can experiment with arena allocation (todo).
16//! * We can add methods to support our custom use cases [1].
17//! * We have support for dynamically-sized types (see from_header_and_iter).
18//! * We have support for thin arcs to unsized types (see ThinArc).
19//! * We have support for references to static data, which don't do any
20//!   refcounting.
21//!
22//! [1]: https://bugzilla.mozilla.org/show_bug.cgi?id=1360883
23

53/// A soft limit on the amount of references that may be made to an `Arc`.
54///
55/// Going above this limit will abort your program (although not
56/// necessarily) at _exactly_ `MAX_REFCOUNT + 1` references.
57const MAX_REFCOUNT: usize = (isize::MAX) as usize;
58
59/// Special refcount value that means the data is not reference counted,
60/// and that the `Arc` is really acting as a read-only static reference.
61const STATIC_REFCOUNT: usize = usize::MAX;

63/// An atomically reference counted shared pointer
64///
65/// See the documentation for [`Arc`] in the standard library. Unlike the
66/// standard library `Arc`, this `Arc` does not support weak reference counting.
67///
68/// See the discussion in https://github.com/rust-lang/rust/pull/60594 for the
69/// usage of PhantomData.
70///
71/// [`Arc`]: https://doc.rust-lang.org/stable/std/sync/struct.Arc.html
72///
73/// cbindgen:derive-eq=false
74/// cbindgen:derive-neq=false
75#[repr(C)]
76pub struct Arc<T: ?Sized> {
77    p: ptr::NonNull<ArcInner<T>>,
78    phantom: PhantomData<T>,
79}

81/// An `Arc` that is known to be uniquely owned
82///
83/// When `Arc`s are constructed, they are known to be
84/// uniquely owned. In such a case it is safe to mutate
85/// the contents of the `Arc`. Normally, one would just handle
86/// this by mutating the data on the stack before allocating the
87/// `Arc`, however it's possible the data is large or unsized
88/// and you need to heap-allocate it earlier in such a way
89/// that it can be freely converted into a regular `Arc` once you're
90/// done.
91///
92/// `UniqueArc` exists for this purpose, when constructed it performs
93/// the same allocations necessary for an `Arc`, however it allows mutable access.
94/// Once the mutation is finished, you can call `.shareable()` and get a regular `Arc`
95/// out of it.
96///
97/// Ignore the doctest below there's no way to skip building with refcount
98/// logging during doc tests (see rust-lang/rust#45599).
99///
100/// ```rust,ignore
101/// # use servo_arc::UniqueArc;
102/// let data = [1, 2, 3, 4, 5];
103/// let mut x = UniqueArc::new(data);
104/// x[4] = 7; // mutate!
105/// let y = x.shareable(); // y is an Arc<T>
106/// ```
107pub struct UniqueArc<T: ?Sized>(Arc<T>);
108
109impl<T> UniqueArc<T> {
110    #[inline]
111    /// Construct a new UniqueArc
112    pub fn new(data: T) -> Self {
113        UniqueArc(Arc::new(data))
114    }
115
116    /// Construct an uninitialized arc
117    #[inline]
118    pub fn new_uninit() -> UniqueArc<mem::MaybeUninit<T>> {
119        unsafe {
120            let layout = Layout::new::<ArcInner<mem::MaybeUninit<T>>>();
121            let ptr = alloc::alloc(layout);
122            let mut p = ptr::NonNull::new(ptr)
123                .unwrap_or_else(|| alloc::handle_alloc_error(layout))
124                .cast::<ArcInner<mem::MaybeUninit<T>>>();
125            ptr::write(&mut p.as_mut().count, atomic::AtomicUsize::new(1));
126
127            #[cfg(feature = "gecko_refcount_logging")]
128            {
129                NS_LogCtor(p.as_ptr() as *mut _, b"ServoArc\0".as_ptr() as *const _, 8)
130            }
131
132            UniqueArc(Arc {
133                p,
134                phantom: PhantomData,
135            })
136        }
137    }
138
139    #[inline]
140    /// Convert to a shareable Arc<T> once we're done mutating it
141    pub fn shareable(self) -> Arc<T> {
142        self.0
143    }
144}

176struct ArcInner<T: ?Sized> {
177    count: atomic::AtomicUsize,
178    data: T,
179}

184/// Computes the offset of the data field within ArcInner.
185fn data_offset<T>() -> usize {
186    let size = size_of::<ArcInner<()>>();
187    let align = align_of::<T>();
188    // https://github.com/rust-lang/rust/blob/1.36.0/src/libcore/alloc.rs#L187-L207
189    size.wrapping_add(align).wrapping_sub(1) & !align.wrapping_sub(1)
190}

192impl<T> Arc<T> {
193    /// Construct an `Arc<T>`
194    #[inline]
195    pub fn new(data: T) -> Self {
196        let ptr = Box::into_raw(Box::new(ArcInner {
197            count: atomic::AtomicUsize::new(1),
198            data,
199        }));
200
201        #[cfg(feature = "gecko_refcount_logging")]
202        unsafe {
203            // FIXME(emilio): Would be so amazing to have
204            // std::intrinsics::type_name() around, so that we could also report
205            // a real size.
206            NS_LogCtor(ptr as *mut _, b"ServoArc\0".as_ptr() as *const _, 8);
207        }
208
209        unsafe {
210            Arc {
211                p: ptr::NonNull::new_unchecked(ptr),
212                phantom: PhantomData,
213            }
214        }
215    }
216
217    /// Construct an intentionally-leaked arc.
218    #[inline]
219    pub fn new_leaked(data: T) -> Self {
220        let arc = Self::new(data);
221        arc.mark_as_intentionally_leaked();
222        arc
223    }
224
225    /// Convert the Arc<T> to a raw pointer, suitable for use across FFI
226    ///
227    /// Note: This returns a pointer to the data T, which is offset in the allocation.
228    #[inline]
229    pub fn into_raw(this: Self) -> *const T {
230        let ptr = unsafe { &((*this.ptr()).data) as *const _ };
231        mem::forget(this);
232        ptr
233    }
234
235    /// Reconstruct the Arc<T> from a raw pointer obtained from into_raw()
236    ///
237    /// Note: This raw pointer will be offset in the allocation and must be preceded
238    /// by the atomic count.
239    #[inline]
240    pub unsafe fn from_raw(ptr: *const T) -> Self {
241        // To find the corresponding pointer to the `ArcInner` we need
242        // to subtract the offset of the `data` field from the pointer.
243        let ptr = (ptr as *const u8).sub(data_offset::<T>());
244        Arc {
245            p: ptr::NonNull::new_unchecked(ptr as *mut ArcInner<T>),
246            phantom: PhantomData,
247        }
248    }
249
250    /// Like from_raw, but returns an addrefed arc instead.
251    #[inline]
252    pub unsafe fn from_raw_addrefed(ptr: *const T) -> Self {
253        let arc = Self::from_raw(ptr);
254        mem::forget(arc.clone());
255        arc
256    }
257
258    /// Create a new static Arc<T> (one that won't reference count the object)
259    /// and place it in the allocation provided by the specified `alloc`
260    /// function.
261    ///
262    /// `alloc` must return a pointer into a static allocation suitable for
263    /// storing data with the `Layout` passed into it. The pointer returned by
264    /// `alloc` will not be freed.
265    #[inline]
266    pub unsafe fn new_static<F>(alloc: F, data: T) -> Arc<T>
267    where
268        F: FnOnce(Layout) -> *mut u8,
269    {
270        let ptr = alloc(Layout::new::<ArcInner<T>>()) as *mut ArcInner<T>;
271
272        let x = ArcInner {
273            count: atomic::AtomicUsize::new(STATIC_REFCOUNT),
274            data,
275        };
276
277        ptr::write(ptr, x);
278
279        Arc {
280            p: ptr::NonNull::new_unchecked(ptr),
281            phantom: PhantomData,
282        }
283    }
284
285    /// Produce a pointer to the data that can be converted back
286    /// to an Arc. This is basically an `&Arc<T>`, without the extra indirection.
287    /// It has the benefits of an `&T` but also knows about the underlying refcount
288    /// and can be converted into more `Arc<T>`s if necessary.
289    #[inline]
290    pub fn borrow_arc<'a>(&'a self) -> ArcBorrow<'a, T> {
291        ArcBorrow(&**self)
292    }
293
294    /// Returns the address on the heap of the Arc itself -- not the T within it -- for memory
295    /// reporting.
296    ///
297    /// If this is a static reference, this returns null.
298    pub fn heap_ptr(&self) -> *const c_void {
299        if self.inner().count.load(Relaxed) == STATIC_REFCOUNT {
300            ptr::null()
301        } else {
302            self.p.as_ptr() as *const ArcInner<T> as *const c_void
303        }
304    }
305}

307impl<T: ?Sized> Arc<T> {
308    #[inline]
309    fn inner(&self) -> &ArcInner<T> {
310        // This unsafety is ok because while this arc is alive we're guaranteed
311        // that the inner pointer is valid. Furthermore, we know that the
312        // `ArcInner` structure itself is `Sync` because the inner data is
313        // `Sync` as well, so we're ok loaning out an immutable pointer to these
314        // contents.
315        unsafe { &*self.ptr() }
316    }
317
318    #[inline(always)]
319    fn record_drop(&self) {
320        #[cfg(feature = "gecko_refcount_logging")]
321        unsafe {
322            NS_LogDtor(self.ptr() as *mut _, b"ServoArc\0".as_ptr() as *const _, 8);
323        }
324    }
325
326    /// Marks this `Arc` as intentionally leaked for the purposes of refcount
327    /// logging.
328    ///
329    /// It's a logic error to call this more than once, but it's not unsafe, as
330    /// it'd just report negative leaks.
331    #[inline(always)]
332    pub fn mark_as_intentionally_leaked(&self) {
333        self.record_drop();
334    }
335
336    // Non-inlined part of `drop`. Just invokes the destructor and calls the
337    // refcount logging machinery if enabled.
338    #[inline(never)]
339    unsafe fn drop_slow(&mut self) {
340        self.record_drop();
341        let _ = Box::from_raw(self.ptr());
342    }
343
344    /// Test pointer equality between the two Arcs, i.e. they must be the _same_
345    /// allocation
346    #[inline]
347    pub fn ptr_eq(this: &Self, other: &Self) -> bool {
348        this.ptr() == other.ptr()
349    }
350
351    fn ptr(&self) -> *mut ArcInner<T> {
352        self.p.as_ptr()
353    }
354}

370impl<T: ?Sized> Clone for Arc<T> {
371    #[inline]
372    fn clone(&self) -> Self {
373        // NOTE(emilio): If you change anything here, make sure that the
374        // implementation in layout/style/ServoStyleConstsInlines.h matches!
375        //
376        // Using a relaxed ordering to check for STATIC_REFCOUNT is safe, since
377        // `count` never changes between STATIC_REFCOUNT and other values.
378        if self.inner().count.load(Relaxed) != STATIC_REFCOUNT {
379            // Using a relaxed ordering is alright here, as knowledge of the
380            // original reference prevents other threads from erroneously deleting
381            // the object.
382            //
383            // As explained in the [Boost documentation][1], Increasing the
384            // reference counter can always be done with memory_order_relaxed: New
385            // references to an object can only be formed from an existing
386            // reference, and passing an existing reference from one thread to
387            // another must already provide any required synchronization.
388            //
389            // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
390            let old_size = self.inner().count.fetch_add(1, Relaxed);
391
392            // However we need to guard against massive refcounts in case someone
393            // is `mem::forget`ing Arcs. If we don't do this the count can overflow
394            // and users will use-after free. We racily saturate to `isize::MAX` on
395            // the assumption that there aren't ~2 billion threads incrementing
396            // the reference count at once. This branch will never be taken in
397            // any realistic program.
398            //
399            // We abort because such a program is incredibly degenerate, and we
400            // don't care to support it.
401            if old_size > MAX_REFCOUNT {
402                process::abort();
403            }
404        }
405
406        unsafe {
407            Arc {
408                p: ptr::NonNull::new_unchecked(self.ptr()),
409                phantom: PhantomData,
410            }
411        }
412    }
413}

424impl<T: Clone> Arc<T> {
425    /// Makes a mutable reference to the `Arc`, cloning if necessary
426    ///
427    /// This is functionally equivalent to [`Arc::make_mut`][mm] from the standard library.
428    ///
429    /// If this `Arc` is uniquely owned, `make_mut()` will provide a mutable
430    /// reference to the contents. If not, `make_mut()` will create a _new_ `Arc`
431    /// with a copy of the contents, update `this` to point to it, and provide
432    /// a mutable reference to its contents.
433    ///
434    /// This is useful for implementing copy-on-write schemes where you wish to
435    /// avoid copying things if your `Arc` is not shared.
436    ///
437    /// [mm]: https://doc.rust-lang.org/stable/std/sync/struct.Arc.html#method.make_mut
438    #[inline]
439    pub fn make_mut(this: &mut Self) -> &mut T {
440        if !this.is_unique() {
441            // Another pointer exists; clone
442            *this = Arc::new((**this).clone());
443        }
444
445        unsafe {
446            // This unsafety is ok because we're guaranteed that the pointer
447            // returned is the *only* pointer that will ever be returned to T. Our
448            // reference count is guaranteed to be 1 at this point, and we required
449            // the Arc itself to be `mut`, so we're returning the only possible
450            // reference to the inner data.
451            &mut (*this.ptr()).data
452        }
453    }
454}

456impl<T: ?Sized> Arc<T> {
457    /// Provides mutable access to the contents _if_ the `Arc` is uniquely owned.
458    #[inline]
459    pub fn get_mut(this: &mut Self) -> Option<&mut T> {
460        if this.is_unique() {
461            unsafe {
462                // See make_mut() for documentation of the threadsafety here.
463                Some(&mut (*this.ptr()).data)
464            }
465        } else {
466            None
467        }
468    }
469
470    /// Whether or not the `Arc` is a static reference.
471    #[inline]
472    pub fn is_static(&self) -> bool {
473        // Using a relaxed ordering to check for STATIC_REFCOUNT is safe, since
474        // `count` never changes between STATIC_REFCOUNT and other values.
475        self.inner().count.load(Relaxed) == STATIC_REFCOUNT
476    }
477
478    /// Whether or not the `Arc` is uniquely owned (is the refcount 1?) and not
479    /// a static reference.
480    #[inline]
481    pub fn is_unique(&self) -> bool {
482        // See the extensive discussion in [1] for why this needs to be Acquire.
483        //
484        // [1] https://github.com/servo/servo/issues/21186
485        self.inner().count.load(Acquire) == 1
486    }
487}

489impl<T: ?Sized> Drop for Arc<T> {
490    #[inline]
491    fn drop(&mut self) {
492        // NOTE(emilio): If you change anything here, make sure that the
493        // implementation in layout/style/ServoStyleConstsInlines.h matches!
494        if self.is_static() {
495            return;
496        }
497
498        // Because `fetch_sub` is already atomic, we do not need to synchronize
499        // with other threads unless we are going to delete the object.
500        if self.inner().count.fetch_sub(1, Release) != 1 {
501            return;
502        }
503
504        // FIXME(bholley): Use the updated comment when [2] is merged.
505        //
506        // This load is needed to prevent reordering of use of the data and
507        // deletion of the data.  Because it is marked `Release`, the decreasing
508        // of the reference count synchronizes with this `Acquire` load. This
509        // means that use of the data happens before decreasing the reference
510        // count, which happens before this load, which happens before the
511        // deletion of the data.
512        //
513        // As explained in the [Boost documentation][1],
514        //
515        // > It is important to enforce any possible access to the object in one
516        // > thread (through an existing reference) to *happen before* deleting
517        // > the object in a different thread. This is achieved by a "release"
518        // > operation after dropping a reference (any access to the object
519        // > through this reference must obviously happened before), and an
520        // > "acquire" operation before deleting the object.
521        //
522        // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
523        // [2]: https://github.com/rust-lang/rust/pull/41714
524        self.inner().count.load(Acquire);
525
526        unsafe {
527            self.drop_slow();
528        }
529    }
530}

665    /// `is_static` indicates whether to create a static Arc.
666    ///
667    /// `alloc` is used to get a pointer to the memory into which the
668    /// dynamically sized ArcInner<HeaderSlice<H, T>> value will be
669    /// written.  If `is_static` is true, then `alloc` must return a
670    /// pointer into some static memory allocation.  If it is false,
671    /// then `alloc` must return an allocation that can be dellocated
672    /// by calling Box::from_raw::<ArcInner<HeaderSlice<H, T>>> on it.
673    #[inline]
674    fn from_header_and_iter_alloc<F, I>(
675        alloc: F,
676        header: H,
677        mut items: I,
678        num_items: usize,
679        is_static: bool,
680    ) -> Self
681    where
682        F: FnOnce(Layout) -> *mut u8,
683        I: Iterator<Item = T>,
684    {
685        assert_ne!(size_of::<T>(), 0, "Need to think about ZST");
686
687        let inner_align = align_of::<ArcInner<HeaderSlice<H, [T; 0]>>>();
688        debug_assert!(inner_align >= align_of::<T>());
689
690        // Compute the required size for the allocation.
691        let size = {
692            // Next, synthesize a totally garbage (but properly aligned) pointer
693            // to a sequence of T.
694            let fake_slice_ptr = inner_align as *const T;
695
696            // Convert that sequence to a fat pointer. The address component of
697            // the fat pointer will be garbage, but the length will be correct.
698            let fake_slice = unsafe { slice::from_raw_parts(fake_slice_ptr, num_items) };
699
700            // Pretend the garbage address points to our allocation target (with
701            // a trailing sequence of T), rather than just a sequence of T.
702            let fake_ptr = fake_slice as *const [T] as *const ArcInner<HeaderSlice<H, [T]>>;
703            let fake_ref: &ArcInner<HeaderSlice<H, [T]>> = unsafe { &*fake_ptr };
704
705            // Use size_of_val, which will combine static information about the
706            // type with the length from the fat pointer. The garbage address
707            // will not be used.
708            mem::size_of_val(fake_ref)
709        };
710
711        let ptr: *mut ArcInner<HeaderSlice<H, [T]>>;
712        unsafe {
713            // Allocate the buffer.
714            let layout = if inner_align <= align_of::<usize>() {
715                Layout::from_size_align_unchecked(size, align_of::<usize>())
716            } else if inner_align <= align_of::<u64>() {
717                // On 32-bit platforms <T> may have 8 byte alignment while usize
718                // has 4 byte aligment.  Use u64 to avoid over-alignment.
719                // This branch will compile away in optimized builds.
720                Layout::from_size_align_unchecked(size, align_of::<u64>())
721            } else {
722                panic!("Over-aligned type not handled");
723            };
724
725            let buffer = alloc(layout);
726
727            // Synthesize the fat pointer. We do this by claiming we have a direct
728            // pointer to a [T], and then changing the type of the borrow. The key
729            // point here is that the length portion of the fat pointer applies
730            // only to the number of elements in the dynamically-sized portion of
731            // the type, so the value will be the same whether it points to a [T]
732            // or something else with a [T] as its last member.
733            let fake_slice: &mut [T] = slice::from_raw_parts_mut(buffer as *mut T, num_items);
734            ptr = fake_slice as *mut [T] as *mut ArcInner<HeaderSlice<H, [T]>>;
735
736            // Write the data.
737            //
738            // Note that any panics here (i.e. from the iterator) are safe, since
739            // we'll just leak the uninitialized memory.
740            let count = if is_static {
741                atomic::AtomicUsize::new(STATIC_REFCOUNT)
742            } else {
743                atomic::AtomicUsize::new(1)
744            };
745            ptr::write(&mut ((*ptr).count), count);
746            ptr::write(&mut ((*ptr).data.header), header);
747            if num_items != 0 {
748                let mut current: *mut T = &mut (*ptr).data.slice[0];
749                for _ in 0..num_items {
750                    ptr::write(
751                        current,
752                        items
753                            .next()
754                            .expect("ExactSizeIterator over-reported length"),
755                    );
756                    current = current.offset(1);
757                }
758                // We should have consumed the buffer exactly, maybe accounting
759                // for some padding from the alignment.
760                debug_assert!(
761                    (buffer.add(size) as usize - current as *mut u8 as usize) < inner_align
762                );
763            }
764            assert!(
765                items.next().is_none(),
766                "ExactSizeIterator under-reported length"
767            );
768        }
769        #[cfg(feature = "gecko_refcount_logging")]
770        unsafe {
771            if !is_static {
772                // FIXME(emilio): Would be so amazing to have
773                // std::intrinsics::type_name() around.
774                NS_LogCtor(ptr as *mut _, b"ServoArc\0".as_ptr() as *const _, 8)
775            }
776        }
777
778        // Return the fat Arc.
779        assert_eq!(
780            size_of::<Self>(),
781            size_of::<usize>() * 2,
782            "The Arc will be fat"
783        );
784        unsafe {
785            Arc {
786                p: ptr::NonNull::new_unchecked(ptr),
787                phantom: PhantomData,
788            }
789        }
790    }
791
792    /// Creates an Arc for a HeaderSlice using the given header struct and iterator to generate the
793    /// slice. Panics if num_items doesn't match the number of items.
794    #[inline]
795    pub fn from_header_and_iter_with_size<I>(header: H, items: I, num_items: usize) -> Self
796    where
797        I: Iterator<Item = T>,
798    {
799        Arc::from_header_and_iter_alloc(
800            |layout| {
801                // align will only ever be align_of::<usize>() or align_of::<u64>()
802                let align = layout.align();
803                unsafe {
804                    if align == mem::align_of::<usize>() {
805                        Self::allocate_buffer::<usize>(layout.size())
806                    } else {
807                        assert_eq!(align, mem::align_of::<u64>());
808                        Self::allocate_buffer::<u64>(layout.size())
809                    }
810                }
811            },
812            header,
813            items,
814            num_items,
815            /* is_static = */ false,
816        )
817    }
818
819    /// Creates an Arc for a HeaderSlice using the given header struct and
820    /// iterator to generate the slice. The resulting Arc will be fat.
821    #[inline]
822    pub fn from_header_and_iter<I>(header: H, items: I) -> Self
823    where
824        I: Iterator<Item = T> + ExactSizeIterator,
825    {
826        let len = items.len();
827        Self::from_header_and_iter_with_size(header, items, len)
828    }
829
830    #[inline]
831    unsafe fn allocate_buffer<W>(size: usize) -> *mut u8 {
832        // We use Vec because the underlying allocation machinery isn't
833        // available in stable Rust. To avoid alignment issues, we allocate
834        // words rather than bytes, rounding up to the nearest word size.
835        let words_to_allocate = divide_rounding_up(size, mem::size_of::<W>());
836        let mut vec = Vec::<W>::with_capacity(words_to_allocate);
837        vec.set_len(words_to_allocate);
838        Box::into_raw(vec.into_boxed_slice()) as *mut W as *mut u8
839    }

863/// A "thin" `Arc` containing dynamically sized data
864///
865/// This is functionally equivalent to Arc<(H, [T])>
866///
867/// When you create an `Arc` containing a dynamically sized type
868/// like `HeaderSlice<H, [T]>`, the `Arc` is represented on the stack
869/// as a "fat pointer", where the length of the slice is stored
870/// alongside the `Arc`'s pointer. In some situations you may wish to
871/// have a thin pointer instead, perhaps for FFI compatibility
872/// or space efficiency.
873///
874/// Note that we use `[T; 0]` in order to have the right alignment for `T`.
875///
876/// `ThinArc` solves this by storing the length in the allocation itself,
877/// via `HeaderSliceWithLength`.
878#[repr(C)]
879pub struct ThinArc<H, T> {
880    ptr: ptr::NonNull<ArcInner<HeaderSliceWithLength<H, [T; 0]>>>,
881    phantom: PhantomData<(H, T)>,
882}

884impl<H: fmt::Debug, T: fmt::Debug> fmt::Debug for ThinArc<H, T> {

886        fmt::Debug::fmt(self.deref(), f)
887    }

893// Synthesize a fat pointer from a thin pointer.
894//
895// See the comment around the analogous operation in from_header_and_iter.
896fn thin_to_thick<H, T>(
897    thin: *mut ArcInner<HeaderSliceWithLength<H, [T; 0]>>,
898) -> *mut ArcInner<HeaderSliceWithLength<H, [T]>> {
899    let len = unsafe { (*thin).data.header.length };
900    let fake_slice: *mut [T] = unsafe { slice::from_raw_parts_mut(thin as *mut T, len) };
901
902    fake_slice as *mut ArcInner<HeaderSliceWithLength<H, [T]>>
903}

906    /// Temporarily converts |self| into a bonafide Arc and exposes it to the
907    /// provided callback. The refcount is not modified.
908    #[inline]
909    pub fn with_arc<F, U>(&self, f: F) -> U
910    where
911        F: FnOnce(&Arc<HeaderSliceWithLength<H, [T]>>) -> U,
912    {
913        // Synthesize transient Arc, which never touches the refcount of the ArcInner.
914        let transient = unsafe {
915            mem::ManuallyDrop::new(Arc {
916                p: ptr::NonNull::new_unchecked(thin_to_thick(self.ptr.as_ptr())),
917                phantom: PhantomData,
918            })
919        };
920
921        // Expose the transient Arc to the callback, which may clone it if it wants.
922        let result = f(&transient);
923
924        // Forward the result.
925        result
926    }

1090/// A "borrowed `Arc`". This is a pointer to
1091/// a T that is known to have been allocated within an
1092/// `Arc`.
1093///
1094/// This is equivalent in guarantees to `&Arc<T>`, however it is
1095/// a bit more flexible. To obtain an `&Arc<T>` you must have
1096/// an `Arc<T>` instance somewhere pinned down until we're done with it.
1097/// It's also a direct pointer to `T`, so using this involves less pointer-chasing
1098///
1099/// However, C++ code may hand us refcounted things as pointers to T directly,
1100/// so we have to conjure up a temporary `Arc` on the stack each time.

1103/// without needing to worry about where the `Arc<T>` is.

1173/// A tagged union that can represent `Arc<A>` or `Arc<B>` while only consuming a
1174/// single word. The type is also `NonNull`, and thus can be stored in an Option
1175/// without increasing size.
1176///
1177/// This is functionally equivalent to
1178/// `enum ArcUnion<A, B> { First(Arc<A>), Second(Arc<B>)` but only takes up
1179/// up a single word of stack space.
1180///
1181/// This could probably be extended to support four types if necessary.
1182pub struct ArcUnion<A, B> {
1183    p: ptr::NonNull<()>,
1184    phantom_a: PhantomData<A>,
1185    phantom_b: PhantomData<B>,
1186}
1187
1188unsafe impl<A: Sync + Send, B: Send + Sync> Send for ArcUnion<A, B> {}

1191impl<A: PartialEq, B: PartialEq> PartialEq for ArcUnion<A, B> {
1192    fn eq(&self, other: &Self) -> bool {
1193        use crate::ArcUnionBorrow::*;
1194        match (self.borrow(), other.borrow()) {
1195            (First(x), First(y)) => x == y,
1196            (Second(x), Second(y)) => x == y,
1197            (_, _) => false,
1198        }
1199    }
1200}

1202/// This represents a borrow of an `ArcUnion`.
1203#[derive(Debug)]
1204pub enum ArcUnionBorrow<'a, A: 'a, B: 'a> {
1205    First(ArcBorrow<'a, A>),
1206    Second(ArcBorrow<'a, B>),
1207}

1209impl<A, B> ArcUnion<A, B> {
1210    unsafe fn new(ptr: *mut ()) -> Self {
1211        ArcUnion {
1212            p: ptr::NonNull::new_unchecked(ptr),
1213            phantom_a: PhantomData,
1214            phantom_b: PhantomData,
1215        }
1216    }

1218    /// Returns true if the two values are pointer-equal.
1219    #[inline]
1220    pub fn ptr_eq(this: &Self, other: &Self) -> bool {
1221        this.p == other.p
1222    }
1223
1224    #[inline]
1225    pub fn ptr(&self) -> ptr::NonNull<()> {
1226        self.p
1227    }

1229    /// Returns an enum representing a borrow of either A or B.
1230    #[inline]
1231    pub fn borrow(&self) -> ArcUnionBorrow<A, B> {
1232        if self.is_first() {
1233            let ptr = self.p.as_ptr() as *const A;
1234            let borrow = unsafe { ArcBorrow::from_ref(&*ptr) };
1235            ArcUnionBorrow::First(borrow)
1236        } else {
1237            let ptr = ((self.p.as_ptr() as usize) & !0x1) as *const B;
1238            let borrow = unsafe { ArcBorrow::from_ref(&*ptr) };
1239            ArcUnionBorrow::Second(borrow)
1240        }
1241    }

1243    /// Creates an `ArcUnion` from an instance of the first type.
1244    pub fn from_first(other: Arc<A>) -> Self {

1246    }

1248    /// Creates an `ArcUnion` from an instance of the second type.
1249    pub fn from_second(other: Arc<B>) -> Self {
1250        unsafe { Self::new(((Arc::into_raw(other) as usize) | 0x1) as *mut _) }
1251    }

1253    /// Returns true if this `ArcUnion` contains the first type.
1254    pub fn is_first(&self) -> bool {
1255        self.p.as_ptr() as usize & 0x1 == 0
1256    }
1257
1258    /// Returns true if this `ArcUnion` contains the second type.
1259    pub fn is_second(&self) -> bool {
1260        !self.is_first()
1261    }

1263    /// Returns a borrow of the first type if applicable, otherwise `None`.
1264    pub fn as_first(&self) -> Option<ArcBorrow<A>> {
1265        match self.borrow() {
1266            ArcUnionBorrow::First(x) => Some(x),
1267            ArcUnionBorrow::Second(_) => None,
1268        }
1269    }
1270
1271    /// Returns a borrow of the second type if applicable, otherwise None.
1272    pub fn as_second(&self) -> Option<ArcBorrow<B>> {
1273        match self.borrow() {
1274            ArcUnionBorrow::First(_) => None,
1275            ArcUnionBorrow::Second(x) => Some(x),
1276        }
1277    }
