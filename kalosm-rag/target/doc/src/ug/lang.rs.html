452    pub fn broadcast<S: Into<Shape>>(arg: Ast, shape: S) -> Result<Ast> {
453        let shape: Shape = shape.into();
454        let dtype = arg.dtype;
455        if arg.shape.rank() > shape.rank() {
456            crate::bail!(
457                "target shape {:?} has less dimensions than original shape {:?}",
458                shape,
459                arg.shape
460            )
461        }
462        let inserted_dims = shape.rank() - arg.shape.rank();
463        let mut broadcasted_dims = (0..inserted_dims).collect::<Vec<_>>();
464        for (dim_idx, &dim_len) in arg.shape.dims().iter().enumerate() {
465            let dim_idx = dim_idx + inserted_dims;
466            if shape.dims()[dim_idx] != dim_len {
467                if dim_len == 1 {
468                    broadcasted_dims.push(dim_idx)
469                } else {
470                    crate::bail!("cannot broadcast from {:?} to {shape:?}", arg.shape)
471                }
472            }
473        }
474        let op = LayoutOp::Broadcast { inserted_dims, broadcasted_dims };
475        let inner = AstInner::Layout { op, arg };
476        Ok(Ast { inner: Arc::new(inner), dtype, shape })
477    }
478
479    pub fn layout<S: Into<Shape>>(op: LayoutOp, arg: Ast, shape: S) -> Result<Ast> {
480        let shape = shape.into();
481        let dtype = arg.dtype;
482        let inner = AstInner::Layout { op, arg };
483        Ok(Ast { inner: Arc::new(inner), dtype, shape })
484    }
485
486    pub fn unary(op: UnaryOp, arg: Ast) -> Result<Ast> {
487        let dtype = match op {
488            UnaryOp::Cast(dtype) => dtype,
489            _ => arg.dtype,
490        };
491        let shape = arg.shape.clone();
492        let inner = AstInner::Unary { op, arg };
493        Ok(Ast { inner: Arc::new(inner), dtype, shape })
494    }
495
496    pub fn reduce(op: ReduceOp, arg: Ast, dim: usize) -> Result<Ast> {
497        let dtype = arg.dtype;
498        let mut shape = arg.shape.dims().to_vec();
499        if dim >= shape.len() {
500            crate::bail!("no dim {dim} in shape {shape:?}")
501        }
502        shape[dim] = 1; // keepdim by default.
503        let inner = AstInner::Reduce { op, arg, dim };
504        Ok(Ast { inner: Arc::new(inner), dtype, shape: Shape::from(shape) })
505    }
506
507    pub fn binary(op: BinaryOp, lhs: Ast, rhs: Ast) -> Result<Ast> {
508        let dtype = if lhs.dtype != rhs.dtype {
509            crate::bail!("dtype mismatch in {op:?}, lhs: {:?}, rhs: {:?}", lhs.dtype, rhs.dtype)
510        } else {
511            lhs.dtype
512        };
513        // Broadcasting has to be done explicitely except for scalar values.
514        if lhs.shape != rhs.shape && lhs.shape.rank() != 0 && rhs.shape.rank() != 0 {
515            crate::bail!("shape mismatch in {op:?}, lhs: {:?}, rhs: {:?}", lhs.shape, rhs.shape)
516        }
517        let shape = lhs.shape.clone();
518        let inner = AstInner::Binary { op, lhs, rhs };
519        Ok(Ast { inner: Arc::new(inner), dtype, shape })
520    }
521
522    pub fn broadcast_binary(op: BinaryOp, lhs: Ast, rhs: Ast) -> Result<Ast> {
523        let shape = lhs.shape.broadcast_shape_binary_op(rhs.shape(), op.as_str())?;
524        let lhs = broadcast(lhs, shape.clone())?;
525        let rhs = broadcast(rhs, shape)?;
526        binary(op, lhs, rhs)
527    }

529    pub fn cst<I: TryInto<Const> + std::fmt::Debug + Copy>(v: I) -> Result<Ast> {
530        let v: Const = match v.try_into() {
531            Err(_) => crate::bail!("unable to create const for {v:?}"),
532            Ok(v) => v,
533        };
534        let dtype = v.dtype();
535        let inner = AstInner::Const(v);
536        Ok(Ast { inner: Arc::new(inner), dtype, shape: Shape::from(()) })
537    }

548        // We use a BTreeSet rather than a HashSet here to ensure consistent ordering
549        // when converting the result to a Vec.
550        pub fn arg_ids(&self) -> std::collections::BTreeSet<ArgId> {
551            fn visit(ast: &Ast, ids: &mut std::collections::BTreeSet<ArgId>) {
552                match ast.inner.as_ref() {
553                    AstInner::Layout { arg, .. }
554                    | AstInner::Reduce { op: _, dim: _, arg }
555                    | AstInner::Unary { op: _, arg } => visit(arg, ids),
556                    AstInner::Binary { op: _, lhs, rhs } => {
557                        visit(lhs, ids);
558                        visit(rhs, ids)
559                    }
560                    AstInner::Load { src, layout: _ } => {
561                        ids.insert(*src);
562                    }
563                    AstInner::Id { .. } | AstInner::Const(_) => {}
564                }
565            }
566            let mut ids = std::collections::BTreeSet::new();
567            visit(self, &mut ids);
568            ids
569        }
570    }

713        instrs: Vec<Instr>,
714        args: Vec<(Arg, usize)>,
715        launch_config: LaunchConfig,
716    }
717
718    impl Kernel {
719        pub fn launch_config(&self) -> &LaunchConfig {
720            &self.launch_config
721        }
722
723        pub fn instrs(&self) -> &Vec<Instr> {
724            &self.instrs
725        }
726
727        pub fn args(&self) -> &[(Arg, usize)] {
728            &self.args
729        }
730
731        pub fn new(
732            instrs: Vec<Instr>,
733            args: Vec<(Arg, usize)>,
734            launch_config: LaunchConfig,
735        ) -> Self {
736            Self { instrs, args, launch_config }
737        }
738
739        pub fn from_instrs(instrs: Vec<Instr>) -> crate::Result<Self> {
740            let mut args = vec![];
741            for (line_no, instr) in instrs.iter().enumerate() {
742                if let Instr::DefineGlobal { index, dtype } = instr {
743                    if *index != args.len() {
744                        crate::bail!("unexpected order for arguments in kernel {instrs:?}")
745                    }
746                    let arg_id = ArgId::new();
747                    let arg = Arg::new(arg_id, Type::Ptr(*dtype));
748                    args.push((arg, line_no));
749                }
750            }
751            let launch_config = LaunchConfig { grid_dim: 1, block_dim: 1, shared_mem: 0 };
752            Ok(Self { instrs, args, launch_config })
753        }
754
755        pub fn flops_mem_per_thread(&self) -> Result<super::FlopsMem> {
756            let mut flops = 0usize;
757            let mut mem = 0usize;
758            let mut mults = vec![];
759            let mut mult = 1usize;
760            for instr in self.instrs.iter() {
761                match instr {
762                    Instr::Load { src: _, offset: _, dtype }
763                    | Instr::Store { dst: _, offset: _, value: _, dtype } => {
764                        mem += mult * dtype.size_in_bytes()
765                    }
766                    Instr::Range { lo, up, step, end_idx: _ } => {
767                        mults.push(mult);
768                        let lo = match lo {
769                            A::Const(Const::I64(lo)) => *lo,
770                            A::Const(Const::I32(lo)) => *lo as i64,
771                            A::Const(_) => {
772                                crate::bail!("range lo is not a const i32/i64")
773                            }
774                            A::Var(lo) => match self.instrs[lo.0] {
775                                Instr::Const(Const::I64(lo)) => lo,
776                                Instr::Const(Const::I32(lo)) => lo as i64,
777                                _ => crate::bail!("range lo is not a const"),
778                            },
779                        };
780                        let up = match up {
781                            A::Const(Const::I32(up)) => *up as i64,
782                            A::Const(Const::I64(up)) => *up,
783                            A::Const(_) => {
784                                crate::bail!("range up is not a const i32/i64")
785                            }
786                            A::Var(up) => match self.instrs[up.0] {
787                                Instr::Const(Const::I32(up)) => up as i64,
788                                Instr::Const(Const::I64(up)) => up,
789                                _ => crate::bail!("range lo is not a const"),
790                            },
791                        };
792                        mult *= (up - lo).max(0) as usize / step;
793                    }
794                    Instr::EndRange { .. } => match mults.pop() {
795                        None => crate::bail!("unexpected EndRange"),
796                        Some(m) => mult = m,
797                    },
798                    Instr::Unary { .. } | Instr::Binary { .. } => flops += mult,
799                    Instr::DefineGlobal { .. }
800                    | Instr::DefineLocal { .. }
801                    | Instr::DefineAcc(_)
802                    | Instr::Special(_)
803                    | Instr::Assign { .. }
804                    | Instr::Const(_)
805                    | Instr::If { .. }
806                    | Instr::EndIf
807                    | Instr::Barrier
808                    | Instr::ReduceLocal { .. } => {}
809                }
810            }
811            Ok(super::FlopsMem { flops, mem_in_bytes: mem })
812        }
813    }
814
815    impl std::fmt::Debug for Kernel {
816        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
817            for (var_id, instr) in self.instrs.iter().enumerate() {
818                writeln!(f, "{var_id:03} {instr:?}")?
819            }
820            Ok(())
821        }
822    }
