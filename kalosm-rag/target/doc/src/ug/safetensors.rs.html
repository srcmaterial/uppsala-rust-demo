16    /// Creates a wrapper around a memory mapped file and deserialize the safetensors header.
17    ///
18    /// # Safety
19    ///
20    /// The unsafe is inherited from [`memmap2::MmapOptions`].
21    pub unsafe fn new<P: AsRef<Path>>(p: P) -> Result<Self> {
22        let p = p.as_ref();
23        let file = std::fs::File::open(p).map_err(|e| Error::from(e).with_path(p))?;
24        let file =
25            memmap2::MmapOptions::new().map(&file).map_err(|e| Error::from(e).with_path(p))?;
26        let safetensors = yoke::Yoke::<SafeTensors_<'static>, memmap2::Mmap>::try_attach_to_cart(
27            file,
28            |data: &[u8]| {
29                let st = safetensors::SafeTensors::deserialize(data)
30                    .map_err(|e| Error::from(e).with_path(p))?;
31                Ok::<_, Error>(SafeTensors_(st))
32            },
33        )?;
34        Ok(Self { safetensors: vec![safetensors], routing: None })
35    }

37    /// Creates a wrapper around multiple memory mapped file and deserialize the safetensors headers.
38    ///
39    /// If a tensor name appears in multiple files, the last entry is returned.
40    ///
41    /// # Safety
42    ///
43    /// The unsafe is inherited from [`memmap2::MmapOptions`].
44    pub unsafe fn multi<P: AsRef<Path>>(paths: &[P]) -> Result<Self> {
45        let mut routing = HashMap::new();
46        let mut safetensors = vec![];
47        for (index, p) in paths.iter().enumerate() {
48            let p = p.as_ref();
49            let file = std::fs::File::open(p).map_err(|e| Error::from(e).with_path(p))?;
50            let file =
51                memmap2::MmapOptions::new().map(&file).map_err(|e| Error::from(e).with_path(p))?;
52            let data = yoke::Yoke::<SafeTensors_<'static>, memmap2::Mmap>::try_attach_to_cart(
53                file,
54                |data: &[u8]| {
55                    let st = safetensors::SafeTensors::deserialize(data)
56                        .map_err(|e| Error::from(e).with_path(p))?;
57                    Ok::<_, Error>(SafeTensors_(st))
58                },
59            )?;
60            for k in data.get().0.names() {
61                routing.insert(k.to_string(), index);
62            }
63            safetensors.push(data)
64        }
65        Ok(Self { safetensors, routing: Some(routing) })
66    }

83    pub fn load_slice<D: Device>(&self, name: &str, device: &D) -> Result<(Shape, D::Slice)> {
84        let view = self.get(name)?;
85        let shape: Shape = view.shape().into();
86        let dtype = match view.dtype() {
87            st::Dtype::BF16 => DType::BF16,
88            st::Dtype::F16 => DType::F16,
89            st::Dtype::F32 => DType::F32,
90            st::Dtype::I32 => DType::I32,
91            st::Dtype::I64 => DType::I64,
92            dtype => crate::bail!("unsupported dtype for {name}: {dtype:?}"),
93        };
94        let mut slice = unsafe { device.allocate_uninit(dtype, shape.num_elements()) }?;
95        match dtype {
96            DType::F16 => {
97                let data = convert_slice::<half::f16>(view.data());
98                slice.copy_host_to_device(&data)?
99            }
100            DType::BF16 => {
101                let data = convert_slice::<half::bf16>(view.data());
102                slice.copy_host_to_device(&data)?
103            }
104            DType::F32 => {
105                let data = convert_slice::<f32>(view.data());
106                slice.copy_host_to_device(&data)?
107            }
108            DType::I32 => {
109                let data = convert_slice::<i32>(view.data());
110                slice.copy_host_to_device(&data)?
111            }
112            DType::I64 => {
113                let data = convert_slice::<i64>(view.data());
114                slice.copy_host_to_device(&data)?
115            }
116        };
117        Ok((shape, slice))
118    }

120    pub fn load_slice_with_cast<D: Device>(
121        &self,
122        name: &str,
123        dst_dtype: DType,
124        device: &D,
125    ) -> Result<(Shape, D::Slice)> {
126        let view = self.get(name)?;
127        let shape: Shape = view.shape().into();
128        let src_dtype = match view.dtype() {
129            st::Dtype::BF16 => DType::BF16,
130            st::Dtype::F16 => DType::F16,
131            st::Dtype::F32 => DType::F32,
132            st::Dtype::I32 => DType::I32,
133            st::Dtype::I64 => DType::I64,
134            dtype => crate::bail!("unsupported dtype for {name}: {dtype:?}"),
135        };
136        if dst_dtype == src_dtype {
137            return self.load_slice(name, device);
138        }
139        let mut slice = unsafe { device.allocate_uninit(dst_dtype, shape.num_elements()) }?;
140        // The pattern matching below is overly verbose but new dtypes are not added that often
141        match (src_dtype, dst_dtype) {
142            // BF16 convs
143            (DType::BF16, DType::BF16) => {
144                let data = convert_slice::<half::bf16>(view.data());
145                slice.copy_host_to_device(&data)?
146            }
147            (DType::BF16, DType::F16) => {
148                let data = convert_slice::<half::bf16>(view.data());
149                let data: Vec<_> =
150                    data.iter().map(|v| half::f16::from_f32(f32::from(*v))).collect();
151                slice.copy_host_to_device(&data)?
152            }
153            (DType::BF16, DType::F32) => {

281    pub fn get(&self, name: &str) -> Result<st::TensorView<'_>> {
282        let index = match &self.routing {
283            None => 0,
284            Some(routing) => {
285                let index = routing
286                    .get(name)
287                    .ok_or_else(|| Error::CannotFindTensor { path: name.to_string() }.bt())?;
288                *index
289            }
290        };
291        Ok(self.safetensors[index].get().0.tensor(name)?)
292    }

295pub fn convert_slice<T: WithDType>(data: &[u8]) -> std::borrow::Cow<'_, [T]> {
296    let size_in_bytes = T::DTYPE.size_in_bytes();
297    let elem_count = data.len() / size_in_bytes;
298    if (data.as_ptr() as usize) % size_in_bytes == 0 {
299        // SAFETY This is safe because we just checked that this
300        // was correctly aligned.
301        let data: &[T] =
302            unsafe { std::slice::from_raw_parts(data.as_ptr() as *const T, elem_count) };
303        std::borrow::Cow::Borrowed(data)
304    } else {
305        // XXX: We need to specify `T` here, otherwise the compiler will infer u8 because of the following cast
306        // Making this vector too small to fit a full f16/f32/f64 weights, resulting in out-of-bounds access
307        let mut c: Vec<T> = Vec::with_capacity(elem_count);
308        // SAFETY: We just created c, so the allocated memory is necessarily
309        // contiguous and non overlapping with the view's data.
310        // We're downgrading the `c` pointer from T to u8, which removes alignment
311        // constraints.
312        unsafe {
313            std::ptr::copy_nonoverlapping(data.as_ptr(), c.as_mut_ptr() as *mut u8, data.len());
314            c.set_len(elem_count)
315        }
316        std::borrow::Cow::Owned(c)
317    }
