5// ssa::Instr are indexed based on their line number which is not convenient when
6// combining blocks of generated instructions
7#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
8pub struct Id(usize);
9impl Id {
10    #[allow(clippy::new_without_default)]
11    pub fn new() -> Self {
12        // https://users.rust-lang.org/t/idiomatic-rust-way-to-generate-unique-id/33805
13        use std::sync::atomic;
14        static COUNTER: atomic::AtomicUsize = atomic::AtomicUsize::new(1);
15        Self(COUNTER.fetch_add(1, atomic::Ordering::Relaxed))
16    }
17
18    pub fn to_varid(self) -> ssa::VarId {
19        ssa::VarId::new(self.0)
20    }
21
22    pub fn to_a(self) -> ssa::A {
23        ssa::A::Var(ssa::VarId::new(self.0))
24    }
25
26    pub fn from_varid(v: ssa::VarId) -> Id {
27        Id(v.as_usize())
28    }
29}
30
31impl From<Id> for ssa::A {
32    fn from(val: Id) -> Self {
33        val.to_a()
34    }
35}

37// ssa like instructions but with explicit dst
38#[derive(Debug)]
39pub struct Block(pub Vec<(Id, SsaI)>);

41#[derive(Debug)]
42pub struct Range {
43    range_id: Id,
44    erange_id: Id,
45}
46
47impl Range {
48    pub fn id(&self) -> Id {
49        self.range_id
50    }
51}
52
53impl Block {
54    pub fn range<L: Into<ssa::A>, U: Into<ssa::A>>(&mut self, lo: L, up: U, step: usize) -> Range {
55        let (range_id, erange_id) = (Id::new(), Id::new());
56        let range =
57            SsaI::Range { lo: lo.into(), up: up.into(), step, end_idx: erange_id.to_varid() };
58        self.0.push((range_id, range));
59        Range { range_id, erange_id }
60    }
61
62    pub fn end_range(&mut self, range: Range) -> Result<()> {
63        let erange = ssa::Instr::EndRange { start_idx: range.range_id.to_varid() };
64        self.0.push((range.erange_id, erange));
65        Ok(())
66    }
67
68    pub fn add(&mut self, src_id: Id, v: i32) -> Id {
69        if v == 0 {
70            src_id
71        } else {
72            let dst_id = Id::new();
73            self.0.push((
74                dst_id,
75                SsaI::Binary {
76                    op: lang::BinaryOp::Add,
77                    lhs: src_id.to_a(),
78                    rhs: v.into(),
79                    dtype: lang::DType::I32,
80                },
81            ));
82            dst_id
83        }
84    }
85
86    pub fn mul(&mut self, src_id: Id, v: i32) -> Id {
87        if v == 1 {
88            src_id
89        } else {
90            let dst_id = Id::new();
91            self.0.push((
92                dst_id,
93                SsaI::Binary {
94                    op: lang::BinaryOp::Mul,
95                    lhs: src_id.to_a(),
96                    rhs: v.into(),
97                    dtype: lang::DType::I32,
98                },
99            ));
100            dst_id
101        }
102    }
103
104    pub fn cst<C: Into<lang::Const>>(&mut self, c: C) -> Id {
105        self.push(SsaI::Const(c.into()))
106    }
107
108    pub fn push(&mut self, inst: SsaI) -> Id {
109        let id = Id::new();
110        self.0.push((id, inst));
111        id
112    }
113
114    pub fn unary<I: Into<ssa::A>>(&mut self, op: lang::UnaryOp, arg: I, dtype: lang::DType) -> Id {
115        let id = Id::new();
116        let op = SsaI::Unary { op, arg: arg.into(), dtype };
117        self.0.push((id, op));
118        id
119    }
120
121    pub fn binary<I1: Into<ssa::A>, I2: Into<ssa::A>>(
122        &mut self,
123        op: lang::BinaryOp,
124        lhs: I1,
125        rhs: I2,
126        dtype: lang::DType,
127    ) -> Id {
128        let id = Id::new();
129        let op = SsaI::Binary { op, lhs: lhs.into(), rhs: rhs.into(), dtype };
130        self.0.push((id, op));
131        id
132    }
133
134    pub fn empty() -> Self {
135        Self(vec![])
136    }
137
138    pub fn new(instrs: Vec<(Id, SsaI)>) -> Self {
139        Self(instrs)
140    }

142    // This switches all the VarId to be in line number rather than "random" unique identifiers.
143    pub fn relocate(&self) -> Result<Vec<SsaI>> {
144        let mut per_id = std::collections::HashMap::new();
145        for (line_idx, (id, _)) in self.0.iter().enumerate() {
146            let line_idx = ssa::VarId::new(line_idx);
147            per_id.insert(id, line_idx);
148        }
149        let mut instrs = vec![];
150        for (_, instr) in self.0.iter() {
151            let get_id = |id: ssa::VarId| match per_id.get(&Id::from_varid(id)) {
152                Some(v) => Ok(*v),
153                None => crate::bail!("id not found {id:?}"),
154            };
155            let get_a = |a: ssa::A| {
156                let a = match a {
157                    ssa::A::Var(v) => ssa::A::Var(get_id(v)?),
158                    ssa::A::Const(c) => ssa::A::Const(c),
159                };
160                Ok::<_, crate::Error>(a)
161            };
162            let instr = match instr {
163                SsaI::Store { dst, offset, value, dtype } => {
164                    let dst = get_id(*dst)?;
165                    let offset = get_a(*offset)?;
166                    let value = get_a(*value)?;
167                    SsaI::Store { dst, offset, value, dtype: *dtype }
168                }
169                SsaI::If { cond, end_idx } => {
170                    let cond = get_a(*cond)?;
171                    let end_idx = get_id(*end_idx)?;
172                    SsaI::If { cond, end_idx }
173                }
174                SsaI::Range { lo, up, step, end_idx } => {
175                    let lo = get_a(*lo)?;
176                    let up = get_a(*up)?;
177                    let end_idx = get_id(*end_idx)?;
178                    SsaI::Range { lo, up, step: *step, end_idx }
179                }
180                SsaI::Load { src, offset, dtype } => {
181                    let src = get_id(*src)?;
182                    let offset = get_a(*offset)?;
183                    SsaI::Load { src, offset, dtype: *dtype }
184                }
185                SsaI::Const(c) => SsaI::Const(*c),
186                SsaI::Binary { op, lhs, rhs, dtype } => {
187                    let lhs = get_a(*lhs)?;
188                    let rhs = get_a(*rhs)?;
189                    SsaI::Binary { op: *op, lhs, rhs, dtype: *dtype }
190                }
191                SsaI::Unary { op, arg, dtype } => {
192                    let arg = get_a(*arg)?;
193                    SsaI::Unary { op: *op, arg, dtype: *dtype }
194                }
195                SsaI::DefineAcc(c) => SsaI::DefineAcc(*c),
196                SsaI::Assign { dst, src } => {
197                    let dst = get_id(*dst)?;
198                    let src = get_a(*src)?;
199                    SsaI::Assign { dst, src }
200                }
201                SsaI::Special(s) => SsaI::Special(*s),
202                SsaI::DefineLocal { size, dtype } => {
203                    SsaI::DefineLocal { size: *size, dtype: *dtype }
204                }
205                SsaI::DefineGlobal { index, dtype } => {
206                    SsaI::DefineGlobal { index: *index, dtype: *dtype }
207                }
208                SsaI::Barrier => SsaI::Barrier,
209                SsaI::ReduceLocal { op, arg, dtype } => {
210                    let arg = get_a(*arg)?;
211                    SsaI::ReduceLocal { op: *op, arg, dtype: *dtype }
212                }
213                SsaI::EndIf => SsaI::EndIf,
214                SsaI::EndRange { start_idx } => {
215                    let start_idx = get_id(*start_idx)?;
216                    SsaI::EndRange { start_idx }
217                }
218            };
219            instrs.push(instr)
220        }
221        Ok(instrs)
222    }
