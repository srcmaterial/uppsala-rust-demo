15/// A metric is a way to compare two pieces of data. It is used to evaluate the performance of a model.
16pub trait Metric<T> {
17    /// The range of values that this metric can return.
18    const RANGE: RangeInclusive<f64> = 0.0..=1.0;
19
20    /// Compute the distance between this piece of data and another piece of data.
21    fn distance(&mut self, first: &T, other: &T) -> impl Future<Output = f64> + Send;
22}

24#[cfg(feature = "bert")]
25/// A metric that uses the Bert model to compute the distance between two strings.
26pub struct BertDistance {
27    bert: Bert,
28}
29
30#[cfg(feature = "bert")]
31impl BertDistance {
32    /// Create a new BertDistance metric.
33    pub fn new(model: Bert) -> Self {
34        BertDistance { bert: model }
35    }
36}
37
38#[cfg(feature = "bert")]
39impl<S: ToString + Send + Sync> Metric<S> for BertDistance {
40    async fn distance(&mut self, first: &S, other: &S) -> f64 {
41        let embeddings = self
42            .bert
43            .embed_vec(vec![first.to_string(), other.to_string()])
44            .await
45            .expect("Failed to embed text with Bert");
46        let [first_embedding, other_embedding] = embeddings
47            .try_into()
48            .expect("Failed to get two embeddings from the batch of two input texts from Bert");
49        first_embedding.cosine_similarity(&other_embedding).into()
50    }
51}

53/// A set of test cases to evaluate a model.
54pub struct TestCases<I> {
55    name: String,
56    tests: Vec<TestCase<I>>,
57}
58
59impl<I> Default for TestCases<I> {
60    #[track_caller]
61    fn default() -> Self {
62        Self::new()
63    }
64}
65
66impl<I> TestCases<I> {
67    /// Create a new set of test cases.
68    #[track_caller]
69    pub fn new() -> Self {
70        TestCases {
71            name: std::panic::Location::caller().to_string(),
72            tests: Vec::new(),
73        }
74    }
75
76    /// Set the name of this set of test cases.
77    pub fn with_name(mut self, name: impl Display) -> Self {
78        self.name = name.to_string();
79        self
80    }
81
82    /// Add a test case to this set of test cases.
83    pub fn with_case(mut self, expected: I, actual: I) -> Self {
84        self.tests.push(TestCase { expected, actual });
85        self
86    }
87
88    /// Push a test case to this set of test cases.
89    pub fn push_case(&mut self, expected: I, actual: I) {
90        self.tests.push(TestCase { expected, actual });
91    }
92
93    /// Evaluate a model using this set of test cases.
94    pub async fn evaluate<M: Metric<I>>(&mut self, metric: &mut M) -> EvaluationResult<'_, I> {
95        let mut values = Vec::new();
96        for case in &self.tests {
97            let TestCase { expected, actual } = case;
98            let distance = metric.distance(expected, actual).await;
99            values.push(TestCaseScored {
100                case,
101                score: distance,
102            });
103        }
104        values.sort_by(|a, b| a.score.partial_cmp(&b.score).unwrap());
105        EvaluationResult {
106            name: self.name.clone(),
107            histogram: OnceLock::new(),
108            tests: values,
109            range: M::RANGE,
110        }
111    }
112}

114/// The result of evaluating a model using a set of test cases.
115#[derive(Clone)]
116pub struct EvaluationResult<'a, I> {
117    name: String,
118    histogram: OnceLock<Histogram<u64>>,
119    tests: Vec<TestCaseScored<'a, I>>,
120    range: RangeInclusive<f64>,
121}
122
123impl<I> EvaluationResult<'_, I> {
124    const SCALE_FACTOR: f64 = 10000.0;
125
126    fn histogram_scale_factor(&self) -> f64 {
127        let min = self.range.start();
128        let max = self.range.end();
129        Self::SCALE_FACTOR / (max - min)
130    }
131
132    fn scale_value(&self, value: f64) -> f64 {
133        let min = self.range.start();
134        let scale_factor = self.histogram_scale_factor();
135        (value - min) * scale_factor
136    }
137
138    fn unscale_value(&self, value: f64) -> f64 {
139        let min = self.range.start();
140        let scale_factor = self.histogram_scale_factor();
141        value / scale_factor + min
142    }
143
144    fn histogram(&self) -> &Histogram<u64> {
145        self.histogram.get_or_init(|| {
146            let mut histogram = Histogram::<u64>::new(3).unwrap();
147            for test in &self.tests {
148                histogram
149                    .record(self.scale_value(test.score) as u64)
150                    .expect("Failed to record score");
151            }
152            histogram
153        })
154    }
155
156    /// Get the mean score of this EvaluationResult.
157    pub fn mean_score(&self) -> f64 {
158        self.unscale_value(self.histogram().mean())
159    }
160
161    /// Get the median score of this EvaluationResult.
162    pub fn median_score(&self) -> f64 {
163        self.unscale_value(self.histogram().value_at_percentile(50.0) as f64)
164    }
165
166    /// Get the minimum score of this EvaluationResult.
167    pub fn min_score(&self) -> f64 {
168        self.unscale_value(self.histogram().min() as f64)
169    }
170
171    /// Get the maximum score of this EvaluationResult.
172    pub fn max_score(&self) -> f64 {
173        self.unscale_value(self.histogram().max() as f64)
174    }
175
176    /// Get the score at a given quantile of this EvaluationResult.
177    pub fn quantile_score(&self, quantile: f64) -> f64 {
178        self.unscale_value(self.histogram().value_at_percentile(quantile * 100.0) as f64)
179    }
180
181    /// Normalize a single score to a value between 0 and 1.
182    pub fn normalize_score(&self, score: f64) -> f64 {
183        let min = self.range.start();
184        let max = self.range.end();
185        (score - min) / (max - min)
186    }
187
188    /// Create a new value from a normalized score.
189    pub fn denormalize_score(&self, score: f64) -> f64 {
190        let min = self.range.start();
191        let max = self.range.end();
192        score * (max - min) + min
193    }
194
195    /// Normalize the EvaluationResult's score to a value between 0 and 1.
196    pub fn normalized(self) -> Self {
197        let mut normalized_values = self.tests;
198        let min = self.range.start();
199        let max = self.range.end();
200        let range = max - min;
201        for test in &mut normalized_values {
202            test.score = (test.score - min) / range;
203        }
204        EvaluationResult {
205            name: self.name,
206            histogram: OnceLock::new(),
207            tests: normalized_values,
208            range: 0.0..=1.0,
209        }
210    }
211}

213impl<I: Display> std::fmt::Display for EvaluationResult<'_, I> {
214    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
215        let histogram = self.histogram();
216
217        let mut statistics = Table::new();
218        statistics.set_header(vec!["Statistic", "Value"]);
219        statistics.add_row(vec![
220            Cell::new("Mean"),
221            Cell::new(format!("{:.2}", self.mean_score())),
222        ]);
223        statistics.add_row(vec![
224            Cell::new("Median"),
225            Cell::new(format!("{:.2}", self.median_score())),
226        ]);
227        statistics.add_row(vec![
228            Cell::new("Min"),
229            Cell::new(format!("{:.2}", self.min_score())),
230        ]);
231        statistics.add_row(vec![
232            Cell::new("Max"),
233            Cell::new(format!("{:.2}", self.max_score())),
234        ]);
235        statistics.add_row(vec![
236            Cell::new("25th Percentile"),
237            Cell::new(format!("{:.2}", self.quantile_score(0.25))),
238        ]);
239        statistics.add_row(vec![
240            Cell::new("75th Percentile"),
241            Cell::new(format!("{:.2}", self.quantile_score(0.75))),
242        ]);
243
244        writeln!(f, "{}", statistics)?;
245
246        let mut table = Table::new();
247        table.set_header(vec!["Expected Output", "Actual Output", "Score"]);
248
249        let bottom_third_of_metric =
250            self.range.start() + (self.range.end() - self.range.start()) / 3.0;
251        let bottom_half_of_metric =
252            self.range.start() + (self.range.end() - self.range.start()) / 2.0;
253
254        fn create_cell(score: f64, quantile: f64) -> Cell {
255            if quantile <= 0.1 {
256                Cell::new(format!("{:.2} (low outlier)", score))
257            } else if quantile <= 0.9 {
258                Cell::new(format!("{:.2}", score))
259            } else {
260                Cell::new(format!("{:.2} (high outlier)", score))
261            }
262        }
263
264        let buckets = [
265            (comfy_table::Color::Red, bottom_third_of_metric),
266            (comfy_table::Color::Yellow, bottom_half_of_metric),
267            (comfy_table::Color::Green, f64::INFINITY),
268        ];
269
270        let mut test_iter = self.tests.iter().peekable();
271        for (color, max) in buckets {
272            let mut count = 0;
273            while let Some(test) = test_iter.next_if(|test| test.score <= max) {
274                let quantile =
275                    histogram.percentile_below((test.score * Self::SCALE_FACTOR) as u64) / 100.0;
276
277                let score_cell = create_cell(test.score, quantile).fg(color);
278                let mut row = Row::new();
279                row.add_cell(Cell::new(&test.case.expected))
280                    .add_cell(Cell::new(&test.case.actual))
281                    .add_cell(score_cell);
282                table.add_row(row);
283                count += 1;
284
285                if count >= 5 {
286                    let mut remaining_matching_tests = 0;
287                    let mut total_score = 0.0;
288                    for test in test_iter.by_ref() {
289                        if test.score > max {
290                            break;
291                        }
292                        total_score += test.score;
293                        remaining_matching_tests += 1;
294                    }
295                    if remaining_matching_tests > 0 {
296                        let mut row = Row::new();
297                        row.add_cell(Cell::new(format!("... {} more", remaining_matching_tests)))
298                            .add_cell(Cell::new(""))
299                            .add_cell(
300                                Cell::new(format!(
301                                    "{:.2} (average)",
302                                    total_score / remaining_matching_tests as f64
303                                ))
304                                .fg(color),
305                            );
306                        table.add_row(row);
307                    }
308                    break;
309                }
310            }
311        }
312
313        writeln!(f, "{}", table)?;
314
315        let mut buckets = [0; 10];
316
317        for test in &self.tests {
318            let normalized_score = self.normalize_score(test.score);
319            let bucket = (normalized_score * 10.0) as usize;
320            buckets[bucket.min(9)] += 1;
321        }
322
323        let max_width = *buckets.iter().max().unwrap();
324
325        // We need to scale the graph to fit in the terminal if it is too wide.
326        let scale_factor = if max_width > 50 {
327            50.0 / max_width as f64
328        } else {
329            1.0
330        };
331
332        let max_width = ((max_width as f64 * scale_factor) as usize).max(3);
333
334        writeln!(f, "| Score Histogram {} |", " ".repeat(max_width - 3))?;
335
336        for (i, bucket) in buckets.iter().enumerate() {
337            let min_bucket = self.denormalize_score(i as f64 / 10.0);
338            let max_bucket = self.denormalize_score((i + 1) as f64 / 10.0);
339            let bucket = (*bucket as f64 * scale_factor) as usize;
340            writeln!(
341                f,
342                "| {:.2} - {:.2}: {}{} |",
343                min_bucket,
344                max_bucket,
345                "*".repeat(bucket),
346                " ".repeat(max_width - bucket)
347            )?;
348        }
349
350        Ok(())
351    }
352}
