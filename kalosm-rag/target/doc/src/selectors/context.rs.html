11/// What kind of selector matching mode we should use.
12///
13/// There are two modes of selector matching. The difference is only noticeable
14/// in presence of pseudo-elements.
15#[derive(Clone, Copy, Debug, PartialEq)]
16pub enum MatchingMode {
17    /// Don't ignore any pseudo-element selectors.
18    Normal,
19
20    /// Ignores any stateless pseudo-element selectors in the rightmost sequence
21    /// of simple selectors.
22    ///
23    /// This is useful, for example, to match against ::before when you aren't a
24    /// pseudo-element yourself.
25    ///
26    /// For example, in presence of `::before:hover`, it would never match, but
27    /// `::before` would be ignored as in "matching".
28    ///
29    /// It's required for all the selectors you match using this mode to have a
30    /// pseudo-element.
31    ForStatelessPseudoElement,
32}
33
34/// The mode to use when matching unvisited and visited links.
35#[derive(Clone, Copy, Debug, Eq, PartialEq)]
36pub enum VisitedHandlingMode {
37    /// All links are matched as if they are unvisted.
38    AllLinksUnvisited,
39    /// All links are matched as if they are visited and unvisited (both :link
40    /// and :visited match).
41    ///
42    /// This is intended to be used from invalidation code, to be conservative
43    /// about whether we need to restyle a link.
44    AllLinksVisitedAndUnvisited,
45    /// A element's "relevant link" is the element being matched if it is a link
46    /// or the nearest ancestor link. The relevant link is matched as though it
47    /// is visited, and all other links are matched as if they are unvisited.
48    RelevantLinkVisited,
49}
50
51impl VisitedHandlingMode {
52    #[inline]
53    pub fn matches_visited(&self) -> bool {
54        matches!(
55            *self,
56            VisitedHandlingMode::RelevantLinkVisited |
57                VisitedHandlingMode::AllLinksVisitedAndUnvisited
58        )
59    }
60
61    #[inline]
62    pub fn matches_unvisited(&self) -> bool {
63        matches!(
64            *self,
65            VisitedHandlingMode::AllLinksUnvisited |
66                VisitedHandlingMode::AllLinksVisitedAndUnvisited
67        )
68    }
69}

87/// Which quirks mode is this document in.
88///
89/// See: https://quirks.spec.whatwg.org/
90#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
91pub enum QuirksMode {
92    /// Quirks mode.
93    Quirks,
94    /// Limited quirks mode.
95    LimitedQuirks,
96    /// No quirks mode.
97    NoQuirks,
98}
99
100impl QuirksMode {
101    #[inline]
102    pub fn classes_and_ids_case_sensitivity(self) -> CaseSensitivity {
103        match self {
104            QuirksMode::NoQuirks | QuirksMode::LimitedQuirks => CaseSensitivity::CaseSensitive,
105            QuirksMode::Quirks => CaseSensitivity::AsciiCaseInsensitive,
106        }
107    }
108}

110/// Whether or not this matching considered relative selector.
111#[derive(Clone, Copy, Debug, PartialEq)]
112pub enum RelativeSelectorMatchingState {
113    /// Was not considered for any relative selector.
114    None,
115    /// Relative selector was considered for a match, but the element to be
116    /// under matching would not anchor the relative selector. i.e. The
117    /// relative selector was not part of the first compound selector (in match
118    /// order).
119    Considered,
120    /// Same as above, but the relative selector was part of the first compound
121    /// selector (in match order).
122    ConsideredAnchor,
123}

125/// Data associated with the matching process for a element.  This context is
126/// used across many selectors for an element, so it's not appropriate for
127/// transient data that applies to only a single selector.
128pub struct MatchingContext<'a, Impl>
129where
130    Impl: SelectorImpl,
131{
132    /// Input with the matching mode we should use when matching selectors.
133    matching_mode: MatchingMode,
134    /// Input with the bloom filter used to fast-reject selectors.
135    pub bloom_filter: Option<&'a BloomFilter>,
136    /// A cache to speed up nth-index-like selectors.
137    pub nth_index_cache: &'a mut NthIndexCache,
138    /// The element which is going to match :scope pseudo-class. It can be
139    /// either one :scope element, or the scoping element.
140    ///
141    /// Note that, although in theory there can be multiple :scope elements,
142    /// in current specs, at most one is specified, and when there is one,
143    /// scoping element is not relevant anymore, so we use a single field for
144    /// them.
145    ///
146    /// When this is None, :scope will match the root element.
147    ///
148    /// See https://drafts.csswg.org/selectors-4/#scope-pseudo
149    pub scope_element: Option<OpaqueElement>,
150
151    /// The current shadow host we're collecting :host rules for.
152    pub current_host: Option<OpaqueElement>,
153
154    /// Controls how matching for links is handled.
155    visited_handling: VisitedHandlingMode,
156
157    /// The current nesting level of selectors that we're matching.
158    nesting_level: usize,
159
160    /// Whether we're inside a negation or not.
161    in_negation: bool,
162
163    /// An optional hook function for checking whether a pseudo-element
164    /// should match when matching_mode is ForStatelessPseudoElement.
165    pub pseudo_element_matching_fn: Option<&'a dyn Fn(&Impl::PseudoElement) -> bool>,
166
167    /// Extra implementation-dependent matching data.
168    pub extra_data: Impl::ExtraMatchingData<'a>,
169
170    /// The current element we're anchoring on for evaluating the relative selector.
171    current_relative_selector_anchor: Option<OpaqueElement>,
172    pub considered_relative_selector: RelativeSelectorMatchingState,
173
174    quirks_mode: QuirksMode,
175    needs_selector_flags: NeedsSelectorFlags,
176
177    /// Whether this match request should ignore nth child selectors (only expected during
178    /// invalidation).
179    ignores_nth_child_selectors_for_invalidation: IgnoreNthChildForInvalidation,
180
181    classes_and_ids_case_sensitivity: CaseSensitivity,
182    _impl: ::std::marker::PhantomData<Impl>,
183}

293    /// Runs F with a deeper nesting level.
294    #[inline]
295    pub fn nest<F, R>(&mut self, f: F) -> R
296    where
297        F: FnOnce(&mut Self) -> R,
298    {
299        self.nesting_level += 1;
300        let result = f(self);
301        self.nesting_level -= 1;
302        result
303    }
304
305    /// Runs F with a deeper nesting level, and marking ourselves in a negation,
306    /// for a :not(..) selector, for example.
307    #[inline]
308    pub fn nest_for_negation<F, R>(&mut self, f: F) -> R
309    where
310        F: FnOnce(&mut Self) -> R,
311    {
312        let old_in_negation = self.in_negation;
313        self.in_negation = true;
314        let result = self.nest(f);
315        self.in_negation = old_in_negation;
316        result
317    }

322    }
323
324    /// Runs F with a different VisitedHandlingMode.
325    #[inline]
326    pub fn with_visited_handling_mode<F, R>(
327        &mut self,
328        handling_mode: VisitedHandlingMode,
329        f: F,
330    ) -> R
331    where
332        F: FnOnce(&mut Self) -> R,
333    {
334        let original_handling_mode = self.visited_handling;
335        self.visited_handling = handling_mode;
336        let result = f(self);
337        self.visited_handling = original_handling_mode;
338        result
339    }
340
341    /// Runs F with a given shadow host which is the root of the tree whose
342    /// rules we're matching.
343    #[inline]
344    pub fn with_shadow_host<F, E, R>(&mut self, host: Option<E>, f: F) -> R
345    where
346        E: Element,
347        F: FnOnce(&mut Self) -> R,
348    {
349        let original_host = self.current_host.take();
350        self.current_host = host.map(|h| h.opaque());
351        let result = f(self);
352        self.current_host = original_host;
353        result
354    }
355
356    /// Returns the current shadow host whose shadow root we're matching rules
357    /// against.
358    #[inline]
359    pub fn shadow_host(&self) -> Option<OpaqueElement> {
360        self.current_host
361    }
362
363    /// Runs F with a deeper nesting level, with the given element as the anchor,
364    /// for a :has(...) selector, for example.
365    #[inline]
366    pub fn nest_for_relative_selector<F, R>(&mut self, anchor: OpaqueElement, f: F) -> R
367    where
368        F: FnOnce(&mut Self) -> R,
369    {
370        debug_assert!(
371            self.current_relative_selector_anchor.is_none(),
372            "Nesting should've been rejected at parse time"
373        );
374        self.current_relative_selector_anchor = Some(anchor);
375        self.considered_relative_selector = RelativeSelectorMatchingState::Considered;
376        let result = self.nest(f);
377        self.current_relative_selector_anchor = None;
378        result
379    }
380
381    /// Returns the current anchor element to evaluate the relative selector against.
382    #[inline]
383    pub fn relative_selector_anchor(&self) -> Option<OpaqueElement> {
384        self.current_relative_selector_anchor
385    }
