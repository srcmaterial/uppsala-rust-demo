9use crate::matching::{ElementSelectorFlags, MatchingContext};
10use crate::parser::SelectorImpl;
11use std::fmt::Debug;
12use std::ptr::NonNull;

14/// Opaque representation of an Element, for identity comparisons.
15#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
16pub struct OpaqueElement(NonNull<()>);
17
18unsafe impl Send for OpaqueElement {}
19
20impl OpaqueElement {
21    /// Creates a new OpaqueElement from an arbitrarily-typed pointer.
22    pub fn new<T>(ptr: &T) -> Self {
23        unsafe {
24            OpaqueElement(NonNull::new_unchecked(
25                ptr as *const T as *const () as *mut (),
26            ))
27        }
28    }
29}
30
31pub trait Element: Sized + Clone + Debug {
32    type Impl: SelectorImpl;
33
34    /// Converts self into an opaque representation.
35    fn opaque(&self) -> OpaqueElement;
36
37    fn parent_element(&self) -> Option<Self>;
38
39    /// Whether the parent node of this element is a shadow root.
40    fn parent_node_is_shadow_root(&self) -> bool;
41
42    /// The host of the containing shadow root, if any.
43    fn containing_shadow_host(&self) -> Option<Self>;
44
45    /// The parent of a given pseudo-element, after matching a pseudo-element
46    /// selector.
47    ///
48    /// This is guaranteed to be called in a pseudo-element.
49    fn pseudo_element_originating_element(&self) -> Option<Self> {
50        debug_assert!(self.is_pseudo_element());
51        self.parent_element()
52    }
53
54    /// Whether we're matching on a pseudo-element.
55    fn is_pseudo_element(&self) -> bool;
56
57    /// Skips non-element nodes
58    fn prev_sibling_element(&self) -> Option<Self>;
59
60    /// Skips non-element nodes
61    fn next_sibling_element(&self) -> Option<Self>;
62
63    /// Skips non-element nodes
64    fn first_element_child(&self) -> Option<Self>;
65
66    fn is_html_element_in_html_document(&self) -> bool;
67
68    fn has_local_name(&self, local_name: &<Self::Impl as SelectorImpl>::BorrowedLocalName) -> bool;
69
70    /// Empty string for no namespace
71    fn has_namespace(&self, ns: &<Self::Impl as SelectorImpl>::BorrowedNamespaceUrl) -> bool;
72
73    /// Whether this element and the `other` element have the same local name and namespace.
74    fn is_same_type(&self, other: &Self) -> bool;
75
76    fn attr_matches(
77        &self,
78        ns: &NamespaceConstraint<&<Self::Impl as SelectorImpl>::NamespaceUrl>,
79        local_name: &<Self::Impl as SelectorImpl>::LocalName,
80        operation: &AttrSelectorOperation<&<Self::Impl as SelectorImpl>::AttrValue>,
81    ) -> bool;
82
83    fn has_attr_in_no_namespace(
84        &self,
85        local_name: &<Self::Impl as SelectorImpl>::LocalName,
86    ) -> bool {
87        self.attr_matches(
88            &NamespaceConstraint::Specific(&crate::parser::namespace_empty_string::<Self::Impl>()),
89            local_name,
90            &AttrSelectorOperation::Exists,
91        )
92    }
93
94    fn match_non_ts_pseudo_class(
95        &self,
96        pc: &<Self::Impl as SelectorImpl>::NonTSPseudoClass,
97        context: &mut MatchingContext<Self::Impl>,
98    ) -> bool;
99
100    fn match_pseudo_element(
101        &self,
102        pe: &<Self::Impl as SelectorImpl>::PseudoElement,
103        context: &mut MatchingContext<Self::Impl>,
104    ) -> bool;
105
106    /// Sets selector flags on the elemnt itself or the parent, depending on the
107    /// flags, which indicate what kind of work may need to be performed when
108    /// DOM state changes.
109    fn apply_selector_flags(&self, flags: ElementSelectorFlags);
110
111    /// Whether this element is a `link`.
112    fn is_link(&self) -> bool;
113
114    /// Returns whether the element is an HTML <slot> element.
115    fn is_html_slot_element(&self) -> bool;
116
117    /// Returns the assigned <slot> element this element is assigned to.
118    ///
119    /// Necessary for the `::slotted` pseudo-class.
120    fn assigned_slot(&self) -> Option<Self> {
121        None
122    }
123
124    fn has_id(
125        &self,
126        id: &<Self::Impl as SelectorImpl>::Identifier,
127        case_sensitivity: CaseSensitivity,
128    ) -> bool;
129
130    fn has_class(
131        &self,
132        name: &<Self::Impl as SelectorImpl>::Identifier,
133        case_sensitivity: CaseSensitivity,
134    ) -> bool;
135
136    /// Returns the mapping from the `exportparts` attribute in the reverse
137    /// direction, that is, in an outer-tree -> inner-tree direction.
138    fn imported_part(
139        &self,
140        name: &<Self::Impl as SelectorImpl>::Identifier,
141    ) -> Option<<Self::Impl as SelectorImpl>::Identifier>;
142
143    fn is_part(&self, name: &<Self::Impl as SelectorImpl>::Identifier) -> bool;
144
145    /// Returns whether this element matches `:empty`.
146    ///
147    /// That is, whether it does not contain any child element or any non-zero-length text node.
148    /// See http://dev.w3.org/csswg/selectors-3/#empty-pseudo
149    fn is_empty(&self) -> bool;
150
151    /// Returns whether this element matches `:root`,
152    /// i.e. whether it is the root element of a document.
153    ///
154    /// Note: this can be false even if `.parent_element()` is `None`
155    /// if the parent node is a `DocumentFragment`.
156    fn is_root(&self) -> bool;
157
158    /// Returns whether this element should ignore matching nth child
159    /// selector.
160    fn ignores_nth_child_selectors(&self) -> bool {
161        false
162    }
163}
