41fn convert_result<O, E>(result: Result<O, E>, input: ParserInput, error_kind: ErrorKind) -> NomResult<O> {
42    result.map(|o| (input, o)).map_err(|_| {
43        // this is a unit bind if NomError = ()

45        nom::Err::Error(err)

99fn integer(input: ParserInput) -> NomResult<i64> {
100    let (i, _) = pair(opt(one_of("+-")), digit1)(input)?;
101
102    let int_input = &input[..input.len() - i.len()];
103    convert_result(i64::from_str(str::from_utf8(int_input).unwrap()), i, ErrorKind::Digit)

106fn real(input: ParserInput) -> NomResult<f32> {
107    let (i, _) = pair(
108        opt(one_of("+-")),
109        alt((
110            map(tuple((digit1, tag(b"."), digit0)), |_| ()),
111            map(pair(tag(b"."), digit1), |_| ()),
112        )),
113    )(input)?;
114
115    let float_input = &input[..input.len() - i.len()];
116    convert_result(f32::from_str(str::from_utf8(float_input).unwrap()), i, ErrorKind::Digit)
117}

119pub(crate) fn hex_char(input: ParserInput) -> NomResult<u8> {
120    map_res(
121        verify(take(2usize), |h: &ParserInput| {
122            h.as_bytes().iter().copied().all(is_hex_digit)
123        }),
124        |x: ParserInput| u8::from_str_radix(str::from_utf8(&x).unwrap(), 16),
125    )(input)
126}

128fn oct_char(input: ParserInput) -> NomResult<u8> {
129    map_res(
130        take_while_m_n(1, 3, is_oct_digit),
131        // Spec requires us to ignore any overflow.
132        |x: ParserInput| u16::from_str_radix(str::from_utf8(&x).unwrap(), 8).map(|o| o as u8),
133    )(input)
134}

136pub(crate) fn name(input: ParserInput) -> NomResult<Vec<u8>> {
137    preceded(
138        tag(b"/"),
139        many0(alt((
140            preceded(tag(b"#"), hex_char),
141            map_opt(take(1usize), |c: ParserInput| {
142                if c[0] != b'#' && is_regular(c[0]) {
143                    Some(c[0])
144                } else {
145                    None
146                }
147            }),
148        ))),
149    )(input)
150}

152fn escape_sequence(input: ParserInput) -> NomResult<Option<u8>> {
153    preceded(
154        tag(b"\\"),
155        alt((
156            map(oct_char, Some),
157            map(eol, |_| None),
158            map(tag(b"n"), |_| Some(b'\n')),
159            map(tag(b"r"), |_| Some(b'\r')),
160            map(tag(b"t"), |_| Some(b'\t')),
161            map(tag(b"b"), |_| Some(b'\x08')),
162            map(tag(b"f"), |_| Some(b'\x0C')),
163            map(take(1usize), |c: ParserInput| Some(c[0])),
164        )),
165    )(input)

168enum InnerLiteralString<'a> {
169    Direct(ParserInput<'a>),
170    Escape(Option<u8>),
171    Eol(ParserInput<'a>),
172    Nested(Vec<u8>),
173}

175impl InnerLiteralString<'_> {
176    fn push(&self, output: &mut Vec<u8>) {
177        match self {
178            InnerLiteralString::Direct(s) | InnerLiteralString::Eol(s) => output.extend_from_slice(s),
179            InnerLiteralString::Escape(e) => output.extend(e),
180            InnerLiteralString::Nested(n) => output.extend_from_slice(n),
181        }
182    }
183}

185fn inner_literal_string(depth: usize) -> impl Fn(ParserInput) -> NomResult<Vec<u8>> {
186    move |input| {
187        fold_many0(
188            alt((
189                map(take_while1(is_direct_literal_string), InnerLiteralString::Direct),
190                map(escape_sequence, InnerLiteralString::Escape),
191                map(eol, InnerLiteralString::Eol),
192                map(nested_literal_string(depth), InnerLiteralString::Nested),
193            )),
194            Vec::new,
195            |mut out: Vec<u8>, value| {
196                value.push(&mut out);
197                out
198            },
199        )(input)
200    }
201}
202
203fn nested_literal_string(depth: usize) -> impl Fn(ParserInput) -> NomResult<Vec<u8>> {
204    move |input| {
205        if depth == 0 {
206            map(verify(tag(b"too deep"), |_| false), |_| vec![])(input)
207        } else {
208            map(
209                delimited(tag(b"("), inner_literal_string(depth - 1), tag(b")")),
210                |mut content| {
211                    content.insert(0, b'(');
212                    content.push(b')');
213                    content
214                },
215            )(input)
216        }
217    }
218}

231fn hexadecimal_string(input: ParserInput) -> NomResult<Object> {
232    map(
233        delimited(
234            tag(b"<"),
235            terminated(
236                fold_many0(
237                    preceded(white_space, hex_digit),
238                    || -> (Vec<u8>, bool) { (Vec::new(), false) },
239                    |state, c| match state {
240                        (mut out, false) => {
241                            out.push(c << 4);
242                            (out, true)
243                        }
244                        (mut out, true) => {
245                            *out.last_mut().unwrap() |= c;
246                            (out, false)
247                        }
248                    },
249                ),
250                white_space,
251            ),
252            tag(b">"),
253        ),
254        |(bytes, _)| Object::String(bytes, StringFormat::Hexadecimal),
255    )(input)
256}

315fn stream<'a>(input: ParserInput<'a>, reader: &Reader, already_seen: &mut HashSet<ObjectId>) -> NomResult<'a, Object> {
316    let (i, dict) = terminated(dictionary, tuple((space, tag(b"stream"), space0, eol)))(input)?;
317
318    if let Ok(length) = dict.get(b"Length").and_then(|value| {
319        if let Ok(id) = value.as_reference() {
320            reader.get_object(id, already_seen).and_then(|value| value.as_i64())
321        } else {
322            value.as_i64()
323        }
324    }) {
325        if length < 0 {
326            // artificial error kind is created to allow descriptive nom errors
327            return Err(nom::Err::Failure(NomError::from_error_kind(i, ErrorKind::LengthValue)));
328        }
329        let (i, data) = terminated(take(length as usize), pair(opt(eol), tag(b"endstream")))(i)?;
330        Ok((i, Object::Stream(Stream::new(dict, data.to_vec()))))
331    } else {
332        // Return position relative to the start of the stream dictionary.
333        Ok((i, Object::Stream(Stream::with_position(dict, input.len() - i.len()))))
334    }
335}

381pub fn indirect_object(
382    input: ParserInput, offset: usize, expected_id: Option<ObjectId>, reader: &Reader,
383    already_seen: &mut HashSet<ObjectId>,
384) -> crate::Result<(ObjectId, Object)> {
385    let (id, mut object) = _indirect_object(input.slice(offset..), offset, expected_id, reader, already_seen)?;
386
387    offset_stream(&mut object, offset);
388
389    Ok((id, object))
390}
391
392fn _indirect_object<'a>(
393    input: ParserInput<'a>, offset: usize, expected_id: Option<ObjectId>, reader: &Reader,
394    already_seen: &mut HashSet<ObjectId>,
395) -> crate::Result<(ObjectId, Object)> {
396    let (i, (_, object_id)) = terminated(tuple((space, object_id)), pair(tag(b"obj"), space))(input)
397        .map_err(|_| Error::IndirectObject { offset })?;
398    if let Some(expected_id) = expected_id {
399        if object_id != expected_id {
400            return Err(crate::error::Error::ObjectIdMismatch);
401        }
402    }
403
404    let object_offset = input.len() - i.len();
405    let (_, mut object) = terminated(
406        |i: ParserInput<'a>| object(i, reader, already_seen),
407        tuple((space, opt(tag(b"endobj")), space)),
408    )(i)
409    .map_err(|_| Error::IndirectObject { offset })?;
410
411    offset_stream(&mut object, object_offset);
412
413    Ok((object_id, object))
414}

438/// Decode CrossReferenceTable
439fn xref(input: ParserInput) -> NomResult<Xref> {
440    let xref_eol = map(alt((tag(b" \r"), tag(b" \n"), tag(b"\r\n"))), |_| ());
441    let xref_entry = pair(
442        separated_pair(unsigned_int, tag(b" "), unsigned_int::<u32>),
443        delimited(tag(b" "), map(one_of("nf"), |k| k == 'n'), xref_eol),
444    );
445
446    let xref_section = pair(
447        separated_pair(unsigned_int::<usize>, tag(b" "), unsigned_int::<u32>),
448        preceded(pair(opt(tag(b" ")), eol), many0(xref_entry)),
449    );
450
451    delimited(
452        pair(tag(b"xref"), eol),
453        fold_many1(
454            xref_section,
455            || -> Xref { Xref::new(0, XrefType::CrossReferenceTable) },
456            |mut xref, ((start, _count), entries)| {
457                for (index, ((offset, generation), is_normal)) in entries.into_iter().enumerate() {
458                    if is_normal {
459                        if let Ok(generation) = generation.try_into() {
460                            xref.insert((start + index) as u32, XrefEntry::Normal { offset, generation });
461                        }
462                    }
463                }
464                xref
465            },
466        ),
467        space,
468    )(input)
469}

475pub fn xref_and_trailer(input: ParserInput, reader: &Reader) -> crate::Result<(Xref, Dictionary)> {
476    let xref_trailer = map(pair(xref, trailer), |(mut xref, trailer)| {
477        xref.size = trailer
478            .get(b"Size")
479            .and_then(Object::as_i64)
480            .map_err(|_| error::ParseError::InvalidTrailer)? as u32;
481        Ok((xref, trailer))
482    });
483    alt((
484        xref_trailer,
485        (|input| {
486            _indirect_object(input, 0, None, reader, &mut HashSet::new())
487                .map(|(_, obj)| {
488                    let res = match obj {
489                        Object::Stream(stream) => decode_xref_stream(stream),
490                        _ => Err(crate::error::ParseError::InvalidXref.into()),
491                    };
492                    (input, res)
493                })
494                .map_err(|_| {
495                    // artificial error kind is created to allow descriptive nom errors
496                    nom::Err::Error(NomError::from_error_kind(input, ErrorKind::Fail))
497                })
498        }),
499    ))(input)
500    .map(|(_, o)| o)
501    .map_err(|_| error::ParseError::InvalidTrailer)?
502}

563    let (input, stream_dict) = inner_dictionary(input)?;

570fn image_data_stream(input: ParserInput, stream_dict: Dictionary) -> crate::Result<(ParserInput, Stream)> {
571    let get_abbr = |key_abbr: &[u8], key: &[u8]| stream_dict.get(key_abbr).or_else(|_| stream_dict.get(key));
572    let width = get_abbr(b"W", b"Width")?.as_i64()? as usize;
573    let height = get_abbr(b"H", b"Height")?.as_i64()? as usize;
574    let bpc = get_abbr(b"BPC", b"BitsPerComponent")?.as_i64()? as usize;
575    let colorspace = get_abbr(b"CS", b"ColorSpace")?.as_name()?;
576    let num_colors = match colorspace {
577        b"DeviceGray" | b"Gray" => 1,
578        b"DeviceRGB" | b"RGB" => 3,
579        b"DeviceRGBA" | b"RGBA" => 4,
580        b"DeviceCMYK" | b"CMYK" => 4,
581        b"Pattern" => {
582            log::warn!("Pattern colorspace is not allowed in inline images");
583            return Err(Error::InvalidInlineImage(String::from(
584                "Pattern colorspace is not allowed in inline images",
585            )));
586        }
587        _ => {
588            log::warn!("Colorspace of inline image not recognized / not yet implemented");
589            return Err(Error::Unimplemented("inline image colorspaces"));
590        }
591    };
592
593    let stride = (width * (num_colors * bpc) + 7) / 8;
594    let length = height * stride;
595
596    let (input, content) = match get_abbr(b"F", b"Filter") {
597        Err(_) => {
598            // no decompression needed as no filter was applied
599            take(length)(input).map_err(|_: nom::Err<()>| crate::error::ParseError::EndOfInput)?
600        }
601        Ok(Object::Name(_filter)) => {
602            log::warn!("Filters for inline images are not yet implemented");
603            return Err(Error::Unimplemented("filters for inline images"));
604        }
605        Ok(Object::Array(_filters)) => {
606            log::warn!("Filters for inline images are not yet implemented");
607            return Err(Error::Unimplemented("filters for inline images"));
608        }
609        Ok(obj) => {
610            log::warn!("Filter must be either a Name or and Array.");
611            return Err(Error::ObjectType {
612                expected: "Name or Array",
613                found: obj.enum_variant(),
614            });
615        }
616    };
617    Ok((input, Stream::new(stream_dict, content.to_vec())))
618}

640        r.ok().and_then(|(i, o)| if !i.is_empty() { None } else { Some(o) })
