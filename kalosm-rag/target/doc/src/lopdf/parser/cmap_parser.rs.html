15impl<E> From<nom::Err<E>> for CMapParseError {
16    fn from(err: nom::Err<E>) -> Self {
17        match err {
18            nom::Err::Incomplete(_) => CMapParseError::Incomplete,
19            // normally nom::Err::Error is a recoverable error, but CMapParseError is the return type
20            // so we assume that there are no more parsing branches to check
21            nom::Err::Failure(_) | nom::Err::Error(_) => CMapParseError::Error,
22        }
23    }
24}

26pub(crate) fn parse(stream_content: ParserInput) -> Result<Vec<CMapSection>, CMapParseError> {
27    let result = cmap_stream(stream_content);
28    let result = result.map_err(CMapParseError::from)?;
29    Ok(result.1)
30}

32fn cmap_stream(input: ParserInput) -> NomResult<Vec<CMapSection>> {
33    delimited(
34        cidinit_procset,
35        cmap_resource_dictionary,
36        tuple((tag(b"end"), multispace0)),
37    )(input)
38}

40fn tuple_discard_result<'a, O, List>(seq: List, input: ParserInput<'a>) -> NomResult<'a, ()>
41where
42    List: nom::sequence::Tuple<ParserInput<'a>, O, NomError<'a>>,
43{
44    let result = tuple(seq)(input);
45    match result {
46        Ok((rest_of_input, _)) => Ok((rest_of_input, ())),
47        Err(e) => Err(e),
48    }
49}

51fn space0(input: ParserInput) -> NomResult<()> {
52    fold_many0(alt((tag(b" "), tag("\t"))), || {}, |_, _| ())(input)
53}
54
55fn space1(input: ParserInput) -> NomResult<()> {
56    fold_many1(alt((tag(b" "), tag("\t"))), || {}, |_, _| ())(input)
57}

59fn parser_discard_result<'a, O>(
60    parser: impl Fn(ParserInput<'a>) -> NomResult<'a, O>,
61) -> impl Fn(ParserInput<'a>) -> NomResult<'a, ()> {
62    move |input| {
63        let result = parser(input);
64        match result {
65            Ok((rest_of_input, _)) => Ok((rest_of_input, ())),
66            Err(e) => Err(e),
67        }
68    }
69}

71fn multispace0(input: ParserInput) -> NomResult<()> {
72    let space = parser_discard_result(tag(b" "));
73    let tab = parser_discard_result(tag("\t"));
74    let eol = parser_discard_result(eol);
75    fold_many0(alt((space, tab, eol, comment)), || {}, |_, _| ())(input)
76}

78fn multispace1(input: ParserInput) -> NomResult<()> {
79    let space = parser_discard_result(tag(b" "));
80    let tab = parser_discard_result(tag("\t"));
81    let eol = parser_discard_result(eol);
82    fold_many1(alt((space, tab, eol, comment)), || {}, |_, _| ())(input)
83}

85fn cidinit_procset(input: ParserInput) -> NomResult<()> {
86    tuple_discard_result(
87        (
88            multispace0,

95            tag(b"begin"),

102fn cmap_resource_dictionary(input: ParserInput) -> NomResult<Vec<CMapSection>> {
103    let begin_parser = tuple((digit1, space1, tag(b"dict"), space1, tag(b"begin"), multispace1));
104    let end_parser = tuple((tag(b"end"), multispace1));
105    delimited(begin_parser, cmap_data, end_parser)(input)
106}

108fn cmap_data(input: ParserInput) -> NomResult<Vec<CMapSection>> {
109    let cmap_end = tuple((
110        tag(b"endcmap"),
111        multispace1,
112        tag(b"CMapName"),
113        space1,
114        tag(b"currentdict"),
115        space1,
116        tag(b"/CMap"),
117        space1,
118        tag(b"defineresource"),
119        space1,
120        tag(b"pop"),
121        multispace1,
122    ));
123    delimited(
124        tuple((tag(b"begincmap"), multispace1)),
125        preceded(cmap_metadata, cmap_codespace_and_mappings),
126        cmap_end,
127    )(input)
128}

130fn cmap_metadata(input: ParserInput) -> NomResult<()> {
131    let metadata_parser = alt((cid_system_info, cmap_name, cmap_type));
132    fold_many_m_n(1, 4, metadata_parser, || (), |_, _| ())(input)
133}

165fn cmap_codespace_and_mappings(input: ParserInput) -> NomResult<Vec<CMapSection>> {
166    many1(alt((codespace_range_section, bf_char_section, bf_range_section)))(input)
167}

169fn codespace_range_section(input: ParserInput) -> NomResult<CMapSection> {
170    let begin_section = tuple((digit1, space1, tag(b"begincodespacerange"), multispace1));
171    let end_section = tuple((tag(b"endcodespacerange"), multispace1));
172    let parse_range = delimited(space0, code_range_pair, multispace1);
173    let (rest_of_input, ranges_result) = delimited(begin_section, many1(parse_range), end_section)(input)?;
174    Ok((rest_of_input, CMapSection::CsRange(ranges_result)))
175}
176
177fn code_range_pair(input: ParserInput) -> NomResult<(SourceCode, SourceCode, CodeLen)> {
178    let (rest_of_input, ((code_begin, code_len_beg), (code_end, code_len_end))) =
179        separated_pair(source_code, space0, source_code)(input)?;
180    if code_len_beg != code_len_end {
181        create_code_len_err(rest_of_input)
182    } else {
183        Ok((rest_of_input, (code_begin, code_end, code_len_beg)))
184    }
185}
186
187fn create_code_len_err<'a, T, E: ParseError<ParserInput<'a>>>(input: ParserInput<'a>) -> Result<T, nom::Err<E>> {
188    Err(nom::Err::Failure(nom::error::make_error(
189        input,
190        nom::error::ErrorKind::LengthValue,
191    )))
192}

194fn source_code(input: ParserInput) -> NomResult<(SourceCode, CodeLen)> {
195    let (rest_of_input, bytes) = delimited(tag(b"<"), many_m_n(1, 4, hex_char), tag(b">"))(input)?;
196    let code_len = bytes.len();
197    let source_code = bytes
198        .into_iter()
199        .rev()
200        .zip(0..4)
201        .map(|(byte, i)| 256u32.pow(i) * byte as u32)
202        .sum();
203    Ok((rest_of_input, (source_code, code_len as u8)))
204}
205
206fn hex_u16(input: ParserInput) -> NomResult<u16> {
207    map(pair(hex_char, hex_char), |(h1, h2)| h1 as u16 * 256 + h2 as u16)(input)
208}

210fn bf_char_section(input: ParserInput) -> NomResult<CMapSection> {
211    let begin_section = tuple((digit1, space1, tag(b"beginbfchar"), multispace1));
212    let end_section = tuple((tag(b"endbfchar"), multispace1));
213    let bf_char_line = delimited(space0, separated_pair(source_code, space0, target_string), multispace1);
214    let (rest_of_input, bf_char_mappings) = delimited(begin_section, many1(bf_char_line), end_section)(input)?;
215    Ok((rest_of_input, CMapSection::BfChar(bf_char_mappings)))
216}
217
218fn target_string(input: ParserInput) -> NomResult<Vec<u16>> {
219    // according to specification dstString can be up to 512 bytes
220    // in ToUnicode cmap these should be 2-byte big endian Unicode values
221    delimited(tag(b"<"), many_m_n(1, 256, terminated(hex_u16, multispace0)), tag(b">"))(input)
222}
223
224fn bf_range_section(input: ParserInput) -> NomResult<CMapSection> {
225    let begin_section = tuple((digit1, space1, tag(b"beginbfrange"), multispace1));
226    let end_section = tuple((tag(b"endbfrange"), multispace1));
227    let (rest_of_input, bf_range_mappings) = delimited(begin_section, many1(bf_range_line), end_section)(input)?;
228    Ok((rest_of_input, CMapSection::BfRange(bf_range_mappings)))
229}
230
231fn bf_range_line(input: ParserInput) -> NomResult<SourceRangeMapping> {
232    let bf_range_parser = separated_pair(
233        code_range_pair,
234        space0,
235        alt((target_string.map(|res| vec![res]), range_target_array)),
236    );
237    delimited(space0, bf_range_parser, multispace1)(input)
238}
239
240fn range_target_array(input: ParserInput) -> NomResult<ArrayOfTargetStrings> {
241    delimited(
242        tuple((tag(b"["), space0)),
243        separated_list1(space1, target_string),
244        tuple((space0, tag(b"]"))),
245    )(input)
246}
