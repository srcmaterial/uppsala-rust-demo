8#[derive(Debug, Error)]
9pub enum Error {
10    /// Lopdf does not (yet) implement a needed feature.
11    #[error("missing feature of lopdf: {0}; please open an issue at https://github.com/J-F-Liu/lopdf/ to let the developers know of your usecase")]
12    Unimplemented(&'static str),
13
14    /// An Object has the wrong type, e.g. the Object is an Array where a Name would be expected.
15    #[error("object has wrong type; expected type {expected} but found type {found}")]
16    ObjectType {
17        expected: &'static str,
18        found: &'static str,
19    },
20    #[error("dictionary has wrong type: ")]
21    DictType { expected: &'static str, found: String },
22    /// The encountered character encoding is invalid.
23    #[error("invalid character encoding")]
24    CharacterEncoding,
25    /// The stream couldn't be decompressed.
26    #[error("couldn't decompress stream {0}")]
27    Decompress(#[from] DecompressError),
28    /// Failed to parse input.
29    #[error("couldn't parse input: {0}")]
30    Parse(#[from] ParseError),
31    /// Error when decrypting the contents of the file
32    #[error("decryption error: {0}")]
33    Decryption(#[from] encryption::DecryptionError),
34    /// Dictionary key was not found.
35    #[error("missing required dictionary key \"{0}\"")]
36    DictKey(String),
37    /// Invalid inline image.
38    #[error("invalid inline image: {0}")]
39    InvalidInlineImage(String),
40    /// Invalid document outline.
41    #[error("invalid document outline: {0}")]
42    InvalidOutline(String),
43    /// Invalid stream.
44    #[error("invalid stream: {0}")]
45    InvalidStream(String),
46    /// Invalid object stream.
47    #[error("invalid object stream: {0}")]
48    InvalidObjectStream(String),
49    /// Byte offset in stream or file is invalid.
50    #[error("invalid byte offset")]
51    InvalidOffset(usize),
52    /// IO error
53    #[error("IO error: {0}")]
54    IO(#[from] std::io::Error),
55    // TODO: Maybe remove, as outline is not required in spec.
56    /// PDF document has no outline.
57    #[error("PDF document does not have an outline")]
58    NoOutline,
59    /// Missing xref entry.
60    #[error("missing xref entry")]
61    MissingXrefEntry,
62    /// The Object ID was not found.
63    #[error("object ID {} {} not found", .0.0, .0.1)]
64    ObjectNotFound(ObjectId),
65    /// Dereferencing object failed due to a reference cycle.
66    #[error("reference cycle with object ID {} {}", .0.0, .0.1)]
67    ReferenceCycle(ObjectId),
68    /// Page number was not found in document.
69    #[error("page number not found")]
70    PageNumberNotFound(u32),
71    /// Numeric type cast failed.
72    #[error("numberic type cast failed: {0}")]
73    NumericCast(String),
74    /// Dereferencing object reached the limit.
75    /// This might indicate a reference loop.
76    #[error("dereferencing object reached limit, may indicate a reference cycle")]
77    ReferenceLimit,
78    /// Decoding text string failed.
79    #[error("decoding text string failed")]
80    TextStringDecode,
81    /// Error while parsing cross reference table.
82    #[error("failed parsing cross reference table: {0}")]
83    Xref(XrefError),
84    /// Invalid indirect object while parsing at offset.
85    #[error("invalid indirect object at byte offset {offset}")]
86    IndirectObject { offset: usize },
87    /// Found object ID does not match expected object ID.
88    #[error("found object ID does not match expected object ID")]
89    ObjectIdMismatch,
90    /// Error when handling images.
91    #[cfg(feature = "embed_image")]
92    #[error("image error: {0}")]
93    Image(#[from] image::ImageError),
94    /// Syntax error while processing the content stream.
95    #[error("syntax error in content stream: {0}")]
96    Syntax(String),
97    /// Could not parse ToUnicodeCMap.
98    #[error("failed parsing ToUnicode CMap: {0}")]
99    ToUnicodeCMap(#[from] UnicodeCMapError),
100}
