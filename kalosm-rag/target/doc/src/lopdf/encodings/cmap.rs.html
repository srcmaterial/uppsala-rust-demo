9/// Unicode Cmap is implemented by 4 maps.
10/// Each map contains a mappings from source codes to unicode values for a different length of codes.
11/// Codes vary from 1 byte to 4 bytes so they are always in limits of u32.
12/// However to map a code to a unicode value an additional knowledge about the number of bytes is needed,
13/// as 2 byte code <0000> shouldn't be matched with a single byte <00> even though they have the same integer value.
14#[derive(Debug, Default)]
15pub struct ToUnicodeCMap {
16    bf_ranges: [RangeInclusiveMap<SourceCode, BfRangeTarget>; 4],
17}
18
19#[derive(Debug, Error)]
20pub enum UnicodeCMapError {
21    #[error("could not parse ToUnicode CMap: {0:#?}")]
22    Parse(CMapParseError),
23    #[error("invalid code range")]
24    InvalidCodeRange,
25}
26
27impl From<CMapParseError> for UnicodeCMapError {
28    fn from(err: CMapParseError) -> Self {
29        UnicodeCMapError::Parse(err)
30    }
31}
32
33impl ToUnicodeCMap {
34    const REPLACEMENT_CHAR: u16 = 0xfffd;
35
36    pub fn new() -> ToUnicodeCMap {
37        ToUnicodeCMap {
38            bf_ranges: [(); 4].map(|_| RangeInclusiveMap::new()),
39        }
40    }
41
42    pub(crate) fn parse(stream_content: Vec<u8>) -> Result<ToUnicodeCMap, UnicodeCMapError> {
43        let cmap_sections = parse(ParserInput::new_extra(&stream_content[..], "cmap"))?;
44        Self::from_sections(cmap_sections)
45    }
46
47    fn from_sections(cmap_sections: Vec<CMapSection>) -> Result<ToUnicodeCMap, UnicodeCMapError> {
48        let mut cmap = Self::new();
49        for section in cmap_sections {
50            match section {
51                CMapSection::CsRange(_) => (), // currently no additional validation is implemented for code ranges
52                CMapSection::BfChar(char_mappings) => {
53                    for ((code, code_len), dst) in char_mappings {
54                        cmap.put_char(code, code_len, dst);
55                    }
56                }
57                CMapSection::BfRange(range_mappings) => {
58                    for ((start, end, code_len), dst_vec) in range_mappings {
59                        if end < start {
60                            return Err(UnicodeCMapError::InvalidCodeRange);
61                        }
62                        match dst_vec.len() {
63                            1 if dst_vec[0].len() == 1 => cmap.put(
64                                start,
65                                end,
66                                code_len,
67                                BfRangeTarget::UTF16CodePoint {
68                                    offset: u32::wrapping_sub(dst_vec[0][0] as u32, start),
69                                },
70                            ),
71                            1 => cmap.put(start, end, code_len, BfRangeTarget::HexString(dst_vec[0].clone())),
72                            0 => return Err(UnicodeCMapError::InvalidCodeRange),
73                            _ => cmap.put(start, end, code_len, BfRangeTarget::ArrayOfHexStrings(dst_vec.clone())),
74                        }
75                    }
76                }
77            }
78        }
79        Ok(cmap)
80    }
81
82    pub fn get(&self, code: SourceCode, code_len: CodeLen) -> Option<Vec<u16>> {
83        if code_len > 4 || code_len == 0 {
84            error!("Code lenght should be between l and 4 bytes, got {code_len}");
85            return None;
86        }
87        use BfRangeTarget::*;
88
89        let bf_ranges_map = &self.bf_ranges[(code_len - 1) as usize];
90
91        bf_ranges_map.get_key_value(&code).map(|(range, value)| match value {
92            HexString(vec) => {
93                let mut ret_vec = vec.clone();
94                *(ret_vec.last_mut().unwrap()) += (code - range.start()) as u16;
95                ret_vec
96            }
97            UTF16CodePoint { offset } => vec![u32::wrapping_add(code, *offset) as u16],
98            ArrayOfHexStrings(vec_of_strings) => vec_of_strings[(code - range.start()) as usize].clone(),
99        })
100    }
101
102    pub fn get_or_replacement_char(&self, code: SourceCode, code_len: CodeLen) -> Vec<u16> {
103        self.get(code, code_len)
104            .unwrap_or(vec![ToUnicodeCMap::REPLACEMENT_CHAR])
105    }
106
107    pub fn put(&mut self, src_code_lo: SourceCode, src_code_hi: SourceCode, code_len: CodeLen, target: BfRangeTarget) {
108        if code_len > 4 || code_len == 0 {
109            error!("Code lenght should be between l and 4 bytes, got {code_len}, ignoring");
110            return;
111        }
112        self.bf_ranges[(code_len - 1) as usize].insert(src_code_lo..=src_code_hi, target)
113    }
114
115    pub fn put_char(&mut self, code: SourceCode, code_len: CodeLen, dst: Vec<u16>) {
116        let target = if dst.len() == 1 {
117            BfRangeTarget::UTF16CodePoint {
118                offset: u32::wrapping_sub(dst[0] as u32, code),
119            }
120        } else {
121            BfRangeTarget::HexString(dst)
122        };
123        self.put(code, code, code_len, target)
124    }
125}
126
127#[derive(Debug, PartialEq, Eq, Clone)]
128pub enum BfRangeTarget {
129    // UTF16-BE encoding is used
130    HexString(Vec<u16>),
131    // don't store the actual codepoint but rather an offset to the src_code_lo
132    // so that consecutive ranges can be mapped to the same value in the range map
133    UTF16CodePoint { offset: u32 },
134    ArrayOfHexStrings(Vec<Vec<u16>>),
135}

137#[cfg(test)]
138mod tests {
139    use super::*;
140
141    #[test]
142    fn put_char_can_be_retrieved() {
143        let mut cmap = ToUnicodeCMap::new();
144        let char_code = 0x01;
145        let char_value = vec![0x1234];
146        cmap.put_char(char_code, 2, char_value.clone());
147
148        assert_eq!(cmap.get(char_code, 2), Some(char_value))
149    }
150
151    #[test]
152    fn char_can_be_retrieved_only_by_appropriate_len() {
153        let mut cmap = ToUnicodeCMap::new();
154        let char_code = 0x1;
155        let code_len = 4;
156        let char_value = vec![0x1234];
157        cmap.put_char(char_code, code_len, char_value.clone());
158
159        for i in 1..=3 {
160            assert_eq!(cmap.get(char_code, i), None);
161        }
162
163        assert_eq!(cmap.get(char_code, code_len), Some(char_value));
164    }
165
166    #[test]
167    fn wrong_code_len_does_not_panic() {
168        let mut cmap = ToUnicodeCMap::new();
169        let char_code = 0x1;
170        let char_value = vec![0x1234];
171
172        cmap.put_char(char_code, 5, char_value.clone());
173        cmap.put_char(char_code, 0, char_value.clone());
174    }
