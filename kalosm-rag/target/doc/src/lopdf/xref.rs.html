1use std::collections::BTreeMap;
2use std::io::{Result, Write};
3
4#[derive(Debug, Clone)]
5pub struct Xref {
6    /// Type of Cross-Reference used in the last incremental version.
7    /// This method of cross-referencing will also be used when saving the file.
8    /// PDFs with Incremental Updates should alway use the same cross-reference type.
9    pub cross_reference_type: XrefType,
10
11    /// Entries for indirect object.
12    pub entries: BTreeMap<u32, XrefEntry>,
13
14    /// Total number of entries (including free entries), equal to the highest object number plus 1.
15    pub size: u32,
16}
17
18#[derive(Debug, Clone, Copy)]
19pub enum XrefType {
20    /// Cross-Reference Streams are supported beginning with PDF 1.5.
21    CrossReferenceStream,
22    /// Cross-Reference Table is older but still frequently used.
23    CrossReferenceTable,
24}
25
26#[derive(Debug, Clone)]
27pub enum XrefEntry {
28    Free, // TODO add generation number
29    UnusableFree,
30    Normal { offset: u32, generation: u16 },
31    Compressed { container: u32, index: u16 },
32}
33
34#[derive(Debug, Clone)]
35pub struct XrefSection {
36    pub starting_id: u32,
37    pub entries: Vec<XrefEntry>,
38}
39
40impl Xref {
41    pub fn new(size: u32, xref_type: XrefType) -> Xref {
42        Xref {
43            cross_reference_type: xref_type,
44            entries: BTreeMap::new(),
45            size,
46        }
47    }
48
49    pub fn get(&self, id: u32) -> Option<&XrefEntry> {
50        self.entries.get(&id)
51    }
52
53    pub fn insert(&mut self, id: u32, entry: XrefEntry) {
54        self.entries.insert(id, entry);
55    }
56
57    /// Combine Xref entries. Only add them if they do not exists already.
58    /// Do not replace existing entries.
59    pub fn merge(&mut self, xref: Xref) {
60        for (id, entry) in xref.entries {
61            self.entries.entry(id).or_insert(entry);
62        }
63    }
64
65    pub fn clear(&mut self) {
66        self.entries.clear()
67    }
68
69    pub fn max_id(&self) -> u32 {
70        match self.entries.keys().max() {
71            Some(&id) => id,
72            None => 0,
73        }
74    }
75}
76
77impl XrefEntry {
78    pub fn is_normal(&self) -> bool {
79        matches!(*self, XrefEntry::Normal { .. })
80    }
81
82    pub fn is_compressed(&self) -> bool {
83        matches!(*self, XrefEntry::Compressed { .. })
84    }
85
86    /// Write Entry in Cross Reference Table.
87    pub fn write_xref_entry(&self, file: &mut dyn Write) -> Result<()> {
88        match self {
89            XrefEntry::Normal { offset, generation } => {
90                writeln!(file, "{:>010} {:>05} n ", offset, generation)?;
91            }
92            XrefEntry::Compressed { container: _, index: _ } => {
93                writeln!(file, "{:>010} {:>05} f ", 0, 65535)?;
94            }
95            XrefEntry::Free => {
96                writeln!(file, "{:>010} {:>05} f ", 0, 0)?;
97            }
98            XrefEntry::UnusableFree => {
99                writeln!(file, "{:>010} {:>05} f ", 0, 65535)?;
100            }
101        }
102        Ok(())
103    }
104}
105
106impl XrefSection {
107    pub fn new(starting_id: u32) -> Self {
108        XrefSection {
109            starting_id,
110            entries: Vec::new(),
111        }
112    }
113
114    pub fn add_entry(&mut self, entry: XrefEntry) {
115        self.entries.push(entry);
116    }
117
118    pub fn add_unusable_free_entry(&mut self) {
119        self.add_entry(XrefEntry::UnusableFree);
120    }
121
122    pub fn is_empty(&self) -> bool {
123        self.entries.is_empty()
124    }
125
126    /// Write Section in Cross Reference Table.
127    pub fn write_xref_section(&self, file: &mut dyn Write) -> Result<()> {
128        if !self.is_empty() {
129            // Write section range
130            writeln!(file, "{} {}", self.starting_id, self.entries.len())?;
131            // Write entries
132            for entry in &self.entries {
133                entry.write_xref_entry(file)?;
134            }
135        }
136        Ok(())
137    }
138}
