1// This module exists because the rust-crypto module is really old and not maintained.
2// Fortunately the RC4 algorithm is very simple to implement.
3pub struct Rc4 {
4    initial_state: [u8; 256],
5}
6
7impl Rc4 {
8    pub fn new<Key: AsRef<[u8]>>(key: Key) -> Self {
9        let key = key.as_ref();
10        assert!(!key.is_empty() && key.len() <= 256);
11
12        let mut initial_state = [0_u8; 256];
13        for (i, v) in initial_state.iter_mut().enumerate() {
14            *v = i as u8;
15        }
16
17        let mut j = 0_u8;
18        for i in 0..256 {
19            j = j.wrapping_add(initial_state[i]).wrapping_add(key[i % key.len()]);
20            initial_state.swap(i, j as usize);
21        }
22
23        Self { initial_state }
24    }
25
26    /// Encrypts/decrypts `input` into `output`.  The shorter of `input` and `output`
27    ///  determine how many bytes are written into `output`.
28    pub fn apply_keystream<'i, 'o, Input, Output>(&self, input: Input, output: Output)
29    where
30        Input: Iterator<Item = &'i u8>,
31        Output: Iterator<Item = &'o mut u8>,
32    {
33        let mut state = self.initial_state;
34        let mut i = 0_u8;
35        let mut j = 0_u8;
36        for (i_byte, o_byte) in input.zip(output) {
37            i = i.wrapping_add(1);
38            j = j.wrapping_add(state[i as usize]);
39            state.swap(i as usize, j as usize);
40            let key_byte = state[(state[i as usize].wrapping_add(state[j as usize])) as usize];
41            *o_byte = i_byte ^ key_byte;
42        }
43    }
44
45    /// Allocates a new Vec<u8> of the same length as `input` and decrypts
46    ///  `input` into it.
47    pub fn decrypt<Input>(&self, input: Input) -> Vec<u8>
48    where
49        Input: AsRef<[u8]>,
50    {
51        let input = input.as_ref();
52        let mut output = vec![0; input.len()];
53        self.apply_keystream(input.iter(), output.iter_mut());
54        output
55    }
56
57    /// Allocates a new Vec<u8> of the same length as `input` and encrypts
58    ///  `input` into it.
59    pub fn encrypt<Input>(&self, input: Input) -> Vec<u8>
60    where
61        Input: AsRef<[u8]>,
62    {
63        // Rc4 is symmetric
64        self.decrypt(input)
65    }
66}
