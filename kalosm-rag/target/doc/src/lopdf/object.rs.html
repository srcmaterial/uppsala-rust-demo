315impl fmt::Debug for Object {
316    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
317        match self {
318            Object::Null => write!(f, "Null"),
319            Object::Boolean(value) => write!(f, "{}", value),
320            Object::Integer(value) => write!(f, "{}", value),
321            Object::Real(value) => write!(f, "{}", value),
322            Object::Name(name) => write!(f, "/{}", String::from_utf8_lossy(name)),
323            Object::String(text, StringFormat::Literal) => write!(f, "({})", String::from_utf8_lossy(text)),
324            Object::String(text, StringFormat::Hexadecimal) => {
325                write!(f, "<")?;
326                for b in text {
327                    write!(f, "{:02x}", b)?;
328                }
329                write!(f, ">")
330            }
331            Object::Array(array) => {
332                let items = array.iter().map(|item| format!("{:?}", item)).collect::<Vec<String>>();
333                write!(f, "[{}]", items.join(" "))
334            }
335            Object::Dictionary(dict) => write!(f, "{:?}", dict),
336            Object::Stream(stream) => write!(f, "{:?}stream...endstream", stream.dict),
337            Object::Reference(id) => write!(f, "{} {} R", id.0, id.1),
338        }
339    }

407    pub fn get_font_encoding(&self, doc: &Document) -> Result<Encoding> {
408        if !self.has_type(b"Font") {
409            return Err(Error::DictType {
410                expected: "Font",
411                found: String::from_utf8_lossy(self.get_type().unwrap_or(b"None")).to_string(),
412            });
413        }
414
415        // Note: currently not all encodings are handled, not implemented:
416        // - dictionary differences encoding
417        // - default base encoding in dictionary differences encoding
418        // - TrueType cmap tables
419        // - DescendantFonts in CID-Keyed fonts
420        // - predefined CJK CMAP other than indicated in SimpleEncoding
421        match self.get(b"Encoding").and_then(Object::as_name) {
422            Ok(b"StandardEncoding") => Ok(Encoding::OneByteEncoding(&encodings::STANDARD_ENCODING)),
423            Ok(b"MacRomanEncoding") => Ok(Encoding::OneByteEncoding(&encodings::MAC_ROMAN_ENCODING)),
424            Ok(b"MacExpertEncoding") => Ok(Encoding::OneByteEncoding(&encodings::MAC_EXPERT_ENCODING)),
425            Ok(b"WinAnsiEncoding") => Ok(Encoding::OneByteEncoding(&encodings::WIN_ANSI_ENCODING)),
426            Ok(b"PDFDocEncoding") => {
427                log::warn!("PDFDocEncoding is not a valid character encoding for a font");
428                Ok(Encoding::OneByteEncoding(&encodings::PDF_DOC_ENCODING))
429            }
430            Ok(b"Identity-H") | Ok(b"Identity-V") => {
431                let stream = self.get_deref(b"ToUnicode", doc)?.as_stream()?;
432                self.get_encoding_from_to_unicode_cmap(stream)
433            }
434            Ok(name) => Ok(Encoding::SimpleEncoding(name)),
435            Err(err) => {
436                warn!(
437                    "Could not parse the encoding, error: {:#?}\nFont: {:#?}\nTrying to retrieve ToUnicode.",
438                    err, self
439                );
440                let stream = self.get_deref(b"ToUnicode", doc).and_then(Object::as_stream);
441                if let Ok(stream) = stream {
442                    return self.get_encoding_from_to_unicode_cmap(stream);
443                }
444
445                warn!("Using standard encoding as a fallback!");
446                Ok(Encoding::OneByteEncoding(&encodings::STANDARD_ENCODING))
447            }
448        }
449    }

457    pub fn extend(&mut self, other: &Dictionary) {
458        let keep_both_objects =
459            |new_dict: &mut IndexMap<Vec<u8>, Object>, key: &Vec<u8>, value: &Object, old_value: Object| {
460                let mut final_array;
461
462                match value {
463                    Object::Array(array) => {
464                        final_array = Vec::with_capacity(array.len() + 1);
465                        final_array.push(old_value);
466                        final_array.extend(array.to_owned());
467                    }
468                    _ => {
469                        final_array = vec![value.to_owned(), old_value];
470                    }
471                }
472
473                new_dict.insert(key.to_owned(), Object::Array(final_array));
474            };
475
476        let mut new_dict = std::mem::take(&mut self.0);
477        new_dict.reserve_exact(other.0.len());
478
479        for (key, value) in other.0.iter() {
480            if let Some(old_value) = new_dict.get(key) {
481                let old_value = old_value.to_owned();
482                match (&old_value, value) {
483                    (Object::Dictionary(old_dict), Object::Dictionary(dict)) => {
484                        let mut replaced_dict = old_dict.to_owned();
485                        replaced_dict.extend(dict);
486                        new_dict.insert(key.to_owned(), Object::Dictionary(replaced_dict));
487                    }
488                    (Object::Array(old_array), Object::Array(array)) => {
489                        let mut replaced_array = old_array.to_owned();
490                        replaced_array.extend(array.to_owned());
491                        new_dict.insert(key.to_owned(), Object::Array(replaced_array));
492                    }
493                    (Object::Integer(old_id), Object::Integer(id)) => {
494                        let array = vec![Object::Integer(*old_id), Object::Integer(*id)];
495                        new_dict.insert(key.to_owned(), Object::Array(array));
496                    }
497                    (Object::Real(old_id), Object::Real(id)) => {
498                        let array = vec![Object::Real(*old_id), Object::Real(*id)];
499                        new_dict.insert(key.to_owned(), Object::Array(array));
500                    }
501                    (Object::String(old_ids, old_format), Object::String(ids, format)) => {
502                        let array = vec![
503                            Object::String(old_ids.to_owned(), old_format.to_owned()),
504                            Object::String(ids.to_owned(), format.to_owned()),
505                        ];
506                        new_dict.insert(key.to_owned(), Object::Array(array));
507                    }
508                    (Object::Reference(old_object_id), Object::Reference(object_id)) => {
509                        let array = vec![Object::Reference(*old_object_id), Object::Reference(*object_id)];
510                        new_dict.insert(key.to_owned(), Object::Array(array));
511                    }
512                    (Object::Null, _) | (Object::Boolean(_), _) | (Object::Name(_), _) | (Object::Stream(_), _) => {
513                        new_dict.insert(key.to_owned(), old_value);
514                    }
515                    (_, _) => keep_both_objects(&mut new_dict, key, value, old_value),
516                }
517            } else {
518                new_dict.insert(key.to_owned(), value.to_owned());
519            }
520        }
521
522        self.0 = new_dict;
523    }

663    pub fn compress(&mut self) -> Result<()> {
664        use flate2::write::ZlibEncoder;
665        use flate2::Compression;
666        use std::io::prelude::*;
667
668        if self.dict.get(b"Filter").is_err() {
669            let mut encoder = ZlibEncoder::new(Vec::new(), Compression::best());
670            encoder.write_all(self.content.as_slice())?;
671            let compressed = encoder.finish()?;
672            if compressed.len() + 19 < self.content.len() {
673                self.dict.set("Filter", "FlateDecode");
674                self.set_content(compressed);
675            }
676        }
677        Ok(())
678    }

680    pub fn decompressed_content(&self) -> Result<Vec<u8>> {
681        let params = self.dict.get(b"DecodeParms").and_then(Object::as_dict).ok();
682        let filters = self.filters()?;
683
684        let mut input = self.content.as_slice();
685        let mut output = vec![];
686
687        // Filters are in decoding order.
688        for filter in filters {
689            output = match filter {
690                b"FlateDecode" => Self::decompress_zlib(input, params)?,
691                b"LZWDecode" => Self::decompress_lzw(input, params)?,
692                b"ASCII85Decode" => Self::decode_ascii85(input)?,
693                _ => return Err(Error::Unimplemented("decompression algorithms")),
694            };
695            input = &output;
696        }
697        Ok(output)
698    }

700    fn decompress_lzw(input: &[u8], params: Option<&Dictionary>) -> Result<Vec<u8>> {
701        use weezl::{decode::Decoder, BitOrder};
702        const MIN_BITS: u8 = 9;
703
704        let early_change = params
705            .and_then(|p| p.get(b"EarlyChange").ok())
706            .and_then(|p| Object::as_i64(p).ok())
707            .map(|v| v != 0)
708            .unwrap_or(true);
709
710        let mut decoder = if early_change {
711            Decoder::with_tiff_size_switch(BitOrder::Msb, MIN_BITS - 1)
712        } else {
713            Decoder::new(BitOrder::Msb, MIN_BITS - 1)
714        };
715
716        let output = Self::decompress_lzw_loop(input, &mut decoder);
717        Self::decompress_predictor(output, params)
718    }
719
720    fn decompress_lzw_loop(input: &[u8], decoder: &mut weezl::decode::Decoder) -> Vec<u8> {
721        let mut output = vec![];
722
723        let result = decoder.into_stream(&mut output).decode_all(input);
724        if let Err(err) = result.status {
725            warn!("{}", err);
726        }
727
728        output
729    }

731    fn decompress_zlib(input: &[u8], params: Option<&Dictionary>) -> Result<Vec<u8>> {
732        use flate2::read::ZlibDecoder;
733        use std::io::prelude::*;
734
735        let mut output = Vec::with_capacity(input.len() * 2);
736        let mut decoder = ZlibDecoder::new(input);
737
738        if !input.is_empty() {
739            decoder.read_to_end(&mut output).unwrap_or_else(|err| {
740                warn!("{}", err);
741                0
742            });
743        }
744        Self::decompress_predictor(output, params)
745    }

747    fn decode_ascii85(input: &[u8]) -> Result<Vec<u8>> {
748        let mut output = vec![];
749        let mut buffer: u32 = 0;
750        let mut count = 0;
751        // Check for EOD marker
752        let input_no_eod = if input.len() >= 2 && &input[input.len() - 2..] == b"~>" {
753            &input[..input.len() - 2]
754        } else {
755            log::warn!("ASCII85 stream is missing its EOD marker");
756            input
757        };
758        for &ch in input_no_eod {
759            if ch == b'z' {
760                if count != 0 {
761                    return Err(DecompressError::Ascii85("z character is not allowed in the middle of a group").into());
762                }
763                output.extend_from_slice(&[0, 0, 0, 0]);
764                continue;
765            }
766
767            if ch.is_ascii_whitespace() {
768                continue;
769            }
770
771            if !(b'!'..=b'u').contains(&ch) {
772                break;
773            }
774            buffer = buffer
775                .checked_mul(85)
776                .ok_or(DecompressError::Ascii85("multiplication overflow"))?;
777            buffer += (ch - b'!') as u32;
778            count += 1;
779
780            if count == 5 {
781                output.extend_from_slice(&buffer.to_be_bytes());
782                buffer = 0;
783                count = 0;
784            }
785        }
786
787        if count > 0 {
788            for _ in count..5 {
789                buffer = buffer
790                    .checked_mul(85)
791                    .ok_or(DecompressError::Ascii85("multiplication overflow"))?;
792                buffer += 84;
793            }
794
795            let bytes = buffer.to_be_bytes();
796            output.extend_from_slice(&bytes[..count - 1]);
797        }
798
799        Ok(output)
800    }

802    fn decompress_predictor(mut data: Vec<u8>, params: Option<&Dictionary>) -> Result<Vec<u8>> {
803        use crate::filters::png;
804
805        if let Some(params) = params {
806            let predictor = params.get(b"Predictor").and_then(Object::as_i64).unwrap_or(1);
807            if (10..=15).contains(&predictor) {
808                let pixels_per_row = max(1, params.get(b"Columns").and_then(Object::as_i64).unwrap_or(1)) as usize;
809                let colors = max(1, params.get(b"Colors").and_then(Object::as_i64).unwrap_or(1)) as usize;
810                let bits = max(8, params.get(b"BitsPerComponent").and_then(Object::as_i64).unwrap_or(8)) as usize;
811                let bytes_per_pixel = colors * bits / 8;
812                data = png::decode_frame(data.as_slice(), bytes_per_pixel, pixels_per_row)?;
813            }
814            Ok(data)
815        } else {
816            Ok(data)
817        }
818    }

820    pub fn decompress(&mut self) -> Result<()> {
821        let data = self.decompressed_content()?;
822        self.dict.remove(b"DecodeParms");
823        self.dict.remove(b"Filter");
824        self.set_content(data);
825        Ok(())
826    }

829        self.dict.get(b"Filter").is_ok()
830    }
