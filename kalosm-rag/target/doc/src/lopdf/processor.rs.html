9    pub fn change_producer(&mut self, producer: &str) {
10        if let Ok(info) = self.trailer.get_mut(b"Info") {
11            if let Some(dict) = match info {
12                Object::Dictionary(dict) => Some(dict),
13                Object::Reference(id) => self.objects.get_mut(id).and_then(|o| o.as_dict_mut().ok()),
14                _ => None,
15            } {
16                dict.set("Producer", Object::string_literal(producer));
17            }
18        }
19    }

21    /// Compress PDF stream objects.
22    pub fn compress(&mut self) {
23        for object in self.objects.values_mut() {
24            if let Object::Stream(stream) = object {
25                if stream.allows_compression {
26                    // Ignore any error and continue to compress other streams.
27                    let _ = stream.compress();
28                }
29            }
30        }
31    }

42    /// Delete pages.
43    pub fn delete_pages(&mut self, page_numbers: &[u32]) {
44        let pages = self.get_pages();
45        for page_number in page_numbers {
46            if let Some(page) = pages.get(page_number).and_then(|page_id| self.delete_object(*page_id)) {
47                let mut page_tree_ref = page
48                    .as_dict()
49                    .and_then(|dict| dict.get(b"Parent"))
50                    .and_then(Object::as_reference);
51                while let Ok(page_tree_id) = page_tree_ref {
52                    if let Some(page_tree) = self.objects.get_mut(&page_tree_id).and_then(|pt| pt.as_dict_mut().ok()) {
53                        if let Ok(count) = page_tree.get(b"Count").and_then(Object::as_i64) {
54                            page_tree.set("Count", count - 1);
55                        }
56                        page_tree_ref = page_tree.get(b"Parent").and_then(Object::as_reference);
57                    } else {
58                        break;
59                    }
60                }
61            }
62        }
63    }

65    /// Prune all unused objects.
66    pub fn prune_objects(&mut self) -> Vec<ObjectId> {
67        let mut ids = vec![];
68        let refs = self.traverse_objects(|_| {});
69        for id in self.objects.keys() {
70            if !refs.contains(id) {
71                ids.push(*id);
72            }
73        }
74
75        for id in &ids {
76            self.objects.remove(id);
77        }
78
79        ids
80    }

82    /// Delete object by object ID.
83    pub fn delete_object(&mut self, id: ObjectId) -> Option<Object> {
84        let action = |object: &mut Object| match object {
85            Object::Array(array) => {
86                if let Some(index) = array.iter().position(|item: &Object| match *item {
87                    Object::Reference(ref_id) => ref_id == id,
88                    _ => false,
89                }) {
90                    array.remove(index);
91                }
92            }
93            Object::Dictionary(dict) => {
94                let keys: Vec<Vec<u8>> = dict
95                    .iter()
96                    .filter(|&(_, item): &(&Vec<u8>, &Object)| match *item {
97                        Object::Reference(ref_id) => ref_id == id,
98                        _ => false,
99                    })
100                    .map(|(k, _)| k.clone())
101                    .collect();
102                for key in keys {
103                    dict.remove(&key);
104                }
105            }
106            _ => {}
107        };
108        self.traverse_objects(action);
109        self.objects.remove(&id)
110    }

114        let mut ids = vec![];
115        for id in self.objects.keys() {

117                .objects
118                .get(id)
119                .and_then(|o| Object::as_stream(o).ok())
120                .map(|stream| stream.content.is_empty())
121                .unwrap_or(false)
122            {
123                ids.push(*id);
124            }
125        }
126

139    fn update_bookmark_pages(&mut self, bookmarks: &[u32], old: &ObjectId, new: &ObjectId) {
140        for id in bookmarks {
141            let (children, page) = match self.bookmark_table.get(id) {
142                Some(n) => (n.children.clone(), n.page),
143                None => return,
144            };
145
146            if page == *old {
147                let bookmark = self.bookmark_table.get_mut(id).unwrap();
148                bookmark.page = *new;
149            }
150
151            if !children.is_empty() {
152                self.update_bookmark_pages(&children[..], old, new);
153            }
154        }
155    }

163    /// Renumber objects with a custom starting id, this is very useful in case of multiple
164    /// document object insertions in a single main document
165    pub fn renumber_objects_with(&mut self, starting_id: u32) {
166        let mut replace = BTreeMap::new();
167        let mut new_id = starting_id;
168        let mut i = 0;
169
170        // Check if we need to order the pages first, as this means the first page doesn't have a lower ID.
171        // So it ends up in a random spot based on its ID. We check first to avoid double traversal, unless we have too.
172
173        let mut page_order: Vec<(i32, (u32, u16))> = self
174            .page_iter()
175            .map(|id| {
176                i += 1;
177                (i, id)
178            })
179            .collect();
180
181        page_order.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
182
183        i = 0;
184
185        let needs_ordering = page_order.iter().any(|a| {
186            i += 1;
187            a.0 != i
188        });
189
190        if needs_ordering {
191            let mut pages = page_order.clone();
192            pages.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
193            let mut objects = BTreeMap::new();
194
195            for (old, new) in pages.iter().zip(page_order) {
196                if let Some(object) = self.objects.remove(&old.1) {
197                    objects.insert((new.1 .0, old.1 .1), object);
198                    replace.insert(old.1, (new.1 .0, old.1 .1));
199                }
200
201                if old.1 != new.1 {
202                    self.renumber_bookmarks(&old.1, &(new.1 .0, old.1 .1));
203                }
204            }
205
206            for (new, object) in objects {
207                self.objects.insert(new, object);
208            }
209
210            let action = |object: &mut Object| {
211                if let Object::Reference(id) = object {
212                    if replace.contains_key(id) {
213                        *id = replace[id];
214                    }
215                }
216            };
217
218            self.traverse_objects(action);
219            replace.clear();
220        }
221
222        let mut ids = self.objects.keys().cloned().collect::<Vec<ObjectId>>();
223        ids.sort_unstable();
224
225        for id in ids {
226            if id.0 != new_id {
227                replace.insert(id, (new_id, id.1));
228            }
229
230            new_id += 1;
231        }
232
233        let mut objects = BTreeMap::new();
234
235        // remove and collect all removed objects
236        for (old, new) in &replace {
237            if let Some(object) = self.objects.remove(old) {
238                objects.insert(*new, object);
239            }
240
241            if old != new {
242                self.renumber_bookmarks(old, new);
243            }
244        }
245
246        // insert new replaced keys objects
247        for (new, object) in objects {
248            self.objects.insert(new, object);
249        }
250
251        let action = |object: &mut Object| {
252            if let Object::Reference(id) = object {
253                if replace.contains_key(id) {
254                    *id = replace[id];
255                }
256            }
257        };
258
259        self.traverse_objects(action);
260
261        self.max_id = new_id - 1;
262    }

264    pub fn change_content_stream(&mut self, stream_id: ObjectId, content: Vec<u8>) {
265        if let Some(Object::Stream(stream)) = self.objects.get_mut(&stream_id) {

272    pub fn change_page_content(&mut self, page_id: ObjectId, content: Vec<u8>) -> Result<()> {
273        let contents = self.get_dictionary(page_id).and_then(|page| page.get(b"Contents"))?;
274        match contents {
275            Object::Reference(id) => self.change_content_stream(*id, content),
276            Object::Array(arr) => {
277                if arr.len() == 1 {
278                    if let Ok(id) = arr[0].as_reference() {
279                        self.change_content_stream(id, content)
280                    }
281                } else {
282                    let new_stream = self.add_object(super::Stream::new(dictionary! {}, content));
283                    if let Ok(Object::Dictionary(dict)) = self.get_object_mut(page_id) {
284                        dict.set("Contents", new_stream);
285                    }
286                }
287            }
288            _ => {}
289        }
290        Ok(())
291    }

293    pub fn extract_stream(&self, stream_id: ObjectId, decompress: bool) -> Result<()> {
294        let mut file = File::create(format!("{:?}.bin", stream_id))?;
295        if let Ok(Object::Stream(stream)) = self.get_object(stream_id) {
296            if decompress {
297                if let Ok(data) = stream.decompressed_content() {
298                    file.write_all(&data)?;
299                } else {
300                    file.write_all(&stream.content)?;
301                }
302            } else {
303                file.write_all(&stream.content)?;
304            }
305        }
306        Ok(())
307    }
