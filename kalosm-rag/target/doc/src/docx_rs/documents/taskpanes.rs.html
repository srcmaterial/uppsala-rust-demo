21impl BuildXML for Taskpanes {
22    fn build(&self) -> Vec<u8> {
23        let b = XMLBuilder::new();
24        let b = b
25            .declaration(Some(true))
26            .open_taskpanes("http://schemas.microsoft.com/office/webextensions/taskpanes/2010/11")
27            .open_taskpane("", "1", "350", "1")
28            .webextensionref(
29                "http://schemas.openxmlformats.org/officeDocument/2006/relationships",
30                "rId1",
31            )
32            .close()
33            .close();
34        b.build()
35    }
36}
