22#[derive(Serialize, Debug, PartialEq, Clone)]
23pub struct WebExtension {
24    pub id: String,
25    pub version: String,
26    pub store: String,
27    pub store_type: String,
28    pub reference_id: String,
29    pub properties: Vec<WebExtensionProperty>,
30}
31
32impl WebExtension {
33    pub fn new(
34        id: impl Into<String>,
35        reference_id: impl Into<String>,
36        version: impl Into<String>,
37        store: impl Into<String>,
38        store_type: impl Into<String>,
39    ) -> Self {
40        Self {
41            id: id.into(),
42            reference_id: reference_id.into(),
43            version: version.into(),
44            store: store.into(),
45            store_type: store_type.into(),
46            properties: vec![],
47        }
48    }
49
50    pub fn property(mut self, name: impl Into<String>, value: impl Into<String>) -> Self {
51        let v = value.into();
52        let v = format!("&quot;{}&quot;", escape(&v).replace("&quot;", "\\&quot;"));
53        self.properties.push(WebExtensionProperty::new(name, &v));
54        self
55    }
56}
57
58impl BuildXML for WebExtension {
59    fn build(&self) -> Vec<u8> {
60        let b = XMLBuilder::new();
61        let mut b = b
62            .declaration(Some(true))
63            .open_webextension(
64                "http://schemas.microsoft.com/office/webextensions/webextension/2010/11",
65                &format!("{{{}}}", &self.id),
66            )
67            .webextension_reference(
68                &self.reference_id,
69                &self.version,
70                &self.store,
71                &self.store_type,
72            )
73            .webextension_alternate_references()
74            .open_webextension_properties();
75
76        for p in self.properties.iter() {
77            b = b.webextension_property(&p.name, &p.value);
78        }
79
80        b.close()
81            .webextension_bindings()
82            .webextension_snapshot(
83                "http://schemas.openxmlformats.org/officeDocument/2006/relationships",
84            )
85            .close()
86            .build()
87    }
88}
