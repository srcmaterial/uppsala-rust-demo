8#[derive(Debug, Clone, PartialEq, Serialize)]
9#[serde(rename_all = "camelCase")]
10pub struct Styles {
11    doc_defaults: DocDefaults,
12    styles: Vec<Style>,
13}
14
15impl Styles {
16    pub fn new() -> Styles {
17        Default::default()
18    }
19
20    pub fn add_style(mut self, style: Style) -> Self {
21        self.styles.push(style);
22        self
23    }
24
25    pub fn default_size(mut self, size: usize) -> Self {
26        self.doc_defaults = self.doc_defaults.size(size);
27        self
28    }
29
30    pub fn default_spacing(mut self, spacing: i32) -> Self {
31        self.doc_defaults = self.doc_defaults.spacing(spacing);
32        self
33    }
34
35    pub fn default_line_spacing(mut self, spacing: LineSpacing) -> Self {
36        self.doc_defaults = self.doc_defaults.line_spacing(spacing);
37        self
38    }
39
40    pub fn default_fonts(mut self, font: RunFonts) -> Self {
41        self.doc_defaults = self.doc_defaults.fonts(font);
42        self
43    }
44
45    pub(crate) fn doc_defaults(mut self, doc_defaults: DocDefaults) -> Self {
46        self.doc_defaults = doc_defaults;
47        self
48    }
49
50    pub fn find_style_by_id(&self, id: &str) -> Option<&Style> {
51        self.styles.iter().find(|s| s.style_id == id)
52    }
53
54    pub fn create_heading_style_map(&self) -> std::collections::HashMap<String, usize> {
55        self.styles
56            .iter()
57            .filter_map(|s| {
58                if s.name.is_heading() {
59                    let n = s.name.get_heading_number();
60                    n.map(|n| (s.style_id.clone(), n))
61                } else {
62                    None
63                }
64            })
65            .collect()
66    }
67}
68
69impl Default for Styles {
70    fn default() -> Self {
71        Self {
72            doc_defaults: DocDefaults::new(),
73            styles: vec![],
74        }
75    }
76}
77
78impl BuildXML for Styles {
79    fn build(&self) -> Vec<u8> {
80        let b = XMLBuilder::new();
81        let normal = Style::new("Normal", StyleType::Paragraph).name("Normal");
82        b.open_styles()
83            .add_child(&self.doc_defaults)
84            .add_child(&normal)
85            .add_children(&self.styles)
86            .close()
87            .build()
88    }
89}
