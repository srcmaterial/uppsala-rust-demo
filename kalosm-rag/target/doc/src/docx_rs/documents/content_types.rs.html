147impl BuildXML for ContentTypes {
148    fn build(&self) -> Vec<u8> {
149        let b = XMLBuilder::new();
150        let mut b = b
151            .declaration(None)
152            .open_types("http://schemas.openxmlformats.org/package/2006/content-types");
153
154        b = b
155            .add_default("png", "image/png")
156            .add_default("jpeg", "image/jpeg")
157            .add_default("jpg", "image/jpg")
158            .add_default("bmp", "image/bmp")
159            .add_default("gif", "image/gif")
160            .add_default(
161                "rels",
162                "application/vnd.openxmlformats-package.relationships+xml",
163            )
164            .add_default("xml", "application/xml");
165
166        for (k, v) in self.types.iter() {
167            b = b.add_override(k, v);
168        }
169        b.close().build()
170    }
171}

173impl FromXML for ContentTypes {
174    fn from_xml<R: Read>(reader: R) -> Result<Self, ReaderError> {
175        let parser = EventReader::new(reader);
176        let mut s = Self::default();
177        let mut depth = 0;
178        for e in parser {
179            match e {
180                Ok(XmlEvent::StartElement { attributes, .. }) => {
181                    if depth == 1 {
182                        let namespace = attributes[0].value.clone();
183                        let path = attributes[1].value.clone();
184                        s = s.add_content(path, namespace);
185                    }
186                    depth += 1;
187                }
188                Ok(XmlEvent::EndElement { .. }) => {
189                    depth -= 1;
190                }
191                Err(_) => {}
192                _ => {}
193            }
194        }
195        Ok(s)
196    }
197}
