7use serde::Serialize;

9#[derive(Debug, Clone, PartialEq, Serialize)]
10#[serde(rename_all = "camelCase")]
11pub struct Settings {
12    default_tab_stop: DefaultTabStop,
13    zoom: Zoom,
14    doc_id: Option<DocId>,
15    doc_vars: Vec<DocVar>,
16    even_and_odd_headers: bool,
17    adjust_line_height_in_table: bool,
18    #[serde(skip_serializing_if = "Option::is_none")]
19    character_spacing_control: Option<CharacterSpacingValues>,
20}
21
22impl Settings {
23    pub fn new() -> Settings {
24        Default::default()
25    }
26
27    pub fn doc_id(mut self, id: impl Into<String>) -> Self {
28        self.doc_id = Some(DocId::new(id.into()));
29        self
30    }
31
32    pub fn default_tab_stop(mut self, tab_stop: usize) -> Self {
33        self.default_tab_stop = DefaultTabStop::new(tab_stop);
34        self
35    }
36
37    pub fn add_doc_var(mut self, name: impl Into<String>, val: impl Into<String>) -> Self {
38        self.doc_vars.push(DocVar::new(name, val));
39        self
40    }
41
42    pub fn even_and_odd_headers(mut self) -> Self {
43        self.even_and_odd_headers = true;
44        self
45    }
46
47    pub fn adjust_line_height_in_table(mut self) -> Self {
48        self.adjust_line_height_in_table = true;
49        self
50    }
51
52    pub fn character_spacing_control(mut self, val: CharacterSpacingValues) -> Self {
53        self.character_spacing_control = Some(val);
54        self
55    }
56}
57
58impl Default for Settings {
59    fn default() -> Self {
60        Self {
61            default_tab_stop: DefaultTabStop::new(840),
62            zoom: Zoom::new(100),
63            doc_id: None,
64            doc_vars: vec![],
65            even_and_odd_headers: false,
66            adjust_line_height_in_table: false,
67            character_spacing_control: None,
68        }
69    }
70}
71
72impl BuildXML for Settings {
73    fn build(&self) -> Vec<u8> {
74        let b = XMLBuilder::new();
75        let mut b = b
76            .declaration(Some(true))
77            .open_settings()
78            .add_child(&self.default_tab_stop)
79            .add_child(&self.zoom)
80            .open_compat()
81            .space_for_ul()
82            .balance_single_byte_double_byte_width()
83            .do_not_leave_backslash_alone()
84            .ul_trail_space()
85            .do_not_expand_shift_return();
86
87        if let Some(v) = self.character_spacing_control {
88            b = b.character_spacing_control(&v.to_string());
89        }
90
91        if self.adjust_line_height_in_table {
92            b = b.adjust_line_height_table();
93        }
94
95        b = b
96            .use_fe_layout()
97            .compat_setting(
98                "compatibilityMode",
99                "http://schemas.microsoft.com/office/word",
100                "15",
101            )
102            .compat_setting(
103                "overrideTableStyleFontSizeAndJustification",
104                "http://schemas.microsoft.com/office/word",
105                "1",
106            )
107            .compat_setting(
108                "enableOpenTypeFeatures",
109                "http://schemas.microsoft.com/office/word",
110                "1",
111            )
112            .compat_setting(
113                "doNotFlipMirrorIndents",
114                "http://schemas.microsoft.com/office/word",
115                "1",
116            )
117            .compat_setting(
118                "differentiateMultirowTableHeaders",
119                "http://schemas.microsoft.com/office/word",
120                "1",
121            )
122            .compat_setting(
123                "useWord2013TrackBottomHyphenation",
124                "http://schemas.microsoft.com/office/word",
125                "0",
126            )
127            .close()
128            .add_optional_child(&self.doc_id);
129
130        if !self.doc_vars.is_empty() {
131            b = b.open_doc_vars();
132            for v in self.doc_vars.iter() {
133                b = b.add_child(v);
134            }
135            b = b.close();
136        }
137
138        if self.even_and_odd_headers {
139            b = b.even_and_odd_headers();
140        }
141        b.close().build()
142    }
143}
