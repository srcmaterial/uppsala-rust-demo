1use super::*;
2use serde::ser::{SerializeStruct, Serializer};

5use crate::documents::BuildXML;

8#[derive(Debug, Clone, Serialize, PartialEq)]
9#[serde(rename_all = "camelCase")]
10pub struct WpsShape {
11    children: Vec<WpsShapeChild>,
12}
13
14#[derive(Debug, Clone, PartialEq)]
15pub enum WpsShapeChild {
16    WpsTextBox(WpsTextBox),
17}
18
19impl Serialize for WpsShapeChild {
20    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
21    where
22        S: Serializer,
23    {
24        match *self {
25            WpsShapeChild::WpsTextBox(ref s) => {
26                let mut t = serializer.serialize_struct("WpsTextBox", 2)?;
27                t.serialize_field("type", "textbox")?;
28                t.serialize_field("data", s)?;
29                t.end()
30            }
31        }
32    }
33}
34
35impl WpsShape {
36    pub fn new() -> WpsShape {
37        Default::default()
38    }
39
40    pub fn add_text_box(mut self, text_box: WpsTextBox) -> Self {
41        self.children.push(WpsShapeChild::WpsTextBox(text_box));
42        self
43    }
44}
45
46impl Default for WpsShape {
47    fn default() -> Self {
48        WpsShape { children: vec![] }
49    }
50}
51
52impl BuildXML for WpsShape {
53    fn build(&self) -> Vec<u8> {
54        let b = XMLBuilder::new();
55        let mut b = b.open_wp_text_box();
56        for c in &self.children {
57            match c {
58                WpsShapeChild::WpsTextBox(t) => b = b.add_child(t),
59            }
60        }
61        b.close().build()
62    }
