11#[derive(Debug, Clone, PartialEq, Serialize)]
12#[serde(rename_all = "camelCase")]
13pub struct TableProperty {
14    width: TableWidth,
15    justification: Justification,
16    borders: TableBorders,
17    #[serde(skip_serializing_if = "Option::is_none")]
18    margins: Option<TableCellMargins>,
19    #[serde(skip_serializing_if = "Option::is_none")]
20    indent: Option<TableIndent>,
21    #[serde(skip_serializing_if = "Option::is_none")]
22    style: Option<TableStyle>,
23    #[serde(skip_serializing_if = "Option::is_none")]
24    layout: Option<TableLayout>,
25    #[serde(skip_serializing_if = "Option::is_none")]
26    position: Option<TablePositionProperty>,
27}
28
29impl Default for TableProperty {
30    fn default() -> Self {
31        TableProperty {
32            width: TableWidth::new(0, WidthType::Auto),
33            justification: Justification::new("left"),
34            borders: TableBorders::new(),
35            margins: None,
36            indent: None,
37            style: None,
38            layout: None,
39            position: None,
40        }
41    }
42}
43
44impl TableProperty {
45    pub fn new() -> TableProperty {
46        Default::default()
47    }
48
49    pub fn without_borders() -> TableProperty {

51            borders: TableBorders::with_empty(),
52            ..Default::default()
53        }
54    }
55
56    pub fn indent(mut self, v: i32) -> TableProperty {
57        self.indent = Some(TableIndent::new(v, WidthType::Dxa));
58        self
59    }
60
61    pub fn width(mut self, v: usize, t: WidthType) -> TableProperty {
62        self.width = TableWidth::new(v, t);
63        self
64    }
65
66    pub fn align(mut self, v: TableAlignmentType) -> TableProperty {
67        self.justification = Justification::new(v.to_string());
68        self
69    }
70
71    pub fn set_margins(mut self, margins: TableCellMargins) -> Self {
72        self.margins = Some(margins);
73        self
74    }
75
76    pub fn cell_margin_top(mut self, v: usize, t: WidthType) -> Self {
77        if let Some(margins) = self.margins {
78            self.margins = Some(margins.margin_top(v, t));
79        } else {
80            let margins = TableCellMargins::new();

82        }
83        self
84    }

96    pub fn cell_margin_bottom(mut self, v: usize, t: WidthType) -> Self {

106    pub fn cell_margin_left(mut self, v: usize, t: WidthType) -> Self {
107        if let Some(margins) = self.margins {
108            self.margins = Some(margins.margin_left(v, t));
109        } else {

111            self.margins = Some(margins.margin_left(v, t));

152impl BuildXML for TableProperty {
153    fn build(&self) -> Vec<u8> {
154        XMLBuilder::new()
155            .open_table_property()
156            .add_child(&self.width)
157            .add_child(&self.justification)
158            .add_child(&self.borders)
159            .add_optional_child(&self.margins)
160            .add_optional_child(&self.indent)
161            .add_optional_child(&self.style)
162            .add_optional_child(&self.layout)
163            .add_optional_child(&self.position)
164            .close()
165            .build()
166    }
167}
