83impl BuildXML for ParagraphBorder {
84    fn build(&self) -> Vec<u8> {
85        let base = XMLBuilder::new();
86        let base = {
87            let val = self.val.to_string();
88            let space = self.space.to_string();
89            let size = self.size.to_string();
90            match self.position {
91                ParagraphBorderPosition::Top => base.paragraph_border_top(&val, &space, &size, &self.color),
92                ParagraphBorderPosition::Left => base.paragraph_border_left(&val, &space, &size, &self.color),
93                ParagraphBorderPosition::Bottom => base.paragraph_border_bottom(&val, &space, &size, &self.color),
94                ParagraphBorderPosition::Right => base.paragraph_border_right(&val, &space, &size, &self.color),
95                ParagraphBorderPosition::Between => base.paragraph_border_between(&val, &space, &size, &self.color),
96                ParagraphBorderPosition::Bar => base.paragraph_border_bar(&val, &space, &size, &self.color),
97            }
98        };
99        base.build()
100    }
101}

103#[derive(Serialize, Debug, Clone, PartialEq)]
104#[serde(rename_all = "camelCase")]
105pub struct ParagraphBorders {
106    left: Option<ParagraphBorder>,
107    right: Option<ParagraphBorder>,
108    top: Option<ParagraphBorder>,
109    bottom: Option<ParagraphBorder>,
110    between: Option<ParagraphBorder>,
111    bar: Option<ParagraphBorder>,
112}
113
114
115impl Default for ParagraphBorders {
116    fn default() -> Self {
117        ParagraphBorders {
118            left: Some(ParagraphBorder::new(ParagraphBorderPosition::Left)),
119            right: Some(ParagraphBorder::new(ParagraphBorderPosition::Right)),
120            top: Some(ParagraphBorder::new(ParagraphBorderPosition::Top)),
121            bottom: Some(ParagraphBorder::new(ParagraphBorderPosition::Bottom)),
122            between: None,
123            bar: None,
124        }
125    }
126}
127
128impl ParagraphBorders {
129    pub fn new() -> Self {
130        Self::default()
131    }
132
133    pub fn with_empty() -> Self {
134        ParagraphBorders {
135            left: None,
136            right: None,
137            top: None,
138            bottom: None,
139            between: None,
140            bar: None,
141        }
142    }
143
144    pub fn set(mut self, border: ParagraphBorder) -> Self {
145        match border.position {
146            ParagraphBorderPosition::Top => self.top = Some(border),
147            ParagraphBorderPosition::Left => self.left = Some(border),
148            ParagraphBorderPosition::Bottom => self.bottom = Some(border),
149            ParagraphBorderPosition::Right => self.right = Some(border),
150            ParagraphBorderPosition::Between => self.between = Some(border),
151            ParagraphBorderPosition::Bar => self.bar = Some(border),
152        };
153        self
154    }
155
156    pub fn clear(mut self, position: ParagraphBorderPosition) -> Self {
157        let nil = ParagraphBorder::new(position.clone()).val(BorderType::Nil);
158        match position {
159            ParagraphBorderPosition::Top => self.top = Some(nil),
160            ParagraphBorderPosition::Left => self.left = Some(nil),
161            ParagraphBorderPosition::Bottom => self.bottom = Some(nil),
162            ParagraphBorderPosition::Right => self.right = Some(nil),
163            ParagraphBorderPosition::Between => self.between = Some(nil),
164            ParagraphBorderPosition::Bar => self.bar = Some(nil),
165        };
166        self
167    }
168
169    pub fn clear_all(mut self) -> Self {
170        self.left = Some(ParagraphBorder::new(ParagraphBorderPosition::Left).val(BorderType::Nil));
171        self.right = Some(ParagraphBorder::new(ParagraphBorderPosition::Right).val(BorderType::Nil));
172        self.top = Some(ParagraphBorder::new(ParagraphBorderPosition::Top).val(BorderType::Nil));
173        self.bottom = Some(ParagraphBorder::new(ParagraphBorderPosition::Bottom).val(BorderType::Nil));
174        self.between = Some(ParagraphBorder::new(ParagraphBorderPosition::Between).val(BorderType::Nil));
175        self.bar = Some(ParagraphBorder::new(ParagraphBorderPosition::Bar).val(BorderType::Nil));
176        self
177    }
178}

180impl BuildXML for ParagraphBorders {
181    fn build(&self) -> Vec<u8> {
182        XMLBuilder::new()
183            .open_paragraph_borders()
184            .add_optional_child(&self.left)
185            .add_optional_child(&self.right)
186            .add_optional_child(&self.top)
187            .add_optional_child(&self.bottom)
188            .add_optional_child(&self.between)
189            .add_optional_child(&self.bar)
190            .close()
191            .build()
192    }
