112impl StructuredDataTag {
113    pub fn new() -> Self {
114        Default::default()
115    }
116
117    pub fn add_run(mut self, run: Run) -> Self {
118        self.children
119            .push(StructuredDataTagChild::Run(Box::new(run)));
120        self
121    }
122
123    pub fn add_paragraph(mut self, p: Paragraph) -> Self {
124        if p.has_numbering {
125            self.has_numbering = true
126        }
127        self.children
128            .push(StructuredDataTagChild::Paragraph(Box::new(p)));
129        self
130    }
131
132    pub fn add_table(mut self, t: Table) -> Self {
133        if t.has_numbering {
134            self.has_numbering = true
135        }
136        self.children
137            .push(StructuredDataTagChild::Table(Box::new(t)));
138        self
139    }
140
141    pub fn data_binding(mut self, d: DataBinding) -> Self {
142        self.property = self.property.data_binding(d);
143        self
144    }
145
146    pub fn alias(mut self, v: impl Into<String>) -> Self {
147        self.property = self.property.alias(v);
148        self
149    }
150
151    fn inner_build(&self) -> Vec<u8> {
152        XMLBuilder::new()
153            .open_structured_tag()
154            .add_child(&self.property)
155            .open_structured_tag_content()
156            .add_children(&self.children)
157            .close()
158            .close()
159            .build()
160    }
161}

163impl BuildXML for StructuredDataTag {
164    fn build(&self) -> Vec<u8> {
165        self.inner_build()
166    }
