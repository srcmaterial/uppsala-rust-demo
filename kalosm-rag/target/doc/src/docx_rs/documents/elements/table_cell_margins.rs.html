8#[derive(Debug, Clone, PartialEq, Serialize)]
9#[serde(rename_all = "camelCase")]
10pub struct TableCellMargins {
11    top: CellMargin,
12    left: CellMargin,
13    bottom: CellMargin,
14    right: CellMargin,
15}
16
17impl Default for TableCellMargins {
18    fn default() -> TableCellMargins {
19        TableCellMargins {
20            top: CellMargin {
21                val: 0,
22                width_type: WidthType::Dxa,
23            },
24            left: CellMargin::default(),
25            bottom: CellMargin {
26                val: 0,
27                width_type: WidthType::Dxa,
28            },
29            right: CellMargin::default(),
30        }
31    }
32}
33
34impl TableCellMargins {
35    pub fn new() -> TableCellMargins {
36        Default::default()
37    }
38
39    pub fn margin(self, top: usize, right: usize, bottom: usize, left: usize) -> TableCellMargins {
40        TableCellMargins {
41            top: CellMargin::new(top, WidthType::Dxa),
42            left: CellMargin::new(left, WidthType::Dxa),
43            bottom: CellMargin::new(bottom, WidthType::Dxa),
44            right: CellMargin::new(right, WidthType::Dxa),
45        }
46    }
47
48    pub fn margin_top(mut self, v: usize, t: WidthType) -> Self {
49        self.top = CellMargin::new(v, t);
50        self
51    }
52
53    pub fn margin_right(mut self, v: usize, t: WidthType) -> Self {
54        self.right = CellMargin::new(v, t);
55        self
56    }
57
58    pub fn margin_left(mut self, v: usize, t: WidthType) -> Self {
59        self.left = CellMargin::new(v, t);
60        self
61    }
62
63    pub fn margin_bottom(mut self, v: usize, t: WidthType) -> Self {
64        self.bottom = CellMargin::new(v, t);
65        self
66    }
67}
68
69impl BuildXML for TableCellMargins {
70    fn build(&self) -> Vec<u8> {
71        XMLBuilder::new()
72            .open_table_cell_margins()
73            .margin_top(self.top.val as i32, self.top.width_type)
74            .margin_left(self.left.val as i32, self.left.width_type)
75            .margin_bottom(self.bottom.val as i32, self.bottom.width_type)
76            .margin_right(self.right.val as i32, self.right.width_type)
77            .close()
78            .build()
79    }
80}
