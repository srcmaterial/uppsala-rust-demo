34        let has_numbering = rows.iter().any(|c| c.has_numbering);

124impl BuildXML for Table {
125    fn build(&self) -> Vec<u8> {
126        let grid = TableGrid::new(self.grid.clone());
127        let b = XMLBuilder::new()
128            .open_table()
129            .add_child(&self.property)
130            .add_child(&grid)
131            .add_children(&self.rows);
132        b.close().build()
133    }
134}

136impl BuildXML for Box<Table> {
137    fn build(&self) -> Vec<u8> {
138        Table::build(self)
139    }
