1use crate::documents::BuildXML;
2use crate::xml_builder::*;
3
4use crate::line_spacing_type::LineSpacingType;
5use serde::*;
6
7#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Default)]
8#[serde(rename_all = "camelCase")]
9pub struct LineSpacing {
10    #[serde(skip_serializing_if = "Option::is_none")]
11    line_rule: Option<LineSpacingType>,
12    #[serde(skip_serializing_if = "Option::is_none")]
13    before: Option<u32>,
14    #[serde(skip_serializing_if = "Option::is_none")]
15    after: Option<u32>,
16    #[serde(skip_serializing_if = "Option::is_none")]
17    before_lines: Option<u32>,
18    #[serde(skip_serializing_if = "Option::is_none")]
19    after_lines: Option<u32>,
20    #[serde(skip_serializing_if = "Option::is_none")]
21    line: Option<i32>,
22}
23
24impl LineSpacing {
25    pub fn new() -> Self {
26        Self::default()
27    }
28
29    pub fn line_rule(mut self, t: LineSpacingType) -> Self {
30        self.line_rule = Some(t);
31        self
32    }
33
34    pub fn before(mut self, before: u32) -> Self {
35        self.before = Some(before);
36        self
37    }
38
39    pub fn after(mut self, after: u32) -> Self {
40        self.after = Some(after);
41        self
42    }
43
44    pub fn before_lines(mut self, before: u32) -> Self {
45        self.before_lines = Some(before);
46        self
47    }
48
49    pub fn after_lines(mut self, after: u32) -> Self {
50        self.after_lines = Some(after);
51        self
52    }
53
54    pub fn line(mut self, line: i32) -> Self {
55        self.line = Some(line as i32);
56        self
57    }
58}
59
60impl BuildXML for LineSpacing {
61    fn build(&self) -> Vec<u8> {
62        let b = XMLBuilder::new();
63        b.line_spacing(
64            self.before,
65            self.after,
66            self.line,
67            self.before_lines,
68            self.after_lines,
69            self.line_rule,
70        )
71        .build()
72    }
73}
