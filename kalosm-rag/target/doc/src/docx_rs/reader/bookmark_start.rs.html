1use std::io::Read;
2use std::str::FromStr;
3
4use xml::attribute::OwnedAttribute;
5use xml::reader::EventReader;
6
7use super::*;

9impl ElementReader for BookmarkStart {
10    fn read<R: Read>(
11        _r: &mut EventReader<R>,
12        attrs: &[OwnedAttribute],
13    ) -> Result<Self, ReaderError> {
14        let mut id: Option<usize> = None;
15        let mut name: Option<String> = None;
16
17        for a in attrs {
18            let local_name = &a.name.local_name;
19            if local_name == "id" {
20                id = Some(usize::from_str(&a.value)?);
21            } else if local_name == "name" {
22                name = Some(a.value.clone());
23            }
24        }
25        if id.is_none() || name.is_none() {
26            return Err(ReaderError::XMLReadError);
27        }
28        Ok(BookmarkStart::new(id.unwrap(), name.unwrap()))
29    }
30}
