11fn read_run_fonts(attributes: &[OwnedAttribute]) -> Result<RunFonts, ReaderError> {
12    let mut f = RunFonts::new();
13    for a in attributes {
14        let local_name = &a.name.local_name;
15        match local_name.as_str() {
16            "asciiTheme" => {
17                f = f.ascii_theme(&a.value);
18            }
19            "eastAsiaTheme" => {
20                f = f.east_asia_theme(&a.value);
21            }
22            "hAnsiTheme" => {
23                f = f.hi_ansi_theme(&a.value);
24            }
25            "cstheme" => {
26                f = f.cs_theme(&a.value);
27            }
28            "ascii" => {
29                f = f.ascii(&a.value);
30            }
31            "eastAsia" => {
32                f = f.east_asia(&a.value);
33            }
34            "hAnsi" => {
35                f = f.hi_ansi(&a.value);
36            }
37            "cs" => {
38                f = f.cs(&a.value);
39            }
40            "hint" => {
41                f = f.hint(&a.value);
42            }
43            _ => {}
44        }
45    }
46    Ok(f)
47}

49impl ElementReader for RunProperty {
50    fn read<R: Read>(
51        r: &mut EventReader<R>,
52        _attrs: &[OwnedAttribute],
53    ) -> Result<Self, ReaderError> {
54        let mut rp = RunProperty::new();
55        loop {
56            let e = r.next();
57            match e {
58                Ok(XmlEvent::StartElement {
59                    attributes, name, ..
60                }) => {
61                    let e = XMLElement::from_str(&name.local_name).unwrap();
62                    match e {
63                        XMLElement::RunStyle => {
64                            if let Some(v) = read_val(&attributes) {
65                                rp = rp.style(&v);
66                            }
67                        }
68                        XMLElement::Bold => {
69                            if !read_bool(&attributes) {
70                                rp = rp.disable_bold();
71                                continue;
72                            }
73                            rp = rp.bold();
74                        }
75                        XMLElement::Caps => {
76                            if !read_bool(&attributes) {
77                                rp.caps = Some(Caps::new().disable());
78                                continue;
79                            }
80                            rp = rp.caps();
81                        }
82                        XMLElement::Highlight => rp = rp.highlight(attributes[0].value.clone()),
83                        XMLElement::Strike => {
84                            if !read_bool(&attributes) {
85                                rp.strike = Some(Strike::new().disable());
86                                continue;
87                            }
88                            rp = rp.strike();
89                        }
90                        XMLElement::VertAlign => {
91                            if let Ok(v) = VertAlignType::from_str(&attributes[0].value) {
92                                rp = rp.vert_align(v)
93                            }
94                        }
95                        XMLElement::Color => rp = rp.color(attributes[0].value.clone()),
96                        XMLElement::Size => {
97                            rp = rp.size(f64::from_str(&attributes[0].value)? as usize)
98                        }
99                        XMLElement::Spacing => {
100                            if let Some(v) = read_val(&attributes) {
101                                if let Ok(s) = f64::from_str(&v) {
102                                    rp = rp.spacing(s as i32)
103                                }
104                            }
105                        }
106                        XMLElement::RunFonts => {
107                            if let Ok(f) = read_run_fonts(&attributes) {
108                                rp = rp.fonts(f);
109                            }
110                        }
111                        XMLElement::Underline => rp = rp.underline(&attributes[0].value.clone()),
112                        XMLElement::Italic => {
113                            if !read_bool(&attributes) {
114                                rp = rp.disable_italic();
115                                continue;
116                            }
117                            rp = rp.italic();
118                        }
119                        XMLElement::Vanish => rp = rp.vanish(),
120                        XMLElement::SpecVanish => rp = rp.spec_vanish(),
121                        XMLElement::TextBorder => {
122                            if let Ok(attr) = read_border(&attributes) {
123                                let mut border = TextBorder::new()
124                                    .border_type(attr.border_type)
125                                    .color(attr.color);
126                                if let Some(size) = attr.size {
127                                    border = border.size(size as usize);
128                                };
129                                rp = rp.text_border(border);
130                                continue;
131                            }
132                        }
133                        XMLElement::Insert => {
134                            if let Ok(ins) = Insert::read(r, &attributes) {
135                                rp = rp.insert(ins);
136                            }
137                        }
138                        XMLElement::Delete => {
139                            if let Ok(del) = Delete::read(r, &attributes) {
140                                rp = rp.delete(del);
141                            }
142                        }
143                        _ => {}
144                    }
145                }
146                Ok(XmlEvent::EndElement { name, .. }) => {
147                    let e = XMLElement::from_str(&name.local_name).unwrap();
148                    if e == XMLElement::RunProperty {
149                        return Ok(rp);
150                    }
151                }
152                Err(_) => return Err(ReaderError::XMLReadError),
153                _ => {}
154            }
155        }
156    }
157}
