10// @See: 20.4.3.4 ST_RelFromH (Horizontal Relative Positioning)
11#[cfg_attr(feature = "wasm", wasm_bindgen)]
12#[derive(Debug, Clone, Copy, PartialEq, Serialize)]
13#[cfg_attr(feature = "wasm", derive(ts_rs::TS))]
14#[cfg_attr(feature = "wasm", ts(export))]
15#[serde(rename_all = "camelCase")]
16pub enum RelativeFromHType {
17    /// Specifies that the horizontal positioning shall be
18    /// relative to the position of the anchor within its run
19    /// content.
20    Character,
21    /// Specifies that the horizontal positioning shall be
22    /// relative to the extents of the column which contains its
23    /// anchor.
24    Column,
25    /// Specifies that the horizontal positioning shall be
26    /// relative to the inside margin of the current page (the
27    /// left margin on odd pages, right on even pages).
28    InsideMargin,
29    /// Specifies that the horizontal positioning shall be
30    /// relative to the left margin of the page.
31    LeftMargin,
32    /// Specifies that the horizontal positioning shall be
33    /// relative to the page margins.
34    Margin,
35    /// Specifies that the horizontal positioning shall be
36    /// relative to the outside margin of the current page (the
37    /// right margin on odd pages, left on even pages).
38    OutsizeMargin,
39    /// Specifies that the horizontal positioning shall be
40    /// relative to the edge of the page.
41    Page,
42    /// Specifies that the horizontal positioning shall be
43    /// relative to the right margin of the page.
44    RightMargin,
45}
46
47impl Default for RelativeFromHType {
48    fn default() -> Self {
49        Self::Margin
50    }
51}
52
53impl fmt::Display for RelativeFromHType {
54    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
55        match *self {
56            RelativeFromHType::Character => write!(f, "character"),
57            RelativeFromHType::Column => write!(f, "column"),
58            RelativeFromHType::InsideMargin => write!(f, "insideMargin"),
59            RelativeFromHType::LeftMargin => write!(f, "leftMargin"),
60            RelativeFromHType::Margin => write!(f, "margin"),
61            RelativeFromHType::OutsizeMargin => write!(f, "outsizeMargin"),
62            RelativeFromHType::Page => write!(f, "page"),
63            RelativeFromHType::RightMargin => write!(f, "rightMargin"),
64        }
65    }
66}
67
68impl FromStr for RelativeFromHType {
69    type Err = errors::TypeError;
70    fn from_str(s: &str) -> Result<Self, Self::Err> {
71        match s {
72            "character" => Ok(RelativeFromHType::Character),
73            "column" => Ok(RelativeFromHType::Column),
74            "insideMargin" => Ok(RelativeFromHType::InsideMargin),
75            "leftMargin" => Ok(RelativeFromHType::LeftMargin),
76            "margin" => Ok(RelativeFromHType::Margin),
77            "outsizeMargin" => Ok(RelativeFromHType::OutsizeMargin),
78            "page" => Ok(RelativeFromHType::Page),
79            "rightMargin" => Ok(RelativeFromHType::RightMargin),
80            _ => Ok(RelativeFromHType::Margin),
81        }
82    }
83}
