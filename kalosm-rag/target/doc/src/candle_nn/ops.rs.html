7/// Applies the softmax function to the input tensor, rescaling the element so that elements on
8/// a slice of fixed index on dimension `dim` are between 0 and 1 and sum to 1.
9///
10/// ```rust
11/// use candle::{Tensor, Device, test_utils::to_vec2_round};
12/// let a = Tensor::new(&[[0f32, 1., 0., 1.], [-2., 2., 3., -3.]], &Device::Cpu)?;
13/// let a = candle_nn::ops::softmax(&a, 1)?;
14/// assert_eq!(
15///     to_vec2_round(&a, 4)?,
16///     &[
17///         [0.1345, 0.3655, 0.1345, 0.3655],
18///         [0.0049, 0.2671, 0.7262, 0.0018]
19///     ]);
20/// # Ok::<(), candle::Error>(())
21/// ```
22pub fn softmax<D: candle::shape::Dim>(xs: &Tensor, dim: D) -> Result<Tensor> {
23    let dim = dim.to_index(xs.shape(), "softmax")?;
24    let max = xs.max_keepdim(dim)?;
25    let diff = xs.broadcast_sub(&max)?;
26    let num = diff.exp()?;
27    let den = num.sum_keepdim(dim)?;
28    num.broadcast_div(&den)
29}

31pub fn log_softmax<D: candle::shape::Dim>(xs: &Tensor, d: D) -> Result<Tensor> {
32    let d = d.to_index(xs.shape(), "log-softmax")?;
33    let max = xs.max_keepdim(d)?;
34    let diff = xs.broadcast_sub(&max)?;
35    let sum_exp = diff.exp()?.sum_keepdim(d)?;
36    let log_sm = diff.broadcast_sub(&sum_exp.log()?)?;
37    Ok(log_sm)
38}

56    fn cpu_fwd(&self, storage: &CpuStorage, layout: &Layout) -> Result<(CpuStorage, Shape)> {
57        use candle::backend::BackendStorage;
58
59        fn fwd<T: num_traits::Float>(v: T) -> T {
60            (v.neg().exp() + T::one()).recip()
61        }
62
63        // FIXME: using `candle::map_dtype` causes compilation errors.
64        let storage = match storage {
65            CpuStorage::BF16(slice) => {
66                CpuStorage::BF16(candle::cpu_backend::unary_map(slice, layout, fwd))
67            }
68            CpuStorage::F16(slice) => {
69                CpuStorage::F16(candle::cpu_backend::unary_map(slice, layout, fwd))
70            }
71            CpuStorage::F32(slice) => {
72                CpuStorage::F32(candle::cpu_backend::unary_map(slice, layout, fwd))
73            }
74            CpuStorage::F64(slice) => {
75                CpuStorage::F64(candle::cpu_backend::unary_map(slice, layout, fwd))
76            }
77            _ => Err(candle::Error::UnsupportedDTypeForOp(
78                storage.dtype(),
79                self.name(),
80            ))?,
81        };
82        Ok((storage, layout.shape().clone()))
83    }

86    fn cuda_fwd(
87        &self,
88        storage: &candle::CudaStorage,
89        layout: &Layout,
90    ) -> Result<(candle::CudaStorage, Shape)> {
91        use candle::backend::BackendStorage;
92        use candle::cuda_backend::cudarc::driver::{
93            CudaSlice, DeviceRepr, LaunchAsync, LaunchConfig, ValidAsZeroBits,

95        use candle::cuda_backend::SlicePtrOrNull;

99        struct S;
100        impl Map1 for S {

102                &self,
103                src: &CudaSlice<T>,
104                dev: &CudaDevice,
105                layout: &Layout,
106            ) -> Result<CudaSlice<T>> {

111                let ds = SlicePtrOrNull::params_from_layout(dev, layout)?;
112                let src = &src.slice(layout.start_offset()..);
113                let func = dev.get_or_load_func(&kernel_name::<T>("usigmoid"), kernels::UNARY)?;
114                // SAFETY: Set later by running the kernel.

118                // SAFETY: ffi.
119                unsafe { func.launch(cfg, params) }.w()?;
120                Ok(out)
121            }
122        }
123
124        let dev = storage.device();
125        let slice = S.map(&storage.slice, dev, layout)?;
126        let dst = candle::CudaStorage {
127            slice,
128            device: dev.clone(),
129        };
130        Ok((dst, layout.shape().clone()))

134    fn metal_fwd(
135        &self,
136        storage: &candle::MetalStorage,
137        layout: &Layout,
138    ) -> Result<(candle::MetalStorage, Shape)> {
139        use candle::backend::BackendStorage;
140        use candle::MetalError;
141        let device = storage.device();
142        let dtype = storage.dtype();
143        let shape = layout.shape();
144        let el_count = shape.elem_count();
145        let buffer = device.new_buffer(el_count, dtype, "sigmoid")?;
146        let command_buffer = device.command_buffer()?;
147        command_buffer.set_label("sigmoid");
148        let src = candle_metal_kernels::BufferOffset {
149            buffer: storage.buffer(),
150            offset_in_bytes: layout.start_offset() * storage.dtype().size_in_bytes(),
151        };
152
153        match (el_count % 2, dtype, layout.is_contiguous()) {
154            (0, DType::BF16 | DType::F16, true) => {
155                use candle_metal_kernels::unary::contiguous_tiled;
156                let kernel_name = match dtype {
157                    DType::F16 => contiguous_tiled::sigmoid::HALF,
158                    DType::F32 => contiguous_tiled::sigmoid::FLOAT,
159                    DType::BF16 => contiguous_tiled::sigmoid::BFLOAT,
160                    dtype => {
161                        candle::bail!(
162                            "Metal contiguous_tiled unary sigmoid {dtype:?} not implemented"
163                        )
164                    }
165                };
166                candle_metal_kernels::call_unary_contiguous_tiled(
167                    device.metal_device(),
168                    &command_buffer,
169                    device.kernels(),
170                    kernel_name,
171                    el_count,
172                    src,
173                    &buffer,
174                )
175                .map_err(MetalError::from)?;
176            }
177            (_, _, true) => {
178                use candle_metal_kernels::unary::contiguous;
179                let kernel_name = match dtype {
180                    DType::F16 => contiguous::sigmoid::HALF,
181                    DType::F32 => contiguous::sigmoid::FLOAT,
182                    DType::BF16 => contiguous::sigmoid::BFLOAT,
183                    dtype => {
184                        candle::bail!("Metal contiguous unary sigmoid {dtype:?} not implemented")
185                    }
186                };
187                candle_metal_kernels::call_unary_contiguous(
188                    device.metal_device(),
189                    &command_buffer,
190                    device.kernels(),
191                    kernel_name,
192                    el_count,
193                    src,
194                    &buffer,
195                )
196                .map_err(MetalError::from)?;
197            }
198            (_, _, false) => {
199                use candle_metal_kernels::unary::strided;
200                let kernel_name = match dtype {
201                    DType::F16 => strided::sigmoid::HALF,
202                    DType::F32 => strided::sigmoid::FLOAT,
203                    DType::BF16 => strided::sigmoid::BFLOAT,
204                    dtype => {
205                        candle::bail!("Metal strided unary sigmoid {dtype:?} not implemented")
206                    }
207                };
208                let dst = candle_metal_kernels::BufferOffset::zero_offset(&buffer);
209                candle_metal_kernels::call_unary_strided(
210                    device.metal_device(),
211                    &command_buffer,
212                    device.kernels(),
213                    kernel_name,
214                    layout.dims(),
215                    src,
216                    layout.stride(),
217                    dst,
218                )
219                .map_err(MetalError::from)?;
220            }
221        }
222
223        let new_storage = candle::MetalStorage::new(buffer, device.clone(), el_count, dtype);
224        Ok((new_storage, layout.shape().clone()))
225    }

227    fn bwd(&self, _arg: &Tensor, res: &Tensor, grad_res: &Tensor) -> Result<Option<Tensor>> {
228        // d/dx sigmoid(x) = (1 - sigmoid(x)) * sigmoid(x)
229        let d_dx_sigmoid = res.ones_like()?.sub(res)?.mul(res)?;
230        Ok(Some(grad_res.mul(&d_dx_sigmoid)?))
231    }

248pub fn dropout(xs: &Tensor, drop_p: f32) -> Result<Tensor> {
249    // This implementation is inefficient as it stores the full mask for the backward pass.
250    // Instead we could just store the seed and have a specialized kernel that would both
251    // generate the random mask and apply it.
252    // Another easier optimization would be to be able to generate boolean mask using just a bit of
253    // entropy per element rather than generating a full float per element.
254    if !(0. ..1.).contains(&drop_p) {
255        candle::bail!("dropout probability has to be in [0, 1), got {drop_p}")
256    }
257    let rand = Tensor::rand(0f32, 1f32, xs.shape(), xs.device())?;
258    let scale = 1.0 / (1.0 - drop_p as f64);
259    let drop_p = Tensor::new(drop_p, xs.device())?.broadcast_as(xs.shape())?;
260    let mask = (rand.ge(&drop_p)?.to_dtype(xs.dtype())? * scale)?;
261    xs * mask
262}

291impl candle::CustomOp1 for SoftmaxLastDim {
292    fn name(&self) -> &'static str {
293        "softmax-last-dim"
294    }

296    fn cpu_fwd(&self, storage: &CpuStorage, layout: &Layout) -> Result<(CpuStorage, Shape)> {
297        fn softmax<T: candle::WithDType + num_traits::Float>(
298            src: &[T],
299            layout: &Layout,
300        ) -> Result<(CpuStorage, Shape)> {
301            let src = match layout.contiguous_offsets() {
302                None => candle::bail!("input has to be contiguous"),
303                Some((o1, o2)) => &src[o1..o2],
304            };
305            let el_count = layout.shape().elem_count();
306            let dims = layout.shape().dims();
307            let dim_m1 = dims[dims.len() - 1];
308            let mut dst = vec![T::zero(); el_count];
309            src.par_chunks(dim_m1)
310                .zip(dst.par_chunks_mut(dim_m1))
311                .for_each(|(src, dst)| {
312                    let mut max = T::neg_infinity();
313                    unsafe { T::vec_reduce_max(src.as_ptr(), &mut max, dim_m1) };
314                    for (s, d) in src.iter().zip(dst.iter_mut()) {
315                        *d = (*s - max).exp();
316                    }
317                    let mut sum_exp = T::zero();
318                    unsafe { T::vec_reduce_sum(dst.as_ptr(), &mut sum_exp, dim_m1) };
319                    for d in dst.iter_mut() {
320                        *d /= sum_exp
321                    }
322                });
323            let storage = candle::WithDType::to_cpu_storage_owned(dst);
324            Ok((storage, Shape::from_dims(dims)))
325        }
326
327        match storage {
328            CpuStorage::BF16(slice) => softmax::<half::bf16>(slice, layout),
329            CpuStorage::F16(slice) => softmax::<half::f16>(slice, layout),
330            CpuStorage::F32(slice) => softmax::<f32>(slice, layout),
331            CpuStorage::F64(slice) => softmax::<f64>(slice, layout),
332            _ => candle::bail!("unsupported dtype for softmax {:?}", storage),
333        }
334    }

392        &self,
393        storage: &candle::MetalStorage,
394        layout: &Layout,
395    ) -> Result<(candle::MetalStorage, Shape)> {
396        use candle::backend::BackendStorage;
397        let device = storage.device();
398        let command_buffer = device.command_buffer()?;
399        let kernels = device.kernels();
400        let name = match storage.dtype() {

402            DType::F16 => "softmax_f16",
403            DType::BF16 => "softmax_bf16",

405        };

409            candle::bail!("Non contiguous softmax-last-dim is not implemented");

413        let elem_count = layout.shape().elem_count();
414        let output = device.new_buffer(elem_count, storage.dtype(), "softmax")?;
415        candle_metal_kernels::call_last_softmax(
416            device.metal_device(),
417            &command_buffer,
418            kernels,
419            name,
420            elem_count,
421            last_dim,
422            storage.buffer(),
423            layout.start_offset() * storage.dtype().size_in_bytes(),
424            &output,
425        )
426        .map_err(candle::Error::wrap)?;
427        let newstorage =
428            candle::MetalStorage::new(output, device.clone(), elem_count, storage.dtype());
429        Ok((newstorage, layout.shape().clone()))
430    }

437#[derive(Debug, Clone)]
438struct RmsNorm {
439    eps: f32,
440}
441
442impl candle::CustomOp2 for RmsNorm {
443    fn name(&self) -> &'static str {
444        "rms-norm"
445    }
446
447    fn cpu_fwd(
448        &self,
449        s1: &CpuStorage,
450        l1: &Layout,
451        s2: &CpuStorage,
452        l2: &Layout,
453    ) -> Result<(CpuStorage, Shape)> {
454        use candle::backend::BackendStorage;
455
456        let eps = self.eps;
457        fn inner<
458            T: candle::WithDType
459                + num_traits::Float
460                + num_traits::AsPrimitive<f32>
461                + num_traits::FromPrimitive,
462        >(
463            src: &[T],
464            layout: &Layout,
465            alpha: &[T],
466            alpha_layout: &Layout,
467            eps: f32,
468        ) -> Result<(CpuStorage, Shape)> {
469            let src = match layout.contiguous_offsets() {
470                None => candle::bail!("input has to be contiguous"),
471                Some((o1, o2)) => &src[o1..o2],
472            };
473            let alpha = match alpha_layout.contiguous_offsets() {
474                None => candle::bail!("alpha has to be contiguous"),
475                Some((o1, o2)) => &alpha[o1..o2],
476            };
477            let el_count = layout.shape().elem_count();
478            let dims = layout.shape().dims();
479            let dim_m1 = dims[dims.len() - 1];
480            let mut dst = vec![T::zero(); el_count];
481            src.par_chunks(dim_m1)
482                .zip(dst.par_chunks_mut(dim_m1))
483                .for_each(|(src, dst)| {
484                    let sum2 = src
485                        .iter()
486                        .map(|&v| {
487                            let v = v.as_();
488                            v * v
489                        })
490                        .sum::<f32>();
491                    let m = (sum2 / dim_m1 as f32 + eps).sqrt();
492                    let m = T::from_f32(m).unwrap_or_else(T::nan);
493                    for ((d, s), alpha) in dst.iter_mut().zip(src.iter()).zip(alpha) {
494                        *d = *s / m * *alpha
495                    }
496                });
497            let storage = candle::WithDType::to_cpu_storage_owned(dst);
498            Ok((storage, Shape::from_dims(dims)))
499        }
500
501        use CpuStorage as C;
502        match (s1, s2) {
503            (C::BF16(s1), C::BF16(s2)) => inner::<half::bf16>(s1, l1, s2, l2, eps),
504            (C::F16(s1), C::F16(s2)) => inner::<half::f16>(s1, l1, s2, l2, eps),
505            (C::F32(s1), C::F32(s2)) => inner::<f32>(s1, l1, s2, l2, eps),
506            _ => candle::bail!("unsupported dtype for rmsnorm {:?}", s1.dtype()),
507        }
508    }

514        l1: &Layout,

582    #[cfg(feature = "metal")]
583    fn metal_fwd(
584        &self,
585        s1: &candle::MetalStorage,
586        l1: &Layout,
587        s2: &candle::MetalStorage,
588        l2: &Layout,
589    ) -> Result<(candle::MetalStorage, Shape)> {
590        use candle::backend::BackendStorage;
591        let device = s1.device();
592        let command_buffer = device.command_buffer()?;
593        let kernels = device.kernels();
594        let name = match (s1.dtype(), s2.dtype()) {
595            (DType::F32, DType::F32) => "rmsnorm_f32",
596            (DType::F16, DType::F16) => "rmsnorm_f16",
597            (DType::BF16, DType::BF16) => "rmsnorm_bf16",
598            (dt1, dt2) => candle::bail!("rmsnorm is not implemented for {dt1:?} {dt2:?}"),
599        };
600
601        if !(l1.is_contiguous() && l2.is_contiguous()) {
602            candle::bail!("Non contiguous rmsnorm is not implemented");
603        }
604
605        let last_dim = l1.dims()[l1.shape().rank() - 1];
606        let elem_count = l1.shape().elem_count();
607        let output = device.new_buffer(elem_count, s1.dtype(), "rmsnorm")?;
608        candle_metal_kernels::call_rms_norm(
609            device.metal_device(),
610            &command_buffer,
611            kernels,
612            name,
613            elem_count,
614            last_dim,
615            self.eps,
616            s1.buffer(),
617            l1.start_offset() * s1.dtype().size_in_bytes(),
618            s2.buffer(),
619            l2.start_offset() * s2.dtype().size_in_bytes(),
620            &output,
621        )
622        .map_err(candle::Error::wrap)?;
623        let newstorage = candle::MetalStorage::new(output, device.clone(), elem_count, s1.dtype());
624        Ok((newstorage, l1.shape().clone()))
625    }

628pub fn rms_norm_slow(x: &Tensor, alpha: &Tensor, eps: f32) -> Result<Tensor> {
629    let x_dtype = x.dtype();
630    let internal_dtype = match x_dtype {
631        DType::F16 | DType::BF16 => DType::F32,
632        d => d,
633    };
634    let hidden_size = x.dim(D::Minus1)?;
635    let x = x.to_dtype(internal_dtype)?;
636    let norm_x = (x.sqr()?.sum_keepdim(D::Minus1)? / hidden_size as f64)?;
637    let x_normed = x.broadcast_div(&(norm_x + eps as f64)?.sqrt()?)?;
638    x_normed.to_dtype(x_dtype)?.broadcast_mul(alpha)
639}

641pub fn rms_norm(xs: &Tensor, alpha: &Tensor, eps: f32) -> Result<Tensor> {
642    let hidden_size_xs = xs.dim(D::Minus1)?;
643    let hidden_size_alpha = alpha.dims1()?;
644    if hidden_size_xs != hidden_size_alpha {
645        candle::bail!(
646            "shape mismatch in rms-norm {:?} {:?}",
647            xs.shape(),
648            alpha.shape()
649        )
650    }
651    xs.apply_op2_no_bwd(alpha, &RmsNorm { eps })
652}

655struct LayerNorm {
656    eps: f32,
657}

659impl candle::CustomOp3 for LayerNorm {
660    fn name(&self) -> &'static str {
661        "layer-norm"
662    }
663
664    fn cpu_fwd(
665        &self,
666        s1: &CpuStorage,
667        l1: &Layout,
668        s2: &CpuStorage,
669        l2: &Layout,
670        s3: &CpuStorage,
671        l3: &Layout,
672    ) -> Result<(CpuStorage, Shape)> {
673        use candle::backend::BackendStorage;
674
675        let eps = self.eps;
676        fn inner<
677            T: candle::WithDType
678                + num_traits::Float
679                + num_traits::AsPrimitive<f32>
680                + num_traits::FromPrimitive,
681        >(
682            src: &[T],
683            layout: &Layout,
684            alpha: &[T],
685            alpha_layout: &Layout,
686            beta: &[T],
687            beta_layout: &Layout,
688            eps: f32,
689        ) -> Result<(CpuStorage, Shape)> {
690            let src = match layout.contiguous_offsets() {
691                None => candle::bail!("input has to be contiguous"),
692                Some((o1, o2)) => &src[o1..o2],
693            };
694            let alpha = match alpha_layout.contiguous_offsets() {
695                None => candle::bail!("alpha has to be contiguous"),
696                Some((o1, o2)) => &alpha[o1..o2],
697            };
698            let beta = match beta_layout.contiguous_offsets() {
699                None => candle::bail!("beta has to be contiguous"),
700                Some((o1, o2)) => &beta[o1..o2],
701            };
702            let el_count = layout.shape().elem_count();
703            let dims = layout.shape().dims();
704            let dim_m1 = dims[dims.len() - 1];
705            let mut dst = vec![T::zero(); el_count];
706            src.par_chunks(dim_m1)
707                .zip(dst.par_chunks_mut(dim_m1))
708                .for_each(|(src, dst)| {
709                    let mut sum = 0f32;
710                    let mut sum2 = 0f32;
711                    for v in src {
712                        let v = v.as_();
713                        sum += v;
714                        sum2 += v * v;
715                    }
716                    let mean = sum / dim_m1 as f32;
717                    let var = sum2 / dim_m1 as f32 - mean * mean;
718                    let inv_std = (var + eps).sqrt().recip();
719                    for ((d, s), (alpha, beta)) in
720                        dst.iter_mut().zip(src.iter()).zip(alpha.iter().zip(beta))
721                    {
722                        let alpha = alpha.as_();
723                        let beta = beta.as_();
724                        let d_ = (s.as_() - mean) * inv_std * alpha + beta;
725                        *d = T::from_f32(d_).unwrap_or_else(T::nan);
726                    }
727                });
728            let storage = candle::WithDType::to_cpu_storage_owned(dst);
729            Ok((storage, Shape::from_dims(dims)))
730        }
731
732        use CpuStorage as C;
733        match (s1, s2, s3) {
734            (C::BF16(s1), C::BF16(s2), C::BF16(s3)) => {
735                inner::<half::bf16>(s1, l1, s2, l2, s3, l3, eps)
736            }
737            (C::F16(s1), C::F16(s2), C::F16(s3)) => inner::<half::f16>(s1, l1, s2, l2, s3, l3, eps),
738            (C::F32(s1), C::F32(s2), C::F32(s3)) => inner::<f32>(s1, l1, s2, l2, s3, l3, eps),
739            _ => candle::bail!("unsupported dtype for rmsnorm {:?}", s1.dtype()),
740        }
741    }

811            }
812        }

815        let dev = s1.device();
816        let slice = S { eps: self.eps }.map(&s1.slice, l1, &s2.slice, l2, &s3.slice, l3, dev)?;
817        let dst = candle::cuda_backend::CudaStorage {

822    }

876pub fn layer_norm_slow(x: &Tensor, alpha: &Tensor, beta: &Tensor, eps: f32) -> Result<Tensor> {
877    let x_dtype = x.dtype();
878    let internal_dtype = match x_dtype {
879        DType::F16 | DType::BF16 => DType::F32,
880        d => d,
881    };
882    let hidden_size = x.dim(D::Minus1)?;
883    let x = x.to_dtype(internal_dtype)?;
884    let x = {
885        let mean_x = (x.sum_keepdim(D::Minus1)? / hidden_size as f64)?;
886        x.broadcast_sub(&mean_x)?
887    };
888    let norm_x = (x.sqr()?.sum_keepdim(D::Minus1)? / hidden_size as f64)?;
889    let x_normed = x.broadcast_div(&(norm_x + eps as f64)?.sqrt()?)?;
890    x_normed
891        .to_dtype(x_dtype)?
892        .broadcast_mul(alpha)?
893        .broadcast_add(beta)
894}

896pub fn layer_norm(xs: &Tensor, alpha: &Tensor, beta: &Tensor, eps: f32) -> Result<Tensor> {
897    let hidden_size_xs = xs.dim(D::Minus1)?;
898    let hidden_size_alpha = alpha.dims1()?;
899    let hidden_size_beta = beta.dims1()?;

902            "shape mismatch in layer-norm src: {:?} alpha: {:?} beta: {:?}",
903            xs.shape(),
904            alpha.shape(),
905            beta.shape()
906        )
907    }
908    xs.apply_op3_no_bwd(alpha, beta, &LayerNorm { eps })
909}

911// https://pytorch.org/docs/stable/generated/torch.nn.PixelShuffle.html
912pub fn pixel_shuffle(xs: &Tensor, upscale_factor: usize) -> Result<Tensor> {
913    let (b_size, c, h, w) = xs.dims4()?;
914    let out_c = c / upscale_factor / upscale_factor;
915    xs.reshape((b_size, out_c, upscale_factor, upscale_factor, h, w))?
916        .permute((0, 1, 4, 2, 5, 3))?
917        .reshape((b_size, out_c, h * upscale_factor, w * upscale_factor))
918}

920pub fn pixel_unshuffle(xs: &Tensor, downscale_factor: usize) -> Result<Tensor> {
921    let (b_size, c, h, w) = xs.dims4()?;
922    let out_c = c * downscale_factor * downscale_factor;
923    xs.reshape((
924        b_size,
925        c,
926        h / downscale_factor,
927        downscale_factor,
928        w / downscale_factor,
929        downscale_factor,
930    ))?
931    .permute((0, 1, 3, 5, 2, 4))?
932    .reshape((b_size, out_c, h / downscale_factor, w / downscale_factor))
933}

935// https://pytorch.org/docs/stable/generated/torch.nn.ReplicationPad2d.html
936pub fn replication_pad2d(xs: &Tensor, pad: usize) -> Result<Tensor> {
937    match pad {
938        0 => Ok(xs.clone()),
939        1 => {
940            let (_b_size, _c, h, w) = xs.dims4()?;
941            let (first, last) = (xs.narrow(3, 0, 1)?, xs.narrow(3, w - 1, 1)?);
942            let xs = Tensor::cat(&[&first, xs, &last], 3)?;
943            let (first, last) = (xs.narrow(2, 0, 1)?, xs.narrow(2, h - 1, 1)?);
944            Tensor::cat(&[&first, &xs, &last], 2)
945        }
946        n => candle::bail!("replication-pad with a size of {n} is not supported"),
947    }
948}

994    #[cfg(feature = "metal")]
995    fn metal_fwd(
996        &self,
997        q: &candle::MetalStorage,
998        q_l: &Layout,
999        k: &candle::MetalStorage,
1000        k_l: &Layout,
1001        v: &candle::MetalStorage,
1002        v_l: &Layout,
1003    ) -> Result<(candle::MetalStorage, Shape)> {
1004        use candle::backend::BackendStorage;
1005        use candle_metal_kernels::SdpaDType;
1006
1007        let device = q.device();
1008
1009        let out_dims = vec![q_l.dim(0)?, q_l.dim(1)?, q_l.dim(2)?, v_l.dim(3)?];
1010        let elem_count: usize = out_dims.iter().product();
1011
1012        let output = device.new_buffer(elem_count, q.dtype(), "sdpa_o")?;
1013
1014        // q,k must have matching emb dim
1015        if q_l.dim(D::Minus1)? != k_l.dim(D::Minus1)? {
1016            candle::bail!("`q` and `k` last dims must match");
1017        }
1018
1019        // k,v must have matching n kv heads
1020        if v_l.dim(D::Minus(3))? != k_l.dim(D::Minus(3))? {
1021            candle::bail!("`k` and `v` head dims must match");
1022        }
1023
1024        // n_heads % n_kv_heads == 0; n_heads >= 1, n_kv_heads >= 1.
1025        if q_l.dim(D::Minus(3))? % k_l.dim(D::Minus(3))? != 0 {
1026            candle::bail!("query `n_heads` must be a multiple of `n_kv_heads`");
1027        }
1028
1029        let k_head = k_l.dim(D::Minus1)?;
1030        let q_head = q_l.dim(D::Minus1)?;
1031        let q_seq = q_l.dim(2)?;
1032
1033        let mut implementation_supports_use_case = q_head == k_head;
1034        let supported_head_dim =
1035            q_head == 32 || q_head == 64 || q_head == 96 || q_head == 128 || q_head == 256;
1036
1037        const SDPA_FULL_THRESHOLD: usize = 2;
1038
1039        let supports_sdpa_full =
1040            q_seq >= SDPA_FULL_THRESHOLD && supported_head_dim && q_head == k_head;
1041        let supports_sdpa_vector = q_seq == 1 && supported_head_dim;
1042
1043        implementation_supports_use_case &= supports_sdpa_full || supports_sdpa_vector;
1044
1045        if !supported_head_dim {
1046            candle::bail!(
1047                "Meta SDPA does not support q head dim {q_head}: q dims {:?}, k dims {:?}, v dims {:?}.",
1048                q_l.dims(),
1049                k_l.dims(),
1050                v_l.dims()
1051            );
1052        }
1053        if !implementation_supports_use_case {
1054            candle::bail!(
1055                "Meta SDPA does not support q dims {:?}, k dims {:?}, v dims {:?}.",
1056                q_l.dims(),
1057                k_l.dims(),
1058                v_l.dims()
1059            );
1060        }
1061
1062        for t in [k.dtype(), v.dtype()] {
1063            if q.dtype() != t {
1064                candle::bail!("all q, k, v dtypes must match.");
1065            }
1066        }
1067
1068        let itype = match q.dtype() {
1069            DType::BF16 => SdpaDType::BF16,
1070            DType::F16 => SdpaDType::F16,
1071            DType::F32 => SdpaDType::F32,
1072            other => candle::bail!("unsupported sdpa type {other:?}"),
1073        };
1074
1075        let command_buffer = q.device().command_buffer()?;
1076        if supports_sdpa_vector {
1077            // Route to the 2 pass fused attention if the k seqlen is large.
1078            // https://github.com/ml-explore/mlx/pull/1597
1079            const TWO_PASS_K_THRESHOLD: usize = 1024;
1080            if k_l.dim(2)? >= TWO_PASS_K_THRESHOLD {
1081                let mut intermediate_shape = [
1082                    &out_dims[0..out_dims.len() - 2],
1083                    &[candle_metal_kernels::SDPA_2PASS_BLOCKS],
1084                    &[out_dims[out_dims.len() - 1]],
1085                ]
1086                .concat();
1087                let intermediate = device.new_buffer(
1088                    intermediate_shape.iter().product::<usize>(),
1089                    DType::F32,
1090                    "sdpa_2pass_intermediate",
1091                )?;
1092                let _ = intermediate_shape.pop().unwrap();
1093                let sums = device.new_buffer(
1094                    intermediate_shape.iter().product::<usize>(),
1095                    DType::F32,
1096                    "sdpa_2pass_sums",
1097                )?;
1098                let maxs = device.new_buffer(
1099                    intermediate_shape.iter().product::<usize>(),
1100                    DType::F32,
1101                    "sdpa_2pass_maxs",
1102                )?;
1103
1104                command_buffer.set_label("vector_attention");
1105                candle_metal_kernels::call_sdpa_vector_2pass(
1106                    q.device().device(),
1107                    &command_buffer,
1108                    q.device().kernels(),
1109                    q_l.start_offset(),
1110                    q_l.dims(),
1111                    q.buffer(),
1112                    k_l.start_offset(),
1113                    k_l.dims(),
1114                    k_l.stride(),
1115                    k.buffer(),
1116                    v_l.start_offset(),
1117                    v_l.stride(),
1118                    v.buffer(),
1119                    &output,
1120                    &intermediate,
1121                    &sums,
1122                    &maxs,
1123                    self.scale,
1124                    self.softcapping,
1125                    itype,
1126                )
1127                .map_err(candle::Error::wrap)?;
1128            } else {
1129                command_buffer.set_label("vector_attention");
1130                candle_metal_kernels::call_sdpa_vector(
1131                    q.device().device(),
1132                    &command_buffer,
1133                    q.device().kernels(),
1134                    q_l.start_offset(),
1135                    q_l.dims(),
1136                    q.buffer(),
1137                    k_l.start_offset(),
1138                    k_l.dims(),
1139                    k_l.stride(),
1140                    k.buffer(),
1141                    v_l.start_offset(),
1142                    v_l.stride(),
1143                    v.buffer(),
1144                    &output,
1145                    self.scale,
1146                    self.softcapping,
1147                    itype,
1148                )
1149                .map_err(candle::Error::wrap)?;
1150            }
1151        } else if supports_sdpa_full {
1152            if q_l.dim(2)? != k_l.dim(2)? {
1153                candle::bail!(
1154                    "query and key sequence length must be equal if using full metal sdpa"
1155                )
1156            }
1157
1158            command_buffer.set_label("full_attention");
1159            candle_metal_kernels::call_sdpa_full(
1160                q.device().device(),
1161                &command_buffer,
1162                q.device().kernels(),
1163                q_l.start_offset(),
1164                q_l.dims(),
1165                q.buffer(),
1166                k_l.start_offset(),
1167                k.buffer(),
1168                v_l.start_offset(),
1169                v.buffer(),
1170                &output,
1171                self.scale,
1172                self.softcapping,
1173                itype,
1174            )
1175            .map_err(candle::Error::wrap)?;
1176        } else {
1177            candle::bail!("must be vector or full sdpa kernel");
1178        }
1179
1180        let newstorage = candle::MetalStorage::new(output, device.clone(), elem_count, q.dtype());
1181        Ok((newstorage, Shape::from_dims(&out_dims)))
1182    }

1185/// Scaled dot product attention with a fused kernel.
1186///
1187/// Computes softmax(qk^T*scale)v.
1188///
1189/// **Inputs shapes:**
1190/// - `q`: (bs, qhead, seq, hidden)
1191/// - `k`: (bs, kv_head, kv_seq, hidden)
1192/// - `k`: (bs, kv_head, kv_seq, v_hidden)
1193/// - `scale` is applied before softmax.
1194/// - If `softcapping` != 1.0:
1195///      - Computation is: softmax(tanh(qk^T*scale/cap)*cap)v
1196///
1197/// **Output shape:** (bs, qhead, seq, v_hidden)
1198///
1199/// **Supported head dims:** 32, 64, 96, 128, 256.
1200///
1201/// ## On Metal:
1202/// - If `seq` == 1:
1203///     - Use a vectorized kernel
1204///     - Supports `seq` != `kv_seq` (cross attn. support)
1205///     - Supports GQA when `qhead` is a multiple of `kv_head`
1206/// - Otherwise:
1207///     - Use an alternate kernel
1208///     - Requires `seq` == `kv_seq`
1209///     - GQA is not supported (requires `qhead` == `kv_head`)
1210pub fn sdpa(q: &Tensor, k: &Tensor, v: &Tensor, scale: f32, softcapping: f32) -> Result<Tensor> {
1211    q.apply_op3_no_bwd(k, v, &Sdpa { scale, softcapping })
1212}
