1//! Linear layer
2//!
3//! This layer applies a linear transformation to the incoming data, `y = x@w.t() + b`.
4//! The bias is optional. The `forward` method can be used to apply the layer, it supports input
5//! with a batch dimension (so of shape `(b_sz, in_c)`) or without (of shape `(in_c,)`), the
6//! output has shape `(b_sz, out_c)` and `(out_c,)` respectively.
7//!
8//! ```rust
9//! use candle::{Tensor, Device::Cpu};
10//! use candle_nn::{Linear, Module};
11//! # fn main() -> candle::Result<()> {
12//!
13//! let w = Tensor::new(&[[1f32, 2.], [3., 4.], [5., 6.]], &Cpu)?;
14//! let layer = Linear::new(w, None); // Use no bias.
15//! let xs = Tensor::new(&[[10f32, 100.]], &Cpu)?;
16//! let ys = layer.forward(&xs)?;
17//! assert_eq!(ys.to_vec2::<f32>()?, &[[210.0, 430.0, 650.0]]);
18//! # Ok(()) }
19//! ```
20use candle::{Result, Tensor};
21
22#[derive(Clone, Debug)]
23pub struct Linear {
24    weight: Tensor,
25    bias: Option<Tensor>,
26}
27
28impl Linear {
29    pub fn new(weight: Tensor, bias: Option<Tensor>) -> Self {
30        Self { weight, bias }
31    }
32
33    pub fn weight(&self) -> &Tensor {
34        &self.weight
35    }
36
37    pub fn bias(&self) -> Option<&Tensor> {
38        self.bias.as_ref()
39    }
40}
41
42impl super::Module for Linear {
43    fn forward(&self, x: &Tensor) -> candle::Result<Tensor> {
44        let w = match *x.dims() {
45            [b1, b2, _, _] => self.weight.broadcast_left((b1, b2))?.t()?,
46            [bsize, _, _] => self.weight.broadcast_left(bsize)?.t()?,
47            _ => self.weight.t()?,
48        };
49        let x = x.matmul(&w)?;
50        match &self.bias {
51            None => Ok(x),
52            Some(bias) => x.broadcast_add(bias),
53        }
54    }
55}
56
57/// Create or initialize a new linear layer.
58///
59/// This uses some default names for weights and biases, namely `"weight"` and `"bias"`.
60pub fn linear(in_dim: usize, out_dim: usize, vb: crate::VarBuilder) -> Result<Linear> {
61    let init_ws = crate::init::DEFAULT_KAIMING_NORMAL;
62    let ws = vb.get_with_hints((out_dim, in_dim), "weight", init_ws)?;
63    let bound = 1. / (in_dim as f64).sqrt();
64    let init_bs = crate::Init::Uniform {
65        lo: -bound,
66        up: bound,
67    };
68    let bs = vb.get_with_hints(out_dim, "bias", init_bs)?;
69    Ok(Linear::new(ws, Some(bs)))
70}
71
72/// Create or initialize a new linear layer without biases.
73pub fn linear_no_bias(in_dim: usize, out_dim: usize, vb: crate::VarBuilder) -> Result<Linear> {
74    let init_ws = crate::init::DEFAULT_KAIMING_NORMAL;
75    let ws = vb.get_with_hints((out_dim, in_dim), "weight", init_ws)?;
76    Ok(Linear::new(ws, None))
77}
78
79pub fn linear_b(
80    in_dim: usize,
81    out_dim: usize,
82    bias: bool,
83    vb: crate::VarBuilder,
84) -> Result<Linear> {
85    if bias {
86        linear(in_dim, out_dim, vb)
87    } else {
88        linear_no_bias(in_dim, out_dim, vb)
89    }
