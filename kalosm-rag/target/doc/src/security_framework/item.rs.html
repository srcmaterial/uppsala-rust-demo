100/// Specifies the number of results returned by a search
101#[derive(Debug, Copy, Clone)]
102pub enum Limit {
103    /// Always return all results
104    All,
105
106    /// Return up to the specified number of results
107    Max(i64),
108}
109
110impl Limit {
111    #[inline]
112    fn to_value(self) -> CFType {
113        match self {
114            Self::All => unsafe { CFString::wrap_under_get_rule(kSecMatchLimitAll).into_CFType() },
115            Self::Max(l) => CFNumber::from(l).into_CFType(),
116        }
117    }
118}
119
120impl From<i64> for Limit {
121    #[inline]
122    fn from(limit: i64) -> Self {
123        Self::Max(limit)
124    }
125}

294    /// Search for objects.
295    pub fn search(&self) -> Result<Vec<SearchResult>> {
296        unsafe {
297            let mut params = vec![];
298
299            if let Some(ref keychains) = self.keychains {
300                params.push((
301                    CFString::wrap_under_get_rule(kSecMatchSearchList),
302                    keychains.as_CFType(),
303                ));
304            }
305
306            if let Some(class) = self.class {
307                params.push((CFString::wrap_under_get_rule(kSecClass), class.to_value()));
308            }
309
310            if let Some(case_insensitive) = self.case_insensitive {
311                params.push((
312                    CFString::wrap_under_get_rule(kSecMatchCaseInsensitive),
313                    CFBoolean::from(case_insensitive).as_CFType()
314                ));
315            }
316
317            if let Some(key_class) = self.key_class {
318                params.push((CFString::wrap_under_get_rule(kSecAttrKeyClass), key_class.to_value()));
319            }
320
321            if self.load_refs {
322                params.push((
323                    CFString::wrap_under_get_rule(kSecReturnRef),
324                    CFBoolean::true_value().into_CFType(),
325                ));
326            }
327
328            if self.load_attributes {
329                params.push((
330                    CFString::wrap_under_get_rule(kSecReturnAttributes),
331                    CFBoolean::true_value().into_CFType(),
332                ));
333            }
334
335            if self.load_data {
336                params.push((
337                    CFString::wrap_under_get_rule(kSecReturnData),
338                    CFBoolean::true_value().into_CFType(),
339                ));
340            }
341
342            if let Some(limit) = self.limit {
343                params.push((
344                    CFString::wrap_under_get_rule(kSecMatchLimit),
345                    limit.to_value(),
346                ));
347            }
348
349            if let Some(ref label) = self.label {
350                params.push((
351                    CFString::wrap_under_get_rule(kSecAttrLabel),
352                    label.as_CFType(),
353                ));
354            }
355
356            if let Some(ref trusted_only) = self.trusted_only {
357                params.push((
358                    CFString::wrap_under_get_rule(kSecMatchTrustedOnly),
359                    if *trusted_only { CFBoolean::true_value().into_CFType() } else { CFBoolean::false_value().into_CFType() },
360                ));
361            }
362
363            if let Some(ref service) = self.service {
364                params.push((
365                    CFString::wrap_under_get_rule(kSecAttrService),
366                    service.as_CFType(),
367                ));
368            }
369            
370            #[cfg(target_os = "macos")]
371            {
372                if let Some(ref subject) = self.subject {
373                    params.push((
374                        CFString::wrap_under_get_rule(kSecMatchSubjectWholeString),
375                        subject.as_CFType(),
376                    ));
377                }
378            }
379
380            if let Some(ref account) = self.account {
381                params.push((
382                    CFString::wrap_under_get_rule(kSecAttrAccount),
383                    account.as_CFType(),
384                ));
385            }
386
387            if let Some(ref access_group) = self.access_group {
388                params.push((
389                    CFString::wrap_under_get_rule(kSecAttrAccessGroup),
390                    access_group.as_CFType(),
391                ));
392            }
393
394            if let Some(ref pub_key_hash) = self.pub_key_hash {
395                params.push((
396                    CFString::wrap_under_get_rule(kSecAttrPublicKeyHash),
397                    pub_key_hash.as_CFType(),
398                ));
399            }
400
401            if let Some(ref app_label) = self.app_label {
402                params.push((
403                    CFString::wrap_under_get_rule(kSecAttrApplicationLabel),
404                    app_label.as_CFType(),
405                ));
406            }
407
408            let params = CFDictionary::from_CFType_pairs(&params);
409
410            let mut ret = ptr::null();
411            cvt(SecItemCopyMatching(params.as_concrete_TypeRef(), &mut ret))?;
412            if ret.is_null() {
413                //  SecItemCopyMatching returns NULL if no load_* was specified,
414                //  causing a segfault.
415                return Ok(vec![]);
416            }
417            let type_id = CFGetTypeID(ret);
418
419            let mut items = vec![];
420
421            if type_id == CFArray::<CFType>::type_id() {
422                let array: CFArray<CFType> = CFArray::wrap_under_create_rule(ret as *mut _);
423                for item in array.iter() {
424                    items.push(get_item(item.as_CFTypeRef()));
425                }
426            } else {
427                items.push(get_item(ret));
428                // This is a bit janky, but get_item uses wrap_under_get_rule
429                // which bumps the refcount but we want create semantics
430                CFRelease(ret);
431            }
432
433            Ok(items)
434        }
435    }
436}

438unsafe fn get_item(item: CFTypeRef) -> SearchResult {
439    let type_id = CFGetTypeID(item);
440
441    if type_id == CFData::type_id() {
442        let data = CFData::wrap_under_get_rule(item as *mut _);
443        let mut buf = Vec::new();
444        buf.extend_from_slice(data.bytes());
445        return SearchResult::Data(buf);
446    }
447
448    if type_id == CFDictionary::<*const u8, *const u8>::type_id() {
449        return SearchResult::Dict(CFDictionary::wrap_under_get_rule(item as *mut _));
450    }
451
452    #[cfg(target_os = "macos")]
453    {
454        use crate::os::macos::keychain_item::SecKeychainItem;
455        if type_id == SecKeychainItem::type_id() {
456            return SearchResult::Ref(Reference::KeychainItem(
457                SecKeychainItem::wrap_under_get_rule(item as *mut _),
458            ));
459        }
460    }
461
462    let reference = if type_id == SecCertificate::type_id() {
463        Reference::Certificate(SecCertificate::wrap_under_get_rule(item as *mut _))
464    } else if type_id == SecKey::type_id() {
465        Reference::Key(SecKey::wrap_under_get_rule(item as *mut _))
466    } else if type_id == SecIdentity::type_id() {
467        Reference::Identity(SecIdentity::wrap_under_get_rule(item as *mut _))
468    } else {
469        panic!("Got bad type from SecItemCopyMatching: {type_id}");
470    };
471
472    SearchResult::Ref(reference)
473}

475/// An enum including all objects whose references can be returned from a search.
476/// Note that generic _Keychain Items_, such as passwords and preferences, do
477/// not have specific object types; they are modeled using dictionaries and so
478/// are available directly as search results in variant `SearchResult::Dict`.
479#[derive(Debug)]
480pub enum Reference {
481    /// A `SecIdentity`.
482    Identity(SecIdentity),
483    /// A `SecCertificate`.
484    Certificate(SecCertificate),
485    /// A `SecKey`.
486    Key(SecKey),
487    /// A `SecKeychainItem`.
488    ///
489    /// Only defined on OSX
490    #[cfg(target_os = "macos")]
491    KeychainItem(crate::os::macos::keychain_item::SecKeychainItem),
492    #[doc(hidden)]
493    __NonExhaustive,
494}

496/// An individual search result.
497pub enum SearchResult {
498    /// A reference to the Security Framework object, if asked for.
499    Ref(Reference),
500    /// A dictionary of data about the Security Framework object, if asked for.
501    Dict(CFDictionary),
502    /// The Security Framework object as bytes, if asked for.
503    Data(Vec<u8>),
504    /// An unknown representation of the Security Framework object.
505    Other,
506}

508impl fmt::Debug for SearchResult {
509    #[cold]
510    fn fmt(&self, fmt: &mut fmt::Formatter<'_>) -> fmt::Result {
511        match *self {
512            Self::Ref(ref reference) => fmt
513                .debug_struct("SearchResult::Ref")
514                .field("reference", reference)
515                .finish(),
516            Self::Data(ref buf) => fmt
517                .debug_struct("SearchResult::Data")
518                .field("data", buf)
519                .finish(),
520            Self::Dict(_) => {
521                let mut debug = fmt.debug_struct("SearchResult::Dict");
522                for (k, v) in self.simplify_dict().unwrap() {
523                    debug.field(&k, &v);
524                }
525                debug.finish()
526            }
527            Self::Other => write!(fmt, "SearchResult::Other"),
528        }
529    }
530}

532impl SearchResult {
533    /// If the search result is a `CFDict`, simplify that to a
534    /// `HashMap<String, String>`. This transformation isn't
535    /// comprehensive, it only supports `CFString`, `CFDate`, and `CFData`
536    /// value types.
537    #[must_use]
538    pub fn simplify_dict(&self) -> Option<HashMap<String, String>> {
539        match *self {
540            Self::Dict(ref d) => unsafe {
541                let mut retmap = HashMap::new();
542                let (keys, values) = d.get_keys_and_values();
543                for (k, v) in keys.iter().zip(values.iter()) {
544                    let keycfstr = CFString::wrap_under_get_rule((*k).cast());
545                    let val: String = match CFGetTypeID(*v) {
546                        cfstring if cfstring == CFString::type_id() => {
547                            format!("{}", CFString::wrap_under_get_rule((*v).cast()))
548                        }
549                        cfdata if cfdata == CFData::type_id() => {
550                            let buf = CFData::wrap_under_get_rule((*v).cast());
551                            let mut vec = Vec::new();
552                            vec.extend_from_slice(buf.bytes());
553                            format!("{}", String::from_utf8_lossy(&vec))
554                        }
555                        cfdate if cfdate == CFDate::type_id() => format!(
556                            "{}",
557                            CFString::wrap_under_create_rule(CFCopyDescription(*v))
558                        ),
559                        _ => String::from("unknown"),
560                    };
561                    retmap.insert(format!("{keycfstr}"), val);
562                }
563                Some(retmap)
564            },
565            _ => None,
566        }
567    }
568}

636    /// Populates a `CFDictionary` to be passed to
637    pub fn to_dictionary(&self) -> CFDictionary {
638        let mut dict = CFMutableDictionary::from_CFType_pairs(&[]);
639
640        let class_opt = match &self.value {
641            ItemAddValue::Ref(ref_) => ref_.class(),
642            ItemAddValue::Data { class, .. } => Some(*class),
643        };
644        if let Some(class) = class_opt {
645            dict.add(&unsafe { kSecClass }.to_void(), &class.0.to_void());
646        }
647
648        let value_pair = match &self.value {
649            ItemAddValue::Ref(ref_) => (unsafe { kSecValueRef }.to_void(), ref_.ref_()),
650            ItemAddValue::Data { data, .. } => (unsafe { kSecValueData }.to_void(), data.to_void()),
651        };
652        dict.add(&value_pair.0, &value_pair.1);
653
654        if let Some(location) = &self.location {
655            match location {
656                #[cfg(any(feature = "OSX_10_15", target_os = "ios", target_os = "tvos", target_os = "watchos", target_os = "visionos"))]
657                Location::DataProtectionKeychain => {
658                    dict.add(
659                        &unsafe { kSecUseDataProtectionKeychain }.to_void(),
660                        &CFBoolean::true_value().to_void(),
661                    );
662                }
663                #[cfg(target_os = "macos")]
664                Location::DefaultFileKeychain => {}
665                #[cfg(target_os = "macos")]
666                Location::FileKeychain(keychain) => {
667                    dict.add(&unsafe { kSecUseKeychain }.to_void(), &keychain.to_void());
668                },
669            }
670        }
671        if let Some(account_name) = &self.account_name {
672            dict.add(&unsafe { kSecAttrAccount }.to_void(), &account_name.to_void());
673        }
674        if let Some(access_group) = &self.access_group {
675            dict.add(&unsafe { kSecAttrAccessGroup }.to_void(), &access_group.to_void());
676        }
677        if let Some(comment) = &self.comment {
678            dict.add(&unsafe { kSecAttrComment }.to_void(), &comment.to_void());
679        }
680        if let Some(description) = &self.description {
681            dict.add(&unsafe { kSecAttrDescription }.to_void(), &description.to_void());
682        }
683        if let Some(label) = &self.label {
684            dict.add(&unsafe { kSecAttrLabel }.to_void(), &label.to_void());
685        }
686        if let Some(service) = &self.service {
687            dict.add(&unsafe { kSecAttrService }.to_void(), &service.to_void());
688        }
689
690        dict.to_immutable()
691    }
692}
693
694/// Value of an item to add to the keychain.
695pub enum ItemAddValue {
696    /// Pass item by Ref (kSecValueRef)
697    Ref(AddRef),
698    /// Pass item by Data (kSecValueData)
699    Data {
700        /// The item class (kSecClass).
701        class: ItemClass,
702        /// The item data.
703        data: CFData,
704    },
705}
706
707/// Type of Ref to add to the keychain.
708pub enum AddRef {
709    /// `SecKey`
710    Key(SecKey),
711    /// `SecIdentity`
712    Identity(SecIdentity),
713    /// `SecCertificate`
714    Certificate(SecCertificate),
715}
716
717impl AddRef {
718    fn class(&self) -> Option<ItemClass> {
719        match self {
720            AddRef::Key(_) => Some(ItemClass::key()),
721            //  kSecClass should not be specified when adding a SecIdentityRef:
722            //  https://developer.apple.com/forums/thread/25751
723            AddRef::Identity(_) => None,
724            AddRef::Certificate(_) => Some(ItemClass::certificate()),
725        }
726    }
727    fn ref_(&self) -> CFTypeRef {
728        match self {
729            AddRef::Key(key) => key.as_CFTypeRef(),
730            AddRef::Identity(id) => id.as_CFTypeRef(),
731            AddRef::Certificate(cert) => cert.as_CFTypeRef(),
732        }
733    }
734}

736/// Which keychain to add an item to.
737///
738/// <https://developer.apple.com/documentation/technotes/tn3137-on-mac-keychains>
739pub enum Location {
740    /// Store the item in the newer DataProtectionKeychain. This is the only
741    /// keychain on iOS. On macOS, this is the newer and more consistent
742    /// keychain implementation. Keys stored in the Secure Enclave _must_ use
743    /// this keychain.
744    ///
745    /// This keychain requires the calling binary to be codesigned with
746    /// entitlements for the KeychainAccessGroups it is supposed to
747    /// access.
748    #[cfg(any(feature = "OSX_10_15", target_os = "ios", target_os = "tvos", target_os = "watchos", target_os = "visionos"))]
749    DataProtectionKeychain,
750    /// Store the key in the default file-based keychain. On macOS, defaults to
751    /// the Login keychain.
752    #[cfg(target_os = "macos")]
753    DefaultFileKeychain,
754    /// Store the key in a specific file-based keychain.
755    #[cfg(target_os = "macos")]
756    FileKeychain(crate::os::macos::keychain::SecKeychain),
757}

759/// Translates to `SecItemAdd`. Use `ItemAddOptions` to build an `add_params`
760/// `CFDictionary`.
761pub fn add_item(add_params: CFDictionary) -> Result<()> {
762    cvt(unsafe { SecItemAdd(add_params.as_concrete_TypeRef(), std::ptr::null_mut()) })
763}
