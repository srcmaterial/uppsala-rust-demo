30bitflags::bitflags! {
31    /// The flags used to specify revocation policy options.
32    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
33    pub struct RevocationPolicy: CFOptionFlags {
34        /// Perform revocation checking using OCSP (Online Certificate Status Protocol).
35        const OCSP_METHOD = kSecRevocationOCSPMethod;
36        /// Perform revocation checking using the CRL (Certification Revocation List) method.
37        const CRL_METHOD = kSecRevocationCRLMethod;
38        /// Prefer CRL revocation checking over OCSP; by default, OCSP is preferred.
39        const PREFER_CRL = kSecRevocationPreferCRL;
40        /// Require a positive response to pass the policy.
41        const REQUIRE_POSITIVE_RESPONSE = kSecRevocationRequirePositiveResponse;
42        /// Consult only locally cached replies; do not use network access.
43        const NETWORK_ACCESS_DISABLED = kSecRevocationNetworkAccessDisabled;
44        /// Perform either OCSP or CRL checking.
45        const USE_ANY_METHOD_AVAILABLE = kSecRevocationUseAnyAvailableMethod;
46    }
47}

49impl SecPolicy {
50    /// Creates a `SecPolicy` for evaluating SSL certificate chains.
51    ///
52    /// The side which you are evaluating should be provided (i.e. pass `SslSslProtocolSide::SERVER` if
53    /// you are a client looking to validate a server's certificate chain).
54    pub fn create_ssl(protocol_side: SslProtocolSide, hostname: Option<&str>) -> Self {
55        let hostname = hostname.map(CFString::new);
56        let hostname = hostname
57            .as_ref()
58            .map(|s| s.as_concrete_TypeRef())
59            .unwrap_or(ptr::null_mut());
60        let is_server = protocol_side == SslProtocolSide::SERVER;
61        unsafe {
62            let policy = SecPolicyCreateSSL(is_server as _, hostname);
63            Self::wrap_under_create_rule(policy)
64        }
65    }
66
67    /// Creates a `SecPolicy` for checking revocation of certificates.
68    ///
69    /// If you do not specify this policy creating a `SecTrust` object, the system defaults
70    /// will be used during evaluation.
71    pub fn create_revocation(options: RevocationPolicy) -> crate::Result<Self> {
72        let policy = unsafe { SecPolicyCreateRevocation(options.bits()) };
73
74        if policy.is_null() {
75            Err(Error::from_code(errSecParam))
76        } else {
77            Ok(unsafe { Self::wrap_under_create_rule(policy) })
78        }
79    }
80
81    /// Returns a policy object for the default X.509 policy.
82    #[must_use]
83    pub fn create_x509() -> Self {
84        unsafe {
85            let policy = SecPolicyCreateBasicX509();
86            Self::wrap_under_create_rule(policy)
87        }
88    }
89}
