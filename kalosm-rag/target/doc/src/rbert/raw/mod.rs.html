26#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize)]
27#[serde(rename_all = "lowercase")]
28pub enum HiddenAct {
29    Gelu,
30    GeluApproximate,
31    Relu,
32}
33
34struct HiddenActLayer {
35    act: HiddenAct,
36    span: tracing::Span,
37}
38
39impl HiddenActLayer {
40    fn new(act: HiddenAct) -> Self {
41        let span = tracing::span!(tracing::Level::TRACE, "hidden-act");
42        Self { act, span }
43    }
44
45    fn forward(&self, xs: &Tensor) -> candle_core::Result<Tensor> {
46        let _enter = self.span.enter();
47        match self.act {
48            // https://github.com/huggingface/transformers/blob/cd4584e3c809bb9e1392ccd3fe38b40daba5519a/src/transformers/activations.py#L213
49            HiddenAct::Gelu => xs.gelu_erf(),
50            HiddenAct::GeluApproximate => xs.gelu(),
51            HiddenAct::Relu => xs.relu(),
52        }
53    }
54}

87/// A raw synchronous Bert model. You should generally use the [`super::Bert`] instead.
88// https://github.com/huggingface/transformers/blob/6eedfa6dd15dc1e22a55ae036f681914e5a0d9a1/src/transformers/models/bert/modeling_bert.py#L874
89pub struct BertModel {
90    embeddings: BertEmbeddings,
91    encoder: BertEncoder,
92    pub(crate) device: Device,
93    span: tracing::Span,
94}
95
96impl BertModel {
97    /// Load a new [`BertModel`] from [`VarBuilder`] with a [`Config`].
98    pub fn load(vb: VarBuilder, config: &Config) -> Result<Self> {
99        let (embeddings, encoder) = match (
100            BertEmbeddings::load(vb.pp("embeddings"), config),
101            BertEncoder::load(vb.pp("encoder"), config),
102        ) {
103            (Ok(embeddings), Ok(encoder)) => (embeddings, encoder),
104            (Err(err), _) | (_, Err(err)) => {
105                if let Some(model_type) = &config.model_type {
106                    if let (Ok(embeddings), Ok(encoder)) = (
107                        BertEmbeddings::load(vb.pp(format!("{model_type}.embeddings")), config),
108                        BertEncoder::load(vb.pp(format!("{model_type}.encoder")), config),
109                    ) {
110                        (embeddings, encoder)
111                    } else {
112                        return Err(err);
113                    }
114                } else {
115                    return Err(err);
116                }
117            }
118        };
119        Ok(Self {
120            embeddings,
121            encoder,
122            device: vb.device().clone(),
123            span: tracing::span!(tracing::Level::TRACE, "model"),
124        })
125    }
126
127    /// Run the bert model with a batch of inputs.
128    ///
129    /// input_ids: The token ids of the input.
130    /// token_type_ids: The token type ids of the input. (this should be a tensor of 0s for embedding tasks)
131    /// attention_mask: The attention mask of the input. This can be None for embedding tasks with a single sentence. If you pad the input with 0s, you will need to create an attention mask.
132    /// train: Whether to run the model in training mode or not.
133    pub fn forward(
134        &self,
135        input_ids: &Tensor,
136        token_type_ids: &Tensor,
137        attention_mask: Option<&Tensor>,
138        train: bool,
139    ) -> Result<Tensor> {
140        let _enter = self.span.enter();
141        let embedding_output = self.embeddings.forward(input_ids, token_type_ids, train)?;
142        let sequence_output = self
143            .encoder
144            .forward(&embedding_output, attention_mask, train)?;
145        Ok(sequence_output)
146    }
147
148    pub(crate) fn max_seq_len(&self) -> usize {
149        self.embeddings.max_seq_len()
150    }
151
152    pub(crate) fn embedding_dim(&self) -> usize {
153        self.embeddings.embedding_dim()
154    }
155}
