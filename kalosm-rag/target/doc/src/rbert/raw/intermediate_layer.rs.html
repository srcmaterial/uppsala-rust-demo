1use candle_core::{Result, Tensor};
2use candle_nn::{Module, VarBuilder};
3use candle_transformers::models::with_tracing::{linear, Linear};
4
5use super::HiddenActLayer;
6
7// https://github.com/huggingface/transformers/blob/6eedfa6dd15dc1e22a55ae036f681914e5a0d9a1/src/transformers/models/bert/modeling_bert.py#L441
8pub(crate) struct BertIntermediate {
9    dense: Linear,
10    intermediate_act: HiddenActLayer,
11    span: tracing::Span,
12}
13
14impl BertIntermediate {
15    pub(crate) fn load(vb: VarBuilder, config: &super::Config) -> Result<Self> {
16        let dense = linear(config.hidden_size, config.intermediate_size, vb.pp("dense"))?;
17        Ok(Self {
18            dense,
19            intermediate_act: HiddenActLayer::new(config.hidden_act),
20            span: tracing::span!(tracing::Level::TRACE, "inter"),
21        })
22    }
23}
24
25impl Module for BertIntermediate {
26    fn forward(&self, hidden_states: &Tensor) -> Result<Tensor> {
27        let _enter = self.span.enter();
28        let hidden_states = self.dense.forward(hidden_states)?;
29        let ys = self.intermediate_act.forward(&hidden_states)?;
30        Ok(ys)
31    }
32}
