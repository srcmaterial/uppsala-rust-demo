1//! Calculates the embeddings for a given input.
2//!
3//! Bert embeddings contain word embeddings, embeddings about the token type and position information.
4
5use candle_core::{Result, Tensor};
6use candle_nn::Dropout;
7use candle_nn::{embedding, Embedding, Module, ModuleT, VarBuilder};
8use candle_transformers::models::with_tracing::{layer_norm, LayerNorm};
9
10// https://github.com/huggingface/transformers/blob/6eedfa6dd15dc1e22a55ae036f681914e5a0d9a1/src/transformers/models/bert/modeling_bert.py#L180
11pub(crate) struct BertEmbeddings {
12    word_embeddings: Embedding,
13    position_embeddings: Option<Embedding>,
14    token_type_embeddings: Embedding,
15    layer_norm: LayerNorm,
16    dropout: Dropout,
17    span: tracing::Span,
18}
19
20impl BertEmbeddings {
21    pub(crate) fn load(vb: VarBuilder, config: &super::Config) -> Result<Self> {
22        let word_embeddings = embedding(
23            config.vocab_size,
24            config.hidden_size,
25            vb.pp("word_embeddings"),
26        )?;
27        let position_embeddings = embedding(
28            config.max_position_embeddings,
29            config.hidden_size,
30            vb.pp("position_embeddings"),
31        )?;
32        let token_type_embeddings = embedding(
33            config.type_vocab_size,
34            config.hidden_size,
35            vb.pp("token_type_embeddings"),
36        )?;
37        let layer_norm = layer_norm(
38            config.hidden_size,
39            config.layer_norm_eps,
40            vb.pp("LayerNorm"),
41        )?;
42        Ok(Self {
43            word_embeddings,
44            position_embeddings: Some(position_embeddings),
45            token_type_embeddings,
46            layer_norm,
47            dropout: Dropout::new(config.hidden_dropout_prob),
48            span: tracing::span!(tracing::Level::TRACE, "embeddings"),
49        })
50    }
51
52    pub(crate) fn forward(
53        &self,
54        input_ids: &Tensor,
55        token_type_ids: &Tensor,
56        train: bool,
57    ) -> Result<Tensor> {
58        let _enter = self.span.enter();
59        let (_bsize, seq_len) = input_ids.dims2()?;
60        let input_embeddings = self.word_embeddings.forward(input_ids)?;
61        let token_type_embeddings = self.token_type_embeddings.forward(token_type_ids)?;
62        let mut embeddings = (&input_embeddings + token_type_embeddings)?;
63        if let Some(position_embeddings) = &self.position_embeddings {
64            let position_ids = Tensor::arange(0, seq_len as u32, input_ids.device())?;
65            embeddings = embeddings.broadcast_add(&position_embeddings.forward(&position_ids)?)?
66        }
67        let embeddings = self.layer_norm.forward(&embeddings)?;
68        let embeddings = self.dropout.forward_t(&embeddings, train)?;
69        Ok(embeddings)
70    }
71
72    pub(crate) fn embedding_dim(&self) -> usize {
73        self.word_embeddings.hidden_size()
74    }
75
76    pub(crate) fn max_seq_len(&self) -> usize {
77        self.position_embeddings
78            .as_ref()
79            .map(|p| p.embeddings().dims()[0])
80            .unwrap_or(0)
81    }
82}
