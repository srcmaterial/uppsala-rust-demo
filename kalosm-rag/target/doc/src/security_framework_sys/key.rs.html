104macro_rules! names {
105    ($( $(# $meta:literal )* $i:ident => $x:ident),*) => {
106        extern "C" {
107            $($(#[cfg(feature = $meta)])* pub static $x: SecKeyAlgorithm;)*
108        }
109
110        #[non_exhaustive]
111        #[derive(Copy, Clone)]
112        pub enum Algorithm {
113            $( $(#[cfg(feature = $meta)])* $i, )*
114        }
115
116        impl From<Algorithm> for SecKeyAlgorithm {
117            fn from(m: Algorithm) -> Self {
118                unsafe { match m {
119                    $( $(#[cfg(feature = $meta)])* Algorithm::$i => $x, )*
120                } }
121            }
122        }
123    }
124}
